79      @PostConstruct
80      public void init() {
81          this.searchFilter = '(' + this.ldapServiceMapper.getIdAttribute() +  "={0})";
82          this.loadFilter = "(objectClass=" + this.ldapServiceMapper.getObjectClass() + ')';
83      }

85      @Override
86      public RegisteredService save(final RegisteredService rs) {
87          if (rs.getId() != RegisteredService.INITIAL_IDENTIFIER_VALUE) {
88              return update(rs);
89          }
90  
91          Connection connection = null;
92          try {
93              connection = this.connectionFactory.getConnection();
94              final AddOperation operation = new AddOperation(connection);
95  
96              final LdapEntry entry = this.ldapServiceMapper.mapFromRegisteredService(this.searchRequest.getBaseDn(), rs);
97              operation.execute(new AddRequest(entry.getDn(), entry.getAttributes()));
98          } catch (final LdapException e) {
99              logger.error(e.getMessage(), e);
100         } finally {
101             LdapUtils.closeConnection(connection);
102         }
103         return rs;
104     }
105 
106     private RegisteredService update(final RegisteredService rs) {
107         Connection searchConnection = null;
108         try {
109             searchConnection = this.connectionFactory.getConnection();
110             final Response<SearchResult> response = searchForServiceById(searchConnection, rs.getId());
111             if (hasResults(response)) {
112                 final String currentDn = response.getResult().getEntry().getDn();
113 
114                 Connection modifyConnection = null;
115                 try {
116                     modifyConnection = this.connectionFactory.getConnection();
117                     final ModifyOperation operation = new ModifyOperation(searchConnection);
118 
119                     final List<AttributeModification> mods = new ArrayList<AttributeModification>();
120 
121                     final LdapEntry entry = this.ldapServiceMapper.mapFromRegisteredService(this.searchRequest.getBaseDn(), rs);
122                     for (final LdapAttribute attr : entry.getAttributes()) {
123                         mods.add(new AttributeModification(AttributeModificationType.REPLACE, attr));
124                     }
125                     final ModifyRequest request = new ModifyRequest(currentDn, mods.toArray(new AttributeModification[] {}));
126                     operation.execute(request);
127                 } finally {
128                     LdapUtils.closeConnection(modifyConnection);
129                 }
130             }
131         } catch (final LdapException e) {
132             logger.error(e.getMessage(), e);
133         } finally {
134             LdapUtils.closeConnection(searchConnection);
135         }
136         return rs;
137     }

139     @Override
140     public boolean delete(final RegisteredService registeredService) {
141         Connection connection = null;
142         try {
143             connection = this.connectionFactory.getConnection();
144 
145             final Response<SearchResult> response = searchForServiceById(connection, registeredService.getId());
146             if (hasResults(response)) {
147                 final LdapEntry entry = response.getResult().getEntry();
148                 final DeleteOperation delete = new DeleteOperation(connection);
149                 final DeleteRequest request = new DeleteRequest(entry.getDn());
150                 final Response<Void> res = delete.execute(request);
151                 return res.getResultCode() == ResultCode.SUCCESS;
152             }
153         } catch (final LdapException e) {
154             logger.error(e.getMessage(), e);
155         } finally {
156             LdapUtils.closeConnection(connection);
157         }
158 
159         return false;
160     }

162     @Override
163     public List<RegisteredService> load() {
164         Connection connection = null;
165         final List<RegisteredService> list = new LinkedList<RegisteredService>();
166         try {
167             connection = this.connectionFactory.getConnection();
168             final Response<SearchResult> response =
169                     executeSearchOperation(connection, new SearchFilter(this.loadFilter));
170             if (hasResults(response)) {
171                 for (final LdapEntry entry : response.getResult().getEntries()) {
172                     final RegisteredService svc = this.ldapServiceMapper.mapToRegisteredService(entry);
173                     list.add(svc);
174                 }
175             }
176         } catch (final LdapException e) {
177             logger.error(e.getMessage(), e);
178         } finally {
179             LdapUtils.closeConnection(connection);
180         }
181         return list;
182     }

184     @Override
185     public RegisteredService findServiceById(final long id) {
186         Connection connection = null;
187         try {
188             connection = this.connectionFactory.getConnection();
189 
190             final Response<SearchResult> response = searchForServiceById(connection, id);
191             if (hasResults(response)) {
192                 return this.ldapServiceMapper.mapToRegisteredService(response.getResult().getEntry());
193             }
194         } catch (final LdapException e) {
195             logger.error(e.getMessage(), e);
196         } finally {
197             LdapUtils.closeConnection(connection);
198         }
199 
200         return null;
201     }

203     private Response<SearchResult> searchForServiceById(final Connection connection, final long id)
204             throws LdapException {
205 
206         final SearchFilter filter = new SearchFilter(this.searchFilter);
207         filter.setParameter(0, id);
208         return executeSearchOperation(connection, filter);
209     }
210 
211     private Response<SearchResult> executeSearchOperation(final Connection connection, final SearchFilter filter)
212             throws LdapException {
213 
214         final SearchOperation searchOperation = new SearchOperation(connection);
215         final SearchRequest request = newRequest(filter);
216         logger.debug("Using search request {}", request.toString());
217         return searchOperation.execute(request);
218     }

232     private boolean hasResults(final Response<SearchResult> response) {
233         return response.getResult() != null && response.getResult().getEntry() != null;
234     }

236     private SearchRequest newRequest(final SearchFilter filter) {
237         final SearchRequest sr = new SearchRequest(this.searchRequest.getBaseDn(), filter);
238         sr.setBinaryAttributes(this.searchRequest.getBinaryAttributes());
239         sr.setDerefAliases(this.searchRequest.getDerefAliases());
240         sr.setSearchEntryHandlers(this.searchRequest.getSearchEntryHandlers());
241         sr.setSearchReferenceHandlers(this.searchRequest.getSearchReferenceHandlers());
242         sr.setFollowReferrals(this.searchRequest.getFollowReferrals());
243         sr.setReturnAttributes(this.searchRequest.getReturnAttributes());
244         sr.setSearchScope(this.searchRequest.getSearchScope());
245         sr.setSizeLimit(this.searchRequest.getSizeLimit());
246         sr.setSortBehavior(this.searchRequest.getSortBehavior());
247         sr.setTimeLimit(this.searchRequest.getTimeLimit());
248         sr.setTypesOnly(this.searchRequest.getTypesOnly());
249         sr.setControls(this.searchRequest.getControls());
250         return sr;
