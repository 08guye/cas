35  /**
36   * Checks if the remote address is in the range of allowed addresses.
37   *
38   * @author David Harrison
39   * @author Scott Battaglia
40   * @since 3.2.1
41   *
42   */
43  public final class RemoteAddressAuthenticationHandler extends AbstractAuthenticationHandler {
44  
45      private final Logger logger = LoggerFactory.getLogger(getClass());
46  
47      /** The network netmask. */
48      @NotNull
49      private InetAddress inetNetmask = null;
50  
51      /** The network base address. */
52      @NotNull
53      private InetAddress inetNetwork = null;
54  
55      @Override
56      public HandlerResult authenticate(final Credential credential) throws GeneralSecurityException {
57          final RemoteAddressCredential c = (RemoteAddressCredential) credential;
58          try {
59              final InetAddress inetAddress = InetAddress.getByName(c.getRemoteAddress().trim());
60              if (containsAddress(this.inetNetwork, this.inetNetmask, inetAddress)) {
61                  return new HandlerResult(this, c, new SimplePrincipal(c.getId()));
62              }
63          } catch (final UnknownHostException e) {
64              logger.debug("Unknown host {}", c.getRemoteAddress());
65          }
66          throw new FailedLoginException(c.getRemoteAddress() + " not in allowed range.");
67      }
68  
69      @Override
70      public boolean supports(final Credential credential) {
71          return credential instanceof RemoteAddressCredential;
72      }
73  
74      /**
75       * The following code is from the Apache Software Foundations's Lenya project
76       * See InetAddressUtil.java
77       * Distributed under the Apache 2.0 software license
78       */
79  
80      /**
81       * Checks if a subnet contains a specific IP address.
82       *
83       * @param network The network address.
84       * @param netmask The subnet mask.
85       * @param ip The IP address to check.
86       * @return A boolean value.
87       */
88      private boolean containsAddress(final InetAddress network, final InetAddress netmask, final InetAddress ip) {
89          logger.debug("Checking IP address: {} in ", ip, network, netmask);
90  
91          byte[] networkBytes = network.getAddress();
92          byte[] netmaskBytes = netmask.getAddress();
93          byte[] ipBytes = ip.getAddress();
94  
95          /* check IPv4/v6-compatibility or parameters: */
96          if(networkBytes.length != netmaskBytes.length
97                  || netmaskBytes.length != ipBytes.length) {
98              logger.debug("Network address {}, subnet mask {} and/or host address {}"
99                      + " have different sizes! (return false ...)", network, netmask, ip);
100             return false;
101         }
102 
103         /* Check if the masked network and ip addresses match: */
104         for(int i=0; i<netmaskBytes.length; i++) {
105             int mask = netmaskBytes[i] & 0xff;
106             if((networkBytes[i] & mask) != (ipBytes[i] & mask)) {
107                 logger.debug("{} is not in {}/{}", ip, network, netmask);
108                 return false;
109             }
110         }
111         logger.debug("{} is in {}/{}", ip, network, netmask);
112         return true;
113     }
114 
115     /**
116      * @param ipAddressRange the IP address range that should be allowed trusted logins     *
117      */
118     public void setIpNetworkRange(final String ipAddressRange) {
119 
120         if(ipAddressRange != null) {
121 
122             final String[] splitAddress = ipAddressRange.split("/");
123 
124             if (splitAddress.length == 2) {
125                 // A valid ip address/netmask was supplied parse values
126                 final String network = splitAddress[0].trim();
127                 final String netmask = splitAddress[1].trim();
128 
129                 try {
130                     this.inetNetwork = InetAddress.getByName(network);
131                     logger.debug("InetAddress network: {}", this.inetNetwork.toString());
132                 } catch (final UnknownHostException e) {
133                     logger.error("The network address was not valid: {}", e.getMessage());
134                 }
135 
136                 try {
137                     this.inetNetmask = InetAddress.getByName(netmask);
138                     logger.debug("InetAddress netmask: {}", this.inetNetmask.toString());
139                 } catch (final UnknownHostException e) {
140                     logger.error("The network netmask was not valid: {}", e.getMessage());
141                 }
142             }
143         }
144     }
