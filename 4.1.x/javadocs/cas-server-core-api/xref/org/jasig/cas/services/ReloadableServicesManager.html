19  package org.jasig.cas.services;
20  
21  /**
22   * Interface to allow for ServicesManagers to attempt to reload their list of
23   * services.
24   *
25   * @author Scott Battaglia
26   * @since 3.1
27   */
28  public interface ReloadableServicesManager extends ServicesManager {
29  
30      /**
31       * Inform the ServicesManager to reload its list of services if its cached
32       * them. Note that this is a suggestion and that ServicesManagers are free
33       * to reload whenever they want.
34       */
35      void reload();
36  
37  }
