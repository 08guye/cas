45  /**
46   * Implementation of a RadiusServer that utilizes the JRadius packages available
47   * at <a href="http://jradius.sf.net">http://jradius.sf.net</a>.
48   *
49   * @author Scott Battaglia
50   * @author Marvin S. Addison
51   * @author Misagh Moayyed
52   * @since 3.1

54  public final class JRadiusServerImpl implements RadiusServer {
55  
56      /**
57       * Default retry count, {@value}.
58       **/
59      public static final int DEFAULT_RETRY_COUNT = 3;
60  
61      /** Logger instance. */
62      private static final Logger LOGGER = LoggerFactory.getLogger(JRadiusServerImpl.class);
63  
64      /** RADIUS protocol. */
65      @NotNull
66      private final RadiusProtocol protocol;
67  
68      /** Produces RADIUS client instances for authentication. */
69      @NotNull
70      private final RadiusClientFactory radiusClientFactory;
71  
72      /** Number of times to retry authentication when no response is received. */
73      @Min(0)
74      private int retries = DEFAULT_RETRY_COUNT;
75  
76      private String nasIpAddress;
77      
78      private String nasIpv6Address;
79      
80      private long nasPort = -1;
81      
82      private long nasPortId = -1;
83      
84      private long nasIdentifier = -1;
85      
86      private long nasRealPort = -1;
87      
88      private long nasPortType = -1;
89      
90      
91      /** Load the dictionary implementation. */
92      static {
93          AttributeFactory
94          .loadAttributeDictionary("net.jradius.dictionary.AttributeDictionaryImpl");
95      }
96  
97      /**
98       * Instantiates a new server implementation
99       * with the radius protocol and client factory specified. 
100      *
101      * @param protocol the protocol
102      * @param clientFactory the client factory
103      */
104     public JRadiusServerImpl(final RadiusProtocol protocol, final RadiusClientFactory clientFactory) {
105         this.protocol = protocol;
106         this.radiusClientFactory = clientFactory;
107     }
108 
109     @Override
110     public RadiusResponse authenticate(final String username, final String password) throws PreventedException {
111 
112         final AttributeList attributeList = new AttributeList();
113         
114         attributeList.add(new Attr_UserName(username));
115         attributeList.add(new Attr_UserPassword(password));
116 
117         if (StringUtils.isNotBlank(this.nasIpAddress)) {
118             attributeList.add(new Attr_NASIPAddress(this.nasIpAddress));
119         }
120         if (StringUtils.isNotBlank(this.nasIpv6Address)) {
121             attributeList.add(new Attr_NASIPv6Address(this.nasIpv6Address));
122         }
123 
124         if (this.nasPort != -1) {
125             attributeList.add(new Attr_NASPort(this.nasPort));
126         }
127         if (this.nasPortId != -1) {
128             attributeList.add(new Attr_NASPortId(this.nasPortId));
129         }
130         if (this.nasIdentifier != -1) {
131             attributeList.add(new Attr_NASIdentifier(this.nasIdentifier));
132         }
133         if (this.nasRealPort != -1) {
134             attributeList.add(new Attr_NASRealPort(this.nasRealPort));
135         }
136         if (this.nasPortType != -1) {
137             attributeList.add(new Attr_NASPortType(this.nasPortType));
138         }
139         
140         RadiusClient client = null;
141         try {
142             client = this.radiusClientFactory.newInstance();
143             final AccessRequest request = new AccessRequest(client, attributeList);
144             final RadiusPacket response = client.authenticate(
145                     request,
146                     RadiusClient.getAuthProtocol(this.protocol.getName()),
147                     this.retries);
148 
149             LOGGER.debug("RADIUS response from {}: {}",
150                     client.getRemoteInetAddress().getCanonicalHostName(),
151                     response.getClass().getName());
152 
153             if (response instanceof AccessAccept) {
154                 final AccessAccept acceptedResponse = (AccessAccept) response;
155                
156                 return new RadiusResponse(acceptedResponse.getCode(),
157                         acceptedResponse.getIdentifier(),
158                         acceptedResponse.getAttributes().getAttributeList());
159             }
160         } catch (final Exception e) {
161             throw new PreventedException(e);            
162         } finally {
163             if (client != null) {
164                 client.close();
165             }
166         }
167         return null;
168     }
