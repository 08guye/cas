19  package org.jasig.cas.adaptors.x509.authentication.handler.support;
20  
21  import org.jasig.cas.adaptors.x509.util.CertUtils;
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  
25  import javax.validation.constraints.Min;
26  import java.security.GeneralSecurityException;
27  import java.security.cert.X509CRL;
28  import java.util.Calendar;
29  
30  
31  /**
32   * Implements a policy to handle expired CRL data whereby expired data is permitted
33   * up to a threshold period of time but not afterward.
34   *
35   * @author Marvin S. Addison
36   * @since 3.4.6
37   *
38   */
39  public final class ThresholdExpiredCRLRevocationPolicy implements RevocationPolicy<X509CRL> {
40      /** Default threshold is 48 hours. */
41      private static final int DEFAULT_THRESHOLD = 172800;
42  
43      /** Logger instance. */
44      private final Logger logger = LoggerFactory.getLogger(getClass());
45  
46  
47      /** Expired threshold period in seconds. */
48      @Min(0)
49      private int threshold = DEFAULT_THRESHOLD;
50  
51  
52      /**
53       * {@inheritDoc}
54       * The CRL next update time is compared against the current time with the threshold
55       * applied and rejected if and only if the next update time is in the past.
56       *
57       * @param crl CRL instance to evaluate.
58       *
59       * @throws GeneralSecurityException On expired CRL data. Check the exception type for exact details
60       *
61       * @see org.jasig.cas.adaptors.x509.authentication.handler.support.RevocationPolicy#apply(java.lang.Object)
62       */
63      @Override
64      public void apply(final X509CRL crl) throws GeneralSecurityException {
65          final Calendar cutoff = Calendar.getInstance();
66          if (CertUtils.isExpired(crl, cutoff.getTime())) {
67              cutoff.add(Calendar.SECOND, -this.threshold);
68              if (CertUtils.isExpired(crl, cutoff.getTime())) {
69                  throw new ExpiredCRLException(crl.toString(), cutoff.getTime(), this.threshold);
70              }
71              logger.info(String.format("CRL expired on %s but is within threshold period, %s seconds.",
72                          crl.getNextUpdate(), this.threshold));
73          }
74      }
75  
76      /**
77       * Sets the threshold period of time after which expired CRL data is rejected.
78       *
79       * @param threshold Number of seconds; MUST be non-negative integer.
80       */
81      public void setThreshold(final int threshold) {
82          this.threshold = threshold;
83      }
84  }
