19  package org.jasig.cas.ticket.registry;
20  
21  import java.util.Collection;
22  
23  import org.jasig.cas.ticket.Ticket;
24  
25  /**
26   * Interface for a registry that stores tickets. The underlying registry can be
27   * backed by anything from a normal HashMap to JGroups for having distributed
28   * registries. It is up to specific implementations to determine their clean up
29   * strategy. Strategies can include a manual clean up by a RegistryCleaner or a
30   * more sophisticated strategy such as LRU.
31   *
32   * @author Scott Battaglia
33  
34   * @since 3.0.0
35   * <p>
36   * This is a published and supported CAS Server 3 API.
37   * </p>
38   */
39  public interface TicketRegistry {
40  
41      /**
42       * Add a ticket to the registry. Ticket storage is based on the ticket id.
43       *
44       * @param ticket The ticket we wish to add to the cache.
45       */
46      void addTicket(Ticket ticket);
47  
48      /**
49       * Retrieve a ticket from the registry. If the ticket retrieved does not
50       * match the expected class, an InvalidTicketException is thrown.
51       *
52       * @param ticketId the id of the ticket we wish to retrieve.
53       * @param clazz The expected class of the ticket we wish to retrieve.
54       * @param <T> the generic ticket type to return that extends {@link Ticket}
55       * @return the requested ticket.
56       */
57      <T extends Ticket> T getTicket(String ticketId, Class<? extends Ticket> clazz);
58  
59      /**
60       * Retrieve a ticket from the registry.
61       *
62       * @param ticketId the id of the ticket we wish to retrieve
63       * @return the requested ticket.
64       */
65      Ticket getTicket(String ticketId);
66  
67      /**
68       * Remove a specific ticket from the registry.
69       *
70       * @param ticketId The id of the ticket to delete.
71       * @return true if the ticket was removed and false if the ticket did not
72       * exist.
73       */
74      boolean deleteTicket(String ticketId);
75  
76      /**
77       * Retrieve all tickets from the registry.
78       *
79       * @return collection of tickets currently stored in the registry. Tickets
80       * might or might not be valid i.e. expired.
81       */
82      Collection<Ticket> getTickets();
83  }
