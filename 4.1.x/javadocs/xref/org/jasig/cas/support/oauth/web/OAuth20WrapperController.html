19  package org.jasig.cas.support.oauth.web;
20  
21  import org.apache.http.HttpStatus;
22  import org.jasig.cas.support.oauth.OAuthConstants;
23  import org.jasig.cas.support.oauth.OAuthUtils;
24  import org.springframework.beans.factory.InitializingBean;
25  import org.springframework.web.servlet.ModelAndView;
26  import org.springframework.web.servlet.mvc.AbstractController;
27  
28  import javax.servlet.http.HttpServletRequest;
29  import javax.servlet.http.HttpServletResponse;
30  
31  /**
32   * This controller is the main entry point for OAuth version 2.0
33   * wrapping in CAS, should be mapped to something like /oauth2.0/*. Dispatch
34   * request to specific controllers : authorize, accessToken...
35   *
36   * @author Jerome Leleu
37   * @since 3.5.0
38   */
39  public final class OAuth20WrapperController extends BaseOAuthWrapperController implements InitializingBean {
40  
41      private AbstractController authorizeController;
42  
43      private AbstractController callbackAuthorizeController;
44  
45      private AbstractController accessTokenController;
46  
47      private AbstractController profileController;
48  
49      @Override
50      public void afterPropertiesSet() throws Exception {
51          authorizeController = new OAuth20AuthorizeController(servicesManager, loginUrl);
52          callbackAuthorizeController = new OAuth20CallbackAuthorizeController();
53          accessTokenController = new OAuth20AccessTokenController(servicesManager, ticketRegistry, timeout);
54          profileController = new OAuth20ProfileController(ticketRegistry);
55      }
56  
57      @Override
58      protected ModelAndView internalHandleRequest(final String method, final HttpServletRequest request,
59              final HttpServletResponse response) throws Exception {
60  
61          // authorize
62          if (OAuthConstants.AUTHORIZE_URL.equals(method)) {
63              return authorizeController.handleRequest(request, response);
64          }
65          // callback on authorize
66          if (OAuthConstants.CALLBACK_AUTHORIZE_URL.equals(method)) {
67              return callbackAuthorizeController.handleRequest(request, response);
68          }
69          //get access token
70          if (OAuthConstants.ACCESS_TOKEN_URL.equals(method)) {
71              return accessTokenController.handleRequest(request, response);
72          }
73          // get profile
74          if (OAuthConstants.PROFILE_URL.equals(method)) {
75              return profileController.handleRequest(request, response);
76          }
77  
78          // else error
79          logger.error("Unknown method : {}", method);
80          OAuthUtils.writeTextError(response, OAuthConstants.INVALID_REQUEST, HttpStatus.SC_OK);
81          return null;
82      }
83  }
