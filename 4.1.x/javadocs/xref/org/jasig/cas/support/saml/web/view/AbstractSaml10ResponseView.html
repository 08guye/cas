19  package org.jasig.cas.support.saml.web.view;
20  
21  import org.jasig.cas.authentication.principal.WebApplicationService;
22  import org.jasig.cas.authentication.support.CasAttributeEncoder;
23  import org.jasig.cas.support.saml.util.Saml10ObjectBuilder;
24  import org.jasig.cas.support.saml.web.support.SamlArgumentExtractor;
25  import org.jasig.cas.web.view.AbstractCasView;
26  import org.joda.time.DateTime;
27  
28  import org.opensaml.saml.saml1.core.Response;
29  
30  import javax.servlet.http.HttpServletRequest;
31  import javax.servlet.http.HttpServletResponse;
32  import javax.validation.constraints.NotNull;
33  import java.util.Map;
34  
35  /**
36   * Base class for all views that render SAML1 SOAP messages directly to the HTTP response stream.
37   *
38   * @author Marvin S. Addison
39   * @since 3.5.1
40   */
41  public abstract class AbstractSaml10ResponseView extends AbstractCasView {
42      private static final String DEFAULT_ENCODING = "UTF-8";
43  
44      /**
45       * The Saml object builder.
46       */
47      protected final Saml10ObjectBuilder samlObjectBuilder = new Saml10ObjectBuilder();
48  
49      private final SamlArgumentExtractor samlArgumentExtractor = new SamlArgumentExtractor();
50  
51      @NotNull
52      private String encoding = DEFAULT_ENCODING;
53  
54      /** Defaults to 0. */
55      private int skewAllowance;
56  
57      /**
58       * Sets the character encoding in the HTTP response.
59       *
60       * @param encoding Response character encoding.
61       */
62      public void setEncoding(final String encoding) {
63          this.encoding = encoding;
64      }
65  
66      /**
67      * Sets the allowance for time skew in seconds
68      * between CAS and the client server.  Default 0s.
69      * This value will be subtracted from the current time when setting the SAML
70      * <code>NotBeforeDate</code> attribute, thereby allowing for the
71      * CAS server to be ahead of the client by as much as the value defined here.
72      *
73      * <p><strong>Note:</strong> Skewing of the issue instant via setting this property
74      * applies to all saml assertions that are issued by CAS and it
75      * currently cannot be controlled on a per relying party basis.
76      * Before configuring this, it is recommended that each service provider
77      * attempt to correctly sync their system time with an NTP server
78      * so as to match the CAS server's issue instant config and to
79      * avoid applying this setting globally. This should only
80      * be used in situations where the NTP server is unresponsive to
81      * sync time on the client, or the client is simply unable
82      * to adjust their server time configuration.</p>
83      *
84      * @param skewAllowance Number of seconds to allow for variance.
85      */
86      public void setSkewAllowance(final int skewAllowance) {
87          logger.debug("Using {} seconds as skew allowance.", skewAllowance);
88          this.skewAllowance = skewAllowance;
89      }
90  
91      @Override
92      protected void renderMergedOutputModel(
93              final Map<String, Object> model, final HttpServletRequest request, final HttpServletResponse response) throws Exception {
94  
95          response.setCharacterEncoding(this.encoding);
96  
97          final WebApplicationService service = this.samlArgumentExtractor.extractService(request);
98          final String serviceId = service != null ? service.getId() : "UNKNOWN";
99  
100         try {
101             final Response samlResponse = this.samlObjectBuilder.newResponse(
102                     this.samlObjectBuilder.generateSecureRandomId(),
103                     DateTime.now().minusSeconds(this.skewAllowance), serviceId, service);
104 
105             prepareResponse(samlResponse, model);
106 
107             this.samlObjectBuilder.encodeSamlResponse(response, request, samlResponse);
108         } catch (final Exception e) {
109             logger.error("Error generating SAML response for service {}.", serviceId);
110             throw e;
111         }
112     }
113 
114     /**
115      * Sets cas attribute encoder.
116      *
117      * @param casAttributeEncoder the cas attribute encoder
118      * @since 4.1
119      */
120     public void setCasAttributeEncoder(@NotNull final CasAttributeEncoder casAttributeEncoder) {
121         this.casAttributeEncoder = casAttributeEncoder;
122     }
123 
124     /**
125      * Subclasses must implement this method by adding child elements (status, assertion, etc) to
126      * the given empty SAML 1 response message.  Impelmenters need not be concerned with error handling.
127      *
128      * @param response SAML 1 response message to be filled.
129      * @param model Spring MVC model map containing data needed to prepare response.
130      */
131     protected abstract void prepareResponse(Response response, Map<String, Object> model);
132 
133 }
