20  package org.jasig.cas.support.spnego.util;
21  
22  import org.apache.commons.lang3.builder.ToStringBuilder;
23  import org.slf4j.Logger;
24  import org.slf4j.LoggerFactory;
25  
26  import java.net.InetAddress;
27  import java.net.UnknownHostException;
28  
29  /**
30   *  Utility class to perform DNS work in a threaded, timeout-able way
31   *  Adapted from: http://thushw.blogspot.com/2009/11/resolving-domain-names-quickly-with.html.
32   *
33   *  @author Sean Baker sean.baker@usuhs.edu
34   *  @author Misagh Moayyed
35   *  @since 4.1
36   */
37  public final class ReverseDNSRunnable implements Runnable {
38  
39      /** Logger instance. **/
40      private static final Logger LOGGER = LoggerFactory.getLogger(ReverseDNSRunnable.class);
41  
42      /** Remote user IP address. **/
43      private final String ipAddress;
44  
45      /** Remote user hostname. **/
46      private String hostName;
47  
48      /**
49       * Simple constructor which also pre-sets hostName attribute for failover situations.
50       * @param ipAddress the ip address on which reverse DNS will be done.
51       */
52      public ReverseDNSRunnable(final String ipAddress) {
53          this.ipAddress = ipAddress;
54          this.hostName = ipAddress;
55      }
56  
57      /**
58       * Runnable implementation to thread the work done in this class, allowing the
59       * implementer to set a thread timeout and thereby short-circuit the lookup.
60       */
61      @Override
62      public void run() {
63          try {
64              LOGGER.debug("Attempting to resolve {}", this.ipAddress);
65              final InetAddress address = InetAddress.getByName(this.ipAddress);
66              set(address.getCanonicalHostName());
67          } catch (final UnknownHostException e) {
68              /** N/A -- Default to IP address, but that's already done. **/
69              LOGGER.debug("Unable to identify the canonical hostname for ip address.", e);
70          }
71      }
72  
73      /**
74       * Glorified setter with logging.
75       * @param hostName the resolved hostname
76       */
77      public synchronized void set(final String hostName) {
78          LOGGER.trace("ReverseDNS -- Found hostName: {}.", hostName);
79          this.hostName = hostName;
80      }
81  
82      /**
83       * Getter method to provide result of lookup.
84       * @return the remote host name, or the IP address if name not found
85       */
86      public synchronized String get() {
87          return this.hostName;
88      }
89  
90      @Override
91      public String toString() {
92          return new ToStringBuilder(this)
93                  .append("ipAddress", this.ipAddress)
94                  .append("hostName", this.hostName)
95                  .toString();
96      }
97  }
