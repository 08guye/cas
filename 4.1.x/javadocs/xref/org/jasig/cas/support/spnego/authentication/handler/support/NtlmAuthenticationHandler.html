19  package org.jasig.cas.support.spnego.authentication.handler.support;
20  
21  import jcifs.Config;
22  import jcifs.UniAddress;
23  import jcifs.netbios.NbtAddress;
24  import jcifs.ntlmssp.Type1Message;
25  import jcifs.ntlmssp.Type2Message;
26  import jcifs.ntlmssp.Type3Message;
27  import jcifs.smb.NtlmPasswordAuthentication;
28  import jcifs.smb.SmbAuthException;
29  import jcifs.smb.SmbSession;
30  import org.jasig.cas.authentication.BasicCredentialMetaData;
31  import org.jasig.cas.authentication.Credential;
32  import org.jasig.cas.authentication.DefaultHandlerResult;
33  import org.jasig.cas.authentication.HandlerResult;
34  import org.jasig.cas.authentication.PreventedException;
35  import org.jasig.cas.authentication.handler.support.AbstractPreAndPostProcessingAuthenticationHandler;
36  import org.jasig.cas.support.spnego.authentication.principal.SpnegoCredential;
37  
38  import javax.security.auth.login.FailedLoginException;
39  import javax.validation.constraints.NotNull;
40  import java.security.GeneralSecurityException;
41  
42  /**
43   * Implementation of an AuthenticationHandler for NTLM supports.
44   *
45   * @author <a href="mailto:julien.henry@capgemini.com">Julien Henry</a>
46   * @author Scott Battaglia
47   * @author Arnaud Lesueur
48   * @since 3.1
49   */
50  
51  public class NtlmAuthenticationHandler extends AbstractPreAndPostProcessingAuthenticationHandler {
52  
53      private static final int NBT_ADDRESS_TYPE = 0x1C;
54      private static final int NTLM_TOKEN_TYPE_FIELD_INDEX = 8;
55      private static final int NTLM_TOKEN_TYPE_ONE = 1;
56      private static final int NTLM_TOKEN_TYPE_THREE = 3;
57      private boolean loadBalance = true;
58  
59      @NotNull
60      private String domainController = Config.getProperty("jcifs.smb.client.domain");
61  
62      private String includePattern;
63  
64      @Override
65      protected final HandlerResult doAuthentication(
66              final Credential credential) throws GeneralSecurityException, PreventedException {
67  
68          final SpnegoCredential ntlmCredential = (SpnegoCredential) credential;
69          final byte[] src = ntlmCredential.getInitToken();
70  
71          UniAddress dc = null;
72  
73          boolean success = false;
74          try {
75              if (this.loadBalance) {
76                  // find the first dc that matches the includepattern
77                  if (this.includePattern != null) {
78                      final NbtAddress[] dcs= NbtAddress.getAllByName(this.domainController, NBT_ADDRESS_TYPE, null, null);
79                      for (final NbtAddress dc2 : dcs) {
80                          if(dc2.getHostAddress().matches(this.includePattern)){
81                              dc = new UniAddress(dc2);
82                              break;
83                          }
84                      }
85                  } else {
86                      dc = new UniAddress(NbtAddress.getByName(this.domainController, NBT_ADDRESS_TYPE, null));
87                  }
88              } else {
89                  dc = UniAddress.getByName(this.domainController, true);
90              }
91              final byte[] challenge = SmbSession.getChallenge(dc);
92  
93              switch (src[NTLM_TOKEN_TYPE_FIELD_INDEX]) {
94                  case NTLM_TOKEN_TYPE_ONE:
95                      logger.debug("Type 1 received");
96                      final Type1Message type1 = new Type1Message(src);
97                      final Type2Message type2 = new Type2Message(type1,
98                              challenge, null);
99                      logger.debug("Type 2 returned. Setting next token.");
100                     ntlmCredential.setNextToken(type2.toByteArray());
101                     break;
102                 case NTLM_TOKEN_TYPE_THREE:
103                     logger.debug("Type 3 received");
104                     final Type3Message type3 = new Type3Message(src);
105                     final byte[] lmResponse = type3.getLMResponse() == null ? new byte[0] : type3.getLMResponse();
106                     final byte[] ntResponse = type3.getNTResponse() == null ? new byte[0] : type3.getNTResponse();
107                     final NtlmPasswordAuthentication ntlm = new NtlmPasswordAuthentication(
108                             type3.getDomain(), type3.getUser(), challenge,
109                             lmResponse, ntResponse);
110                     logger.debug("Trying to authenticate {} with domain controller", type3.getUser());
111                     try {
112                         SmbSession.logon(dc, ntlm);
113                         ntlmCredential.setPrincipal(this.principalFactory.createPrincipal(type3.getUser()));
114                         success = true;
115                     } catch (final SmbAuthException sae) {
116                         throw new FailedLoginException(sae.getMessage());
117                     }
118                     break;
119                 default:
120                     logger.debug("Unknown type: {}", src[NTLM_TOKEN_TYPE_FIELD_INDEX]);
121             }
122         } catch (final Exception e) {
123             throw new FailedLoginException(e.getMessage());
124         }
125 
126         if (!success) {
127             throw new FailedLoginException();
128         }
129         return new DefaultHandlerResult(this, new BasicCredentialMetaData(ntlmCredential), ntlmCredential.getPrincipal());
130     }
131 
132     @Override
133     public boolean supports(final Credential credential) {
134         return credential instanceof SpnegoCredential;
135     }
136 
137     public void setLoadBalance(final boolean loadBalance) {
138         this.loadBalance = loadBalance;
139     }
140 
141     public void setDomainController(final String domainController) {
142         this.domainController = domainController;
143     }
144 
145     public void setIncludePattern(final String includePattern) {
146         this.includePattern = includePattern;
147     }
148 
149 }
