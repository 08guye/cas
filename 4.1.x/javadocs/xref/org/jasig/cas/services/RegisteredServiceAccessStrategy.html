20  package org.jasig.cas.services;
21  
22  import java.io.Serializable;
23  import java.net.URI;
24  import java.util.Map;
25  
26  /**
27   * This is {@link RegisteredServiceAccessStrategy}
28   * that can decide if a service is recognized and authorized to participate
29   * in the CAS protocol flow during authentication/validation events.
30   *
31   * @author Misagh Moayyed mmoayyed@unicon.net
32   * @since 4.1
33   */
34  public interface RegisteredServiceAccessStrategy extends Serializable {
35  
36      /**
37       * Verify is the service is enabled and recognized by CAS.
38       *
39       * @return true/false if service is enabled
40       */
41      boolean isServiceAccessAllowed();
42  
43      /**
44       * Assert that the service can participate in sso.
45       *
46       * @return true/false if service can participate in sso
47       */
48      boolean isServiceAccessAllowedForSso();
49  
50      /**
51       * Verify authorization policy by checking the pre-configured rules
52       * that may depend on what the principal might be carrying.
53       *
54       * @param principalAttributes the principal attributes. Rather than passing the principal
55       *                            directly, we are only allowing principal attributes
56       *                            given they may be coming from a source external to the principal
57       *                            itself. (Cached principal attributes, etc)
58       * @return true/false if service access can be granted to principal
59       */
60      boolean doPrincipalAttributesAllowServiceAccess(Map<String, Object> principalAttributes);
61  
62      /**
63       * Redirect the request to a separate and possibly external URL
64       * in case authorization fails for this service. If no URL is
65       * specified, CAS shall redirect the request by default to a generic
66       * page that describes the authorization failed attempt.
67       * @return the redirect url
68       * @since 4.2
69       */
70      URI getUnauthorizedRedirectUrl();
71  }
