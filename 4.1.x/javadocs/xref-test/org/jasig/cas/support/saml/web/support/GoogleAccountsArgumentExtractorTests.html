47  public class GoogleAccountsArgumentExtractorTests extends AbstractOpenSamlTests {
48  
49      private GoogleAccountsArgumentExtractor extractor;
50  
51      @Before
52      public void setUp() throws Exception {
53          final PublicKeyFactoryBean pubKeyFactoryBean = new PublicKeyFactoryBean();
54          final PrivateKeyFactoryBean privKeyFactoryBean = new PrivateKeyFactoryBean();
55  
56          pubKeyFactoryBean.setAlgorithm("DSA");
57          privKeyFactoryBean.setAlgorithm("DSA");
58  
59          final ClassPathResource pubKeyResource = new ClassPathResource("DSAPublicKey01.key");
60          final ClassPathResource privKeyResource = new ClassPathResource("DSAPrivateKey01.key");
61  
62          pubKeyFactoryBean.setLocation(pubKeyResource);
63          privKeyFactoryBean.setLocation(privKeyResource);
64          assertTrue(privKeyFactoryBean.getObjectType().equals(PrivateKey.class));
65          assertTrue(pubKeyFactoryBean.getObjectType().equals(PublicKey.class));
66          pubKeyFactoryBean.afterPropertiesSet();
67          privKeyFactoryBean.afterPropertiesSet();
68  
69          final ServicesManager servicesManager = mock(ServicesManager.class);
70          
71          this.extractor = new GoogleAccountsArgumentExtractor(pubKeyFactoryBean.getObject(),
72                  privKeyFactoryBean.getObject());
73      }
74  
75      @Test
76      public void verifyNoService() {
77          assertNull(this.extractor.extractService(new MockHttpServletRequest()));
78      }
79  
80      @Test
81      public void verifyAuthnRequest() throws Exception {
82          final MockHttpServletRequest request = new MockHttpServletRequest();
83          final String samlRequest = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>"
84                  + "<samlp:AuthnRequest xmlns:samlp=\"urn:oasis:names:tc:SAML:2.0:protocol\" "
85                  + "ID=\"5545454455\" Version=\"2.0\" IssueInstant=\"Value\" "
86                  + "ProtocolBinding=\"urn:oasis:names.tc:SAML:2.0:bindings:HTTP-Redirect\" "
87                  + "ProviderName=\"https://localhost:8443/myRutgers\" AssertionConsumerServiceURL=\"https://localhost:8443/myRutgers\"/>";
88          request.setParameter(SamlProtocolConstants.PARAMETER_SAML_REQUEST, encodeMessage(samlRequest));
89  
90          this.extractor.setSkewAllowance(200);
91  
92          final GoogleAccountsService service = (GoogleAccountsService) this.extractor.extractService(request);
93          service.setPrincipal(TestUtils.getPrincipal());
94          assertNotNull(service);
95          final Response response = service.getResponse("SAMPLE_TICKET");
96          assertNotNull(response);
97          assertEquals(service.getSkewAllowance(), 200);
98      }
99  
100     private static String encodeMessage(final String xmlString) throws IOException {
101         return CompressionUtils.deflate(xmlString);
102     }
103 }
