108     @Test
109     public void verifyEncodeDecodeTGTImpl() throws Exception {
110         final Credential userPassCredential = new UsernamePasswordCredential(USERNAME, PASSWORD);
111         final AuthenticationBuilder bldr = new DefaultAuthenticationBuilder(
112                 new DefaultPrincipalFactory()
113                         .createPrincipal("user", Collections.unmodifiableMap(this.principalAttributes)));
114         bldr.setAttributes(Collections.unmodifiableMap(this.principalAttributes));
115         bldr.setAuthenticationDate(new Date());
116         bldr.addCredential(new BasicCredentialMetaData(userPassCredential));
117         bldr.addFailure("error", AccountNotFoundException.class);
118         bldr.addSuccess("authn", new DefaultHandlerResult(
119                 new AcceptUsersAuthenticationHandler(),
120                 new BasicCredentialMetaData(userPassCredential)));
121 
122         final TicketGrantingTicket parent =
123                 new TicketGrantingTicketImpl(TGT_ID, TestUtils.getService(), null, bldr.build(),
124                         new NeverExpiresExpirationPolicy());
125 
126         final TicketGrantingTicket expectedTGT =
127                 new TicketGrantingTicketImpl(TGT_ID, TestUtils.getService(),
128                         null, bldr.build(),
129                         new NeverExpiresExpirationPolicy());
130 
131         final ServiceTicket ticket = expectedTGT.grantServiceTicket(ST_ID,
132                 TestUtils.getService(),
133                 new NeverExpiresExpirationPolicy(), false);
134         CachedData result = transcoder.encode(expectedTGT);
135         final TicketGrantingTicket resultTicket = (TicketGrantingTicket) transcoder.decode(result);
136 
137         assertEquals(expectedTGT, resultTicket);
138         result = transcoder.encode(ticket);
139         final ServiceTicket resultStTicket = (ServiceTicket) transcoder.decode(result);
140         assertEquals(ticket, resultStTicket);
141 
142     }

300     static class MockTicketGrantingTicket implements TicketGrantingTicket {
301 
302         private static final long serialVersionUID = 4829406617873497061L;
303 
304         private final String id;
305 
306         private int usageCount;
307 
308         private Service proxiedBy;
309 
310         private final Date creationDate = new Date();
311 
312         private final Authentication authentication;
313 
314         /** Factory to create the principal type. **/
315         @NotNull
316         private final PrincipalFactory principalFactory = new DefaultPrincipalFactory();
317 
318         /** Constructor for serialization support. */
319         MockTicketGrantingTicket() {
320             this.id = null;
321             this.authentication = null;
322         }
323 
324         MockTicketGrantingTicket(final String id, final Credential credential, final Map<String, Object> principalAttributes) {
325             this.id = id;
326             final CredentialMetaData credentialMetaData = new BasicCredentialMetaData(credential);
327             final DefaultAuthenticationBuilder builder = new DefaultAuthenticationBuilder();
328             builder.setPrincipal(this.principalFactory.createPrincipal(USERNAME, principalAttributes));
329             builder.setAuthenticationDate(new Date());
330             builder.addCredential(credentialMetaData);
331             builder.addAttribute(RememberMeCredential.AUTHENTICATION_ATTRIBUTE_REMEMBER_ME, Boolean.TRUE);
332             final AuthenticationHandler handler = new MockAuthenticationHandler();
333             try {
334                 builder.addSuccess(handler.getName(), handler.authenticate(credential));
335             } catch (final Exception e) {
336                 throw new RuntimeException(e);
337             }
338             builder.addFailure(handler.getName(), FailedLoginException.class);
339             this.authentication = builder.build();
340         }
341 
342         @Override
343         public Authentication getAuthentication() {
344             return this.authentication;
345         }
346 
347         @Override
348         public List<Authentication> getSupplementalAuthentications() {
349             return Collections.emptyList();
350         }
351 
352         @Override
353         public ServiceTicket grantServiceTicket(
354                 final String id,
355                 final Service service,
356                 final ExpirationPolicy expirationPolicy,
357                 final boolean credentialsProvided) {
358             this.usageCount++;
359             return new MockServiceTicket(id);
360         }
361 
362         @Override
363         public Service getProxiedBy() {
364             return proxiedBy;
365         }
366 
367         @Override
368         public Map<String, Service> getServices() {
369             return Collections.emptyMap();
370         }
371 
372         @Override
373         public void removeAllServices() {}
374 
375         @Override
376         public void markTicketExpired() {}
377 
378         @Override
379         public boolean isRoot() {
380             return true;
381         }
382 
383         @Override
384         public TicketGrantingTicket getRoot() {
385             return this;
386         }
387 
388         @Override
389         public List<Authentication> getChainedAuthentications() {
390             return Collections.emptyList();
391         }
392 
393         @Override
394         public String getId() {
395             return this.id;
396         }
397 
398         @Override
399         public boolean isExpired() {
400             return false;
401         }
402 
403         @Override
404         public TicketGrantingTicket getGrantingTicket() {
405             return this;
406         }
407 
408         @Override
409         public long getCreationTime() {
410             return this.creationDate.getTime();
411         }
412 
413         @Override
414         public int getCountOfUses() {
415             return this.usageCount;
416         }
417 
418         @Override
419         public boolean equals(final Object other) {
420             return other instanceof MockTicketGrantingTicket
421                     && ((MockTicketGrantingTicket) other).getId().equals(this.id)
422                     && ((MockTicketGrantingTicket) other).getCountOfUses() == this.usageCount
423                     && ((MockTicketGrantingTicket) other).getCreationTime() == this.creationDate.getTime()
424                     && ((MockTicketGrantingTicket) other).getAuthentication().equals(this.authentication);
425         }
426 
427         @Override
428         public int hashCode() {
429             final HashCodeBuilder bldr = new HashCodeBuilder(17, 33);
430             return bldr.append(this.id)
431                         .append(this.usageCount)
432                         .append(this.creationDate.getTime())
433                         .append(this.authentication).toHashCode();
434         }

437     public static class MockAuthenticationHandler implements AuthenticationHandler {
438 
439         @Override
440         public DefaultHandlerResult authenticate(final Credential credential) throws GeneralSecurityException, PreventedException {
441             if (credential instanceof HttpBasedServiceCredential) {
442                 return new DefaultHandlerResult(this, (HttpBasedServiceCredential) credential);
443             } else {
444                 return new DefaultHandlerResult(this, new BasicCredentialMetaData(credential));
445             }
446         }
447 
448         @Override
449         public boolean supports(final Credential credential) {
450             return true;
451         }
452 
453         @Override
454         public String getName() {
455             return "MockAuthenticationHandler";
456         }
