19  package org.jasig.cas.ticket.proxy.support;
20  
21  import org.jasig.cas.TestUtils;
22  import org.jasig.cas.authentication.HttpBasedServiceCredential;
23  import org.jasig.cas.ticket.TicketGrantingTicket;
24  import org.jasig.cas.util.DefaultUniqueTicketIdGenerator;
25  import org.jasig.cas.util.http.HttpClient;
26  import org.jasig.cas.util.http.SimpleHttpClientFactoryBean;
27  import org.junit.Before;
28  import org.junit.Test;
29  import org.mockito.Mock;
30  import org.mockito.MockitoAnnotations;
31  
32  import java.net.URL;
33  
34  import static org.junit.Assert.*;
35  import static org.mockito.Mockito.*;
36  
37  /**
38   * @author Scott Battaglia
39  
40   * @since 3.0.0
41   */
42  public class Cas20ProxyHandlerTests {
43  
44      private Cas20ProxyHandler handler;
45  
46      @Mock
47      private TicketGrantingTicket proxyGrantingTicket;
48  
49      public Cas20ProxyHandlerTests() {
50          MockitoAnnotations.initMocks(this);
51      }
52      @Before
53      public void setUp() throws Exception {
54          this.handler = new Cas20ProxyHandler();
55  
56          final SimpleHttpClientFactoryBean factory = new SimpleHttpClientFactoryBean();
57          factory.setConnectionTimeout(10000);
58          factory.setReadTimeout(10000);
59          this.handler.setHttpClient(factory.getObject());
60          this.handler.setUniqueTicketIdGenerator(new DefaultUniqueTicketIdGenerator());
61          when(this.proxyGrantingTicket.getId()).thenReturn("proxyGrantingTicket");
62      }
63  
64      @Test
65      public void verifyValidProxyTicketWithoutQueryString() throws Exception {
66          assertNotNull(this.handler.handle(new HttpBasedServiceCredential(
67              new URL("https://www.google.com/"), TestUtils.getRegisteredService("https://some.app.edu")), proxyGrantingTicket));
68      }
69  
70      @Test
71      public void verifyValidProxyTicketWithQueryString() throws Exception {
72          assertNotNull(this.handler.handle(new HttpBasedServiceCredential(
73              new URL("https://www.google.com/?test=test"), TestUtils.getRegisteredService("https://some.app.edu")),
74                  proxyGrantingTicket));
75      }
76  
77      @Test
78      public void verifyNonValidProxyTicket() throws Exception {
79          final SimpleHttpClientFactoryBean clientFactory = new SimpleHttpClientFactoryBean();
80          clientFactory.setAcceptableCodes(new int[] {900});
81          final HttpClient httpClient = clientFactory.getObject();
82          this.handler.setHttpClient(httpClient);
83          assertNull(this.handler.handle(new HttpBasedServiceCredential(new URL(
84              "http://www.rutgers.edu"), TestUtils.getRegisteredService("https://some.app.edu")), proxyGrantingTicket));
85      }
86  }
