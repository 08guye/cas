19  package org.jasig.cas.authentication.handler.support;
20  
21  import org.jasig.cas.TestUtils;
22  import org.jasig.cas.util.http.HttpClient;
23  import org.jasig.cas.util.http.SimpleHttpClientFactoryBean;
24  import org.junit.Before;
25  import org.junit.Test;
26  
27  import javax.security.auth.login.FailedLoginException;
28  
29  import static org.junit.Assert.*;
30  
31  /**
32   * @author Scott Battaglia
33   * @since 3.0.0
34   */
35  public final class HttpBasedServiceCredentialsAuthenticationHandlerTests {
36  
37      private HttpBasedServiceCredentialsAuthenticationHandler authenticationHandler;
38  
39      @Before
40      public void setUp() throws Exception {
41          this.authenticationHandler = new HttpBasedServiceCredentialsAuthenticationHandler();
42          this.authenticationHandler.setHttpClient(new SimpleHttpClientFactoryBean().getObject());
43      }
44  
45      @Test
46      public void verifySupportsProperUserCredentials() {
47          assertTrue(this.authenticationHandler.supports(TestUtils.getHttpBasedServiceCredentials()));
48      }
49  
50      @Test
51      public void verifyDoesntSupportBadUserCredentials() {
52          assertFalse(this.authenticationHandler.supports(TestUtils.getCredentialsWithSameUsernameAndPassword()));
53      }
54  
55      @Test
56      public void verifyAcceptsProperCertificateCredentials() throws Exception {
57          assertNotNull(this.authenticationHandler.authenticate(TestUtils.getHttpBasedServiceCredentials()));
58      }
59  
60      @Test(expected = FailedLoginException.class)
61      public void verifyRejectsInProperCertificateCredentials() throws Exception {
62          this.authenticationHandler.authenticate(
63                  TestUtils.getHttpBasedServiceCredentials("https://clearinghouse.ja-sig.org"));
64      }
65  
66      @Test
67      public void verifyAcceptsNonHttpsCredentials() throws Exception {
68          this.authenticationHandler.setHttpClient(new SimpleHttpClientFactoryBean().getObject());
69          assertNotNull(this.authenticationHandler.authenticate(
70                  TestUtils.getHttpBasedServiceCredentials("http://www.google.com")));
71      }
72  
73      @Test(expected = FailedLoginException.class)
74      public void verifyNoAcceptableStatusCode() throws Exception {
75          this.authenticationHandler.authenticate(
76                  TestUtils.getHttpBasedServiceCredentials("https://clue.acs.rutgers.edu"));
77      }
78  
79      @Test(expected = FailedLoginException.class)
80      public void verifyNoAcceptableStatusCodeButOneSet() throws Exception {
81          final SimpleHttpClientFactoryBean clientFactory = new SimpleHttpClientFactoryBean();
82          clientFactory.setAcceptableCodes(new int[] {900});
83          final HttpClient httpClient = clientFactory.getObject();
84          this.authenticationHandler.setHttpClient(httpClient);
85          this.authenticationHandler.authenticate(TestUtils.getHttpBasedServiceCredentials("https://www.ja-sig.org"));
86      }
87  }
