81      public TicketRegistry getNewTicketRegistry() throws Exception {
82          this.context = new ClassPathXmlApplicationContext(
83                  APPLICATION_CONTEXT_FILE_NAME);
84          this.registry = (JBossCacheTicketRegistry) context
85                  .getBean(APPLICATION_CONTEXT_CACHE_BEAN_NAME);

88          this.treeCache.removeNode("/ticket");
89          return this.registry;
90      }

225     @Test
226     public void verifyGetTicketsFromRegistryEqualToTicketsAdded() {
227         final Collection<Ticket> tickets = new ArrayList<>();
228         final MockHttpServletRequest request = new MockHttpServletRequest();
229         request.addParameter("service", "test");
230 
231         for (int i = 0; i < TICKETS_IN_REGISTRY; i++) {
232             final TicketGrantingTicket ticketGrantingTicket = new TicketGrantingTicketImpl(
233                     "TEST" + i, TestUtils.getAuthentication(),
234                     new NeverExpiresExpirationPolicy());
235             final ServiceTicket st = ticketGrantingTicket.grantServiceTicket(
236                     "tests" + i, SimpleWebApplicationServiceImpl.createServiceFrom(request),
237                     new NeverExpiresExpirationPolicy(), false);
238             tickets.add(ticketGrantingTicket);
239             tickets.add(st);
240             this.ticketRegistry.addTicket(ticketGrantingTicket);
241             this.ticketRegistry.addTicket(st);
242         }
243 
244         try {
245             final Collection<Ticket> ticketRegistryTickets = this.ticketRegistry.getTickets();
246             assertEquals(
247                     "The size of the registry is not the same as the collection.",
248                     ticketRegistryTickets.size(), tickets.size());
249 
250             for (final Ticket ticket : tickets) {
251                 if (!ticketRegistryTickets.contains(ticket)) {
252                     fail("Ticket was added to registry but was not found in retrieval of collection of all tickets.");
253                 }
254             }
255         } catch (final Exception e) {
256             fail("Caught an exception. But no exception should have been thrown.");
257         }
258     }
