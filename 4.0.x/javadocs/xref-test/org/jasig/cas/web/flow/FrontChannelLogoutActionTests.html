117     @Test
118     public void testLogoutOneLogoutRequestNotAttempted() throws Exception {
119         final String FAKE_URL = "http://url";
120         LogoutRequest logoutRequest = new LogoutRequest(TICKET_ID, new SimpleWebApplicationServiceImpl(FAKE_URL));
121         WebUtils.putLogoutRequests(this.requestContext, Arrays.asList(logoutRequest));
122         this.requestContext.getFlowScope().put(FrontChannelLogoutAction.LOGOUT_INDEX, 0);
123         final Event event = this.frontChannelLogoutAction.doExecute(this.requestContext);
124         assertEquals(FrontChannelLogoutAction.REDIRECT_APP_EVENT, event.getId());
125         List<LogoutRequest> list = WebUtils.getLogoutRequests(this.requestContext);
126         assertEquals(1, list.size());
127         final String url = (String) event.getAttributes().get("logoutUrl");
128         assertTrue(url.startsWith(FAKE_URL + "?SAMLRequest="));
129         final byte[] samlMessage = Base64.decodeBase64(URLDecoder.decode(StringUtils.substringAfter(url,  "?SAMLRequest="), "UTF-8"));
130         final Inflater decompresser = new Inflater();
131         decompresser.setInput(samlMessage);
132         final byte[] result = new byte[1000];
133         decompresser.inflate(result);
134         decompresser.end();
135         final String message = new String(result);
136         assertTrue(message.startsWith("<samlp:LogoutRequest xmlns:samlp=\"urn:oasis:names:tc:SAML:2.0:protocol\" ID=\""));
137         assertTrue(message.indexOf("<samlp:SessionIndex>" + TICKET_ID + "</samlp:SessionIndex>") >= 0);
138     }
