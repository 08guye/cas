41  /**
42   * Provides a simple {@link UserDetailsService} implementation that obtains user details from an LDAP search.
43   * Two searches are performed by this component for every user details lookup:
44   *
45   * <ol>
46   *     <li>Search for an entry to resolve the username. In most cases the search should return exactly one result,
47   *     but the {@link #setAllowMultipleResults(boolean)} property may be toggled to change that behavior.</li>
48   *     <li>Search for groups of which the user is a member. This search commonly occurs on a separate directory
49   *     branch than that of the user search.</li>
50   * </ol>
51   *
52   * @author Marvin S. Addison
53   * @author Misagh Moayyed
54   * @since 4.0
55   */
56  public class LdapUserDetailsService implements UserDetailsService {
57  
58      /** Default role prefix. */
59      public static final String DEFAULT_ROLE_PREFIX = "ROLE_";
60  
61      /** Placeholder for unknown password given to user details. */
62      public static final String UNKNOWN_PASSWORD = "<UNKNOWN>";
63  
64      /** Logger instance. */
65      private final Logger logger = LoggerFactory.getLogger(getClass());
66  
67      /** Source of LDAP connections. */
68      @NotNull
69      private final ConnectionFactory connectionFactory;
70  
71      /** Executes the search query for user data. */
72      @NotNull
73      private final SearchExecutor userSearchExecutor;
74  
75      /** Executes the search query for roles. */
76      @NotNull
77      private final SearchExecutor roleSearchExecutor;
78  
79      /** Specify the name of LDAP attribute to use as principal identifier. */
80      @NotNull
81      private final String userAttributeName;
82  
83      /** Specify the name of LDAP attribute to be used as the basis for role granted authorities. */
84      @NotNull
85      private final String roleAttributeName;
86  
87      /** Prefix appended to the uppercased {@link #roleAttributeName} per the normal Spring Security convention. */
88      @NotNull
89      private String rolePrefix = DEFAULT_ROLE_PREFIX;
90  
91      /** Flag that indicates whether multiple search results are allowed for a given credential. */
92      private boolean allowMultipleResults = false;
93  
94      /**
95       * Creates a new instance with the given required parameters.
96       *
97       * @param  factory  Source of LDAP connections for searches.
98       * @param  userSearchExecutor  Executes the LDAP search for user data.
99       * @param roleSearchExecutor  Executes the LDAP search for role data.
100      * @param userAttributeName  Name of LDAP attribute that contains username for user details.
101      * @param roleAttributeName  Name of LDAP attribute that contains role membership data for the user.
102      */
103     public LdapUserDetailsService(
104             final ConnectionFactory factory,
105             final SearchExecutor userSearchExecutor,
106             final SearchExecutor roleSearchExecutor,
107             final String userAttributeName,
108             final String roleAttributeName) {
109 
110         this.connectionFactory = factory;
111         this.userSearchExecutor = userSearchExecutor;
112         this.roleSearchExecutor = roleSearchExecutor;
113         this.userAttributeName = userAttributeName;
114         this.roleAttributeName = roleAttributeName;
115     }

118     /**
119      * Sets the prefix appended to the uppercase {@link #roleAttributeName} per the normal Spring Security convention.
120      * The default value {@value #DEFAULT_ROLE_PREFIX} is sufficient in most cases.
121      *
122      * @param  rolePrefix  Role prefix.
123      */
124     public void setRolePrefix(final String rolePrefix) {
125         this.rolePrefix = rolePrefix;
126     }
127 
128 
129     /**
130      * Sets whether to allow multiple search results for user details given a username.
131      * This is false by default, which is sufficient and secure for more deployments.
132      * Setting this to true may have security consequences.
133      *
134      * @param  allowMultiple  True to allow multiple search results in which case the first result
135      *                        returned is used to construct user details, or false to indicate that
136      *                        a runtime exception should be raised on multiple search results for user details.
137      */
138     public void setAllowMultipleResults(final boolean allowMultiple) {
139         this.allowMultipleResults = allowMultiple;
140     }
141 
142     @Override
143     public UserDetails loadUserByUsername(final String username) {
144         final SearchResult userResult;
145         try {
146             logger.debug("Attempting to get details for user {}.", username);
147             final Response<SearchResult> response = this.userSearchExecutor.search(
148                     this.connectionFactory,
149                     createSearchFilter(this.userSearchExecutor, username));
150             logger.debug("LDAP user search response: {}", response);
151             userResult = response.getResult();
152         } catch (final LdapException e) {
153             throw new RuntimeException("LDAP error fetching details for user.", e);
154         }
155         if (userResult.size() == 0) {
156             throw new UsernameNotFoundException(username + " not found.");
157         }
158         if (userResult.size() > 1 && !this.allowMultipleResults) {
159             throw new IllegalStateException(
160                     "Found multiple results for user which is not allowed (allowMultipleResults=false).");
161         }
162         final String userDn = userResult.getEntry().getDn();
163         final LdapAttribute userAttribute = userResult.getEntry().getAttribute(this.userAttributeName);
164         if (userAttribute == null) {
165             throw new IllegalStateException(this.userAttributeName + " attribute not found in results.");
166         }
167         final String id = userAttribute.getStringValue();
168 
169         final SearchResult roleResult;
170         try {
171             logger.debug("Attempting to get roles for user {}.", userDn);
172             final Response<SearchResult> response = this.roleSearchExecutor.search(
173                     this.connectionFactory,
174                     createSearchFilter(this.roleSearchExecutor, userDn));
175             logger.debug("LDAP role search response: {}", response);
176             roleResult = response.getResult();
177         } catch (final LdapException e) {
178             throw new RuntimeException("LDAP error fetching roles for user.", e);
179         }
180         LdapAttribute roleAttribute;
181         final Collection<SimpleGrantedAuthority> roles = new ArrayList<SimpleGrantedAuthority>(roleResult.size());
182         for (final LdapEntry entry : roleResult.getEntries()) {
183             roleAttribute = entry.getAttribute(this.roleAttributeName);
184             if (roleAttribute == null) {
185                 logger.warn("Role attribute not found on entry {}", entry);
186                 continue;
187             }
188             roles.add(new SimpleGrantedAuthority(this.rolePrefix + roleAttribute.getStringValue().toUpperCase()));
189         }
190 
191         return new User(id, UNKNOWN_PASSWORD, roles);
192     }
193 
194     /**
195      * Constructs a new search filter using {@link SearchExecutor#searchFilter} as a template and
196      * the username as a parameter.
197      *
198      * @param  username  Username parameter of search query.
199      *
200      * @return  Search filter with parameters applied.
201      */
202     private SearchFilter createSearchFilter(final SearchExecutor executor, final String username) {
203         final SearchFilter filter = new SearchFilter();
204         filter.setFilter(executor.getSearchFilter().getFilter());
205         filter.setParameter(0, username);
206 
207         logger.debug("Constructed LDAP search filter [{}]", filter.format());
208         return filter;
209     }
