47  /**
48   * Default account state handler.
49   *
50   * @author Marvin S. Addison
51   * @since 4.0.0
52   */
53  public class DefaultAccountStateHandler implements AccountStateHandler {
54      /** Map of account state error to CAS authentication exception. */
55      protected static final Map<AccountState.Error, LoginException> ERROR_MAP;
56  
57      /** Logger instance. */
58      protected final Logger logger = LoggerFactory.getLogger(getClass());
59  
60      /**
61       * Instantiates a new account state handler, that populates
62       * the error map with LDAP error codes and corresponding exceptions.
63       */
64      static {
65          ERROR_MAP = new HashMap<>();
66          ERROR_MAP.put(ActiveDirectoryAccountState.Error.ACCOUNT_DISABLED, new AccountDisabledException());
67          ERROR_MAP.put(ActiveDirectoryAccountState.Error.ACCOUNT_LOCKED_OUT, new AccountLockedException());
68          ERROR_MAP.put(ActiveDirectoryAccountState.Error.INVALID_LOGON_HOURS, new InvalidLoginTimeException());
69          ERROR_MAP.put(ActiveDirectoryAccountState.Error.INVALID_WORKSTATION, new InvalidLoginLocationException());
70          ERROR_MAP.put(ActiveDirectoryAccountState.Error.PASSWORD_MUST_CHANGE, new AccountPasswordMustChangeException());
71          ERROR_MAP.put(ActiveDirectoryAccountState.Error.PASSWORD_EXPIRED, new CredentialExpiredException());
72          ERROR_MAP.put(EDirectoryAccountState.Error.ACCOUNT_EXPIRED, new AccountExpiredException());
73          ERROR_MAP.put(EDirectoryAccountState.Error.LOGIN_LOCKOUT, new AccountLockedException());
74          ERROR_MAP.put(EDirectoryAccountState.Error.LOGIN_TIME_LIMITED, new InvalidLoginTimeException());
75          ERROR_MAP.put(EDirectoryAccountState.Error.PASSWORD_EXPIRED, new CredentialExpiredException());
76          ERROR_MAP.put(PasswordExpirationAccountState.Error.PASSWORD_EXPIRED, new CredentialExpiredException());
77          ERROR_MAP.put(PasswordPolicyControl.Error.ACCOUNT_LOCKED, new AccountLockedException());
78          ERROR_MAP.put(PasswordPolicyControl.Error.PASSWORD_EXPIRED, new CredentialExpiredException());
79          ERROR_MAP.put(PasswordPolicyControl.Error.CHANGE_AFTER_RESET, new CredentialExpiredException());
80      }
81  
82      @Override
83      public List<Message> handle(final AuthenticationResponse response, final LdapPasswordPolicyConfiguration configuration)
84              throws LoginException {
85  
86          final AccountState state = response.getAccountState();
87          final AccountState.Error error;
88          final AccountState.Warning warning;
89          if (state != null) {
90              error = state.getError();
91              warning = state.getWarning();
92          } else {
93              logger.debug("Account state not defined");
94              error = null;
95              warning = null;
96          }
97          final List<Message> messages = new ArrayList<>();
98          handleError(error, response, configuration, messages);
99          handleWarning(warning, response, configuration, messages);
100         return messages;
101     }
102 
103     /**
104      * Handle an account state error produced by ldaptive account state machinery.
105      * <p>
106      * Override this method to provide custom error handling.
107      *
108      * @param error Account state error.
109      * @param response Ldaptive authentication response.
110      * @param configuration Password policy configuration.
111      * @param messages Container for messages produced by account state error handling.
112      *
113      * @throws LoginException On errors that should be communicated as login exceptions.
114      */
115     protected void handleError(
116             final AccountState.Error error,
117             final AuthenticationResponse response,
118             final LdapPasswordPolicyConfiguration configuration,
119             final List<Message> messages)
120             throws LoginException {
121 
122         logger.debug("Handling {}", error);
123         final LoginException ex = ERROR_MAP.get(error);
124         if (ex != null) {
125             throw ex;
126         }
127         logger.debug("No LDAP error mapping defined for {}", error);
128     }
129 
130 
131     /**
132      * Handle an account state warning produced by ldaptive account state machinery.
133      * <p>
134      * Override this method to provide custom warning message handling.
135      *
136      * @param warning the account state warning messages.
137      * @param response Ldaptive authentication response.
138      * @param configuration Password policy configuration.
139      * @param messages Container for messages produced by account state warning handling.
140      */
141     protected void handleWarning(
142             final AccountState.Warning warning,
143             final AuthenticationResponse response,
144             final LdapPasswordPolicyConfiguration configuration,
145             final List<Message> messages) {
146 
147         if (warning == null) {
148             logger.debug("Account state warning not defined");
149             return;
150         }
151 
152         final Calendar expDate = warning.getExpiration();
153         final Days ttl = Days.daysBetween(Instant.now(), new Instant(expDate));
154         logger.debug(
155                 "Password expires in {} days. Expiration warning threshold is {} days.",
156                 ttl.getDays(),
157                 configuration.getPasswordWarningNumberOfDays());
158         if (configuration.isAlwaysDisplayPasswordExpirationWarning()
159                 || ttl.getDays() < configuration.getPasswordWarningNumberOfDays()) {
160             messages.add(new PasswordExpiringWarningMessage(
161                     "Password expires in {0} days. Please change your password at <href=\"{1}\">{1}</a>",
162                     ttl.getDays(),
163                     configuration.getPasswordPolicyUrl()));
164         }
165         if (warning.getLoginsRemaining() > 0) {
166             messages.add(new Message(
167                     "password.expiration.loginsRemaining",
168                     "You have {0} logins remaining before you MUST change your password.",
169                     warning.getLoginsRemaining()));
170 
171         }
172     }
