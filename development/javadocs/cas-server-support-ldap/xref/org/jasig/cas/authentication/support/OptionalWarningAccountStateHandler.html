19  package org.jasig.cas.authentication.support;
20  
21  import java.util.List;
22  
23  import javax.validation.constraints.NotNull;
24  
25  import org.jasig.cas.Message;
26  import org.ldaptive.LdapAttribute;
27  import org.ldaptive.auth.AccountState;
28  import org.ldaptive.auth.AuthenticationResponse;
29  
30  /**
31   * The component supports both opt-in and opt-out warnings on a per-user basis using a simple algorithm of three
32   * variables:
33   * <ol>
34   *     <li>{@link #setWarningAttributeName(String) warningAttributeName}</li>
35   *     <li>{@link #setWarningAttributeValue(String)} warningAttributeValue}</li>
36   *     <li>{@link #setDisplayWarningOnMatch(boolean) displayWarningOnMatch}</li>
37   * </ol>
38   * The first two parameters define an attribute on the user entry to match on, and the third parameter determines
39   * whether password expiration warnings should be displayed on match.
40   * <p>
41   * Deployers MUST configure LDAP components to provide <code>warningAttributeName</code> in the set of attributes
42   * returned from the LDAP query for user details.
43   *
44   * @author Marvin S. Addison
45   * @since 4.0.0
46   */
47  public class OptionalWarningAccountStateHandler extends DefaultAccountStateHandler {
48  
49      /** Name of user attribute that describes whether or not to display expiration warnings. */
50      @NotNull
51      private String warningAttributeName;
52  
53      /** Attribute value to match. */
54      @NotNull
55      private String warningAttributeValue;
56  
57      /** True to opt into password expiration warnings on match, false to opt out on match. */
58      private boolean displayWarningOnMatch = true;
59  
60  
61      /**
62       * Sets the user attribute used to determine whether to display password expiration warnings.
63       *
64       * @param warningAttributeName Attribute on authenticated user entry.
65       */
66      public void setWarningAttributeName(final String warningAttributeName) {
67          this.warningAttributeName = warningAttributeName;
68      }
69  
70      /**
71       * Sets the value of {@link #warningAttributeName} used as basis of comparison.
72       *
73       * @param warningAttributeValue Value to match against.
74       */
75      public void setWarningAttributeValue(final String warningAttributeValue) {
76          this.warningAttributeValue = warningAttributeValue;
77      }
78  
79      /**
80       * Determines whether password expiration warnings are opt-in or opt-out.
81       *
82       * @param displayWarningOnMatch True to opt into password expiration warnings on match, false to opt out on match.
83       *                              Default is true.
84       */
85      public void setDisplayWarningOnMatch(final boolean displayWarningOnMatch) {
86          this.displayWarningOnMatch = displayWarningOnMatch;
87      }
88  
89      @Override
90      protected void handleWarning(
91              final AccountState.Warning warning,
92              final AuthenticationResponse response,
93              final LdapPasswordPolicyConfiguration configuration,
94              final List<Message> messages) {
95  
96          final LdapAttribute attribute = response.getLdapEntry().getAttribute(this.warningAttributeName);
97          boolean matches = false;
98          if (attribute != null) {
99              logger.debug("Found warning attribute {} with value {}", attribute.getName(), attribute.getStringValue());
100             matches = this.warningAttributeValue.equals(attribute.getStringValue());
101         }
102         logger.debug("matches={}, displayWarningOnMatch={}", matches, displayWarningOnMatch);
103         if (displayWarningOnMatch == matches) {
104             super.handleWarning(warning, response, configuration, messages);
105         }
106     }
107 }
