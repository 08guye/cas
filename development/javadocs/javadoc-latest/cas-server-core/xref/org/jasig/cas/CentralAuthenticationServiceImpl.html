73  /**
74   * Concrete implementation of a CentralAuthenticationService, and also the
75   * central, organizing component of CAS's internal implementation.
76   * <p>
77   * This class is threadsafe.
78   * <p>
79   * This class has the following properties that must be set:
80   * <ul>
81   * <li> <code>ticketRegistry</code> - The Ticket Registry to maintain the list
82   * of available tickets.</li>
83   * <li> <code>serviceTicketRegistry</code> - Provides an alternative to configure separate registries for
84   * TGTs and ST in order to store them in different locations (i.e. long term memory or short-term)</li>
85   * <li> <code>authenticationManager</code> - The service that will handle
86   * authentication.</li>
87   * <li> <code>ticketGrantingTicketUniqueTicketIdGenerator</code> - Plug in to
88   * generate unique secure ids for TicketGrantingTickets.</li>
89   * <li> <code>serviceTicketUniqueTicketIdGenerator</code> - Plug in to
90   * generate unique secure ids for ServiceTickets.</li>
91   * <li> <code>ticketGrantingTicketExpirationPolicy</code> - The expiration
92   * policy for TicketGrantingTickets.</li>
93   * <li> <code>serviceTicketExpirationPolicy</code> - The expiration policy for
94   * ServiceTickets.</li>
95   * </ul>
96   *
97   * @author William G. Thompson, Jr.
98   * @author Scott Battaglia
99   * @author Dmitry Kopylenko
100  * @since 3.0.0
101  */

174     public CentralAuthenticationServiceImpl(final TicketRegistry ticketRegistry,
175                                             final TicketRegistry serviceTicketRegistry,
176                                             final AuthenticationManager authenticationManager,
177                                             final UniqueTicketIdGenerator ticketGrantingTicketUniqueTicketIdGenerator,
178                                             final Map<String, UniqueTicketIdGenerator> uniqueTicketIdGeneratorsForService,

188         }
189         this.authenticationManager = authenticationManager;
190         this.ticketGrantingTicketUniqueTicketIdGenerator = ticketGrantingTicketUniqueTicketIdGenerator;
191         this.uniqueTicketIdGeneratorsForService = uniqueTicketIdGeneratorsForService;
192         this.ticketGrantingTicketExpirationPolicy = ticketGrantingTicketExpirationPolicy;
193         this.serviceTicketExpirationPolicy = serviceTicketExpirationPolicy;
194         this.servicesManager = servicesManager;
195         this.logoutManager = logoutManager;

198     /**
199      * {@inheritDoc}
200      * Destroy a TicketGrantingTicket and perform back channel logout. This has the effect of invalidating any
201      * Ticket that was derived from the TicketGrantingTicket being destroyed. May throw an
202      * {@link IllegalArgumentException} if the TicketGrantingTicket ID is null.
203      *
204      * @param ticketGrantingTicketId the id of the ticket we want to destroy
205      * @return the logout requests.
206      */
207     @Audit(
208             action="TICKET_GRANTING_TICKET_DESTROYED",
209             actionResolverName="DESTROY_TICKET_GRANTING_TICKET_RESOLVER",
210             resourceResolverName="DESTROY_TICKET_GRANTING_TICKET_RESOURCE_RESOLVER")
211     @Profiled(tag = "DESTROY_TICKET_GRANTING_TICKET", logFailuresSeparately = false)
212     @Transactional(readOnly = false)
213     @Override
214     public List<LogoutRequest> destroyTicketGrantingTicket(final String ticketGrantingTicketId) {
215         try {
216             logger.debug("Removing ticket [{}] from registry...", ticketGrantingTicketId);
217             final TicketGrantingTicket ticket = getTicket(ticketGrantingTicketId, TicketGrantingTicket.class);
218             logger.debug("Ticket found. Processing logout requests and then deleting the ticket...");
219             final List<LogoutRequest> logoutRequests = logoutManager.performLogout(ticket);
220             this.ticketRegistry.deleteTicket(ticketGrantingTicketId);
221             return logoutRequests;
222         } catch (final InvalidTicketException e) {
223             logger.debug("TicketGrantingTicket [{}] cannot be found in the ticket registry.", ticketGrantingTicketId);
224         }
225         return Collections.emptyList();
226     }

228     @Audit(
229         action="SERVICE_TICKET",
230         actionResolverName="GRANT_SERVICE_TICKET_RESOLVER",
231         resourceResolverName="GRANT_SERVICE_TICKET_RESOURCE_RESOLVER")
232     @Profiled(tag="GRANT_SERVICE_TICKET", logFailuresSeparately = false)
233     @Transactional(readOnly = false)
234     @Override
235     public ServiceTicket grantServiceTicket(
236             final String ticketGrantingTicketId, final Service service, final Credential... credentials)
237             throws AuthenticationException, TicketException {
238 
239         Assert.notNull(service, "service cannot be null");
240 
241         final TicketGrantingTicket ticketGrantingTicket = getTicket(ticketGrantingTicketId, TicketGrantingTicket.class);
242         final RegisteredService registeredService = this.servicesManager.findServiceBy(service);
243 
244         verifyRegisteredServiceProperties(registeredService, service);
245 
246         Authentication currentAuthentication = null;
247         if (credentials != null && credentials.length > 0) {
248             currentAuthentication = this.authenticationManager.authenticate(credentials);
249             final Authentication original = ticketGrantingTicket.getAuthentication();
250             if (!currentAuthentication.getPrincipal().equals(original.getPrincipal())) {
251                 throw new MixedPrincipalException(
252                         currentAuthentication, currentAuthentication.getPrincipal(), original.getPrincipal());
253             }
254             ticketGrantingTicket.getSupplementalAuthentications().add(currentAuthentication);
255         }
256         
257         if (!registeredService.isSsoEnabled() && currentAuthentication == null) {
258             logger.warn("ServiceManagement: Service [{}] is not allowed to use SSO.", service.getId());
259             throw new UnauthorizedSsoServiceException();
260         }
261 
262         //CAS-1019
263         final List<Authentication> authns = ticketGrantingTicket.getChainedAuthentications();
264         if(authns.size() > 1) {
265             if (!registeredService.getProxyPolicy().isAllowedToProxy()) {
266                 final String message = String.
267                         format("ServiceManagement: Proxy attempt by service [%s] (registered service [%s]) is not allowed.",
268                         service.getId(), registeredService.toString());
269                 logger.warn(message);
270                 throw new UnauthorizedProxyingException(message);
271             }
272         }
273 
274         // Perform security policy check by getting the authentication that satisfies the configured policy
275         // This throws if no suitable policy is found
276         getAuthenticationSatisfiedByPolicy(ticketGrantingTicket, new ServiceContext(service, registeredService));
277 
278         final String uniqueTicketIdGenKey = service.getClass().getName();
279         if (!this.uniqueTicketIdGeneratorsForService.containsKey(uniqueTicketIdGenKey)) {
280             logger.warn("Cannot create service ticket because the key [{}] for service [{}] is not linked to a ticket id generator",
281                     uniqueTicketIdGenKey, service.getId());
282             throw new UnauthorizedSsoServiceException();
283         }
284         
285         final UniqueTicketIdGenerator serviceTicketUniqueTicketIdGenerator =
286                 this.uniqueTicketIdGeneratorsForService.get(uniqueTicketIdGenKey);
287 
288         final List<Authentication> authentications = ticketGrantingTicket.getChainedAuthentications();
289         final String ticketPrefix = authentications.size() == 1 ? ServiceTicket.PREFIX : ServiceTicket.PROXY_TICKET_PREFIX;
290         final String ticketId = serviceTicketUniqueTicketIdGenerator.getNewTicketId(ticketPrefix);
291         final ServiceTicket serviceTicket = ticketGrantingTicket.grantServiceTicket(
292                 ticketId,
293                 service,
294                 this.serviceTicketExpirationPolicy,
295                 currentAuthentication != null);
296 
297         this.serviceTicketRegistry.addTicket(serviceTicket);
298 
299         final String principalId = authentications.get(authentications.size() - 1).getPrincipal().getId();
300         logger.info("Granted ticket [{}] for service [{}] for user [{}]",
301                 serviceTicket.getId(), service.getId(), principalId);
302 
303         return serviceTicket;
304     }

322     @Audit(
323         action="PROXY_GRANTING_TICKET",
324         actionResolverName="GRANT_PROXY_GRANTING_TICKET_RESOLVER",
325         resourceResolverName="GRANT_PROXY_GRANTING_TICKET_RESOURCE_RESOLVER")
326     @Profiled(tag="GRANT_PROXY_GRANTING_TICKET", logFailuresSeparately = false)
327     @Transactional(readOnly = false)
328     @Override
329     public TicketGrantingTicket delegateTicketGrantingTicket(final String serviceTicketId, final Credential... credentials)
330             throws AuthenticationException, TicketException {
331 
332         Assert.notNull(serviceTicketId, "serviceTicketId cannot be null");
333         Assert.notNull(credentials, "credentials cannot be null");
334 
335         final ServiceTicket serviceTicket =  this.serviceTicketRegistry.getTicket(serviceTicketId, ServiceTicket.class);
336 
337         if (serviceTicket == null || serviceTicket.isExpired()) {
338             logger.debug("ServiceTicket [{}] has expired or cannot be found in the ticket registry", serviceTicketId);
339             throw new InvalidTicketException(serviceTicketId);
340         }
341 
342         final RegisteredService registeredService = this.servicesManager
343                 .findServiceBy(serviceTicket.getService());
344 
345         verifyRegisteredServiceProperties(registeredService, serviceTicket.getService());
346         
347         if (!registeredService.getProxyPolicy().isAllowedToProxy()) {
348             logger.warn("ServiceManagement: Service [{}] attempted to proxy, but is not allowed.", serviceTicket.getService().getId());
349             throw new UnauthorizedProxyingException();
350         }
351 
352         final Authentication authentication = this.authenticationManager.authenticate(credentials);
353 
354         final String pgtId = this.ticketGrantingTicketUniqueTicketIdGenerator.getNewTicketId(
355                 TicketGrantingTicket.PROXY_GRANTING_TICKET_PREFIX);
356         final TicketGrantingTicket proxyGrantingTicket = serviceTicket.grantTicketGrantingTicket(pgtId,
357                                     authentication, this.ticketGrantingTicketExpirationPolicy);
358 
359         logger.debug("Generated proxy granting ticket [{}] based off of [{}]", proxyGrantingTicket, serviceTicketId);
360         this.ticketRegistry.addTicket(proxyGrantingTicket);
361 
362         return proxyGrantingTicket;
363     }

365     @Audit(
366         action="SERVICE_TICKET_VALIDATE",
367         actionResolverName="VALIDATE_SERVICE_TICKET_RESOLVER",
368         resourceResolverName="VALIDATE_SERVICE_TICKET_RESOURCE_RESOLVER")
369     @Profiled(tag="VALIDATE_SERVICE_TICKET", logFailuresSeparately = false)
370     @Transactional(readOnly = false)
371     @Override
372     public Assertion validateServiceTicket(final String serviceTicketId, final Service service) throws TicketException {
373         Assert.notNull(serviceTicketId, "serviceTicketId cannot be null");
374         Assert.notNull(service, "service cannot be null");
375  
376         final ServiceTicket serviceTicket =  this.serviceTicketRegistry.getTicket(serviceTicketId, ServiceTicket.class);
377 
378         if (serviceTicket == null) {
379             logger.info("Service ticket [{}] does not exist.", serviceTicketId);
380             throw new InvalidTicketException(serviceTicketId);
381         }
382 
383         final RegisteredService registeredService = this.servicesManager.findServiceBy(service);
384 
385         verifyRegisteredServiceProperties(registeredService, serviceTicket.getService());
386         
387         try {
388             synchronized (serviceTicket) {
389                 if (serviceTicket.isExpired()) {
390                     logger.info("ServiceTicket [{}] has expired.", serviceTicketId);
391                     throw new InvalidTicketException(serviceTicketId);
392                 }
393 
394                 if (!serviceTicket.isValidFor(service)) {
395                     logger.error("Service ticket [{}] with service [{}] does not match supplied service [{}]",
396                             serviceTicketId, serviceTicket.getService().getId(), service);
397                     throw new TicketValidationException(serviceTicket.getService());
398                 }
399             }
400 
401             final TicketGrantingTicket root = serviceTicket.getGrantingTicket().getRoot();
402             final Authentication authentication = getAuthenticationSatisfiedByPolicy(
403                     root, new ServiceContext(serviceTicket.getService(), registeredService));
404             final Principal principal = authentication.getPrincipal();
405 
406             final AttributeReleasePolicy attributePolicy = registeredService.getAttributeReleasePolicy();
407             logger.debug("Attribute policy [{}] is associated with service [{}]", attributePolicy, registeredService);
408             
409             @SuppressWarnings("unchecked")
410             final Map<String, Object> attributesToRelease = attributePolicy != null
411                     ? attributePolicy.getAttributes(principal) : Collections.EMPTY_MAP;
412             
413             final String principalId = registeredService.getUsernameAttributeProvider().resolveUsername(principal, service);
414             final Principal modifiedPrincipal = this.principalFactory.createPrincipal(principalId, attributesToRelease);
415             final AuthenticationBuilder builder = AuthenticationBuilder.newInstance(authentication);
416             builder.setPrincipal(modifiedPrincipal);
417 
418             return new ImmutableAssertion(
419                     builder.build(),
420                     serviceTicket.getGrantingTicket().getChainedAuthentications(),
421                     serviceTicket.getService(),
422                     serviceTicket.isFromNewLogin());
423         } finally {
424             if (serviceTicket.isExpired()) {
425                 this.serviceTicketRegistry.deleteTicket(serviceTicketId);
426             }
427         }
428     }

453     /**
454      * {@inheritDoc}
455      */
456     @Transactional(readOnly = true)
457     @Override
458     public <T extends Ticket> T getTicket(@NotNull final String ticketId, @NotNull final Class<? extends Ticket> clazz)
459             throws InvalidTicketException {
460         Assert.notNull(ticketId, "ticketId cannot be null");
461         final Ticket ticket = this.ticketRegistry.getTicket(ticketId, clazz);
462 
463         if (ticket == null) {
464             logger.debug("Ticket [{}] by type [{}] cannot be found in the ticket registry.", ticketId, clazz.getSimpleName());
465             throw new InvalidTicketException(ticketId);
466         }
467 
468         if (ticket instanceof TicketGrantingTicket) {
469             synchronized (ticket) {
470                 if (ticket.isExpired()) {
471                     this.ticketRegistry.deleteTicket(ticketId);
472                     logger.debug("Ticket [{}] has expired and is now deleted from the ticket registry.", ticketId);
473                     throw new InvalidTicketException(ticketId);
474                 }
475             }
476         }
477         return (T) ticket;
478     }
479     /**
480      * {@inheritDoc}
481      */
482     @Transactional(readOnly = true)
483     @Override
484     public Collection<Ticket> getTickets(@NotNull final Predicate predicate) {
485         final Collection<Ticket> c = new HashSet<>(this.ticketRegistry.getTickets());
486         final Iterator<Ticket> it = c.iterator();
487         while (it.hasNext()) {
488             if (!predicate.evaluate(it.next())) {
489                 it.remove();
490             }
491         }
492         return c;
493     }

535     /**
536      * Gets the authentication satisfied by policy.
537      *
538      * @param ticket the ticket
539      * @param context the context
540      * @return the authentication satisfied by policy
541      * @throws org.jasig.cas.ticket.TicketException the ticket exception
542      */
543     private Authentication getAuthenticationSatisfiedByPolicy(
544             final TicketGrantingTicket ticket, final ServiceContext context) throws TicketException {
545 
546         final ContextualAuthenticationPolicy<ServiceContext> policy =
547                 serviceContextAuthenticationPolicyFactory.createPolicy(context);
548         if (policy.isSatisfiedBy(ticket.getAuthentication())) {
549             return ticket.getAuthentication();
550         }
551         for (final Authentication auth : ticket.getSupplementalAuthentications()) {
552             if (policy.isSatisfiedBy(auth)) {
553                 return auth;
554             }
555         }
556         throw new UnsatisfiedAuthenticationPolicyException(policy);
557     }

559     /**
560      * Ensure that the service is found and enabled in the service registry.
561      * @param registeredService the located entry in the registry
562      * @param service authenticating service
563      * @throws UnauthorizedServiceException
564      */
565     private void verifyRegisteredServiceProperties(final RegisteredService registeredService, final Service service) {
566         if (registeredService == null) {
567             final String msg = String.format("ServiceManagement: Unauthorized Service Access. "
568                     + "Service [%s] is not found in service registry.", service.getId());
569             logger.warn(msg);
570             throw new UnauthorizedServiceException(UnauthorizedServiceException.CODE_UNAUTHZ_SERVICE, msg);
571         }
572         if (!registeredService.isEnabled()) {
573             final String msg = String.format("ServiceManagement: Unauthorized Service Access. "
574                     + "Service %s] is not enabled in service registry.", service.getId());
575             
576             logger.warn(msg);
577             throw new UnauthorizedServiceException(UnauthorizedServiceException.CODE_UNAUTHZ_SERVICE, msg);
578         }
