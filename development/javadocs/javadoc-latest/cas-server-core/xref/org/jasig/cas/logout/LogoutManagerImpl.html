41  /**
42   * This logout manager handles the Single Log Out process.
43   *
44   * @author Jerome Leleu
45   * @since 4.0.0
46   */
47  public final class LogoutManagerImpl implements LogoutManager {
48  
49      /** The logger. */
50      private static final Logger LOGGER = LoggerFactory.getLogger(LogoutManagerImpl.class);
51  
52      /** The parameter name that contains the logout request. */
53      private static final String LOGOUT_PARAMETER_NAME = "logoutRequest";
54  
55      /** ASCII character set. */
56      private static final Charset ASCII = Charset.forName("ASCII");
57  
58      /** The services manager. */
59      @NotNull
60      private final ServicesManager servicesManager;
61  
62      /** An HTTP client. */
63      @NotNull
64      private final HttpClient httpClient;
65  
66      @NotNull
67      private final LogoutMessageCreator logoutMessageBuilder;
68      
69      /** Whether single sign out is disabled or not. */
70      private boolean singleLogoutCallbacksDisabled;
71      
72      /** 
73       * Whether messages to endpoints would be sent in an asynchronous fashion.
74       * True by default.
75       **/
76      private boolean asynchronous = true;
77      
78      /**
79       * Build the logout manager.
80       * @param servicesManager the services manager.
81       * @param httpClient an HTTP client.
82       * @param logoutMessageBuilder the builder to construct logout messages.
83       */
84      public LogoutManagerImpl(final ServicesManager servicesManager, final HttpClient httpClient,
85                               final LogoutMessageCreator logoutMessageBuilder) {
86          this.servicesManager = servicesManager;
87          this.httpClient = httpClient;
88          this.logoutMessageBuilder = logoutMessageBuilder;
89      }
90  
91      /**
92       * Set if messages are sent in an asynchronous fashion.
93       *
94       * @param asyncCallbacks if message is synchronously sent
95       * @since 4.1.0
96       */
97      public void setAsynchronous(final boolean asyncCallbacks) {
98          this.asynchronous = asyncCallbacks;
99      }
100     
101     /**
102      * Set if messages are sent in an asynchronous fashion.
103      *
104      * @param asyncCallbacks if message is synchronously sent
105      * @deprecated As of 4.1. Use {@link #setAsynchronous(boolean)} instead
106      */
107     @Deprecated
108     public void setIssueAsynchronousCallbacks(final boolean asyncCallbacks) {
109         this.asynchronous = asyncCallbacks;
110         LOGGER.warn("setIssueAsynchronousCallbacks() is deprecated. Use setAsynchronous() instead.");
111     }
112     
113     /**
114      * Perform a back channel logout for a given ticket granting ticket and returns all the logout requests.
115      *
116      * @param ticket a given ticket granting ticket.
117      * @return all logout requests.
118      */
119     @Override
120     public List<LogoutRequest> performLogout(final TicketGrantingTicket ticket) {
121         final Map<String, Service> services;
122         // synchronize the retrieval of the services and their cleaning for the TGT
123         // to avoid concurrent logout mess ups
124         synchronized (ticket) {
125             services = ticket.getServices();
126             ticket.removeAllServices();
127         }
128         ticket.markTicketExpired();
129 
130         final List<LogoutRequest> logoutRequests = new ArrayList<>();
131         // if SLO is not disabled
132         if (!this.singleLogoutCallbacksDisabled) {
133             // through all services
134             for (final Map.Entry<String, Service> entry : services.entrySet()) {
135                 // it's a SingleLogoutService, else ignore
136                 final Service service = entry.getValue();
137                 if (service instanceof SingleLogoutService) {
138                     final LogoutRequest logoutRequest = handleLogoutForSloService((SingleLogoutService) service, entry.getKey());
139                     if (logoutRequest != null) {
140                         logoutRequests.add(logoutRequest);
141                     }
142                 }
143             }
144         }
145 
146         return logoutRequests;
147     }
148 
149     /**
150      * Service supports back channel single logout?
151      * Service must be found in the registry. enabled and logout type must not be {@link LogoutType#NONE}.
152      * @param registeredService the registered service
153      * @return true, if support is available.
154      */
155     private boolean serviceSupportsSingleLogout(final RegisteredService registeredService) {
156         return registeredService != null
157                 && registeredService.getAccessStrategy().isServiceAccessAllowed()
158                 && registeredService.getLogoutType() != LogoutType.NONE;
159     }
160 
161     /**
162      * Handle logout for slo service.
163      *
164      * @param singleLogoutService the service
165      * @param ticketId the ticket id
166      * @return the logout request
167      */
168     private LogoutRequest handleLogoutForSloService(final SingleLogoutService singleLogoutService, final String ticketId) {
169         if (!singleLogoutService.isLoggedOutAlready()) {
170 
171             final RegisteredService registeredService = servicesManager.findServiceBy(singleLogoutService);
172             if (serviceSupportsSingleLogout(registeredService)) {
173                 final LogoutRequest logoutRequest = new LogoutRequest(ticketId, singleLogoutService);
174                 final LogoutType type = registeredService.getLogoutType() == null
175                         ? LogoutType.BACK_CHANNEL : registeredService.getLogoutType();
176 
177                 switch (type) {
178                     case BACK_CHANNEL:
179                         if (performBackChannelLogout(logoutRequest)) {
180                             logoutRequest.setStatus(LogoutRequestStatus.SUCCESS);
181                         } else {
182                             logoutRequest.setStatus(LogoutRequestStatus.FAILURE);
183                             LOGGER.warn("Logout message not sent to [{}]; Continuing processing...", singleLogoutService.getId());
184                         }
185                         break;
186                     default:
187                         logoutRequest.setStatus(LogoutRequestStatus.NOT_ATTEMPTED);
188                         break;
189                 }
190                 return logoutRequest;
191             }
192 
193         }
194         return null;
195     }
196     /**
197      * Log out of a service through back channel.
198      *
199      * @param request the logout request.
200      * @return if the logout has been performed.
201      */
202     private boolean performBackChannelLogout(final LogoutRequest request) {
203         try {
204             final String logoutRequest = this.logoutMessageBuilder.create(request);
205             request.getService().setLoggedOutAlready(true);
206     
207             LOGGER.debug("Sending logout request for: [{}]", request.getService().getId());
208             final String originalUrl = request.getService().getOriginalUrl();        
209             final LogoutHttpMessage sender = new LogoutHttpMessage(new URL(originalUrl), logoutRequest);
210 
211             return this.httpClient.sendMessageToEndPoint(sender);
212         } catch (final Exception e) {
213             LOGGER.error(e.getMessage(), e);
214         }
215         return false;
216     }
217 
218     /**
219      * Create a logout message for front channel logout.
220      *
221      * @param logoutRequest the logout request.
222      * @return a front SAML logout message.
223      */
224     public String createFrontChannelLogoutMessage(final LogoutRequest logoutRequest) {
225         final String logoutMessage = this.logoutMessageBuilder.create(logoutRequest);
226         LOGGER.trace("Attempting to deflate the logout message [{}]", logoutMessage);
227         return CompressionUtils.deflate(logoutMessage);
228     }
229 
230     /**
231      * Set if the logout is disabled.
232      *
233      * @param singleLogoutCallbacksDisabled if the logout is disabled.
234      */
235     public void setSingleLogoutCallbacksDisabled(final boolean singleLogoutCallbacksDisabled) {
236         this.singleLogoutCallbacksDisabled = singleLogoutCallbacksDisabled;
237     }
238            
239     /**
240      * A logout http message that is accompanied by a special content type
241      * and formatting.
242      * @since 4.1.0
243      */
244     private final class LogoutHttpMessage extends HttpMessage {
245         
246         /**
247          * Constructs a logout message, whose method of submission
248          * is controlled by the {@link LogoutManagerImpl#asynchronous}.
249          * 
250          * @param url The url to send the message to
251          * @param message Message to send to the url
252          */
253         public LogoutHttpMessage(final URL url, final String message) {
254             super(url, message, LogoutManagerImpl.this.asynchronous);
255             setContentType(MediaType.APPLICATION_FORM_URLENCODED_VALUE);
256         }
257 
258         /**
259          * {@inheritDoc}.
260          * Prepends the string "<code>logoutRequest=</code>" to the message body.
261          */
262         @Override
263         protected String formatOutputMessageInternal(final String message) {
264             return LOGOUT_PARAMETER_NAME + "=" + super.formatOutputMessageInternal(message);
265         }        
266     }
