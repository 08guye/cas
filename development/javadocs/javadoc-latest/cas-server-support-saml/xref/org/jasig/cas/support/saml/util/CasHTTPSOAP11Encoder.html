19  package org.jasig.cas.support.saml.util;
20  
21  import org.opensaml.Configuration;
22  import org.opensaml.common.SAMLObject;
23  import org.opensaml.saml1.binding.encoding.HTTPSOAP11Encoder;
24  import org.opensaml.ws.soap.common.SOAPObjectBuilder;
25  import org.opensaml.ws.soap.soap11.Body;
26  import org.opensaml.ws.soap.soap11.Envelope;
27  import org.opensaml.ws.soap.util.SOAPConstants;
28  import org.opensaml.xml.XMLObjectBuilderFactory;
29  
30  /**
31   * Override OpenSAML {@link HTTPSOAP11Encoder} such that SOAP-ENV XML namespace prefix is used for SOAP envelope
32   * elements.  This is needed for backward compatibility with certain CAS clients (e.g. Java CAS client).
33   *
34   * @author Marvin S. Addison
35   * @since 3.5.1
36   * @deprecated As of 4.1, the class name is abbreviated in a way that is not per camel-casing standards and will be renamed in the future.
37   */
38  @Deprecated
39  public final class CasHTTPSOAP11Encoder extends HTTPSOAP11Encoder {
40      private static final String OPENSAML_11_SOAP_NS_PREFIX = "SOAP-ENV";
41  
42      @Override
43      protected Envelope buildSOAPMessage(final SAMLObject samlMessage) {
44          final XMLObjectBuilderFactory builderFactory = Configuration.getBuilderFactory();
45  
46          final SOAPObjectBuilder<Envelope> envBuilder =
47                  (SOAPObjectBuilder<Envelope>) builderFactory.getBuilder(Envelope.DEFAULT_ELEMENT_NAME);
48          final Envelope envelope = envBuilder.buildObject(
49                  SOAPConstants.SOAP11_NS, Envelope.DEFAULT_ELEMENT_LOCAL_NAME, OPENSAML_11_SOAP_NS_PREFIX);
50  
51          final SOAPObjectBuilder<Body> bodyBuilder =
52                  (SOAPObjectBuilder<Body>) builderFactory.getBuilder(Body.DEFAULT_ELEMENT_NAME);
53          final Body body = bodyBuilder.buildObject(
54                  SOAPConstants.SOAP11_NS, Body.DEFAULT_ELEMENT_LOCAL_NAME, OPENSAML_11_SOAP_NS_PREFIX);
55  
56          body.getUnknownXMLObjects().add(samlMessage);
57          envelope.setBody(body);
58  
59          return envelope;
60      }
61  }
