19  package org.jasig.cas.support.oauth.web;
20  
21  import com.fasterxml.jackson.core.JsonFactory;
22  import com.fasterxml.jackson.core.JsonGenerator;
23  import com.fasterxml.jackson.databind.ObjectMapper;
24  import org.apache.commons.lang3.StringUtils;
25  import org.jasig.cas.authentication.principal.Principal;
26  import org.jasig.cas.support.oauth.OAuthConstants;
27  import org.jasig.cas.ticket.TicketGrantingTicket;
28  import org.jasig.cas.ticket.registry.TicketRegistry;
29  import org.slf4j.Logger;
30  import org.slf4j.LoggerFactory;
31  import org.springframework.web.servlet.ModelAndView;
32  import org.springframework.web.servlet.mvc.AbstractController;
33  import javax.servlet.http.HttpServletRequest;
34  import javax.servlet.http.HttpServletResponse;
35  import java.util.Map;
36  
37  /**
38   * This controller returns a profile for the authenticated user
39   * (identifier + attributes), found with the access token (CAS granting
40   * ticket).
41   *
42   * @author Jerome Leleu
43   * @since 3.5.0
44   */
45  public final class OAuth20ProfileController extends AbstractController {
46  
47      private static final Logger LOGGER = LoggerFactory.getLogger(OAuth20ProfileController.class);
48  
49      private static final String ID = "id";
50  
51      private static final String ATTRIBUTES = "attributes";
52  
53      private final TicketRegistry ticketRegistry;
54  
55      private final JsonFactory jsonFactory = new JsonFactory(new ObjectMapper());
56  
57      /**
58       * Instantiates a new o auth20 profile controller.
59       *
60       * @param ticketRegistry the ticket registry
61       */
62      public OAuth20ProfileController(final TicketRegistry ticketRegistry) {
63          this.ticketRegistry = ticketRegistry;
64      }
65  
66      @Override
67      protected ModelAndView handleRequestInternal(final HttpServletRequest request, final HttpServletResponse response) throws Exception {
68          String accessToken = request.getParameter(OAuthConstants.ACCESS_TOKEN);
69          if (StringUtils.isBlank(accessToken)) {
70              final String authHeader = request.getHeader("Authorization");
71              if (StringUtils.isNotBlank(authHeader) && authHeader.startsWith(OAuthConstants.BEARER_TOKEN + " ")) {
72                  accessToken = authHeader.substring(OAuthConstants.BEARER_TOKEN.length() + 1);
73              }
74          }
75          LOGGER.debug("{} : {}", OAuthConstants.ACCESS_TOKEN, accessToken);
76  
77          try (final JsonGenerator jsonGenerator = this.jsonFactory.createJsonGenerator(response.getWriter())) {
78              response.setContentType("application/json");
79              // accessToken is required
80              if (StringUtils.isBlank(accessToken)) {
81                  LOGGER.error("Missing {}", OAuthConstants.ACCESS_TOKEN);
82                  jsonGenerator.writeStartObject();
83                  jsonGenerator.writeStringField("error", OAuthConstants.MISSING_ACCESS_TOKEN);
84                  jsonGenerator.writeEndObject();
85                  return null;
86              }
87              // get ticket granting ticket
88              final TicketGrantingTicket ticketGrantingTicket = (TicketGrantingTicket) this.ticketRegistry.getTicket(accessToken);
89              if (ticketGrantingTicket == null || ticketGrantingTicket.isExpired()) {
90                  LOGGER.error("expired accessToken : {}", accessToken);
91                  jsonGenerator.writeStartObject();
92                  jsonGenerator.writeStringField("error", OAuthConstants.EXPIRED_ACCESS_TOKEN);
93                  jsonGenerator.writeEndObject();
94                  return null;
95              }
96              // generate profile : identifier + attributes
97              final Principal principal = ticketGrantingTicket.getAuthentication().getPrincipal();
98              jsonGenerator.writeStartObject();
99              jsonGenerator.writeStringField(ID, principal.getId());
100             jsonGenerator.writeArrayFieldStart(ATTRIBUTES);
101             final Map<String, Object> attributes = principal.getAttributes();
102             for (final Map.Entry<String, Object> entry : attributes.entrySet()) {
103                 jsonGenerator.writeStartObject();
104                 jsonGenerator.writeObjectField(entry.getKey(), entry.getValue());
105                 jsonGenerator.writeEndObject();
106             }
107             jsonGenerator.writeEndArray();
108             jsonGenerator.writeEndObject();
109             return null;
110         } finally {
111             response.flushBuffer();
112         }
113     }
114 }
