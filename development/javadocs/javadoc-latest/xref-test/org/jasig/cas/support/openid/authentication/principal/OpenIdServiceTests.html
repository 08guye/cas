19  package org.jasig.cas.support.openid.authentication.principal;
20  
21  import org.jasig.cas.CentralAuthenticationService;
22  import org.jasig.cas.authentication.principal.Response;
23  import org.jasig.cas.util.ApplicationContextProvider;
24  import org.junit.Before;
25  import org.junit.Test;
26  import org.openid4java.association.Association;
27  import org.openid4java.server.ServerAssociationStore;
28  import org.openid4java.server.ServerManager;
29  import org.slf4j.Logger;
30  import org.slf4j.LoggerFactory;
31  import org.springframework.context.ApplicationContext;
32  import org.springframework.mock.web.MockHttpServletRequest;
33  
34  import static org.junit.Assert.*;
35  import static org.mockito.Mockito.*;
36  
37  /**
38   * @author Scott Battaglia
39   * @since 3.1
40   */
41  public class OpenIdServiceTests {
42  
43      private static final Logger LOGGER = LoggerFactory.getLogger(OpenIdServiceTests.class);
44  
45      private OpenIdService openIdService;
46      private ApplicationContext context;
47      private CentralAuthenticationService cas;
48      private ServerManager manager;
49      private ServerAssociationStore sharedAssociations;
50      private final MockHttpServletRequest request = new MockHttpServletRequest();
51  
52      @Before
53      public void setUp() throws Exception {
54          request.addParameter("openid.identity", "http://openid.ja-sig.org/battags");
55          request.addParameter("openid.return_to", "http://www.ja-sig.org/?service=fa");
56          request.addParameter("openid.mode", "checkid_setup");
57          sharedAssociations = mock(ServerAssociationStore.class);
58          manager = new ServerManager();
59          manager.setOPEndpointUrl("https://localshot:8443/cas/login");
60          manager.setEnforceRpId(false);
61          manager.setSharedAssociations(sharedAssociations);
62          context = mock(ApplicationContext.class);
63          cas = mock(CentralAuthenticationService.class);
64  
65          when(context.getBean("serverManager")).thenReturn(manager);
66          when(context.getBean("centralAuthenticationService", CentralAuthenticationService.class)).thenReturn(cas);
67  
68          final ApplicationContextProvider contextProvider = new ApplicationContextProvider();
69          contextProvider.setApplicationContext(context);
70      }
71  
72      @Test
73      public void verifyGetResponse() {
74          openIdService = OpenIdService.createServiceFrom(request);
75          final Response response = this.openIdService.getResponse("test");
76          try {
77              verify(cas, never()).validateServiceTicket("test", openIdService);
78          } catch (final Exception e) {
79              LOGGER.debug("Exception during verification of service ticket", e);
80          }
81          assertNotNull(response);
82  
83          assertEquals("test", response.getAttributes().get("openid.assoc_handle"));
84          assertEquals("http://www.ja-sig.org/?service=fa", response.getAttributes().get("openid.return_to"));
85          assertEquals("http://openid.ja-sig.org/battags", response.getAttributes().get("openid.identity"));
86  
87          final Response response2 = this.openIdService.getResponse(null);
88          assertEquals("cancel", response2.getAttributes().get("openid.mode"));
89      }
90  
91      @Test
92      public void verifySmartModeGetResponse() {
93          request.addParameter("openid.assoc_handle", "test");
94          openIdService = OpenIdService.createServiceFrom(request);
95          Association association = null;
96          try {
97              association = Association.generate(Association.TYPE_HMAC_SHA1, "test", 60);
98          } catch (final Exception e) {
99              fail("Could not generate association");
100         }
101         when(sharedAssociations.load("test")).thenReturn(association);
102         final Response response = this.openIdService.getResponse("test");
103         try {
104             verify(cas).validateServiceTicket("test", openIdService);
105         } catch (final Exception e) {
106             fail("Error while validating ticket");
107         }
108 
109         request.removeParameter("openid.assoc_handle");
110         assertNotNull(response);
111 
112         assertEquals("test", response.getAttributes().get("openid.assoc_handle"));
113         assertEquals("http://www.ja-sig.org/?service=fa", response.getAttributes().get("openid.return_to"));
114         assertEquals("http://openid.ja-sig.org/battags", response.getAttributes().get("openid.identity"));
115     }
116 
117     @Test
118     public void verifyExpiredAssociationGetResponse() {
119         request.addParameter("openid.assoc_handle", "test");
120         openIdService = OpenIdService.createServiceFrom(request);
121         Association association = null;
122         try {
123             association = Association.generate(Association.TYPE_HMAC_SHA1, "test", 2);
124         } catch (final Exception e) {
125             fail("Could not generate association");
126         }
127         when(sharedAssociations.load("test")).thenReturn(association);
128         synchronized (this) {
129             try {
130                 this.wait(3000);
131             } catch (final InterruptedException ie) {
132                 fail("Could not wait long enough to check association expiry date");
133             }
134         }
135         final Response response = this.openIdService.getResponse("test");
136         request.removeParameter("openid.assoc_handle");
137         assertNotNull(response);
138 
139         assertEquals(1, response.getAttributes().size());
140         assertEquals("cancel", response.getAttributes().get("openid.mode"));
141     }
142 
143     @Test
144     public void verifyEquals() {
145         final MockHttpServletRequest request1 = new MockHttpServletRequest();
146         request1.addParameter("openid.identity", "http://openid.ja-sig.org/battags");
147         request1.addParameter("openid.return_to", "http://www.ja-sig.org/?service=fa");
148         request1.addParameter("openid.mode", "openid.checkid_setup");
149 
150         final MockHttpServletRequest request2 = new MockHttpServletRequest();
151         request2.addParameter("openid.identity", "http://openid.ja-sig.org/battags");
152         request2.addParameter("openid.return_to", "http://www.ja-sig.org/?service=fa");
153 
154         final OpenIdService o1 = OpenIdService.createServiceFrom(request1);
155         final OpenIdService o2 = OpenIdService.createServiceFrom(request2);
156 
157         assertTrue(o1.equals(o2));
158         assertFalse(o1.equals(new Object()));
159     }
