59  public class FrontChannelLogoutActionTests {
60  
61      private static final String FLOW_EXECUTION_KEY = "12234";
62  
63      private static final String TICKET_ID = "ST-XXX";
64  
65      private FrontChannelLogoutAction frontChannelLogoutAction;
66  
67      private MockHttpServletRequest request;
68  
69      private MockHttpServletResponse response;
70  
71      private RequestContext requestContext;
72  
73      @Before
74      public void onSetUp() throws Exception {
75          final LogoutManager logoutManager = new LogoutManagerImpl(mock(ServicesManager.class),
76                  new SimpleHttpClientFactoryBean().getObject(), new SamlCompliantLogoutMessageCreator());
77          this.frontChannelLogoutAction = new FrontChannelLogoutAction(logoutManager);
78  
79          this.request = new MockHttpServletRequest();
80          this.response = new MockHttpServletResponse();
81          this.requestContext = mock(RequestContext.class);
82          final ServletExternalContext servletExternalContext = mock(ServletExternalContext.class);
83          when(this.requestContext.getExternalContext()).thenReturn(servletExternalContext);
84          when(servletExternalContext.getNativeRequest()).thenReturn(request);
85          when(servletExternalContext.getNativeResponse()).thenReturn(response);
86          final LocalAttributeMap flowScope = new LocalAttributeMap();
87          when(this.requestContext.getFlowScope()).thenReturn(flowScope);
88          final MockFlowExecutionKey mockFlowExecutionKey = new MockFlowExecutionKey(FLOW_EXECUTION_KEY);
89          final MockFlowExecutionContext mockFlowExecutionContext = new MockFlowExecutionContext();
90          mockFlowExecutionContext.setKey(mockFlowExecutionKey);
91          when(this.requestContext.getFlowExecutionContext()).thenReturn(mockFlowExecutionContext);
92      }
93  
94      @Test
95      public void verifyLogoutNoRequest() throws Exception {
96          this.requestContext.getFlowScope().put(FrontChannelLogoutAction.LOGOUT_INDEX, 0);
97          final Event event = this.frontChannelLogoutAction.doExecute(this.requestContext);
98          assertEquals(FrontChannelLogoutAction.FINISH_EVENT, event.getId());
99      }
100 
101     @Test
102     public void verifyLogoutNoIndex() throws Exception {
103         WebUtils.putLogoutRequests(this.requestContext, Collections.<LogoutRequest>emptyList());
104         final Event event = this.frontChannelLogoutAction.doExecute(this.requestContext);
105         assertEquals(FrontChannelLogoutAction.FINISH_EVENT, event.getId());
106     }
107 
108     @Test
109     public void verifyLogoutOneLogoutRequestSuccess() throws Exception {
110         final LogoutRequest logoutRequest = new LogoutRequest("", null);
111         logoutRequest.setStatus(LogoutRequestStatus.SUCCESS);
112         WebUtils.putLogoutRequests(this.requestContext, Collections.<LogoutRequest>emptyList());
113         this.requestContext.getFlowScope().put(FrontChannelLogoutAction.LOGOUT_INDEX, 0);
114         final Event event = this.frontChannelLogoutAction.doExecute(this.requestContext);
115         assertEquals(FrontChannelLogoutAction.FINISH_EVENT, event.getId());
116     }
117 
118     @Test
119     public void verifyLogoutOneLogoutRequestNotAttempted() throws Exception {
120         final String fakeUrl = "http://url";
121         final LogoutRequest logoutRequest = new LogoutRequest(TICKET_ID, new SimpleWebApplicationServiceImpl(fakeUrl));
122         WebUtils.putLogoutRequests(this.requestContext, Arrays.asList(logoutRequest));
123         this.requestContext.getFlowScope().put(FrontChannelLogoutAction.LOGOUT_INDEX, 0);
124         final Event event = this.frontChannelLogoutAction.doExecute(this.requestContext);
125         assertEquals(FrontChannelLogoutAction.REDIRECT_APP_EVENT, event.getId());
126         final List<LogoutRequest> list = WebUtils.getLogoutRequests(this.requestContext);
127         assertEquals(1, list.size());
128         final String url = (String) event.getAttributes().get("logoutUrl");
129         assertTrue(url.startsWith(fakeUrl + "?SAMLRequest="));
130         final byte[] samlMessage = CompressionUtils.decodeBase64ToByteArray(
131                 URLDecoder.decode(StringUtils.substringAfter(url, "?SAMLRequest="), "UTF-8"));
132         final Inflater decompresser = new Inflater();
133         decompresser.setInput(samlMessage);
134         final byte[] result = new byte[1000];
135         decompresser.inflate(result);
136         decompresser.end();
137         final String message = new String(result);
138         assertTrue(message.startsWith("<samlp:LogoutRequest xmlns:samlp=\"urn:oasis:names:tc:SAML:2.0:protocol\" ID=\""));
139         assertTrue(message.indexOf("<samlp:SessionIndex>" + TICKET_ID + "</samlp:SessionIndex>") >= 0);
140     }
