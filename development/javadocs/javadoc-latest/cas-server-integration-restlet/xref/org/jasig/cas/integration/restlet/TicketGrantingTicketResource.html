38  import org.springframework.beans.factory.annotation.Autowired;

40  /**
41   * Implementation of a Restlet resource for creating Service Tickets from a
42   * {@link org.jasig.cas.ticket.TicketGrantingTicket}, as well as deleting a
43   * {@link org.jasig.cas.ticket.TicketGrantingTicket}.
44   *
45   * @author Scott Battaglia
46   * @since 3.3
47   * @deprecated As of 4.1. Use the {@link TicketResource} implementation from cas-server-support-rest module
48   */
49  @Deprecated
50  public final class TicketGrantingTicketResource extends ServerResource {
51      private static final Logger LOGGER = LoggerFactory.getLogger(TicketGrantingTicketResource.class);
52  
53      @Autowired
54      private CentralAuthenticationService centralAuthenticationService;
55  
56      private String ticketGrantingTicketId;
57  
58      @Override
59      public void init(final Context context, final Request request, final Response response) {
60          super.init(context, request, response);
61          this.ticketGrantingTicketId = (String) request.getAttributes().get("ticketGrantingTicketId");
62          this.setNegotiated(false);
63          this.getVariants().add(new Variant(MediaType.APPLICATION_WWW_FORM));
64      }
65  
66      /**
67       * Removes the TGT.
68       */
69      @Delete
70      public void removeRepresentations() {
71          this.centralAuthenticationService.destroyTicketGrantingTicket(this.ticketGrantingTicketId);
72          getResponse().setStatus(Status.SUCCESS_OK);
73      }
74  
75      /**
76       * Accept service and attempt to grant service ticket.
77       *
78       * @param entity the entity
79       */
80      @Post
81      public void acceptRepresentation(final Representation entity) {
82          final Form form = new Form(entity);
83          final String serviceUrl = form.getFirstValue("service");
84          try {
85              final ServiceTicket serviceTicketId = this.centralAuthenticationService.grantServiceTicket(
86                      this.ticketGrantingTicketId,
87                      new SimpleWebApplicationServiceImpl(serviceUrl));
88              getResponse().setEntity(serviceTicketId.getId(), MediaType.TEXT_PLAIN);
89          } catch (final InvalidTicketException e) {
90              getResponse().setStatus(Status.CLIENT_ERROR_NOT_FOUND, "TicketGrantingTicket could not be found.");
91          } catch (final Exception e) {
92              LOGGER.error(e.getMessage(), e);
93              getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
94          }
95      }
96  }
