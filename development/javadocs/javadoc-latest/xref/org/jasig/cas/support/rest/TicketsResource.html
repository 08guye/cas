19  package org.jasig.cas.support.rest;
20  
21  import org.jasig.cas.CentralAuthenticationService;
22  import org.jasig.cas.authentication.Credential;
23  import org.jasig.cas.authentication.UsernamePasswordCredential;
24  import org.jasig.cas.authentication.principal.SimpleWebApplicationServiceImpl;
25  import org.jasig.cas.ticket.InvalidTicketException;
26  import org.jasig.cas.ticket.ServiceTicket;
27  import org.jasig.cas.ticket.TicketGrantingTicket;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  import org.springframework.beans.factory.annotation.Autowired;
31  import org.springframework.http.HttpHeaders;
32  import org.springframework.http.HttpStatus;
33  import org.springframework.http.MediaType;
34  import org.springframework.http.ResponseEntity;
35  import org.springframework.util.MultiValueMap;
36  import org.springframework.web.bind.annotation.PathVariable;
37  import org.springframework.web.bind.annotation.RequestBody;
38  import org.springframework.web.bind.annotation.RequestMapping;
39  import org.springframework.web.bind.annotation.RequestMethod;
40  import org.springframework.web.bind.annotation.RestController;
41  import javax.servlet.http.HttpServletRequest;
42  import java.net.URI;
43  import java.util.Formatter;
44  
45  /**
46   * {@link org.springframework.web.bind.annotation.RestController} implementation of CAS' REST API.
47   *
48   * This class implements main CAS RESTful resource for vending/deleting TGTs and vending STs:
49   *
50   * <ul>
51   *     <li>{@code POST /v1/tickets}</li>
52   *     <li>{@code POST /v1/tickets/{TGT-id}}</li>
53   *     <li>{@code DELETE /v1/tickets/{TGT-id}}</li>
54   * </ul>
55   *
56   * @author Dmitriy Kopylenko
57   * @since 4.1.0
58   */
59  @RestController("/v1")
60  public class TicketsResource {
61  
62      private static final Logger LOGGER = LoggerFactory.getLogger(TicketsResource.class);
63  
64      @Autowired
65      private CentralAuthenticationService cas;
66  
67      /**
68       * Create new ticket granting ticket.
69       *
70       * @param requestBody username and password application/x-www-form-urlencoded values
71       * @param request raw HttpServletRequest used to call this method
72       * @return ResponseEntity representing RESTful response
73       */
74      @RequestMapping(value = "/tickets", method = RequestMethod.POST, consumes = MediaType.APPLICATION_FORM_URLENCODED_VALUE)
75      public final ResponseEntity<String> createTicketGrantingTicket(@RequestBody final MultiValueMap<String, String> requestBody,
76                                                                     final HttpServletRequest request) {
77          try (Formatter fmt = new Formatter()) {
78              final TicketGrantingTicket tgtId = this.cas.createTicketGrantingTicket(obtainCredential(requestBody));
79              final URI ticketReference = new URI(request.getRequestURL().toString() + "/" + tgtId.getId());
80              final HttpHeaders headers = new HttpHeaders();
81              headers.setLocation(ticketReference);
82              headers.setContentType(MediaType.TEXT_HTML);
83              fmt.format("<!DOCTYPE HTML PUBLIC \\\"-//IETF//DTD HTML 2.0//EN\\\"><html><head><title>");
84              //IETF//DTD HTML 2.0//EN\\\"><html><head><title>");
85              fmt.format("%s %s", HttpStatus.CREATED, HttpStatus.CREATED.getReasonPhrase())
86                      .format("</title></head><body><h1>TGT Created</h1><form action=\"%s", ticketReference.toString())
87                      .format("\" method=\"POST\">Service:<input type=\"text\" name=\"service\" value=\"\">")
88                      .format("<br><input type=\"submit\" value=\"Submit\"></form></body></html>");
89              return new ResponseEntity<String>(fmt.toString(), headers, HttpStatus.CREATED);
90          } catch (final Throwable e) {
91              LOGGER.error(e.getMessage(), e);
92              return new ResponseEntity<String>(e.getMessage(), HttpStatus.BAD_REQUEST);
93          }
94      }
95  
96      /**
97       * Create new service ticket.
98       *
99       * @param requestBody service application/x-www-form-urlencoded value
100      * @param tgtId ticket granting ticket id URI path param
101      * @return {@link ResponseEntity} representing RESTful response
102      */
103     @RequestMapping(value = "/tickets/{tgtId:.+}", method = RequestMethod.POST, consumes = MediaType.APPLICATION_FORM_URLENCODED_VALUE)
104     public final ResponseEntity<String> createServiceTicket(@RequestBody final MultiValueMap<String, String> requestBody,
105                                                             @PathVariable("tgtId") final String tgtId) {
106         try {
107             final ServiceTicket serviceTicketId = this.cas.grantServiceTicket(tgtId,
108                     new SimpleWebApplicationServiceImpl(requestBody.getFirst("service")));
109             return new ResponseEntity<String>(serviceTicketId.getId(), HttpStatus.OK);
110         } catch (final InvalidTicketException e) {
111             return new ResponseEntity<String>("TicketGrantingTicket could not be found", HttpStatus.NOT_FOUND);
112         } catch (final Exception e) {
113             LOGGER.error(e.getMessage(), e);
114             return new ResponseEntity<String>(e.getMessage(), HttpStatus.BAD_REQUEST);
115         }
116     }
117 
118     /**
119      * Destroy ticket granting ticket.
120      *
121      * @param tgtId ticket granting ticket id URI path param
122      * @return {@link ResponseEntity} representing RESTful response. Signals
123      * {@link HttpStatus#OK} when successful.
124      */
125     @RequestMapping(value = "/tickets/{tgtId:.+}", method = RequestMethod.DELETE)
126     public final ResponseEntity<String> deleteTicketGrantingTicket(@PathVariable("tgtId") final String tgtId) {
127         this.cas.destroyTicketGrantingTicket(tgtId);
128         return new ResponseEntity<String>(tgtId, HttpStatus.OK);
129     }
130 
131     /**
132      * Obtain credential from the request. Could be overridden by subclasses.
133      *
134      * @param requestBody raw entity request body
135      * @return the credential instance
136      */
137     protected Credential obtainCredential(final MultiValueMap<String, String> requestBody) {
138         return new UsernamePasswordCredential(requestBody.getFirst("username"), requestBody.getFirst("password"));
139     }
140 }
