40  /**
41   * Authentication Handler that accepts X509 Certificates, determines their
42   * validity and ensures that they were issued by a trusted issuer. (targeted at
43   * X509v3) Optionally checks KeyUsage extension in the user certificate
44   * (container should do that too). Note that this handler trusts the servlet
45   * container to do some initial checks like path validation. Deployers can
46   * supply an optional pattern to match subject dns against to further restrict
47   * certificates in case they are not using their own issuer. It's also possible
48   * to specify a maximum pathLength for the SUPPLIED certificates. (note that
49   * this does not include a pathLength check for the root certificate)
50   * [PathLength is 0 for the CA certificate that issues the end-user certificate]
51   *
52   * @author Scott Battaglia
53   * @author Jan Van der Velpen
54   * @since 3.0.0.4
55   */
56  public class X509CredentialsAuthenticationHandler extends AbstractPreAndPostProcessingAuthenticationHandler {
57  
58      /** Default setting to limit the number of intermediate certificates. */
59      private static final int DEFAULT_MAXPATHLENGTH = 1;
60  
61      /** Default setting whether to allow unspecified number of intermediate certificates. */
62      private static final boolean DEFAULT_MAXPATHLENGTH_ALLOW_UNSPECIFIED = false;
63  
64      /** Default setting to check keyUsage extension. */
65      private static final boolean DEFAULT_CHECK_KEYUSAGE = false;
66  
67      /** Default setting to force require "KeyUsage" extension. */
68      private static final boolean DEFAULT_REQUIRE_KEYUSAGE = false;
69  
70      /** Default subject pattern match. */
71      private static final Pattern DEFAULT_SUBJECT_DN_PATTERN = Pattern.compile(".*");
72  
73      /** OID for KeyUsage X.509v3 extension field. */
74      private static final String KEY_USAGE_OID = "2.5.29.15";
75  
76      /** Instance of Logging. */
77      private final Logger logger = LoggerFactory.getLogger(getClass());
78  
79      /** The compiled pattern supplied by the deployer. */
80      @NotNull
81      private Pattern regExTrustedIssuerDnPattern;
82  
83      /**
84       * Deployer supplied setting for maximum pathLength in a SUPPLIED
85       * certificate.
86       */
87      private int maxPathLength = DEFAULT_MAXPATHLENGTH;
88  
89      /**
90       * Deployer supplied setting to allow unlimited pathLength in a SUPPLIED
91       * certificate.
92       */
93      private boolean maxPathLengthAllowUnspecified = DEFAULT_MAXPATHLENGTH_ALLOW_UNSPECIFIED;
94  
95      /** Deployer supplied setting to check the KeyUsage extension. */
96      private boolean checkKeyUsage = DEFAULT_CHECK_KEYUSAGE;
97  
98      /**
99       * Deployer supplied setting to force require the correct KeyUsage
100      * extension.
101      */
102     private boolean requireKeyUsage = DEFAULT_REQUIRE_KEYUSAGE;
103 
104     /** The compiled pattern for trusted DN's supplied by the deployer. */
105     @NotNull
106     private Pattern regExSubjectDnPattern = DEFAULT_SUBJECT_DN_PATTERN;
107 
108     /** Certificate revocation checker component. */
109     @NotNull
110     private RevocationChecker revocationChecker = new NoOpRevocationChecker();
111 
112 
113     @Override
114     public boolean supports(final Credential credential) {
115         return credential != null
116                 && X509CertificateCredential.class.isAssignableFrom(credential
117                         .getClass());
118     }
119 
120     /**
121      * {@inheritDoc}
122      */
123     @Override
124     protected final HandlerResult doAuthentication(final Credential credential) throws GeneralSecurityException, PreventedException {
125 
126         final X509CertificateCredential x509Credential = (X509CertificateCredential) credential;
127         final X509Certificate[] certificates = x509Credential.getCertificates();
128 
129         X509Certificate clientCert = null;
130         boolean hasTrustedIssuer = false;
131         for (int i = certificates.length - 1; i >= 0; i--) {
132             final X509Certificate certificate = certificates[i];
133             logger.debug("Evaluating {}", CertUtils.toString(certificate));
134 
135             validate(certificate);
136 
137             if (!hasTrustedIssuer) {
138                 hasTrustedIssuer = isCertificateFromTrustedIssuer(certificate);
139             }
140 
141             // getBasicConstraints returns pathLenContraint which is generally
142             // >=0 when this is a CA cert and -1 when it's not
143             final int pathLength = certificate.getBasicConstraints();
144             if (pathLength < 0) {
145                 logger.debug("Found valid client certificate");
146                 clientCert = certificate;
147             } else {
148                 logger.debug("Found valid CA certificate");
149             }
150         }
151         if (hasTrustedIssuer && clientCert != null) {
152             x509Credential.setCertificate(clientCert);
153             return new HandlerResult(this, x509Credential, new SimplePrincipal(x509Credential.getId()));
154         }
155         throw new FailedLoginException();
156     }

194     /**
195      * Sets the component responsible for evaluating certificate revocation status for client
196      * certificates presented to handler. The default checker is a NO-OP implementation
197      * for backward compatibility with previous versions that do not perform revocation
198      * checking.
199      *
200      * @param checker Revocation checker component.
201      */
202     public void setRevocationChecker(final RevocationChecker checker) {
203         this.revocationChecker = checker;
204     }
205 
206     /**
207      * Validate the X509Certificate received.
208      *
209      * @param cert the cert
210      * @throws GeneralSecurityException the general security exception
211      */
212     private void validate(final X509Certificate cert) throws GeneralSecurityException {
213         cert.checkValidity();
214         this.revocationChecker.check(cert);
215 
216         final int pathLength = cert.getBasicConstraints();
217         if (pathLength < 0) {
218             if (!isCertificateAllowed(cert)) {
219                 throw new FailedLoginException(
220                         "Certificate subject does not match pattern " + this.regExSubjectDnPattern.pattern());
221             }
222             if (this.checkKeyUsage && !isValidKeyUsage(cert)) {
223                 throw new FailedLoginException(
224                         "Certificate keyUsage constraint forbids SSL client authentication.");
225             }
226         } else {
227             // Check pathLength for CA cert
228             if (pathLength == Integer.MAX_VALUE && !this.maxPathLengthAllowUnspecified) {
229                 throw new FailedLoginException("Unlimited certificate path length not allowed by configuration.");
230             } else if (pathLength > this.maxPathLength && pathLength < Integer.MAX_VALUE) {
231                 throw new FailedLoginException(String.format(
232                         "Certificate path length %s exceeds maximum value %s.", pathLength, this.maxPathLength));
233             }
234         }
235     }
236 
237     /**
238      * Checks if is valid key usage. <p>
239      * KeyUsage ::= BIT STRING { digitalSignature (0), nonRepudiation (1),
240      * keyEncipherment (2), dataEncipherment (3), keyAgreement (4),
241      * keyCertSign (5), cRLSign (6), encipherOnly (7), decipherOnly (8) }
242      *          
243      * @param certificate the certificate
244      * @return true, if  valid key usage
245      */
246     private boolean isValidKeyUsage(final X509Certificate certificate) {
247         logger.debug("Checking certificate keyUsage extension");
248         final boolean[] keyUsage = certificate.getKeyUsage();
249         if (keyUsage == null) {
250             logger.warn("Configuration specifies checkKeyUsage but keyUsage extension not found in certificate.");
251             return !this.requireKeyUsage;
252         }
253 
254         final boolean valid;
255         if (isCritical(certificate, KEY_USAGE_OID) || this.requireKeyUsage) {
256             logger.debug("KeyUsage extension is marked critical or required by configuration.");
257             valid = keyUsage[0];
258         } else {
259             logger.debug(
260                "KeyUsage digitalSignature=%s, Returning true since keyUsage validation not required by configuration.");
261             valid = true;
262         }
263         return valid;
264     }
265 
266     /**
267      * Checks if critical extension oids contain the extension oid.
268      *
269      * @param certificate the certificate
270      * @param extensionOid the extension oid
271      * @return true, if  critical
272      */
273     private boolean isCritical(final X509Certificate certificate, final String extensionOid) {
274         final Set<String> criticalOids = certificate.getCriticalExtensionOIDs();
275 
276         if (criticalOids == null || criticalOids.isEmpty()) {
277             return false;
278         }
279 
280         return criticalOids.contains(extensionOid);
281     }
282 
283     /**
284      * Checks if is certificate allowed based no the pattern given.
285      *
286      * @param cert the cert
287      * @return true, if  certificate allowed
288      */
289     private boolean isCertificateAllowed(final X509Certificate cert) {
290         return doesNameMatchPattern(cert.getSubjectDN(), this.regExSubjectDnPattern);
291     }
292 
293     /**
294      * Checks if is certificate from trusted issuer based on the regex pattern.
295      *
296      * @param cert the cert
297      * @return true, if  certificate from trusted issuer
298      */
299     private boolean isCertificateFromTrustedIssuer(final X509Certificate cert) {
300         return doesNameMatchPattern(cert.getIssuerDN(), this.regExTrustedIssuerDnPattern);
301     }
302 
303     /**
304      * Does principal name match pattern?
305      *
306      * @param principal the principal
307      * @param pattern the pattern
308      * @return true, if successful
309      */
310     private boolean doesNameMatchPattern(final Principal principal,
311             final Pattern pattern) {
312         final String name = principal.getName();
313         final boolean result = pattern.matcher(name).matches();
314         logger.debug(String.format("%s matches %s == %s", pattern.pattern(), name, result));
315         return result;
316     }
