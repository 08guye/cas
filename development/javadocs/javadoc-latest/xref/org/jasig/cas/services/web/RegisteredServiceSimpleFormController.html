51  /**
52   * SimpleFormController to handle adding/editing of RegisteredServices.
53   *
54   * @author Scott Battaglia
55   * @since 3.1
56   */
57  @Controller
58  public final class RegisteredServiceSimpleFormController {
59  
60      private static final Logger LOGGER = LoggerFactory.getLogger(RegisteredServiceSimpleFormController.class);
61      
62      private static final String COMMAND_NAME = "registeredService";
63      private static final String VIEW_NAME = "editServiceView";
64      
65      /** Instance of ServiceRegistryManager. */
66      @NotNull
67      private final ServicesManager servicesManager;
68  
69      /** Instance of AttributeRegistry. */
70      @NotNull
71      private final IPersonAttributeDao personAttributeDao;
72      
73      @NotNull
74      @Resource(name="registeredServiceValidator")
75      private Validator validator;
76  
77      /**
78       * Instantiates a new registered service simple form controller.
79       *
80       * @param servicesManager the services manager
81       * @param personAttributeDao the attribute repository
82       */
83      @Autowired
84      public RegisteredServiceSimpleFormController(final ServicesManager servicesManager, 
85              final IPersonAttributeDao personAttributeDao) {
86          this.personAttributeDao = personAttributeDao;
87          this.servicesManager = servicesManager;
88      }
89      
90      /**
91       * Instantiates a new registered service simple form controller.
92       *
93       * @param servicesManager the services manager
94       * @param personAttributeDao the person attribute dao
95       * @param validator the validator
96       */
97      RegisteredServiceSimpleFormController(final ServicesManager servicesManager, 
98              final IPersonAttributeDao personAttributeDao, final Validator validator) {
99          this(servicesManager, personAttributeDao);
100         this.validator = validator;
101     }
102     
103     /**
104      * Sets the require fields and the disallowed fields from the
105      * HttpServletRequest.
106      *
107      * @param request the request
108      * @param binder the binder
109      * @throws Exception the exception
110      */
111     @InitBinder
112     protected void initBinder(final HttpServletRequest request, final ServletRequestDataBinder binder) throws Exception {
113         binder.setRequiredFields("description", "serviceId",
114                 "name", "enabled", "ssoEnabled",
115                 "anonymousAccess", "evaluationOrder");
116         binder.setDisallowedFields("id");
117         binder.registerCustomEditor(String.class, new StringTrimmerEditor(true));
118     }
119 
120     /**
121      * Adds the service to the ServiceRegistry via the ServiceRegistryManager.
122      *
123      * @param service the service
124      * @param result the binding result
125      * @param model the page model
126      * @param request the http request
127      * @return the model and view
128      * @throws Exception the exception
129      */
130     @RequestMapping(method = RequestMethod.POST)
131     protected ModelAndView onSubmit(@ModelAttribute(COMMAND_NAME) final RegisteredService service,
132             final BindingResult result, final ModelMap model, final HttpServletRequest request) throws Exception {
133         
134         updateModelMap(model, request);
135         
136         this.validator.validate(service, result);
137         if (result.hasErrors()) {
138             model.addAttribute("validationErrors", result.getAllErrors());
139             return render(request, model);
140         }
141        
142         RegisteredService svcToUse = service;
143         if (service.getServiceId().startsWith("^") && service instanceof RegisteredServiceImpl) {
144             LOGGER.debug("Detected regular expression starting with ^");
145             final RegexRegisteredService regexService = new RegexRegisteredService();
146             regexService.copyFrom(service);
147             svcToUse = regexService;
148         } else if (!service.getServiceId().startsWith("^") && service instanceof RegexRegisteredService) {
149             LOGGER.debug("Detected ant expression {}", service.getServiceId());
150             final RegisteredServiceImpl regexService = new RegisteredServiceImpl();
151             regexService.copyFrom(service);
152             svcToUse = regexService;
153         } 
154 
155         this.servicesManager.save(svcToUse);
156         LOGGER.info("Saved changes to service {}", svcToUse.getId());
157 
158         final ModelAndView modelAndView = new ModelAndView(new RedirectView(
159                 "manage.html#" + svcToUse.getId(), true));
160         modelAndView.addObject("action", "add");
161         modelAndView.addObject("id", svcToUse.getId());
162 
163         
164         
165         return modelAndView;
166     }
167 
168     /**
169      * Render the page noted by {@link #VIEW_NAME}.
170      * The view is first updated by {@link #updateModelMap(ModelMap, HttpServletRequest)}.
171      * @param request the request
172      * @param model the model
173      * @return the model and view
174      * @throws Exception the exception
175      */
176     @RequestMapping(method=RequestMethod.GET, value= {"add.html", "edit.html"})
177     protected ModelAndView render(final HttpServletRequest request, final ModelMap model)
178             throws Exception {
179         updateModelMap(model, request);
180         return new ModelAndView(VIEW_NAME);
181     }
182     
183     /**
184      * Updates model map. The following objects will be available in the model:
185      * 
186      * <ul>
187      *  <li><code>availableAttributes</code></li>
188      *  <li><code>availableUsernameAttributes</code></li>
189      *  <li><code>pageTitle</code></li>
190      * </ul>
191      *
192      * @param model the model
193      * @param request the request
194      */
195     private void updateModelMap(final ModelMap model, final HttpServletRequest request) {
196         final List<String> possibleAttributeNames = new ArrayList<>();
197         possibleAttributeNames.addAll(this.personAttributeDao.getPossibleUserAttributeNames());
198         Collections.sort(possibleAttributeNames);
199         model.addAttribute("availableAttributes", possibleAttributeNames);
200 
201         final List<String> possibleUsernameAttributeNames = new ArrayList<>();
202         possibleUsernameAttributeNames.addAll(possibleAttributeNames);
203         possibleUsernameAttributeNames.add(0, "");
204         model.addAttribute("availableUsernameAttributes", possibleUsernameAttributeNames);
205         
206         final String path = request.getServletPath().replace("/", "").replace(".html", "").concat("ServiceView");
207         model.addAttribute("pageTitle", path);    
208     }
209     
210     /**
211      * Determines the registered service to be used. 
212      * If no <code>id</code> is specified, a service of type
213      * {@link RegisteredServiceImpl} will be created by default.
214      *
215      * @param id the id
216      * @return the service
217      */
218     @ModelAttribute(COMMAND_NAME)
219     public RegisteredService getCommand(@RequestParam(value="id", required=false) final String id) {
220 
221         if (!StringUtils.hasText(id)) {
222             final RegisteredService service = new RegisteredServiceImpl();
223             LOGGER.debug("Created new service of type {}", service.getClass().getName());
224             return service;
225         }
226 
227         final RegisteredService service = this.servicesManager.findServiceBy(Long.parseLong(id));
228 
229         if (service != null) {
230             LOGGER.debug("Loaded service {}", service.getServiceId());
231         } else {
232             LOGGER.debug("Invalid service id specified.");
233         }
234 
235         return service;
236     }
