19  package org.jasig.cas.support.spnego.web.flow;
20  
21  import org.jasig.cas.authentication.Credential;
22  import org.jasig.cas.support.spnego.authentication.principal.SpnegoCredential;
23  import org.jasig.cas.support.spnego.util.SpnegoConstants;
24  import org.jasig.cas.util.CompressionUtils;
25  import org.jasig.cas.web.flow.AbstractNonInteractiveCredentialsAction;
26  import org.jasig.cas.web.support.WebUtils;
27  import org.springframework.util.StringUtils;
28  import org.springframework.webflow.execution.RequestContext;
29  
30  import javax.servlet.http.HttpServletRequest;
31  import javax.servlet.http.HttpServletResponse;
32  import java.nio.charset.Charset;
33  
34  /**
35   * Second action of a SPNEGO flow : decode the gssapi-data and build a new
36   * {@link org.jasig.cas.support.spnego.authentication.principal.SpnegoCredential}.<br>
37   * Once AbstractNonInteractiveCredentialsAction has executed the authentication
38   * procedure, this action check whether a principal is present in Credential and
39   * add corresponding response headers.
40   *
41   * @author Arnaud Lesueur
42   * @author Marc-Antoine Garrigue
43   * @see <a href="http://ietfreport.isoc.org/idref/rfc4559/#page-2">RFC 4559</a>
44   * @since 3.1
45   */
46  public final class SpnegoCredentialsAction extends AbstractNonInteractiveCredentialsAction {
47  
48  
49      private boolean ntlm;
50  
51      private String messageBeginPrefix = constructMessagePrefix();
52  
53      /**
54       * Behavior in case of SPNEGO authentication failure :<br>
55       * <ul><li>True : if SPNEGO is the last authentication method with no fallback.</li>
56       * <li>False : if an interactive view (eg: login page) should be send to user as SPNEGO failure fallback</li>
57       * </ul>
58       */
59      private boolean send401OnAuthenticationFailure = true;
60  
61      @Override
62      protected Credential constructCredentialsFromRequest(
63              final RequestContext context) {
64          final HttpServletRequest request = WebUtils
65                  .getHttpServletRequest(context);
66  
67          final String authorizationHeader = request
68                  .getHeader(SpnegoConstants.HEADER_AUTHORIZATION);
69  
70          if (StringUtils.hasText(authorizationHeader)
71                  && authorizationHeader.startsWith(this.messageBeginPrefix)
72                  && authorizationHeader.length() > this.messageBeginPrefix.length()) {
73  
74              logger.debug("SPNEGO Authorization header found with {} bytes",
75                      authorizationHeader.length() - this.messageBeginPrefix.length());
76  
77              final byte[] token = CompressionUtils.decodeBase64ToByteArray(authorizationHeader.substring(this.messageBeginPrefix.length()));
78              logger.debug("Obtained token: {}", new String(token, Charset.defaultCharset()));
79              return new SpnegoCredential(token);
80          }
81  
82          return null;
83      }
84  
85      /**
86       * Construct message prefix.
87       *
88       * @return the string
89       */
90      protected String constructMessagePrefix() {
91          return (this.ntlm ? SpnegoConstants.NTLM : SpnegoConstants.NEGOTIATE)
92                  + " ";
93      }
94  
95      @Override
96      protected void onError(final RequestContext context,
97              final Credential credential) {
98          setResponseHeader(context, credential);
99      }
100 
101     @Override
102     protected void onSuccess(final RequestContext context,
103             final Credential credential) {
104         setResponseHeader(context, credential);
105     }
106 
107     /**
108      * Sets the response header based on the retrieved tocken.
109      *
110      * @param context the context
111      * @param credential the credential
112      */
113     private void setResponseHeader(final RequestContext context,
114             final Credential credential) {
115         if (credential == null) {
116             return;
117         }
118 
119         final HttpServletResponse response = WebUtils
120                 .getHttpServletResponse(context);
121         final SpnegoCredential spnegoCredentials = (SpnegoCredential) credential;
122         final byte[] nextToken = spnegoCredentials.getNextToken();
123         if (nextToken != null) {
124             logger.debug("Obtained output token: {}", new String(nextToken, Charset.defaultCharset()));
125             response.setHeader(SpnegoConstants.HEADER_AUTHENTICATE, (this.ntlm
126                     ? SpnegoConstants.NTLM : SpnegoConstants.NEGOTIATE)
127                     + " " + CompressionUtils.encodeBase64(nextToken));
128         } else {
129             logger.debug("Unable to obtain the output token required.");
130         }
131 
132         if (spnegoCredentials.getPrincipal() == null && send401OnAuthenticationFailure) {
133             logger.debug("Setting HTTP Status to 401");
134             response.setStatus(HttpServletResponse.SC_UNAUTHORIZED);
135         }
136     }
137 
138     /**
139      * Sets the ntlm and generate message prefix.
140      *
141      * @param ntlm the new ntlm
142      */
143     public void setNtlm(final boolean ntlm) {
144         this.ntlm = ntlm;
145         this.messageBeginPrefix = constructMessagePrefix();
146     }
147 
148     public void setSend401OnAuthenticationFailure(final boolean send401OnAuthenticationFailure) {
149         this.send401OnAuthenticationFailure = send401OnAuthenticationFailure;
150     }
151 
152 }
