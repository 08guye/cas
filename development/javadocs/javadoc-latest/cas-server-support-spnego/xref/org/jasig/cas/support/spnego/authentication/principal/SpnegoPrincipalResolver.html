19  package org.jasig.cas.support.spnego.authentication.principal;
20  
21  import org.jasig.cas.authentication.Credential;
22  import org.jasig.cas.authentication.principal.PersonDirectoryPrincipalResolver;
23  
24  import javax.validation.constraints.NotNull;
25  import java.util.Locale;
26  
27  /**
28   * Implementation of a CredentialToPrincipalResolver that takes a
29   * SpnegoCredential and returns a SimplePrincipal.
30   *
31   * @author Arnaud Lesueur
32   * @author Marc-Antoine Garrigue
33   * @since 3.1
34   */
35  public final class SpnegoPrincipalResolver extends PersonDirectoryPrincipalResolver {
36  
37      /** Tranformation types. **/
38      public static enum Transform {NONE, UPPERCASE, LOWERCASE}
39  
40      @NotNull
41      private Transform transformPrincipalId = Transform.NONE;
42  
43      @Override
44      protected String extractPrincipalId(final Credential credential) {
45          final SpnegoCredential c = (SpnegoCredential) credential;
46          final String id = c.getPrincipal().getId();
47  
48          switch (this.transformPrincipalId) {
49          case UPPERCASE:
50              return id.toUpperCase(Locale.ENGLISH);
51          case LOWERCASE:
52              return id.toLowerCase(Locale.ENGLISH);
53          default:
54              return id;
55          }
56      }
57  
58      @Override
59      public boolean supports(final Credential credential) {
60          return credential != null
61                  && SpnegoCredential.class.equals(credential.getClass());
62      }
63  
64      public void setTransformPrincipalId(final Transform transform) {
65          this.transformPrincipalId = transform;
66      }
67  }
