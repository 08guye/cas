65  @RunWith(Parameterized.class)
66  public class KryoTranscoderTests {
67  
68      private static final String ST_ID = "ST-1234567890ABCDEFGHIJKLMNOPQRSTUVWXYZ01234567890ABCDEFGHIJK";
69      private static final String TGT_ID = "TGT-1234567890ABCDEFGHIJKLMNOPQRSTUVWXYZ01234567890ABCDEFGHIJK-cas1";
70      
71      private final KryoTranscoder transcoder;
72  
73      public KryoTranscoderTests(final int bufferSize) {
74          transcoder = new KryoTranscoder(bufferSize);
75          final Map<Class<?>, Serializer> serializerMap = new HashMap<Class<?>, Serializer>();
76          serializerMap.put(
77                  MockServiceTicket.class,
78                  new FieldSerializer(transcoder.getKryo(), MockServiceTicket.class));
79          serializerMap.put(
80                  MockTicketGrantingTicket.class,
81                  new FieldSerializer(transcoder.getKryo(), MockTicketGrantingTicket.class));
82          transcoder.setSerializerMap(serializerMap);
83          transcoder.initialize();
84      }
85  
86      @Parameterized.Parameters
87      public static List<Object[]> getTestParms() {
88          final List<Object[]> params = new ArrayList<Object[]>(6);
89  
90          // Test case #1 - Buffer is bigger than encoded data
91          params.add(new Object[] {1024});
92  
93          // Test case #2 - Buffer overflow case
94          params.add(new Object[] {10});
95          return params;
96      }
97  
98      @Test
99      public void verifyEncodeDecode() throws Exception {
100         final ServiceTicket expectedST =
101                 new MockServiceTicket(ST_ID);
102         assertEquals(expectedST, transcoder.decode(transcoder.encode(expectedST)));
103 
104         final Credential userPassCredential = new UsernamePasswordCredential("handymanbob", "foo");
105         final TicketGrantingTicket expectedTGT =
106                 new MockTicketGrantingTicket(TGT_ID, userPassCredential);
107         expectedTGT.grantServiceTicket(ST_ID, null, null, false);
108         assertEquals(expectedTGT, transcoder.decode(transcoder.encode(expectedTGT)));
109 
110         internalProxyTest("http://localhost");
111         internalProxyTest("https://localhost:8080/path/file.html?p1=v1&p2=v2#fragment");
112     }
113 
114     private void internalProxyTest(final String proxyUrl) throws MalformedURLException {
115         final RegisteredServiceImpl svc = new RegisteredServiceImpl();
116         svc.setServiceId("https://some.app.edu");
117         final Credential proxyCredential = new HttpBasedServiceCredential(new URL(proxyUrl), svc);
118         final TicketGrantingTicket expectedTGT =
119                 new MockTicketGrantingTicket(TGT_ID, proxyCredential);
120         expectedTGT.grantServiceTicket(ST_ID, null, null, false);
121         assertEquals(expectedTGT, transcoder.decode(transcoder.encode(expectedTGT)));        
122     }

185     static class MockTicketGrantingTicket implements TicketGrantingTicket {
186 
187         private static final long serialVersionUID = 4829406617873497061L;
188 
189         private final String id;
190 
191         private int usageCount;
192 
193         private final Date creationDate = new Date();
194 
195         private final Authentication authentication;
196 
197         /** Factory to create the principal type. **/
198         @NotNull
199         private PrincipalFactory principalFactory = new DefaultPrincipalFactory();
200 
201         /** Constructor for serialization support. */
202         MockTicketGrantingTicket() {
203             this.id = null;
204             this.authentication = null;
205         }
206 
207         public MockTicketGrantingTicket(final String id, final Credential credential) {
208             this.id = id;
209             final CredentialMetaData credentialMetaData = new BasicCredentialMetaData(credential);
210             final AuthenticationBuilder builder = new AuthenticationBuilder();
211             final Map<String, Object> attributes = new HashMap<String, Object>();
212             attributes.put("nickname", "bob");
213             builder.setPrincipal(this.principalFactory.createPrincipal("handymanbob", attributes));
214             builder.setAuthenticationDate(new Date());
215             builder.addCredential(credentialMetaData);
216             final AuthenticationHandler handler = new MockAuthenticationHandler();
217             try {
218                 builder.addSuccess(handler.getName(), handler.authenticate(credential));
219             } catch (final Exception e) {
220                 throw new RuntimeException(e);
221             }
222             builder.addFailure(handler.getName(), FailedLoginException.class);
223             this.authentication = builder.build();
224         }
225 
226         @Override
227         public Authentication getAuthentication() {
228             return this.authentication;
229         }
230 
231         @Override
232         public List<Authentication> getSupplementalAuthentications() {
233             return Collections.emptyList();
234         }
235 
236         @Override
237         public ServiceTicket grantServiceTicket(
238                 final String id,
239                 final Service service,
240                 final ExpirationPolicy expirationPolicy,
241                 final boolean credentialsProvided) {
242             this.usageCount++;
243             return new MockServiceTicket(id);
244         }
245 
246         @Override
247         public Map<String, Service> getServices() {
248             return Collections.emptyMap();
249         }

270         }
271

276 
277         @Override

279             return false;
280         }

284             return this;

297         @Override
298         public boolean equals(final Object other) {
299             return other instanceof MockTicketGrantingTicket
300                     && ((MockTicketGrantingTicket) other).getId().equals(this.id)
301                     && ((MockTicketGrantingTicket) other).getCountOfUses() == this.usageCount
302                     && ((MockTicketGrantingTicket) other).getCreationTime() == this.creationDate.getTime()
303                     && ((MockTicketGrantingTicket) other).getAuthentication().equals(this.authentication);
304         }
305 
306         @Override

308             final HashCodeBuilder bldr = new HashCodeBuilder(17, 33);
309             return bldr.append(this.id)

311                         .append(this.creationDate.getTime())
312                         .append(this.authentication).toHashCode();
313         }

316     public static class MockAuthenticationHandler implements AuthenticationHandler {
317 
318         @Override
319         public HandlerResult authenticate(final Credential credential) throws GeneralSecurityException, PreventedException {
320             if (credential instanceof HttpBasedServiceCredential) {
321                 return new HandlerResult(this, (HttpBasedServiceCredential) credential);
322             } else {
323                 return new HandlerResult(this, new BasicCredentialMetaData(credential));
324             }
325         }
326 
327         @Override
328         public boolean supports(final Credential credential) {
329             return true;
330         }
331 
332         @Override
333         public String getName() {
334             return "MockAuthenticationHandler";
335         }
