54  /**
55   * This is {@link AbstractSaml20ObjectBuilder}.
56   * to build saml2 objects.
57   * @author Misagh Moayyed mmoayyed@unicon.net
58   * @since 4.1
59   */
60  public abstract class AbstractSaml20ObjectBuilder extends AbstractSamlObjectBuilder {
61      private static final int HEX_HIGH_BITS_BITWISE_FLAG = 0x0f;
62  
63      /**
64       * Gets name id.
65       *
66       * @param nameIdFormat the name id format
67       * @param nameIdValue the name id value
68       * @return the name iD
69       */
70      protected NameID getNameID(final String nameIdFormat, final String nameIdValue) {
71          final NameID nameId = newSamlObject(NameID.class);
72          nameId.setFormat(nameIdFormat);
73          nameId.setValue(nameIdValue);
74          return nameId;
75      }

77      /**
78       * Create a new SAML response object.
79       * @param id the id
80       * @param issueInstant the issue instant
81       * @param recipient the recipient
82       * @param service the service
83       * @return the response
84       */
85      public Response newResponse(final String id, final DateTime issueInstant,
86                                  final String recipient, final WebApplicationService service) {
87  
88          final Response samlResponse = newSamlObject(Response.class);
89          samlResponse.setID(id);
90          samlResponse.setIssueInstant(issueInstant);
91          samlResponse.setVersion(SAMLVersion.VERSION_20);
92          if (service instanceof SamlService) {
93              final SamlService samlService = (SamlService) service;
94  
95              if (samlService.getRequestID() != null) {
96                  samlResponse.setInResponseTo(samlService.getRequestID());
97              }
98          }
99          return samlResponse;
100     }

102     /**
103      * Create a new SAML status object.
104      *
105      * @param codeValue the code value
106      * @param statusMessage the status message
107      * @return the status
108      */
109     public Status newStatus(final String codeValue, final String statusMessage) {
110         final Status status = newSamlObject(Status.class);
111         final StatusCode code = newSamlObject(StatusCode.class);
112         code.setValue(codeValue);
113         status.setStatusCode(code);
114         if (StringUtils.hasText(statusMessage)) {
115             final StatusMessage message = newSamlObject(StatusMessage.class);
116             message.setMessage(statusMessage);
117             status.setStatusMessage(message);
118         }
119         return status;
120     }

122     /**
123      * Create a new SAML1 response object.
124      *
125      * @param authnStatement the authn statement
126      * @param issuer the issuer
127      * @param issuedAt the issued at
128      * @param id the id
129      * @return the assertion
130      */
131     public Assertion newAssertion(final AuthnStatement authnStatement, final String issuer,
132                                   final DateTime issuedAt, final String id) {
133         final Assertion assertion = newSamlObject(Assertion.class);
134         assertion.setID(id);
135         assertion.setIssueInstant(issuedAt);
136         assertion.setIssuer(newIssuer(issuer));
137         assertion.getAuthnStatements().add(authnStatement);
138         return assertion;
139     }

153     /**
154      * New authn statement.
155      *
156      * @param contextClassRef the context class ref such as {@link AuthnContext#PASSWORD_AUTHN_CTX}
157      * @param authnInstant the authn instant
158      * @return the authn statement
159      */
160     public AuthnStatement newAuthnStatement(final String contextClassRef, final DateTime authnInstant) {
161         final AuthnStatement stmt = newSamlObject(AuthnStatement.class);
162         final AuthnContext ctx = newSamlObject(AuthnContext.class);
163 
164         final AuthnContextClassRef classRef = newSamlObject(AuthnContextClassRef.class);
165         classRef.setAuthnContextClassRef(contextClassRef);
166 
167         ctx.setAuthnContextClassRef(classRef);
168         stmt.setAuthnContext(ctx);
169         stmt.setAuthnInstant(authnInstant);
170 
171         return stmt;
172     }

174     /**
175      * New conditions element.
176      *
177      * @param notBefore the not before
178      * @param notOnOrAfter the not on or after
179      * @param audienceUri the service id
180      * @return the conditions
181      */
182     public Conditions newConditions(final DateTime notBefore, final DateTime notOnOrAfter, final String audienceUri) {
183         final Conditions conditions = newSamlObject(Conditions.class);
184         conditions.setNotBefore(notBefore);
185         conditions.setNotOnOrAfter(notOnOrAfter);
186 
187         final AudienceRestriction audienceRestriction = newSamlObject(AudienceRestriction.class);
188         final Audience audience = newSamlObject(Audience.class);
189         audience.setAudienceURI(audienceUri);
190         audienceRestriction.getAudiences().add(audience);
191         conditions.getAudienceRestrictions().add(audienceRestriction);
192         return conditions;
193     }

195     /**
196      * New subject element.
197      *
198      * @param nameIdFormat the name id format
199      * @param nameIdValue the name id value
200      * @param recipient the recipient
201      * @param notOnOrAfter the not on or after
202      * @param inResponseTo the in response to
203      * @return the subject
204      */
205     public Subject newSubject(final String nameIdFormat, final String nameIdValue,
206                               final String recipient, final DateTime notOnOrAfter,
207                               final String inResponseTo) {
208 
209         final SubjectConfirmation confirmation = newSamlObject(SubjectConfirmation.class);
210         confirmation.setMethod(SubjectConfirmation.METHOD_BEARER);
211 
212         final SubjectConfirmationData data = newSamlObject(SubjectConfirmationData.class);
213         data.setRecipient(recipient);
214         data.setNotOnOrAfter(notOnOrAfter);
215         data.setInResponseTo(inResponseTo);
216 
217         confirmation.setSubjectConfirmationData(data);
218 
219         final Subject subject = newSamlObject(Subject.class);
220         subject.setNameID(getNameID(nameIdFormat, nameIdValue));
221         subject.getSubjectConfirmations().add(confirmation);
222         return subject;
223     }

225     @Override
226     public String generateSecureRandomId() {
227         final SecureRandom generator = new SecureRandom();
228         final char[] charMappings = {
229                 'a', 'b', 'c', 'd', 'e', 'f', 'g',
230                 'h', 'i', 'j', 'k', 'l', 'm', 'n', 'o',
231                 'p'};
232 
233         final int charsLength = 40;
234         final int generatorBytesLength = 20;
235         final int shiftLength = 4;
236 
237         // 160 bits
238         final byte[] bytes = new byte[generatorBytesLength];
239         generator.nextBytes(bytes);
240 
241         final char[] chars = new char[charsLength];
242         for (int i = 0; i < bytes.length; i++) {
243             final int left = bytes[i] >> shiftLength & HEX_HIGH_BITS_BITWISE_FLAG;
244             final int right = bytes[i] & HEX_HIGH_BITS_BITWISE_FLAG;
245             chars[i * 2] = charMappings[left];
246             chars[i * 2 + 1] = charMappings[right];
247         }
248         return String.valueOf(chars);
249     }

251     /**
252      * Deflate the given bytes using zlib.
253      *
254      * @param bytes the bytes
255      * @return the converted string
256      */
257     private static String zlibDeflate(final byte[] bytes) {
258         final int bufferLength = 1024;
259         final ByteArrayInputStream bais = new ByteArrayInputStream(bytes);
260         final ByteArrayOutputStream baos = new ByteArrayOutputStream();
261         final InflaterInputStream iis = new InflaterInputStream(bais);
262         final byte[] buf = new byte[bufferLength];
263 
264         try {
265             int count = iis.read(buf);
266             while (count != -1) {
267                 baos.write(buf, 0, count);
268                 count = iis.read(buf);
269             }
270             return new String(baos.toByteArray());
271         } catch (final Exception e) {
272             return null;
273         } finally {
274             IOUtils.closeQuietly(iis);
275         }
276     }

278     /**
279      * Base64 decode.
280      *
281      * @param xml the xml
282      * @return the byte[]
283      */
284     private static byte[] base64Decode(final String xml) {
285         try {
286             final byte[] xmlBytes = xml.getBytes("UTF-8");
287             return Base64.decodeBase64(xmlBytes);
288         } catch (final Exception e) {
289             return null;
290         }
291     }

293     /**
294      * Inflate the given byte array.
295      *
296      * @param bytes the bytes
297      * @return the string
298      */
299     private static String inflate(final byte[] bytes) {
300         final int bufferLength = 10000;
301         final Inflater inflater = new Inflater(true);
302         final byte[] xmlMessageBytes = new byte[bufferLength];
303 
304         final byte[] extendedBytes = new byte[bytes.length + 1];
305         System.arraycopy(bytes, 0, extendedBytes, 0, bytes.length);
306         extendedBytes[bytes.length] = 0;
307 
308         inflater.setInput(extendedBytes);
309 
310         try {
311             final int resultLength = inflater.inflate(xmlMessageBytes);
312             inflater.end();
313 
314             if (!inflater.finished()) {
315                 throw new RuntimeException("buffer not large enough.");
316             }
317 
318             inflater.end();
319             return new String(xmlMessageBytes, 0, resultLength, "UTF-8");
320         } catch (final DataFormatException e) {
321             return null;
322         } catch (final UnsupportedEncodingException e) {
323             throw new RuntimeException("Cannot find encoding: UTF-8", e);
324         }
325     }

329     /**
330      * Decode authn request xml.
331      *
332      * @param encodedRequestXmlString the encoded request xml string
333      * @return the request
334      */
335     public String decodeSamlAuthnRequest(final String encodedRequestXmlString) {
336         if (encodedRequestXmlString == null) {
337             return null;
338         }
339 
340         final byte[] decodedBytes = base64Decode(encodedRequestXmlString);
341         if (decodedBytes == null) {
342             return null;
343         }
344 
345         final String inflated = inflate(decodedBytes);
346         if (inflated != null) {
347             return inflated;
348         }
349 
350         return zlibDeflate(decodedBytes);
351     }
