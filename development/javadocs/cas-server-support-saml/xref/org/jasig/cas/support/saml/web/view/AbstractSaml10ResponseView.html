51  /**
52   * Base class for all views that render SAML1 SOAP messages directly to the HTTP response stream.
53   *
54   * @author Marvin S. Addison
55   * @since 3.5.1
56   */
57  public abstract class AbstractSaml10ResponseView extends AbstractCasView {

59      private static final String DEFAULT_ELEMENT_NAME_FIELD = "DEFAULT_ELEMENT_NAME";

61      private static final String DEFAULT_ENCODING = "UTF-8";

63      private final SamlArgumentExtractor samlArgumentExtractor = new SamlArgumentExtractor();

65      private final HTTPSOAP11Encoder encoder = new CasHTTPSOAP11Encoder();

74      /**
75       * Sets the character encoding in the HTTP response.
76       *
77       * @param encoding Response character encoding.
78       */
79      public void setEncoding(final String encoding) {
80          this.encoding = encoding;
81      }

83      /**
84      * Sets the allowance for time skew in seconds
85      * between CAS and the client server.  Default 0s.
86      * This value will be subtracted from the current time when setting the SAML
87      * <code>NotBeforeDate</code> attribute, thereby allowing for the
88      * CAS server to be ahead of the client by as much as the value defined here.
89      *
90      * <p><strong>Note:</strong> Skewing of the issue instant via setting this property
91      * applies to all saml assertions that are issued by CAS and it
92      * currently cannot be controlled on a per relying party basis.
93      * Before configuring this, it is recommended that each service provider
94      * attempt to correctly sync their system time with an NTP server
95      * so as to match the CAS server's issue instant config and to
96      * avoid applying this setting globally. This should only
97      * be used in situations where the NTP server is unresponsive to
98      * sync time on the client, or the client is simply unable
99      * to adjust their server time configuration.</p>
100     *
101     * @param skewAllowance Number of seconds to allow for variance.
102     */
103     public void setSkewAllowance(final int skewAllowance) {
104         logger.debug("Using {} seconds as skew allowance.", skewAllowance);
105         this.skewAllowance = skewAllowance;
106     }

118     /**
119      * Instantiates a new abstract saml10 response view.
120      */
121     protected AbstractSaml10ResponseView() {
122         try {
123             this.idGenerator = new SecureRandomIdentifierGenerator();
124         } catch (final NoSuchAlgorithmException e) {
125             throw new IllegalStateException("Cannot create secure random ID generator for SAML message IDs.");
126         }
127     }

129     @Override
130     protected void renderMergedOutputModel(
131             final Map<String, Object> model, final HttpServletRequest request, final HttpServletResponse response) throws Exception {
132 
133         response.setCharacterEncoding(this.encoding);
134 
135         final WebApplicationService service = this.samlArgumentExtractor.extractService(request);
136         final String serviceId = service != null ? service.getId() : "UNKNOWN";
137 
138         try {
139             final Response samlResponse = newSamlObject(Response.class);
140             samlResponse.setID(generateId());
141             samlResponse.setIssueInstant(DateTime.now().minusSeconds(skewAllowance));
142             samlResponse.setVersion(SAMLVersion.VERSION_11);
143             samlResponse.setRecipient(serviceId);
144             if (service instanceof SamlService) {
145                 final SamlService samlService = (SamlService) service;
146 
147                 if (samlService.getRequestID() != null) {
148                     samlResponse.setInResponseTo(samlService.getRequestID());
149                 }
150             }
151             prepareResponse(samlResponse, model);
152 
153             final BasicSAMLMessageContext messageContext = new BasicSAMLMessageContext();
154             messageContext.setOutboundMessageTransport(new HttpServletResponseAdapter(response, request.isSecure()));
155             messageContext.setOutboundSAMLMessage(samlResponse);
156             this.encoder.encode(messageContext);
157         } catch (final Exception e) {
158             logger.error("Error generating SAML response for service {}.", serviceId);
159             throw e;
160         }
161     }

182     /**
183      * Create a new SAML object.
184      *
185      * @param <T> the generic type
186      * @param objectType the object type
187      * @return the t
188      */
189     protected final <T extends SAMLObject> T newSamlObject(final Class<T> objectType) {
190         final QName qName;
191         try {
192             final Field f = objectType.getField(DEFAULT_ELEMENT_NAME_FIELD);
193             qName = (QName) f.get(null);
194         } catch (final NoSuchFieldException e) {
195             throw new IllegalStateException("Cannot find field " + objectType.getName() + "." + DEFAULT_ELEMENT_NAME_FIELD);
196         } catch (final IllegalAccessException e) {
197             throw new IllegalStateException("Cannot access field " + objectType.getName() + "." + DEFAULT_ELEMENT_NAME_FIELD);
198         }
199         final SAMLObjectBuilder<T> builder = (SAMLObjectBuilder<T>) Configuration.getBuilderFactory().getBuilder(qName);
200         if (builder == null) {
201             throw new IllegalStateException("No SAMLObjectBuilder registered for class " + objectType.getName());
202         }
203         return objectType.cast(builder.buildObject());
204     }

206     /**
207      * Create a new SAML status object.
208      *
209      * @param codeValue the code value
210      * @param statusMessage the status message
211      * @return the status
212      */
213     protected final Status newStatus(final QName codeValue, final String statusMessage) {
214         final Status status = newSamlObject(Status.class);
215         final StatusCode code = newSamlObject(StatusCode.class);
216         code.setValue(codeValue);
217         status.setStatusCode(code);
218         if (statusMessage != null) {
219             final StatusMessage message = newSamlObject(StatusMessage.class);
220             message.setMessage(statusMessage);
221             status.setStatusMessage(message);
222         }
223         return status;
224     }
