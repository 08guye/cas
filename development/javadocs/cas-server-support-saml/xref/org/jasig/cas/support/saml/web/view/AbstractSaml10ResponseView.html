19  package org.jasig.cas.support.saml.web.view;
20  
21  import org.jasig.cas.authentication.principal.WebApplicationService;
22  import org.jasig.cas.support.saml.util.Saml10ObjectBuilder;
23  import org.jasig.cas.support.saml.web.support.SamlArgumentExtractor;
24  import org.jasig.cas.web.view.AbstractCasView;
25  import org.joda.time.DateTime;
26  
27  import org.opensaml.core.config.InitializationException;
28  import org.opensaml.core.config.InitializationService;
29  import org.opensaml.core.xml.config.XMLObjectProviderRegistrySupport;
30  import org.opensaml.saml.saml1.core.Response;
31  
32  import javax.servlet.http.HttpServletRequest;
33  import javax.servlet.http.HttpServletResponse;
34  import javax.validation.constraints.NotNull;
35  import java.util.Map;
36  
37  /**
38   * Base class for all views that render SAML1 SOAP messages directly to the HTTP response stream.
39   *
40   * @author Marvin S. Addison
41   * @since 3.5.1
42   */
43  public abstract class AbstractSaml10ResponseView extends AbstractCasView {
44      private static final String DEFAULT_ENCODING = "UTF-8";
45  
46      /**
47       * The Saml object builder.
48       */
49      protected final Saml10ObjectBuilder samlObjectBuilder = new Saml10ObjectBuilder();
50  
51      private final SamlArgumentExtractor samlArgumentExtractor = new SamlArgumentExtractor();
52  
53      @NotNull
54      private String encoding = DEFAULT_ENCODING;
55  
56      /** Defaults to 0. */
57      private int skewAllowance;
58  
59      static {
60          try {
61  
62              if (XMLObjectProviderRegistrySupport.getParserPool() == null) {
63                  InitializationService.initialize();
64              }
65          } catch (final InitializationException e) {
66              throw new IllegalStateException("Error initializing OpenSAML library.", e);
67          }
68      }
69  
70      /**
71       * Sets the character encoding in the HTTP response.
72       *
73       * @param encoding Response character encoding.
74       */
75      public void setEncoding(final String encoding) {
76          this.encoding = encoding;
77      }
78  
79      /**
80      * Sets the allowance for time skew in seconds
81      * between CAS and the client server.  Default 0s.
82      * This value will be subtracted from the current time when setting the SAML
83      * <code>NotBeforeDate</code> attribute, thereby allowing for the
84      * CAS server to be ahead of the client by as much as the value defined here.
85      *
86      * <p><strong>Note:</strong> Skewing of the issue instant via setting this property
87      * applies to all saml assertions that are issued by CAS and it
88      * currently cannot be controlled on a per relying party basis.
89      * Before configuring this, it is recommended that each service provider
90      * attempt to correctly sync their system time with an NTP server
91      * so as to match the CAS server's issue instant config and to
92      * avoid applying this setting globally. This should only
93      * be used in situations where the NTP server is unresponsive to
94      * sync time on the client, or the client is simply unable
95      * to adjust their server time configuration.</p>
96      *
97      * @param skewAllowance Number of seconds to allow for variance.
98      */
99      public void setSkewAllowance(final int skewAllowance) {
100         logger.debug("Using {} seconds as skew allowance.", skewAllowance);
101         this.skewAllowance = skewAllowance;
102     }
103 
104     @Override
105     protected void renderMergedOutputModel(
106             final Map<String, Object> model, final HttpServletRequest request, final HttpServletResponse response) throws Exception {
107 
108         response.setCharacterEncoding(this.encoding);
109 
110         final WebApplicationService service = this.samlArgumentExtractor.extractService(request);
111         final String serviceId = service != null ? service.getId() : "UNKNOWN";
112 
113         try {
114             final Response samlResponse = this.samlObjectBuilder.newResponse(
115                     this.samlObjectBuilder.generateSecureRandomId(),
116                     DateTime.now().minusSeconds(this.skewAllowance), serviceId, service);
117 
118             prepareResponse(samlResponse, model);
119 
120             this.samlObjectBuilder.encodeSamlResponse(response, request, samlResponse);
121         } catch (final Exception e) {
122             logger.error("Error generating SAML response for service {}.", serviceId);
123             throw e;
124         }
125     }
126 
127     /**
128      * Subclasses must implement this method by adding child elements (status, assertion, etc) to
129      * the given empty SAML 1 response message.  Impelmenters need not be concerned with error handling.
130      *
131      * @param response SAML 1 response message to be filled.
132      * @param model Spring MVC model map containing data needed to prepare response.
133      */
134     protected abstract void prepareResponse(Response response, Map<String, Object> model);
135 
136 }
