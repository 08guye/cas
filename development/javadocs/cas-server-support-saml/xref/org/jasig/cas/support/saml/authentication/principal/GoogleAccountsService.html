153     /**
154      * Creates the service from request.
155      *
156      * @param request the request
157      * @param privateKey the private key
158      * @param publicKey the public key
159      * @param servicesManager the services manager
160      * @return the google accounts service
161      */
162     public static GoogleAccountsService createServiceFrom(
163             final HttpServletRequest request, final PrivateKey privateKey,
164             final PublicKey publicKey, final ServicesManager servicesManager) {
165         final String relayState = request.getParameter(CONST_RELAY_STATE);
166 
167         final String xmlRequest = decodeAuthnRequestXML(request.getParameter(CONST_PARAM_SERVICE));
168 
169         if (!StringUtils.hasText(xmlRequest)) {
170             return null;
171         }
172 
173         final Document document = SamlUtils.constructDocumentFromXmlString(xmlRequest);
174 
175         if (document == null) {
176             return null;
177         }
178 
179         final String assertionConsumerServiceUrl = document.getRootElement().getAttributeValue("AssertionConsumerServiceURL");
180         final String requestId = document.getRootElement().getAttributeValue("ID");
181 
182         return new GoogleAccountsService(assertionConsumerServiceUrl,
183                 relayState, requestId, privateKey, publicKey, servicesManager);
184     }

186     @Override
187     public Response getResponse(final String ticketId) {
188         final Map<String, String> parameters = new HashMap<String, String>();
189         final String samlResponse = constructSamlResponse();
190         final String signedResponse = SamlUtils.signSamlResponse(samlResponse,
191                 this.privateKey, this.publicKey);
192         parameters.put("SAMLResponse", signedResponse);
193         parameters.put("RelayState", this.relayState);
194 
195         return Response.getPostResponse(getOriginalUrl(), parameters);
196     }

208     /**
209      * Construct SAML response.
210      *
211      * @return the SAML response
212      */
213     private String constructSamlResponse() {
214         String samlResponse = TEMPLATE_SAML_RESPONSE;
215 
216         final Calendar c = Calendar.getInstance();
217         c.setTime(new Date());
218         c.add(Calendar.YEAR, 1);
219         
220         final RegisteredService svc = this.servicesManager.findServiceBy(this);
221         final String userId = svc.getUsernameAttributeProvider().resolveUsername(getPrincipal(), this);
222 
223         final String currentDateTime = new ISOStandardDateFormat().getCurrentDateAndTime();
224         samlResponse = samlResponse.replace("<USERNAME_STRING>", userId);
225         samlResponse = samlResponse.replace("<RESPONSE_ID>", createID());
226         samlResponse = samlResponse.replace("<ISSUE_INSTANT>", currentDateTime);
227         samlResponse = samlResponse.replace("<AUTHN_INSTANT>", currentDateTime);
228         samlResponse = samlResponse.replaceAll("<NOT_ON_OR_AFTER>", currentDateTime);
229         samlResponse = samlResponse.replace("<ASSERTION_ID>", createID());
230         samlResponse = samlResponse.replaceAll("<ACS_URL>", getId());
231         samlResponse = samlResponse.replace("<REQUEST_ID>", this.requestId);
232 
233         return samlResponse;
234     }

236     /**
237      * Creates the SAML id.
238      *
239      * @return the id
240      */
241     private static String createID() {
242         final byte[] bytes = new byte[20]; // 160 bits
243         RANDOM_GENERATOR.nextBytes(bytes);
244 
245         final char[] chars = new char[40];
246 
247         for (int i = 0; i < bytes.length; i++) {
248             final int left = bytes[i] >> 4 & 0x0f;
249             final int right = bytes[i] & 0x0f;
250             chars[i * 2] = CHAR_MAPPINGS[left];
251             chars[i * 2 + 1] = CHAR_MAPPINGS[right];
252         }
253 
254         return String.valueOf(chars);
255     }

257     /**
258      * Decode authn request xml.
259      *
260      * @param encodedRequestXmlString the encoded request xml string
261      * @return the request
262      */
263     private static String decodeAuthnRequestXML(
264             final String encodedRequestXmlString) {
265         if (encodedRequestXmlString == null) {
266             return null;
267         }
268 
269         final byte[] decodedBytes = base64Decode(encodedRequestXmlString);
270 
271         if (decodedBytes == null) {
272             return null;
273         }
274 
275         final String inflated = inflate(decodedBytes);
276 
277         if (inflated != null) {
278             return inflated;
279         }
280 
281         return zlibDeflate(decodedBytes);
282     }
283 
284     /**
285      * Deflate the given bytes using zlib.
286      *
287      * @param bytes the bytes
288      * @return the converted string
289      */
290     private static String zlibDeflate(final byte[] bytes) {
291         final ByteArrayInputStream bais = new ByteArrayInputStream(bytes);
292         final ByteArrayOutputStream baos = new ByteArrayOutputStream();
293         final InflaterInputStream iis = new InflaterInputStream(bais);
294         final byte[] buf = new byte[1024];
295 
296         try {
297             int count = iis.read(buf);
298             while (count != -1) {
299                 baos.write(buf, 0, count);
300                 count = iis.read(buf);
301             }
302             return new String(baos.toByteArray());
303         } catch (final Exception e) {
304             return null;
305         } finally {
306             IOUtils.closeQuietly(iis);
307         }
308     }

310     /**
311      * Base64 decode.
312      *
313      * @param xml the xml
314      * @return the byte[]
315      */
316     private static byte[] base64Decode(final String xml) {
317         try {
318             final byte[] xmlBytes = xml.getBytes("UTF-8");
319             return Base64.decodeBase64(xmlBytes);
320         } catch (final Exception e) {
321             return null;
322         }
323     }
324 
325     /**
326      * Inflate the given byte array.
327      *
328      * @param bytes the bytes
329      * @return the string
330      */
331     private static String inflate(final byte[] bytes) {
332         final Inflater inflater = new Inflater(true);
333         final byte[] xmlMessageBytes = new byte[10000];
334 
335         final byte[] extendedBytes = new byte[bytes.length + 1];
336         System.arraycopy(bytes, 0, extendedBytes, 0, bytes.length);
337         extendedBytes[bytes.length] = 0;
338 
339         inflater.setInput(extendedBytes);
340 
341         try {
342             final int resultLength = inflater.inflate(xmlMessageBytes);
343             inflater.end();
344 
345             if (!inflater.finished()) {
346                 throw new RuntimeException("buffer not large enough.");
347             }
348 
349             inflater.end();
350             return new String(xmlMessageBytes, 0, resultLength, "UTF-8");
351         } catch (final DataFormatException e) {
352             return null;
353         } catch (final UnsupportedEncodingException e) {
354             throw new RuntimeException("Cannot find encoding: UTF-8", e);
355         }
356     }
