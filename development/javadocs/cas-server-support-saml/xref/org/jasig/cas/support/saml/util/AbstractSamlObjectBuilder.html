75  /**
76   * An abstract builder to serve as the template handler
77   * for SAML1 and SAML2 responses.
78   *
79   * @author Misagh Moayyed mmoayyed@unicon.net
80   * @since 4.1
81   */
82  public abstract class AbstractSamlObjectBuilder {
83      /**
84       * The constant DEFAULT_ELEMENT_NAME_FIELD.
85       */
86      protected static final String DEFAULT_ELEMENT_NAME_FIELD = "DEFAULT_ELEMENT_NAME";
87  
88      /**
89       * The constant DEFAULT_ELEMENT_LOCAL_NAME_FIELD.
90       */
91      protected static final String DEFAULT_ELEMENT_LOCAL_NAME_FIELD = "DEFAULT_ELEMENT_LOCAL_NAME";
92  
93      /** Logger instance. **/
94      protected final Logger logger = LoggerFactory.getLogger(this.getClass());
95  
96      static {
97          try {
98              // Initialize OpenSAML default configuration
99              // (only needed once per classloader)
100             DefaultBootstrap.bootstrap();
101         } catch (final ConfigurationException e) {
102             throw new IllegalStateException("Error initializing OpenSAML library.", e);
103         }
104     }

106     /**
107      * Create a new SAML object.
108      *
109      * @param <T> the generic type
110      * @param objectType the object type
111      * @return the t
112      */
113     public final <T extends SAMLObject> T newSamlObject(final Class<T> objectType) {
114         final QName qName = getSamlObjectQName(objectType);
115         final SAMLObjectBuilder<T> builder = (SAMLObjectBuilder<T>) Configuration.getBuilderFactory().getBuilder(qName);
116         if (builder == null) {
117             throw new IllegalStateException("No SAMLObjectBuilder registered for class " + objectType.getName());
118         }
119         return objectType.cast(builder.buildObject(qName));
120     }

122     /**
123      * Gets saml object QName.
124      *
125      * @param objectType the object type
126      * @return the saml object QName
127      * @throws RuntimeException the exception
128      */
129     public QName getSamlObjectQName(final Class objectType) throws RuntimeException {
130         try {
131             final Field f = objectType.getField(DEFAULT_ELEMENT_NAME_FIELD);
132             final QName qName = (QName) f.get(null);
133             return qName;
134         } catch (final NoSuchFieldException e) {
135             throw new IllegalStateException("Cannot find field " + objectType.getName() + "." + DEFAULT_ELEMENT_NAME_FIELD);
136         } catch (final IllegalAccessException e) {
137             throw new IllegalStateException("Cannot access field " + objectType.getName() + "." + DEFAULT_ELEMENT_NAME_FIELD);
138         }
139     }

141     /**
142      * Build the saml object based on its QName.
143      *
144      * @param objectType the object
145      * @param qName the QName
146      * @param <T> the object type
147      * @return the saml object
148      */
149     private <T extends SAMLObject> T newSamlObject(final Class<T> objectType, final QName qName) {
150         final SAMLObjectBuilder<T> builder = (SAMLObjectBuilder<T>) Configuration.getBuilderFactory().getBuilder(qName);
151         if (builder == null) {
152             throw new IllegalStateException("No SAMLObjectBuilder registered for class " + objectType.getName());
153         }
154         return objectType.cast(builder.buildObject());
155     }

157     /**
158      * New attribute value.
159      *
160      * @param value the value
161      * @param elementName the element name
162      * @return the xS string
163      */
164     protected final XSString newAttributeValue(final Object value, final QName elementName) {
165         final XSStringBuilder attrValueBuilder = new XSStringBuilder();
166         final XSString stringValue = attrValueBuilder.buildObject(elementName, XSString.TYPE_NAME);
167         if (value instanceof String) {
168             stringValue.setValue((String) value);
169         } else {
170             stringValue.setValue(value.toString());
171         }
172         return stringValue;
173     }

189     /**
190      * Marshal the saml xml object to raw xml.
191      *
192      * @param object the object
193      * @param writer the writer
194      * @return the xml string
195      */
196     public String marshalSamlXmlObject(final XMLObject object, final StringWriter writer)  {
197         try {
198             final MarshallerFactory marshallerFactory = Configuration.getMarshallerFactory();
199             final Marshaller marshaller = marshallerFactory.getMarshaller(object);
200             final Element element = marshaller.marshall(object);
201             element.setAttributeNS("http://www.w3.org/2000/xmlns/", "xmlns", SAMLConstants.SAML20_NS);
202             element.setAttributeNS("http://www.w3.org/2000/xmlns/", "xmlns:xenc", "http://www.w3.org/2001/04/xmlenc#");
203 
204             final TransformerFactory transFactory = TransformerFactory.newInstance();
205             final Transformer transformer = transFactory.newTransformer();
206             transformer.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION, "yes");
207             transformer.setOutputProperty(OutputKeys.INDENT, "yes");
208             transformer.transform(new DOMSource(element), new StreamResult(writer));
209             return writer.toString();
210         } catch (final Exception e) {
211             throw new IllegalStateException("An error has occurred while marshalling SAML object to xml", e);
212         }
213     }

215     /**
216      * Sign SAML response.
217      *
218      * @param samlResponse the SAML response
219      * @param privateKey the private key
220      * @param publicKey the public key
221      * @return the response
222      */
223     public final String signSamlResponse(final String samlResponse,
224                                          final PrivateKey privateKey, final PublicKey publicKey) {
225         final Document doc = constructDocumentFromXml(samlResponse);
226 
227         if (doc != null) {
228             final org.jdom.Element signedElement = signSamlElement(doc.getRootElement(),
229                     privateKey, publicKey);
230             doc.setRootElement((org.jdom.Element) signedElement.detach());
231             return new XMLOutputter().outputString(doc);
232         }
233         throw new RuntimeException("Error signing SAML Response: Null document");
234     }

236     /**
237      * Construct document from xml string.
238      *
239      * @param xmlString the xml string
240      * @return the document
241      */
242     public static Document constructDocumentFromXml(final String xmlString) {
243         try {
244             final SAXBuilder builder = new SAXBuilder();
245             builder.setFeature("http://xml.org/sax/features/external-general-entities", false);
246             builder.setFeature("http://apache.org/xml/features/disallow-doctype-decl", true);
247             return builder
248                     .build(new ByteArrayInputStream(xmlString.getBytes(Charset.defaultCharset())));
249         } catch (final Exception e) {
250             return null;
251         }
252     }

254     /**
255      * Sign SAML element.
256      *
257      * @param element the element
258      * @param privKey the priv key
259      * @param pubKey the pub key
260      * @return the element
261      */
262     private static org.jdom.Element signSamlElement(final org.jdom.Element element, final PrivateKey privKey,
263                                                     final PublicKey pubKey) {
264         try {
265             final String providerName = System.getProperty("jsr105Provider",
266                     "org.jcp.xml.dsig.internal.dom.XMLDSigRI");
267 
268             final XMLSignatureFactory sigFactory = XMLSignatureFactory
269                     .getInstance("DOM", (Provider) Class.forName(providerName)
270                             .newInstance());
271 
272             final List<Transform> envelopedTransform = Collections
273                     .singletonList(sigFactory.newTransform(Transform.ENVELOPED,
274                             (TransformParameterSpec) null));
275 
276             final Reference ref = sigFactory.newReference("", sigFactory
277                             .newDigestMethod(DigestMethod.SHA1, null), envelopedTransform,
278                     null, null);
279 
280             // Create the SignatureMethod based on the type of key
281             SignatureMethod signatureMethod;
282             if (pubKey instanceof DSAPublicKey) {
283                 signatureMethod = sigFactory.newSignatureMethod(
284                         SignatureMethod.DSA_SHA1, null);
285             } else if (pubKey instanceof RSAPublicKey) {
286                 signatureMethod = sigFactory.newSignatureMethod(
287                         SignatureMethod.RSA_SHA1, null);
288             } else {
289                 throw new RuntimeException("Error signing SAML element: Unsupported type of key");
290             }
291 
292             final CanonicalizationMethod canonicalizationMethod = sigFactory
293                     .newCanonicalizationMethod(
294                             CanonicalizationMethod.INCLUSIVE_WITH_COMMENTS,
295                             (C14NMethodParameterSpec) null);
296 
297             // Create the SignedInfo
298             final SignedInfo signedInfo = sigFactory.newSignedInfo(
299                     canonicalizationMethod, signatureMethod, Collections
300                             .singletonList(ref));
301 
302             // Create a KeyValue containing the DSA or RSA PublicKey
303             final KeyInfoFactory keyInfoFactory = sigFactory
304                     .getKeyInfoFactory();
305             final KeyValue keyValuePair = keyInfoFactory.newKeyValue(pubKey);
306 
307             // Create a KeyInfo and add the KeyValue to it
308             final KeyInfo keyInfo = keyInfoFactory.newKeyInfo(Collections
309                     .singletonList(keyValuePair));
310             // Convert the JDOM document to w3c (Java XML signature API requires
311             // w3c
312             // representation)
313             final org.w3c.dom.Element w3cElement = toDom(element);
314 
315             // Create a DOMSignContext and specify the DSA/RSA PrivateKey and
316             // location of the resulting XMLSignature's parent element
317             final DOMSignContext dsc = new DOMSignContext(privKey, w3cElement);
318 
319             final org.w3c.dom.Node xmlSigInsertionPoint = getXmlSignatureInsertLocation(w3cElement);
320             dsc.setNextSibling(xmlSigInsertionPoint);
321 
322             // Marshal, generate (and sign) the enveloped signature
323             final XMLSignature signature = sigFactory.newXMLSignature(signedInfo,
324                     keyInfo);
325             signature.sign(dsc);
326 
327             return toJdom(w3cElement);
328 
329         } catch (final Exception e) {
330             throw new RuntimeException("Error signing SAML element: "
331                     + e.getMessage(), e);
332         }
333     }

335     /**
336      * Gets the xml signature insert location.
337      *
338      * @param elem the elem
339      * @return the xml signature insert location
340      */
341     private static Node getXmlSignatureInsertLocation(final org.w3c.dom.Element elem) {
342         org.w3c.dom.Node insertLocation = null;
343         org.w3c.dom.NodeList nodeList = elem.getElementsByTagNameNS(
344                 SAMLConstants.SAML20P_NS, "Extensions");
345         if (nodeList.getLength() != 0) {
346             insertLocation = nodeList.item(nodeList.getLength() - 1);
347         } else {
348             nodeList = elem.getElementsByTagNameNS(SAMLConstants.SAML20P_NS, "Status");
349             insertLocation = nodeList.item(nodeList.getLength() - 1);
350         }
351         return insertLocation;
352     }

364     /**
365      * Convert the received jdom doc to a Document element.
366      *
367      * @param doc the doc
368      * @return the org.w3c.dom. document
369      */
370     private static org.w3c.dom.Document toDom(final Document doc) {
371         try {
372             final XMLOutputter xmlOutputter = new XMLOutputter();
373             final StringWriter elemStrWriter = new StringWriter();
374             xmlOutputter.output(doc, elemStrWriter);
375             final byte[] xmlBytes = elemStrWriter.toString().getBytes(Charset.defaultCharset());
376             final DocumentBuilderFactory dbf = DocumentBuilderFactory
377                     .newInstance();
378             dbf.setNamespaceAware(true);
379             return dbf.newDocumentBuilder().parse(
380                     new ByteArrayInputStream(xmlBytes));
381         } catch (final Exception e) {
382             return null;
383         }
384     }
