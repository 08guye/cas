93      /**
94       * Creates the SAML service from the request.
95       *
96       * @param request the request
97       * @return the SAML service
98       */
99      public static SamlService createServiceFrom(
100             final HttpServletRequest request) {
101         final String service = request.getParameter(CONST_PARAM_SERVICE);
102         final String artifactId;
103         final String requestBody = getRequestBody(request);
104         final String requestId;
105 
106         if (!StringUtils.hasText(service) && !StringUtils.hasText(requestBody)) {
107             return null;
108         }
109 
110         final String id = cleanupUrl(service);
111 
112         if (StringUtils.hasText(requestBody)) {
113 
114             final String tagStart;
115             final String tagEnd;
116             if (requestBody.contains(CONST_START_ARTIFACT_XML_TAG)) {
117                 tagStart = CONST_START_ARTIFACT_XML_TAG;
118                 tagEnd = CONST_END_ARTIFACT_XML_TAG;
119             } else {
120                 tagStart = CONST_START_ARTIFACT_XML_TAG_NO_NAMESPACE;
121                 tagEnd = CONST_END_ARTIFACT_XML_TAG_NO_NAMESPACE;
122             }
123             final int startTagLocation = requestBody.indexOf(tagStart);
124             final int artifactStartLocation = startTagLocation + tagStart.length();
125             final int endTagLocation = requestBody.indexOf(tagEnd);
126 
127             artifactId = requestBody.substring(artifactStartLocation, endTagLocation).trim();
128 
129             // is there a request id?
130             requestId = extractRequestId(requestBody);
131         } else {
132             artifactId = null;
133             requestId = null;
134         }
135 
136         LOGGER.debug("Attempted to extract Request from HttpServletRequest. Results:");
137         LOGGER.debug(String.format("Request Body: %s", requestBody));
138         LOGGER.debug(String.format("Extracted ArtifactId: %s", artifactId));
139         LOGGER.debug(String.format("Extracted Request Id: %s", requestId));
140 
141         return new SamlService(id, service, artifactId, requestId);
142     }

144     @Override
145     public Response getResponse(final String ticketId) {
146         final Map<String, String> parameters = new HashMap<String, String>();
147 
148         parameters.put(CONST_PARAM_TICKET, ticketId);
149         parameters.put(CONST_PARAM_SERVICE, getOriginalUrl());
150 
151         return Response.getRedirectResponse(getOriginalUrl(), parameters);
152     }

154     /**
155      * Extract request id from the body.
156      *
157      * @param requestBody the request body
158      * @return the string
159      */
160     protected static String extractRequestId(final String requestBody) {
161         if (!requestBody.contains("RequestID")) {
162             return null;
163         }
164 
165         try {
166             final int position = requestBody.indexOf("RequestID=\"") + 11;
167             final int nextPosition = requestBody.indexOf("\"", position);
168 
169             return requestBody.substring(position,  nextPosition);
170         } catch (final Exception e) {
171             LOGGER.debug("Exception parsing RequestID from request.", e);
172             return null;
173         }
174     }
175 
176     /**
177      * Gets the request body from the request.
178      *
179      * @param request the request
180      * @return the request body
181      */
182     protected static String getRequestBody(final HttpServletRequest request) {
183         final StringBuilder builder = new StringBuilder();
184         BufferedReader reader = null;
185         try {
186             reader = request.getReader();
187 
188             String line;
189             while ((line = reader.readLine()) != null) {
190                 builder.append(line);
191             }
192             return builder.toString();
193         } catch (final Exception e) {
194             return null;
195         } finally {
196             IOUtils.closeQuietly(reader);
197         }
198     }
