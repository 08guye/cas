51  /**
52   * Implementation of a view to return a SAML SOAP response and assertion, based on
53   * the SAML 1.1 specification.
54   * <p>
55   * If an AttributePrincipal is supplied, then the assertion will include the
56   * attributes from it (assuming a String key/Object value pair). The only
57   * Authentication attribute it will look at is the authMethod (if supplied).
58   * <p>
59   * Note that this class will currently not handle proxy authentication.
60   * <p>
61   *
62   * @author Scott Battaglia
63   * @author Marvin S. Addison
64   * @since 3.1
65   */
66  public final class Saml10SuccessResponseView extends AbstractSaml10ResponseView {
67      /** Namespace for custom attributes in the saml validation payload. */
68      private static final String VALIDATION_SAML_ATTRIBUTE_NAMESPACE = "http://www.ja-sig.org/products/cas/";
69  
70      private static final String CONFIRMATION_METHOD = "urn:oasis:names:tc:SAML:1.0:cm:artifact";
71  
72      private final XSStringBuilder attrValueBuilder = new XSStringBuilder();
73  
74      /** The issuer, generally the hostname. */
75      @NotNull
76      private String issuer;
77  
78      /** The amount of time in milliseconds this is valid for. */
79      @Min(1000)
80      private long issueLength = 30000;
81  
82      @NotNull
83      private String rememberMeAttributeName = CasProtocolConstants.VALIDATION_REMEMBER_ME_ATTRIBUTE_NAME;
84  
85      @Override
86      protected void prepareResponse(final Response response, final Map<String, Object> model) {
87  
88          final DateTime issuedAt = response.getIssueInstant();
89          final Service service = getAssertionFrom(model).getService();
90  
91          // Build up the SAML assertion containing AuthenticationStatement and AttributeStatement
92          final Assertion assertion = newSamlObject(Assertion.class);
93          assertion.setID(generateId());
94          assertion.setIssueInstant(issuedAt);
95          assertion.setIssuer(this.issuer);
96          assertion.setConditions(newConditions(issuedAt, service.getId()));
97          final AuthenticationStatement authnStatement = newAuthenticationStatement(model);
98          assertion.getAuthenticationStatements().add(authnStatement);
99  
100         final Subject subject = newSubject(getPrincipal(model).getId());
101         final Map<String, Object> attributesToSend = prepareSamlAttributes(model);
102 
103         if (!attributesToSend.isEmpty()) {
104             assertion.getAttributeStatements().add(newAttributeStatement(subject, attributesToSend));
105         }
106 
107         response.setStatus(newStatus(StatusCode.SUCCESS, null));
108         response.getAssertions().add(assertion);
109     }

112     /**
113      * Prepare saml attributes. Combines both principal and authentication
114      * attributes. If the authentication is to be remembered, uses {@link #setRememberMeAttributeName(String)}
115      * for the remember-me attribute name.
116      *
117      * @param model the model
118      * @return the final map
119      * @since 4.1
120      */
121     private Map<String, Object> prepareSamlAttributes(final Map<String, Object> model) {
122         final Map<String, Object> authnAttributes = new HashMap<String, Object>(getAuthenticationAttributesAsMultiValuedAttributes(model));
123         if (isRememberMeAuthentication(model)) {
124             authnAttributes.remove(RememberMeCredential.AUTHENTICATION_ATTRIBUTE_REMEMBER_ME);
125             authnAttributes.put(this.rememberMeAttributeName, Boolean.TRUE.toString());
126         }
127         final Map<String, Object> attributesToReturn = new HashMap<String, Object>();
128         attributesToReturn.putAll(getPrincipalAttributesAsMultiValuedAttributes(model));
129         attributesToReturn.putAll(authnAttributes);
130         return attributesToReturn;
131     }

133     /**
134      * New conditions element.
135      *
136      * @param issuedAt the issued at
137      * @param serviceId the service id
138      * @return the conditions
139      */
140     private Conditions newConditions(final DateTime issuedAt, final String serviceId) {
141         final Conditions conditions = newSamlObject(Conditions.class);
142         conditions.setNotBefore(issuedAt);
143         conditions.setNotOnOrAfter(issuedAt.plus(this.issueLength));
144         final AudienceRestrictionCondition audienceRestriction = newSamlObject(AudienceRestrictionCondition.class);
145         final Audience audience = newSamlObject(Audience.class);
146         audience.setUri(serviceId);
147         audienceRestriction.getAudiences().add(audience);
148         conditions.getAudienceRestrictionConditions().add(audienceRestriction);
149         return conditions;
150     }

152     /**
153      * New subject element.
154      *
155      * @param identifier the identifier
156      * @return the subject
157      */
158     private Subject newSubject(final String identifier) {
159         final SubjectConfirmation confirmation = newSamlObject(SubjectConfirmation.class);
160         final ConfirmationMethod method = newSamlObject(ConfirmationMethod.class);
161         method.setConfirmationMethod(CONFIRMATION_METHOD);
162         confirmation.getConfirmationMethods().add(method);
163         final NameIdentifier nameIdentifier = newSamlObject(NameIdentifier.class);
164         nameIdentifier.setNameIdentifier(identifier);
165         final Subject subject = newSamlObject(Subject.class);
166         subject.setNameIdentifier(nameIdentifier);
167         subject.setSubjectConfirmation(confirmation);
168         return subject;
169     }

171     /**
172      * New authentication statement.
173      *
174      * @param model the model
175      * @return the authentication statement
176      */
177     private AuthenticationStatement newAuthenticationStatement(final Map<String, Object> model) {
178         final Authentication authentication = getPrimaryAuthenticationFrom(model);
179         final String authenticationMethod = (String) authentication.getAttributes().get(
180                 SamlAuthenticationMetaDataPopulator.ATTRIBUTE_AUTHENTICATION_METHOD);
181         final AuthenticationStatement authnStatement = newSamlObject(AuthenticationStatement.class);
182         authnStatement.setAuthenticationInstant(new DateTime(authentication.getAuthenticationDate()));
183         authnStatement.setAuthenticationMethod(
184                 authenticationMethod != null
185                 ? authenticationMethod
186                         : SamlAuthenticationMetaDataPopulator.AUTHN_METHOD_UNSPECIFIED);
187         authnStatement.setSubject(newSubject(getPrincipal(model).getId()));
188         return authnStatement;
189     }

191     /**
192      * New attribute statement.
193      *
194      * @param subject the subject
195      * @param attributes the attributes
196      * @return the attribute statement
197      */
198     private AttributeStatement newAttributeStatement(
199             final Subject subject, final Map<String, Object> attributes) {
200 
201         final AttributeStatement attrStatement = newSamlObject(AttributeStatement.class);
202         attrStatement.setSubject(subject);
203         for (final Entry<String, Object> e : attributes.entrySet()) {
204             if (e.getValue() instanceof Collection<?> && ((Collection<?>) e.getValue()).isEmpty()) {
205                 // bnoordhuis: don't add the attribute, it causes a org.opensaml.MalformedException
206                 logger.info("Skipping attribute {} because it does not have any values.", e.getKey());
207                 continue;
208             }
209             final Attribute attribute = newSamlObject(Attribute.class);
210             attribute.setAttributeName(e.getKey());
211             attribute.setAttributeNamespace(VALIDATION_SAML_ATTRIBUTE_NAMESPACE);
212             if (e.getValue() instanceof Collection<?>) {
213                 final Collection<?> c = (Collection<?>) e.getValue();
214                 for (final Object value : c) {
215                     attribute.getAttributeValues().add(newAttributeValue(value));
216                 }
217             } else {
218                 attribute.getAttributeValues().add(newAttributeValue(e.getValue()));
219             }
220             attrStatement.getAttributes().add(attribute);
221         }
222 
223         return attrStatement;
224     }

226     /**
227      * New attribute value.
228      *
229      * @param value the value
230      * @return the xS string
231      */
232     private XSString newAttributeValue(final Object value) {
233         final XSString stringValue = this.attrValueBuilder.buildObject(AttributeValue.DEFAULT_ELEMENT_NAME, XSString.TYPE_NAME);
234         if (value instanceof String) {
235             stringValue.setValue((String) value);
236         } else {
237             stringValue.setValue(value.toString());
238         }
239         return stringValue;
240     }
