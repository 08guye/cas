30  import org.slf4j.helpers.Util;

32  import java.net.URL;
33  import java.util.Map;
34  import java.util.Set;
35  import java.util.concurrent.ConcurrentHashMap;

37  /**
38   * An implementation of {@link org.slf4j.ILoggerFactory} that is looked up via the
39   * {@link org.slf4j.impl.StaticLoggerBinder} of CAS itself. It is responsible for
40   * creating {@link org.slf4j.Logger} instances and passing them back to the slf4j engine.
41   * @author Misagh Moayyed
42   * @since 4.1
43   */
44  public final class CasLoggerFactory implements ILoggerFactory {
45  
46      private static final String PACKAGE_TO_SCAN = "org.slf4j.impl";
47  
48      private final Map<String, CasDelegatingLogger> loggerMap;
49  
50      private final Class<? extends ILoggerFactory> realLoggerFactoryClass;
51      /**
52       * Instantiates a new Cas logger factory.
53       * Configures the reflection scanning engine to be prepared to scan <code>org.slf4j.impl</code>
54       * in order to find other avaliable factories.
55       */
56      public CasLoggerFactory() {
57          this.loggerMap = new ConcurrentHashMap<String, CasDelegatingLogger>();
58          final Set<URL> set = ClasspathHelper.forPackage(PACKAGE_TO_SCAN);
59          final Reflections reflections = new Reflections(new ConfigurationBuilder().addUrls(set).setScanners(new SubTypesScanner()));
60  
61          final Set<Class<? extends ILoggerFactory>> subTypesOf = reflections.getSubTypesOf(ILoggerFactory.class);
62          subTypesOf.remove(this.getClass());
63  
64          if (subTypesOf.size() > 1) {
65              Util.report("Multiple ILoggerFactory bindings are found on the classpath:");
66              for (final Class<? extends ILoggerFactory> c : subTypesOf) {
67                  Util.report("* " + c.getCanonicalName());
68              }
69          }
70  
71          if (subTypesOf.isEmpty()) {
72              final RuntimeException e = new RuntimeException("No ILoggerFactory could be found on the classpath."
73                      + " CAS cannot determine the logging framework."
74                      + " Examine the project dependencies and ensure there is one and only one logging framework is available.");
75  
76              Util.report(e.getMessage(), e);
77              throw e;
78          }
79          this.realLoggerFactoryClass = subTypesOf.iterator().next();
80          Util.report("ILoggerFactory to be used for logging is: " + this.realLoggerFactoryClass.getName());
81      }
82  
83      /**
84       * {@inheritDoc}
85       * <p>Attempts to find the <strong>real</strong> <code>Logger</code> instance that
86       * is doing the heavy lifting and routes the request to an instance of
87       * {@link CasDelegatingLogger}. The instance is cached by the logger name.</p>
88       */
89      @Override
90      public Logger getLogger(final String name) {
91          if (StringUtils.isBlank(name)) {
92              return NOPLogger.NOP_LOGGER;
93          }
94          synchronized (loggerMap) {
95              if (!loggerMap.containsKey(name)) {
96                  final Logger logger = getRealLoggerInstance(name);
97                  loggerMap.put(name, new CasDelegatingLogger(logger));
98              }
99              return loggerMap.get(name);
100         }
101     }
102 
103     /**
104      * Find the actual <code>Logger</code> instance that is available on the classpath.
105      * This is usually the logger adapter that is provided by the real logging framework,
106      * such as log4j, etc. The method will scan the runtime to find logger factories that
107      * are of type {@link org.slf4j.ILoggerFactory}. It will remove itself from this list
108      * first and then attempts to locate the next best factory from which real logger instances
109      * can be created.
110      * @param name requested logger name
111      * @return the logger instance created by the logger factory available on the classpath during runtime, or null.
112      */
113     private Logger getRealLoggerInstance(final String name) {
114         try {
115             final ILoggerFactory factInstance = this.realLoggerFactoryClass.newInstance();
116             return factInstance.getLogger(name);
117         } catch (final Exception e) {
118             throw new RuntimeException(e.getMessage(), e);
119         }
120     }
