69  /**
70   * Concrete implementation of a CentralAuthenticationService, and also the
71   * central, organizing component of CAS's internal implementation.
72   * <p>
73   * This class is threadsafe.
74   * <p>
75   * This class has the following properties that must be set:
76   * <ul>
77   * <li> <code>ticketRegistry</code> - The Ticket Registry to maintain the list
78   * of available tickets.</li>
79   * <li> <code>serviceTicketRegistry</code> - Provides an alternative to configure separate registries for

85   * <li> <code>serviceTicketUniqueTicketIdGenerator</code> - Plug in to
86   * generate unique secure ids for ServiceTickets.</li>
87   * <li> <code>ticketGrantingTicketExpirationPolicy</code> - The expiration
88   * policy for TicketGrantingTickets.</li>
89   * <li> <code>serviceTicketExpirationPolicy</code> - The expiration policy for
90   * ServiceTickets.</li>
91   * </ul>
92   *
93   * @author William G. Thompson, Jr.
94   * @author Scott Battaglia
95   * @author Dmitry Kopylenko

194     /**
195      * {@inheritDoc}
196      * Destroy a TicketGrantingTicket and perform back channel logout. This has the effect of invalidating any
197      * Ticket that was derived from the TicketGrantingTicket being destroyed. May throw an
198      * {@link IllegalArgumentException} if the TicketGrantingTicket ID is null.
199      *
200      * @param ticketGrantingTicketId the id of the ticket we want to destroy
201      * @return the logout requests.
202      */
203     @Audit(
204             action="TICKET_GRANTING_TICKET_DESTROYED",
205             actionResolverName="DESTROY_TICKET_GRANTING_TICKET_RESOLVER",
206             resourceResolverName="DESTROY_TICKET_GRANTING_TICKET_RESOURCE_RESOLVER")
207     @Profiled(tag = "DESTROY_TICKET_GRANTING_TICKET", logFailuresSeparately = false)
208     @Transactional(readOnly = false)
209     @Override
210     public List<LogoutRequest> destroyTicketGrantingTicket(final String ticketGrantingTicketId) {
211         Assert.notNull(ticketGrantingTicketId);
212 
213         logger.debug("Removing ticket [{}] from registry.", ticketGrantingTicketId);
214         final TicketGrantingTicket ticket = this.ticketRegistry.getTicket(ticketGrantingTicketId,
215                 TicketGrantingTicket.class);
216 
217         if (ticket == null) {
218             logger.debug("TicketGrantingTicket [{}] cannot be found in the ticket registry.", ticketGrantingTicketId);
219             return Collections.emptyList();
220         }
221 
222         logger.debug("Ticket found. Processing logout requests and then deleting the ticket...");
223         final List<LogoutRequest> logoutRequests = logoutManager.performLogout(ticket);
224         this.ticketRegistry.deleteTicket(ticketGrantingTicketId);
225 
226         return logoutRequests;
227     }

229     @Audit(
230         action="SERVICE_TICKET",
231         actionResolverName="GRANT_SERVICE_TICKET_RESOLVER",
232         resourceResolverName="GRANT_SERVICE_TICKET_RESOURCE_RESOLVER")
233     @Profiled(tag="GRANT_SERVICE_TICKET", logFailuresSeparately = false)
234     @Transactional(readOnly = false)
235     @Override
236     public String grantServiceTicket(
237             final String ticketGrantingTicketId, final Service service, final Credential... credentials)
238             throws AuthenticationException, TicketException {
239         Assert.notNull(ticketGrantingTicketId, "ticketGrantingticketId cannot be null");
240         Assert.notNull(service, "service cannot be null");
241 
242         final TicketGrantingTicket ticketGrantingTicket = this.ticketRegistry.getTicket(ticketGrantingTicketId, TicketGrantingTicket.class);
243 
244         if (ticketGrantingTicket == null) {
245             logger.debug("TicketGrantingTicket [{}] cannot be found in the ticket registry.", ticketGrantingTicketId);
246             throw new InvalidTicketException(ticketGrantingTicketId);
247         }
248 
249         synchronized (ticketGrantingTicket) {
250             if (ticketGrantingTicket.isExpired()) {
251                 this.ticketRegistry.deleteTicket(ticketGrantingTicketId);
252                 logger.debug("TicketGrantingTicket [{}] has expired and is now deleted from the ticket registry.", ticketGrantingTicketId);
253                 throw new InvalidTicketException(ticketGrantingTicketId);
254             }
255         }
256 
257         final RegisteredService registeredService = this.servicesManager.findServiceBy(service);
258 
259         verifyRegisteredServiceProperties(registeredService, service);
260         
261         if (!registeredService.isSsoEnabled() && credentials == null
262             && ticketGrantingTicket.getCountOfUses() > 0) {
263             logger.warn("ServiceManagement: Service [{}] is not allowed to use SSO.", service.getId());
264             throw new UnauthorizedSsoServiceException();
265         }
266 
267         //CAS-1019
268         final List<Authentication> authns = ticketGrantingTicket.getChainedAuthentications();
269         if(authns.size() > 1) {
270             if (!registeredService.getProxyPolicy().isAllowedToProxy()) {
271                 final String message = String.
272                         format("ServiceManagement: Proxy attempt by service [%s] (registered service [%s]) is not allowed.",
273                         service.getId(), registeredService.toString());
274                 logger.warn(message);
275                 throw new UnauthorizedProxyingException(message);
276             }
277         }
278 
279         if (credentials != null) {
280             final Authentication current = this.authenticationManager.authenticate(credentials);
281             final Authentication original = ticketGrantingTicket.getAuthentication();
282             if (!current.getPrincipal().equals(original.getPrincipal())) {
283                 throw new MixedPrincipalException(current, current.getPrincipal(), original.getPrincipal());
284             }
285             ticketGrantingTicket.getSupplementalAuthentications().add(current);
286         }
287 
288         // Perform security policy check by getting the authentication that satisfies the configured policy
289         // This throws if no suitable policy is found
290         getAuthenticationSatisfiedByPolicy(ticketGrantingTicket, new ServiceContext(service, registeredService));
291 
292         final String uniqueTicketIdGenKey = service.getClass().getName();
293         if (!this.uniqueTicketIdGeneratorsForService.containsKey(uniqueTicketIdGenKey)) {
294             logger.warn("Cannot create service ticket because the key [{}] for service [{}] is not linked to a ticket id generator",
295                     uniqueTicketIdGenKey, service.getId());
296             throw new UnauthorizedSsoServiceException();
297         }
298         
299         final UniqueTicketIdGenerator serviceTicketUniqueTicketIdGenerator =
300                 this.uniqueTicketIdGeneratorsForService.get(uniqueTicketIdGenKey);
301 
302         final List<Authentication> authentications = ticketGrantingTicket.getChainedAuthentications();
303         final String ticketPrefix = authentications.size() == 1 ? ServiceTicket.PREFIX : ServiceTicket.PROXY_TICKET_PREFIX;
304         final String ticketId = serviceTicketUniqueTicketIdGenerator.getNewTicketId(ticketPrefix);
305         final ServiceTicket serviceTicket = ticketGrantingTicket.grantServiceTicket(ticketId, service,
306                 this.serviceTicketExpirationPolicy, credentials != null);
307 
308         this.serviceTicketRegistry.addTicket(serviceTicket);
309 
310         final String principalId = authentications.get(authentications.size() - 1).getPrincipal().getId();
311         logger.info("Granted ticket [{}] for service [{}] for user [{}]",
312                 serviceTicket.getId(), service.getId(), principalId);
313 
314         return serviceTicket.getId();
315     }

333     @Audit(
334         action="PROXY_GRANTING_TICKET",
335         actionResolverName="GRANT_PROXY_GRANTING_TICKET_RESOLVER",
336         resourceResolverName="GRANT_PROXY_GRANTING_TICKET_RESOURCE_RESOLVER")
337     @Profiled(tag="GRANT_PROXY_GRANTING_TICKET", logFailuresSeparately = false)
338     @Transactional(readOnly = false)
339     @Override
340     public String delegateTicketGrantingTicket(final String serviceTicketId, final Credential... credentials)
341             throws AuthenticationException, TicketException {
342 
343         Assert.notNull(serviceTicketId, "serviceTicketId cannot be null");
344         Assert.notNull(credentials, "credentials cannot be null");
345 
346         final ServiceTicket serviceTicket =  this.serviceTicketRegistry.getTicket(serviceTicketId, ServiceTicket.class);
347 
348         if (serviceTicket == null || serviceTicket.isExpired()) {
349             logger.debug("ServiceTicket [{}] has expired or cannot be found in the ticket registry", serviceTicketId);
350             throw new InvalidTicketException(serviceTicketId);
351         }
352 
353         final RegisteredService registeredService = this.servicesManager
354                 .findServiceBy(serviceTicket.getService());
355 
356         verifyRegisteredServiceProperties(registeredService, serviceTicket.getService());
357         
358         if (!registeredService.getProxyPolicy().isAllowedToProxy()) {
359             logger.warn("ServiceManagement: Service [{}] attempted to proxy, but is not allowed.", serviceTicket.getService().getId());
360             throw new UnauthorizedProxyingException();
361         }
362 
363         final Authentication authentication = this.authenticationManager.authenticate(credentials);
364 
365         final String pgtId = this.ticketGrantingTicketUniqueTicketIdGenerator.getNewTicketId(
366                 TicketGrantingTicket.PROXY_GRANTING_TICKET_PREFIX);
367         final TicketGrantingTicket proxyGrantingTicket = serviceTicket.grantTicketGrantingTicket(pgtId,
368                                     authentication, this.ticketGrantingTicketExpirationPolicy);
369 
370         logger.debug("Generated proxy granting ticket [{}] based off of [{}]", proxyGrantingTicket, serviceTicketId);
371         this.ticketRegistry.addTicket(proxyGrantingTicket);
372 
373         return proxyGrantingTicket.getId();
374     }

376     @Audit(
377         action="SERVICE_TICKET_VALIDATE",
378         actionResolverName="VALIDATE_SERVICE_TICKET_RESOLVER",
379         resourceResolverName="VALIDATE_SERVICE_TICKET_RESOURCE_RESOLVER")
380     @Profiled(tag="VALIDATE_SERVICE_TICKET", logFailuresSeparately = false)
381     @Transactional(readOnly = false)
382     @Override
383     public Assertion validateServiceTicket(final String serviceTicketId, final Service service) throws TicketException {
384         Assert.notNull(serviceTicketId, "serviceTicketId cannot be null");
385         Assert.notNull(service, "service cannot be null");
386  
387         final ServiceTicket serviceTicket =  this.serviceTicketRegistry.getTicket(serviceTicketId, ServiceTicket.class);
388 
389         if (serviceTicket == null) {
390             logger.info("ServiceTicket [{}] does not exist.", serviceTicketId);
391             throw new InvalidTicketException(serviceTicketId);
392         }
393 
394         final RegisteredService registeredService = this.servicesManager.findServiceBy(service);
395 
396         verifyRegisteredServiceProperties(registeredService, serviceTicket.getService());
397         
398         try {
399             synchronized (serviceTicket) {
400                 if (serviceTicket.isExpired()) {
401                     logger.info("ServiceTicket [{}] has expired.", serviceTicketId);
402                     throw new InvalidTicketException(serviceTicketId);
403                 }
404 
405                 if (!serviceTicket.isValidFor(service)) {
406                     logger.error("ServiceTicket [{}] with service [{}] does not match supplied service [{}]",
407                             serviceTicketId, serviceTicket.getService().getId(), service);
408                     throw new TicketValidationException(serviceTicket.getService());
409                 }
410             }
411 
412             final TicketGrantingTicket root = serviceTicket.getGrantingTicket().getRoot();
413             final Authentication authentication = getAuthenticationSatisfiedByPolicy(
414                     root, new ServiceContext(serviceTicket.getService(), registeredService));
415             final Principal principal = authentication.getPrincipal();
416 
417             final AttributeReleasePolicy attributePolicy = registeredService.getAttributeReleasePolicy();
418             logger.debug("Attribute policy [{}] is associated with service [{}]", attributePolicy, registeredService);
419             
420             @SuppressWarnings("unchecked")
421             final Map<String, Object> attributesToRelease = attributePolicy != null
422                     ? attributePolicy.getAttributes(principal) : Collections.EMPTY_MAP;
423                     
424             final String principalId = determinePrincipalIdForRegisteredService(principal, registeredService, serviceTicket);
425             final Principal modifiedPrincipal = new SimplePrincipal(principalId, attributesToRelease);
426             final AuthenticationBuilder builder = AuthenticationBuilder.newInstance(authentication);
427             builder.setPrincipal(modifiedPrincipal);
428 
429             return new ImmutableAssertion(
430                     builder.build(),
431                     serviceTicket.getGrantingTicket().getChainedAuthentications(),
432                     serviceTicket.getService(),
433                     serviceTicket.isFromNewLogin());
434         } finally {
435             if (serviceTicket.isExpired()) {
436                 this.serviceTicketRegistry.deleteTicket(serviceTicketId);
437             }
438         }
439     }

441     /**
442      * Determines the principal id to use for a {@link RegisteredService} using the following rules:
443      *
444      * <ul>
445      *  <li> If the service is marked to allow anonymous access, a persistent id is returned. </li>
446      *  <li> If the {@link org.jasig.cas.services.RegisteredService#getUsernameAttribute()} is blank, then the default
447      *  principal id is returned.</li>
448      *  <li>If the username attribute is available as part of the principal's attributes,
449      *  the corresponding attribute value will be returned.
450      *  </li>
451      *   <li>Otherwise, the default principal's id is returned as the username attribute
452      *   with an additional warning.</li>
453      * </ul>
454      *
455      * @param principal The principal object to be validated and constructed
456      * @param registeredService Requesting service for which a principal is being validated.
457      * @param serviceTicket An instance of the service ticket used for validation
458      *
459      * @return The principal id to use for the requesting registered service
460      */
461     private String determinePrincipalIdForRegisteredService(final Principal principal,
462                                                             final RegisteredService registeredService,
463                                                             final ServiceTicket serviceTicket) {
464         String principalId = null;
465         final String serviceUsernameAttribute = registeredService.getUsernameAttribute();
466 
467         if (registeredService.isAnonymousAccess()) {
468             principalId = this.persistentIdGenerator.generate(principal, serviceTicket.getService());
469         } else if (StringUtils.isBlank(serviceUsernameAttribute)) {
470             principalId = principal.getId();
471         } else {
472             if (principal.getAttributes().containsKey(serviceUsernameAttribute)) {
473                 principalId = principal.getAttributes().get(serviceUsernameAttribute).toString();
474             } else {
475                 principalId = principal.getId();
476                 final Object[] errorLogParameters = new Object[] {
477                         principalId,
478                         registeredService.getUsernameAttribute(),
479                         principal.getAttributes(),
480                         registeredService.getServiceId(),
481                         principalId };
482                 logger.warn("Principal [{}] did not have attribute [{}] among attributes [{}] so CAS cannot "
483                         + "provide on the validation response the user attribute the registered service [{}] expects. "
484                         + "CAS will instead return the default username attribute [{}]", errorLogParameters);
485             }
486 
487         }
488 
489         logger.debug("Principal id to return for service [{}] is [{}]. The default principal id is [{}].",
490                 new Object[]{registeredService.getName(), principal.getId(), principalId});
491         return principalId;
492     }

540     /**
541      * Gets the authentication satisfied by policy.
542      *
543      * @param ticket the ticket
544      * @param context the context
545      * @return the authentication satisfied by policy
546      * @throws TicketException the ticket exception
547      */
548     private Authentication getAuthenticationSatisfiedByPolicy(
549             final TicketGrantingTicket ticket, final ServiceContext context) throws TicketException {
550 
551         final ContextualAuthenticationPolicy<ServiceContext> policy =
552                 serviceContextAuthenticationPolicyFactory.createPolicy(context);
553         if (policy.isSatisfiedBy(ticket.getAuthentication())) {
554             return ticket.getAuthentication();
555         }
556         for (final Authentication auth : ticket.getSupplementalAuthentications()) {
557             if (policy.isSatisfiedBy(auth)) {
558                 return auth;
559             }
560         }
561         throw new UnsatisfiedAuthenticationPolicyException(policy);
562     }

564     /**
565      * Ensure that the service is found and enabled in the service registry.
566      * @param registeredService the located entry in the registry
567      * @param service authenticating service
568      * @throws UnauthorizedServiceException
569      */
570     private void verifyRegisteredServiceProperties(final RegisteredService registeredService, final Service service) {
571         if (registeredService == null) {
572             final String msg = String.format("ServiceManagement: Unauthorized Service Access. "
573                     + "Service [%s] is not found in service registry.", service.getId());
574             logger.warn(msg);
575             throw new UnauthorizedServiceException(UnauthorizedServiceException.CODE_UNAUTHZ_SERVICE, msg);
576         }
577         if (!registeredService.isEnabled()) {
