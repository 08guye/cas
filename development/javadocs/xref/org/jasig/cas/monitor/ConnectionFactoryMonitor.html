19  package org.jasig.cas.monitor;
20  
21  import org.ldaptive.Connection;
22  import org.ldaptive.ConnectionFactory;
23  import org.ldaptive.LdapException;
24  import org.ldaptive.pool.Validator;
25  import org.springframework.beans.factory.annotation.Autowired;
26  import org.springframework.beans.factory.annotation.Qualifier;
27  import org.springframework.stereotype.Component;
28  
29  import javax.annotation.Nullable;
30  
31  /**
32   * Monitors an ldaptive {@link ConnectionFactory}.  While this class can be used with instances of
33   * {@link org.ldaptive.pool.PooledConnectionFactory}, the {@link PooledConnectionFactoryMonitor} class is preferable.
34   *
35   * @author Marvin S. Addison
36   * @since 4.0.0
37   */
38  @Component("ldapConnectionFactoryMonitor")
39  public class ConnectionFactoryMonitor extends AbstractNamedMonitor<Status> {
40  
41      /** OK status. */
42      private static final Status OK = new Status(StatusCode.OK);
43  
44      /** Error status. */
45      private static final Status ERROR = new Status(StatusCode.ERROR);
46  
47      /** Source of connections to validate. */
48      @Nullable
49      @Autowired(required=false)
50      @Qualifier("connectionFactoryMonitorConnectionFactory")
51      private ConnectionFactory connectionFactory;
52  
53      /** Connection validator. */
54      @Nullable
55      @Autowired(required=false)
56      @Qualifier("connectionFactoryMonitorValidator")
57      private Validator<Connection> validator;
58  
59  
60      /**
61       * Instantiates a new Connection factory monitor.
62       */
63      public ConnectionFactoryMonitor() {}
64  
65      /**
66       * Creates a new instance that monitors the given connection factory.
67       *
68       * @param  factory  Connection factory to monitor.
69       * @param  validator  Validates connections from the factory.
70       */
71      public ConnectionFactoryMonitor(final ConnectionFactory factory, final Validator<Connection> validator) {
72          this.connectionFactory = factory;
73          this.validator = validator;
74      }
75  
76  
77      /**
78       * Gets a connection from the underlying connection factory and attempts to validate it.
79       *
80       * @return  Status with code {@link StatusCode#OK} on success otherwise {@link StatusCode#ERROR}.
81       */
82      @Override
83      public Status observe() {
84          if (this.connectionFactory != null && this.validator != null) {
85              try (final Connection conn = this.connectionFactory.getConnection()) {
86                  if (!conn.isOpen()) {
87                      conn.open();
88                  }
89                  return this.validator.validate(conn) ? OK : ERROR;
90              } catch (final LdapException e) {
91                  logger.warn("Validation failed with error.", e);
92              }
93              return ERROR;
94          }
95          return Status.UNKNOWN;
96      }
97  }
