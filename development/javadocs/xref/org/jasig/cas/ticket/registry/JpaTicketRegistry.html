63  /**
64   * JPA implementation of a CAS {@link TicketRegistry}. This implementation of
65   * ticket registry is suitable for HA environments.
66   *
67   * @author Scott Battaglia
68   * @author Marvin S. Addison
69   *
70   * @since 3.2.1
71   *
72   */
73  @Component("jpaTicketRegistry")
74  public final class JpaTicketRegistry extends AbstractDistributedTicketRegistry implements Job {
75  
76      @Value("${ticket.registry.cleaner.repeatinterval:5000000}")
77      private int refreshInterval;
78  
79      @Value("${ticket.registry.cleaner.startdelay:20000}")
80      private int startDelay;
81  
82      @Autowired
83      @NotNull
84      private ApplicationContext applicationContext;
85  
86      @Autowired
87      @Qualifier("logoutManager")
88      private LogoutManager logoutManager;
89  
90      @Autowired
91      @Qualifier("jpaLockingStrategy")
92      private LockingStrategy jpaLockingStrategy;
93  
94      @NotNull
95      @PersistenceContext(unitName = "ticketEntityManagerFactory")
96      private EntityManager entityManager;
97  
98      @NotNull
99      private String ticketGrantingTicketPrefix = TicketGrantingTicket.PREFIX;
100 
101     @Override
102     protected void updateTicket(final Ticket ticket) {
103         entityManager.merge(ticket);
104         logger.debug("Updated ticket [{}].", ticket);
105     }
106 
107     @Override
108     public void addTicket(final Ticket ticket) {
109         entityManager.persist(ticket);
110         logger.debug("Added ticket [{}] to registry.", ticket);
111     }
112 
113     @Override
114     public boolean deleteTicket(final String ticketId) {
115         final Ticket ticket = getRawTicket(ticketId);
116 
117         if (ticket == null) {
118             return false;
119         }
120 
121         if (ticket instanceof ServiceTicket) {
122             removeTicket(ticket);
123             logger.debug("Deleted ticket [{}] from the registry.", ticket);
124             return true;
125         }
126 
127         deleteTicketAndChildren(ticket);
128         logger.debug("Deleted ticket [{}] and its children from the registry.", ticket);
129         return true;
130     }
131 
132     /**
133      * Delete the TGt and all of its service tickets.
134      *
135      * @param ticket the ticket
136      */
137     private void deleteTicketAndChildren(final Ticket ticket) {
138         final List<TicketGrantingTicketImpl> ticketGrantingTicketImpls = entityManager
139             .createQuery("select t from TicketGrantingTicketImpl t where t.ticketGrantingTicket.id = :id",
140                     TicketGrantingTicketImpl.class)
141             .setLockMode(LockModeType.PESSIMISTIC_WRITE)
142             .setParameter("id", ticket.getId())
143             .getResultList();
144         final List<ServiceTicketImpl> serviceTicketImpls = entityManager
145                 .createQuery("select s from ServiceTicketImpl s where s.ticketGrantingTicket.id = :id",
146                         ServiceTicketImpl.class)
147                 .setParameter("id", ticket.getId())
148                 .getResultList();
149 
150         for (final ServiceTicketImpl s : serviceTicketImpls) {
151             removeTicket(s);
152         }
153 
154         for (final TicketGrantingTicketImpl t : ticketGrantingTicketImpls) {
155             deleteTicketAndChildren(t);
156         }
157 
158         removeTicket(ticket);
159     }
160 
161     /**
162      * Removes the ticket.
163      *
164      * @param ticket the ticket
165      */
166     private void removeTicket(final Ticket ticket) {
167         try {
168             if (logger.isDebugEnabled()) {
169                 final Date creationDate = new Date(ticket.getCreationTime());
170                 logger.debug("Removing Ticket [{}] created: {}", ticket, creationDate.toString());
171              }
172             entityManager.remove(ticket);
173         } catch (final Exception e) {
174             logger.error("Error removing {} from registry.", ticket, e);
175         }
176     }
177 
178     @Override
179     public Ticket getTicket(final String ticketId) {
180         return getProxiedTicketInstance(getRawTicket(ticketId));
181     }
182 
183     /**
184      * Gets the ticket from the database, as is.
185      *
186      * @param ticketId the ticket id
187      * @return the raw ticket
188      */
189     private Ticket getRawTicket(final String ticketId) {
190         try {
191             if (ticketId.startsWith(this.ticketGrantingTicketPrefix)) {
192                 return entityManager.find(TicketGrantingTicketImpl.class, ticketId);
193             }
194 
195             return entityManager.find(ServiceTicketImpl.class, ticketId);
196         } catch (final Exception e) {
197             logger.error("Error getting ticket {} from registry.", ticketId, e);
198         }
199         return null;
200     }
201 
202     @Override
203     public Collection<Ticket> getTickets() {
204         final List<TicketGrantingTicketImpl> tgts = entityManager
205             .createQuery("select t from TicketGrantingTicketImpl t", TicketGrantingTicketImpl.class)
206             .getResultList();
207         final List<ServiceTicketImpl> sts = entityManager
208             .createQuery("select s from ServiceTicketImpl s", ServiceTicketImpl.class)
209             .getResultList();
210 
211         final List<Ticket> tickets = new ArrayList<>();
212         tickets.addAll(tgts);
213         tickets.addAll(sts);
214 
215         return tickets;
216     }

238     /**
239      * Count the result into a numeric value.
240      *
241      * @param result the result
242      * @return the int
243      */
244     private int countToInt(final Object result) {
245         final int intval;
246         if (result instanceof Long) {
247             intval = ((Long) result).intValue();
248         } else if (result instanceof Integer) {
249             intval = (Integer) result;
250         } else {
251             // Must be a Number of some kind
252             intval = ((Number) result).intValue();
253         }
254         return intval;
255     }

258     /**
259      * Schedule reloader job.
260      */
261     @PostConstruct
262     public void scheduleCleanerJob() {
263         try {
264 
265             logger.info("Preparing to schedule cleaner job");
266 
267             final JobDetail job = JobBuilder.newJob(this.getClass())
268                 .withIdentity(this.getClass().getSimpleName().concat(UUID.randomUUID().toString()))
269                 .build();
270 
271             final Trigger trigger = TriggerBuilder.newTrigger()
272                 .withIdentity(this.getClass().getSimpleName().concat(UUID.randomUUID().toString()))
273                 .startAt(new Date(System.currentTimeMillis() + this.startDelay))
274                 .withSchedule(SimpleScheduleBuilder.simpleSchedule()
275                     .withIntervalInMinutes(this.refreshInterval)
276                     .repeatForever()).build();
277 
278             final JobFactory jobFactory = new CasSpringBeanJobFactory(this.applicationContext);
279             final SchedulerFactory schFactory = new StdSchedulerFactory();
280             final Scheduler sch = schFactory.getScheduler();
281             sch.setJobFactory(jobFactory);
282             sch.start();
283             logger.debug("Started {} scheduler", this.getClass().getName());
284             sch.scheduleJob(job, trigger);
285             logger.info("{} will clean tickets every {} seconds",
286                 this.getClass().getSimpleName(),
287                 TimeUnit.MILLISECONDS.toSeconds(this.refreshInterval));
288 
289         } catch (final Exception e){
290             logger.warn(e.getMessage(), e);
291         }
292 
293     }
294 
295     @Override
296     public void execute(final JobExecutionContext jobExecutionContext) throws JobExecutionException {
297         SpringBeanAutowiringSupport.processInjectionBasedOnCurrentContext(this);
298 
299         try {
300 
301             logger.info("Beginning ticket cleanup.");
302             logger.debug("Attempting to acquire ticket cleanup lock.");
303             if (!this.jpaLockingStrategy.acquire()) {
304                 logger.info("Could not obtain lock.  Aborting cleanup.");
305                 return;
306             }
307             logger.debug("Acquired lock.  Proceeding with cleanup.");
308 
309             logger.info("Beginning ticket cleanup...");
310             final Collection<Ticket> ticketsToRemove = Collections2.filter(this.getTickets(), new Predicate<Ticket>() {
311                 @Override
312                 public boolean apply(@Nullable final Ticket ticket) {
313                     if (ticket.isExpired()) {
314                         if (ticket instanceof TicketGrantingTicket) {
315                             logger.debug("Cleaning up expired ticket-granting ticket [{}]", ticket.getId());
316                             logoutManager.performLogout((TicketGrantingTicket) ticket);
317                             deleteTicket(ticket.getId());
318                         } else if (ticket instanceof ServiceTicket) {
319                             logger.debug("Cleaning up expired service ticket [{}]", ticket.getId());
320                             deleteTicket(ticket.getId());
321                         } else {
322                             logger.warn("Unknown ticket type [{} found to clean", ticket.getClass().getSimpleName());
323                         }
324                         return true;
325                     }
326                     return false;
327                 }
328             });
329             logger.info("{} expired tickets found and removed.", ticketsToRemove.size());
330         } catch (final Exception e) {
331             logger.error(e.getMessage(), e);
332         } finally {
333             logger.debug("Releasing ticket cleanup lock.");
334             this.jpaLockingStrategy.release();
335             logger.info("Finished ticket cleanup.");
336         }
337 
338     }
