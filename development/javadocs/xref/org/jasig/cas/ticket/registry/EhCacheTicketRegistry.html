36  /**
37   * <p>
38   * <a href="http://ehcache.org/">Ehcache</a> based distributed ticket registry.
39   * </p>
40   * <p>
41   * Use distinct caches for ticket granting tickets (TGT) and service tickets (ST) for:
42   * <ul>
43   * <li>Tuning : use cache level time to live with different values for TGT an ST.</li>
44   * <li>Monitoring : follow separately the number of TGT and ST.</li>
45   * </ul>
46   * </p>
47   *
48   * @author <a href="mailto:cleclerc@xebia.fr">Cyrille Le Clerc</a>
49   * @author Adam Rybicki
50   * @author Andrew Tillinghast
51   */
52  public final class EhCacheTicketRegistry extends AbstractDistributedTicketRegistry implements InitializingBean {
53  
54      private Cache serviceTicketsCache = null;
55  
56      private Cache ticketGrantingTicketsCache = null;
57  
58      /** @see #setSupportRegistryState(boolean)*/
59      private boolean supportRegistryState = true;
60  
61      /**
62       * Instantiates a new EhCache ticket registry.
63       */
64      public EhCacheTicketRegistry() {
65          
66      }

68      /**
69       * Instantiates a new EhCache ticket registry.
70       *
71       * @param serviceTicketsCache the service tickets cache
72       * @param ticketGrantingTicketsCache the ticket granting tickets cache
73       */
74      public EhCacheTicketRegistry(final Cache serviceTicketsCache, final Cache ticketGrantingTicketsCache) {
75          super();
76          setServiceTicketsCache(serviceTicketsCache);
77          setTicketGrantingTicketsCache(ticketGrantingTicketsCache);
78      }
79  
80      /**
81       * Instantiates a new EhCache ticket registry.
82       *
83       * @param serviceTicketsCache the service tickets cache
84       * @param ticketGrantingTicketsCache the ticket granting tickets cache
85       * @param supportRegistryState the support registry state
86       */
87      public EhCacheTicketRegistry(final Cache serviceTicketsCache, final Cache ticketGrantingTicketsCache,
88              final boolean supportRegistryState) {
89          this(serviceTicketsCache, ticketGrantingTicketsCache);
90          setSupportRegistryState(supportRegistryState);
91      }
92  
93      @Override
94      public void addTicket(final Ticket ticket) {
95          final Element element = new Element(ticket.getId(), ticket);
96          if (ticket instanceof ServiceTicket) {
97              logger.debug("Adding service ticket {} to the cache", ticket.getId(), this.serviceTicketsCache.getName());
98              this.serviceTicketsCache.put(element);
99          } else if (ticket instanceof TicketGrantingTicket) {
100             logger.debug("Adding ticket granting ticket {} to the cache {}", ticket.getId(),
101                     this.ticketGrantingTicketsCache.getName());
102             this.ticketGrantingTicketsCache.put(element);
103         } else {
104             throw new IllegalArgumentException("Invalid ticket type " + ticket);
105         }
106     }
107 
108     @Override
109     public boolean deleteTicket(final String ticketId) {
110         if (StringUtils.isBlank(ticketId)) {
111             return false;
112         }
113         return this.serviceTicketsCache.remove(ticketId) || this.ticketGrantingTicketsCache.remove(ticketId);
114     }
115 
116     @Override
117     public Ticket getTicket(final String ticketId) {
118         if (ticketId == null) {
119             return null;
120         }
121 
122         Element element = this.serviceTicketsCache.get(ticketId);
123         if (element == null) {
124             element = this.ticketGrantingTicketsCache.get(ticketId);
125         }
126         return element == null ? null : getProxiedTicketInstance((Ticket) element.getObjectValue());
127     }
128 
129     @Override
130     public Collection<Ticket> getTickets() {
131         final Collection<Element> serviceTickets = this.serviceTicketsCache.getAll(
132                 this.serviceTicketsCache.getKeysWithExpiryCheck()).values();
133         final Collection<Element> tgtTicketsTickets = this.ticketGrantingTicketsCache.getAll(
134                 this.ticketGrantingTicketsCache.getKeysWithExpiryCheck()).values();
135 
136         final Collection<Ticket> allTickets = new HashSet<Ticket>(serviceTickets.size() + tgtTicketsTickets.size());
137 
138         for (final Element ticket : serviceTickets) {
139             allTickets.add((Ticket) ticket.getObjectValue());
140         }
141 
142         for (final Element ticket : tgtTicketsTickets) {
143             allTickets.add((Ticket) ticket.getObjectValue());
144         }
145 
146         return allTickets;
147     }

157     @Override
158     public String toString() {

160                 .append("serviceTicketsCache", this.serviceTicketsCache).toString();
161     }

163     @Override
164     protected void updateTicket(final Ticket ticket) {

173     /**
174      * Flag to indicate whether this registry instance should participate in reporting its state with
175      * default value set to <code>true</code>.
176      * Based on the <a href="http://ehcache.org/apidocs/net/sf/ehcache/Ehcache.html#getKeysWithExpiryCheck()">EhCache documentation</a>,
177      * determining the number of service tickets and the total session count from the cache can be considered
178      * an expensive operation with the time taken as O(n), where n is the number of elements in the cache.
179      *
180      * <p>Therefore, the flag provides a level of flexibility such that depending on the cache and environment
181      * settings, reporting statistics
182      * can be set to false and disabled.</p>
183      *
184      * @param supportRegistryState true, if the registry is to support registry state
185      * @see #sessionCount()
186      * @see #serviceTicketCount()
187      * @see org.jasig.cas.monitor.SessionMonitor
188      */
189     public void setSupportRegistryState(final boolean supportRegistryState) {
190         this.supportRegistryState = supportRegistryState;
191     }

193     @Override
194     public void afterPropertiesSet() throws Exception {
195         if (this.serviceTicketsCache == null || this.ticketGrantingTicketsCache == null) {
196             throw new BeanInstantiationException(this.getClass(),
197                     "Both serviceTicketsCache and ticketGrantingTicketsCache are required properties.");
198         }
199 
200         if (logger.isDebugEnabled()) {
201             CacheConfiguration config = this.serviceTicketsCache.getCacheConfiguration();
202             logger.debug("serviceTicketsCache.maxElementsInMemory={}", config.getMaxEntriesLocalHeap());
203             logger.debug("serviceTicketsCache.maxElementsOnDisk={}", config.getMaxElementsOnDisk());
204             logger.debug("serviceTicketsCache.isOverflowToDisk={}", config.isOverflowToDisk());
205             logger.debug("serviceTicketsCache.timeToLive={}", config.getTimeToLiveSeconds());
206             logger.debug("serviceTicketsCache.timeToIdle={}", config.getTimeToIdleSeconds());
207             logger.debug("serviceTicketsCache.cacheManager={}", this.serviceTicketsCache.getCacheManager().getName());
208 
209             config = this.ticketGrantingTicketsCache.getCacheConfiguration();
210             logger.debug("ticketGrantingTicketsCache.maxElementsInMemory={}", config.getMaxEntriesLocalHeap());
211             logger.debug("ticketGrantingTicketsCache.maxElementsOnDisk={}", config.getMaxElementsOnDisk());
212             logger.debug("ticketGrantingTicketsCache.isOverflowToDisk={}", config.isOverflowToDisk());
213             logger.debug("ticketGrantingTicketsCache.timeToLive={}", config.getTimeToLiveSeconds());
214             logger.debug("ticketGrantingTicketsCache.timeToIdle={}", config.getTimeToIdleSeconds());
215             logger.debug("ticketGrantingTicketsCache.cacheManager={}", this.ticketGrantingTicketsCache.getCacheManager()
216                     .getName());
217         }
218     }
