19  package org.jasig.cas.ticket.registry;
20  
21  import org.jasig.cas.ticket.ServiceTicket;
22  import org.jasig.cas.ticket.Ticket;
23  import org.jasig.cas.ticket.TicketGrantingTicket;
24  import org.springframework.util.Assert;
25  
26  import java.util.Collection;
27  import java.util.Collections;
28  import java.util.Map;
29  import java.util.concurrent.ConcurrentHashMap;
30  
31  /**
32   * Implementation of the TicketRegistry that is backed by a ConcurrentHashMap.
33   *
34   * @author Scott Battaglia
35   * @since 3.0
36   */
37  public final class DefaultTicketRegistry extends AbstractTicketRegistry  {
38  
39      /** A HashMap to contain the tickets. */
40      private final Map<String, Ticket> cache;
41  
42      /**
43       * Instantiates a new default ticket registry.
44       */
45      public DefaultTicketRegistry() {
46          this.cache = new ConcurrentHashMap<String, Ticket>();
47      }
48  
49      /**
50       * Creates a new, empty registry with the specified initial capacity, load
51       * factor, and concurrency level.
52       *
53       * @param initialCapacity - the initial capacity. The implementation
54       * performs internal sizing to accommodate this many elements.
55       * @param loadFactor - the load factor threshold, used to control resizing.
56       * Resizing may be performed when the average number of elements per bin
57       * exceeds this threshold.
58       * @param concurrencyLevel - the estimated number of concurrently updating
59       * threads. The implementation performs internal sizing to try to
60       * accommodate this many threads.
61       */
62      public DefaultTicketRegistry(final int initialCapacity, final float loadFactor, final int concurrencyLevel) {
63          this.cache = new ConcurrentHashMap<String, Ticket>(initialCapacity, loadFactor, concurrencyLevel);
64      }
65  
66      /**
67       * {@inheritDoc}
68       * @throws IllegalArgumentException if the Ticket is null.
69       */
70      @Override
71      public void addTicket(final Ticket ticket) {
72          Assert.notNull(ticket, "ticket cannot be null");
73  
74          logger.debug("Added ticket [{}] to registry.", ticket.getId());
75          this.cache.put(ticket.getId(), ticket);
76      }
77  
78      @Override
79      public Ticket getTicket(final String ticketId) {
80          if (ticketId == null) {
81              return null;
82          }
83  
84          logger.debug("Attempting to retrieve ticket [{}]", ticketId);
85          final Ticket ticket = this.cache.get(ticketId);
86  
87          if (ticket != null) {
88              logger.debug("Ticket [{}] found in registry.", ticketId);
89          }
90  
91          return ticket;
92      }
93  
94      @Override
95      public boolean deleteTicket(final String ticketId) {
96          if (ticketId == null) {
97              return false;
98          }
99          logger.debug("Removing ticket [{}] from registry", ticketId);
100         return (this.cache.remove(ticketId) != null);
101     }
102 
103     public Collection<Ticket> getTickets() {
104         return Collections.unmodifiableCollection(this.cache.values());
105     }
106 
107     @Override
108     public int sessionCount() {
109         int count = 0;
110         for (Ticket t : this.cache.values()) {
111             if (t instanceof TicketGrantingTicket) {
112                 count++;
113             }
114         }
115         return count;
116     }
117 
118     @Override
119     public int serviceTicketCount() {
120         int count = 0;
121         for (Ticket t : this.cache.values()) {
122             if (t instanceof ServiceTicket) {
123                 count++;
124             }
125         }
126         return count;
127     }
128 }
