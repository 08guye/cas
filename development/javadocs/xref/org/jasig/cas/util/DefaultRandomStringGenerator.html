19  package org.jasig.cas.util;
20  
21  import java.security.SecureRandom;
22  
23  /**
24   * Implementation of the RandomStringGenerator that allows you to define the
25   * length of the random part.
26   *
27   * @author Scott Battaglia
28  
29   * @since 3.0
30   */
31  public final class DefaultRandomStringGenerator implements
32      RandomStringGenerator {
33  
34      /** The array of printable characters to be used in our random string. */
35      private static final char[] PRINTABLE_CHARACTERS = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ012345679"
36          .toCharArray();
37  
38      /** The default maximum length. */
39      protected static final int DEFAULT_MAX_RANDOM_LENGTH = 35;
40  
41      /** An instance of secure random to ensure randomness is secure. */
42      private SecureRandom randomizer = new SecureRandom();
43  
44      /** The maximum length the random string can be. */
45      private final int maximumRandomLength;
46  
47      /**
48       * Instantiates a new default random string generator
49       * with length set to {@link #DEFAULT_MAX_RANDOM_LENGTH}.
50       */
51      public DefaultRandomStringGenerator() {
52          this.maximumRandomLength = DEFAULT_MAX_RANDOM_LENGTH;
53      }
54  
55      /**
56       * Instantiates a new default random string generator.
57       *
58       * @param maxRandomLength the max random length
59       */
60      public DefaultRandomStringGenerator(final int maxRandomLength) {
61          this.maximumRandomLength = maxRandomLength;
62      }
63  
64      public int getMinLength() {
65          return this.maximumRandomLength;
66      }
67  
68      public int getMaxLength() {
69          return this.maximumRandomLength;
70      }
71  
72      @Override
73      public String getNewString() {
74          final byte[] random = getNewStringAsBytes();
75  
76          return convertBytesToString(random);
77      }
78  
79      @Override
80      public byte[] getNewStringAsBytes() {
81          final byte[] random = new byte[this.maximumRandomLength];
82  
83          this.randomizer.nextBytes(random);
84  
85          return random;
86      }
87  
88      /**
89       * Convert bytes to string, taking into account {@link #PRINTABLE_CHARACTERS}.
90       *
91       * @param random the random
92       * @return the string
93       */
94      private String convertBytesToString(final byte[] random) {
95          final char[] output = new char[random.length];
96          for (int i = 0; i < random.length; i++) {
97              final int index = Math.abs(random[i] % PRINTABLE_CHARACTERS.length);
98              output[i] = PRINTABLE_CHARACTERS[index];
99          }
100 
101         return new String(output);
102     }
103 }
