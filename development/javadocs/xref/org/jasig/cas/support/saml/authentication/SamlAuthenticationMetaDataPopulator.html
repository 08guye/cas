19  package org.jasig.cas.support.saml.authentication;
20  
21  import org.jasig.cas.authentication.AuthenticationBuilder;
22  import org.jasig.cas.authentication.AuthenticationMetaDataPopulator;
23  import org.jasig.cas.authentication.Credential;
24  import org.jasig.cas.authentication.HttpBasedServiceCredential;
25  import org.jasig.cas.authentication.UsernamePasswordCredential;
26  
27  import java.util.HashMap;
28  import java.util.Map;
29  
30  /**
31   * AuthenticationMetaDataPopulator to retrieve the Authentication Type.
32   * <p>
33   * Note: Authentication Methods are exposed under the key:
34   * <code>samlAuthenticationStatement::authMethod</code> in the Authentication
35   * attributes map.
36   *
37   * @author Scott Battaglia
38   * @author Marvin S. Addison
39   * @since 3.1
40   */
41  public class SamlAuthenticationMetaDataPopulator implements AuthenticationMetaDataPopulator {
42  
43      /** The Constant ATTRIBUTE_AUTHENTICATION_METHOD. */
44      public static final String ATTRIBUTE_AUTHENTICATION_METHOD = "samlAuthenticationStatementAuthMethod";
45  
46      /** The Constant AUTHN_METHOD_PASSWORD. */
47      public static final String AUTHN_METHOD_PASSWORD = "urn:oasis:names:tc:SAML:1.0:am:password";
48  
49      /** The Constant AUTHN_METHOD_SSL_TLS_CLIENT. */
50      public static final String AUTHN_METHOD_SSL_TLS_CLIENT = "urn:ietf:rfc:2246";
51  
52      /** The Constant AUTHN_METHOD_X509_PUBLICKEY. */
53      public static final String AUTHN_METHOD_X509_PUBLICKEY = "urn:oasis:names:tc:SAML:1.0:am:X509-PKI";
54  
55      /** The Constant AUTHN_METHOD_UNSPECIFIED. */
56      public static final String AUTHN_METHOD_UNSPECIFIED = "urn:oasis:names:tc:SAML:1.0:am:unspecified";
57  
58      private final Map<String, String> authenticationMethods = new HashMap<String, String>();
59  
60      /**
61       * Instantiates a new SAML authentication meta data populator.
62       */
63      public SamlAuthenticationMetaDataPopulator() {
64          this.authenticationMethods.put(
65                  HttpBasedServiceCredential.class.getName(),
66                  AUTHN_METHOD_SSL_TLS_CLIENT);
67          this.authenticationMethods.put(
68                  UsernamePasswordCredential.class.getName(),
69                  AUTHN_METHOD_PASSWORD);
70  
71          // Next two classes are in other modules, so avoid using Class#getName() to prevent circular dependency
72          this.authenticationMethods.put(
73                  "org.jasig.cas.adaptors.trusted.authentication.principal.PrincipalBearingCredentials",
74                  AUTHN_METHOD_UNSPECIFIED);
75          this.authenticationMethods.put(
76                  "org.jasig.cas.adaptors.x509.authentication.principal.X509CertificateCredentials",
77                  AUTHN_METHOD_X509_PUBLICKEY);
78      }
79  
80      @Override
81      public final void populateAttributes(final AuthenticationBuilder builder, final Credential credential) {
82  
83          final String credentialsClass = credential.getClass().getName();
84          final String authenticationMethod = this.authenticationMethods.get(credentialsClass);
85  
86          builder.addAttribute(ATTRIBUTE_AUTHENTICATION_METHOD, authenticationMethod);
87      }
88  
89      @Override
90      public boolean supports(final Credential credential) {
91          return true;
92      }
93  
94      /**
95       * Map of user-defined mappings. Note it is possible to override the
96       * defaults. Mapping should be of the following type:
97       * <pre>Package/Class Name as String -> Name SAML Type</pre>
98       * <p>
99       * Example: (<code>"org.jasig.cas.authentication.HttpBasedServiceCredential"
100      * -> SAMLAuthenticationStatement.AuthenticationMethod_SSL_TLS_Client</code>)
101      *
102      * @param userDefinedMappings map of user defined authentication types.
103      */
104     public void setUserDefinedMappings(final Map<String, String> userDefinedMappings) {
105         this.authenticationMethods.putAll(userDefinedMappings);
106     }
107 }
