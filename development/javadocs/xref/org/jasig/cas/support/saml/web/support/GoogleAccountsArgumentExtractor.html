19  package org.jasig.cas.support.saml.web.support;
20  
21  import org.jasig.cas.authentication.principal.WebApplicationService;
22  import org.jasig.cas.services.ServicesManager;
23  import org.jasig.cas.support.saml.authentication.principal.GoogleAccountsService;
24  import org.jasig.cas.web.support.AbstractArgumentExtractor;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  
28  import javax.servlet.http.HttpServletRequest;
29  import javax.validation.constraints.NotNull;
30  import java.security.PrivateKey;
31  import java.security.PublicKey;
32  
33  /**
34   * Constructs a GoogleAccounts compatible service and provides the public and
35   * private keys.
36   *
37   * @author Scott Battaglia
38   * @since 3.1
39   */
40  public final class GoogleAccountsArgumentExtractor extends AbstractArgumentExtractor {
41  
42      private static final Logger LOGGER = LoggerFactory.getLogger(GoogleAccountsArgumentExtractor.class);
43  
44      @NotNull
45      private final PublicKey publicKey;
46  
47      @NotNull
48      private final PrivateKey privateKey;
49  
50      @NotNull
51      private final ServicesManager servicesManager;
52      
53      @Override
54      public WebApplicationService extractServiceInternal(final HttpServletRequest request) {
55          return GoogleAccountsService.createServiceFrom(request,
56                  this.privateKey, this.publicKey, this.servicesManager);
57      }
58  
59      /**
60       * Instantiates a new google accounts argument extractor.
61       *
62       * @param publicKey the public key
63       * @param privateKey the private key
64       * @param servicesManager the services manager
65       */
66      public GoogleAccountsArgumentExtractor(final PublicKey publicKey,
67              final PrivateKey privateKey, final ServicesManager servicesManager) {
68          this.publicKey = publicKey;
69          this.privateKey = privateKey;
70          this.servicesManager = servicesManager;
71      }
72  
73      /**
74       * @deprecated As of 4.1. Use Ctors instead.
75       * @param privateKey the private key object
76       */
77      @Deprecated
78      public void setPrivateKey(final PrivateKey privateKey) {
79          LOGGER.warn("setPrivateKey() is deprecated and has no effect. Consider using constructors instead.");
80      }
81  
82      /**
83       * @deprecated As of 4.1. Use Ctors instead.
84       * @param publicKey the public key object
85       */
86      @Deprecated
87      public void setPublicKey(final PublicKey publicKey) {
88          LOGGER.warn("setPublicKey() is deprecated and has no effect. Consider using constructors instead.");
89      }
90  
91      /**
92       * @deprecated As of 4.1. The behavior is controlled by the service registry instead.
93       * Sets an alternate username to send to Google (i.e. fully qualified email address).  Relies on an appropriate
94       * attribute available for the user.
95       * <p>
96       * Note that this is optional and the default is to use the normal identifier.
97       *
98       * @param alternateUsername the alternate username. This is OPTIONAL.
99       */
100     @Deprecated
101     public void setAlternateUsername(final String alternateUsername) {
102         LOGGER.warn("setAlternateUsername() is deprecated and has no effect. Instead use the configuration in service registry.");
103     }
104 }
