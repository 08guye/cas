118     /**
119      * Creates the service from request.
120      *
121      * @param request the request
122      * @param privateKey the private key
123      * @param publicKey the public key
124      * @param servicesManager the services manager
125      * @return the google accounts service
126      */
127     public static GoogleAccountsService createServiceFrom(
128             final HttpServletRequest request, final PrivateKey privateKey,
129             final PublicKey publicKey, final ServicesManager servicesManager) {
130         final String relayState = request.getParameter(SamlProtocolConstants.PARAMETER_SAML_RELAY_STATE);
131 
132         final String xmlRequest = BUILDER.decodeSamlAuthnRequest(
133                 request.getParameter(SamlProtocolConstants.PARAMETER_SAML_REQUEST));
134 
135         if (!StringUtils.hasText(xmlRequest)) {
136             return null;
137         }
138 
139         final Document document = AbstractSaml20ObjectBuilder.constructDocumentFromXml(xmlRequest);
140 
141         if (document == null) {
142             return null;
143         }
144 
145         final Element root = document.getRootElement();
146         final String assertionConsumerServiceUrl = root.getAttributeValue("AssertionConsumerServiceURL");
147         final String requestId = root.getAttributeValue("ID");
148 
149         return new GoogleAccountsService(assertionConsumerServiceUrl,
150                 relayState, requestId, privateKey, publicKey, servicesManager);
151     }

153     @Override
154     public Response getResponse(final String ticketId) {
155         final Map<String, String> parameters = new HashMap<>();
156         final String samlResponse = constructSamlResponse();
157         final String signedResponse = BUILDER.signSamlResponse(samlResponse,
158                 this.privateKey, this.publicKey);
159         parameters.put(SamlProtocolConstants.PARAMETER_SAML_RESPONSE, signedResponse);
160         parameters.put(SamlProtocolConstants.PARAMETER_SAML_RELAY_STATE, this.relayState);
161 
162         return Response.getPostResponse(getOriginalUrl(), parameters);
163     }

175     /**
176      * Construct SAML response.
177      * <a href="http://bit.ly/1uI8Ggu">See this reference for more info.</a>
178      * @return the SAML response
179      */
180     private String constructSamlResponse() {
181         final DateTime currentDateTime = DateTime.parse(new ISOStandardDateFormat().getCurrentDateAndTime());
182         final DateTime notBeforeIssueInstant = DateTime.parse("2003-04-17T00:46:02Z");
183 
184         final RegisteredService svc = this.servicesManager.findServiceBy(this);
185         final String userId = svc.getUsernameAttributeProvider().resolveUsername(getPrincipal(), this);
186 
187         final org.opensaml.saml.saml2.core.Response response = BUILDER.newResponse(
188                 BUILDER.generateSecureRandomId(),
189                 currentDateTime,
190                 getId(), this);
191         response.setStatus(BUILDER.newStatus(StatusCode.SUCCESS, null));
192 
193         final AuthnStatement authnStatement = BUILDER.newAuthnStatement(
194                 AuthnContext.PASSWORD_AUTHN_CTX, currentDateTime);
195         final Assertion assertion = BUILDER.newAssertion(authnStatement,
196                 "https://www.opensaml.org/IDP",
197                 notBeforeIssueInstant, BUILDER.generateSecureRandomId());
198 
199         final Conditions conditions = BUILDER.newConditions(notBeforeIssueInstant,
200                 currentDateTime, getId());
201         assertion.setConditions(conditions);
202 
203         final Subject subject = BUILDER.newSubject(NameID.EMAIL, userId,
204                 getId(), currentDateTime, this.requestId);
205         assertion.setSubject(subject);
206 
207         response.getAssertions().add(assertion);
208 
209         final StringWriter writer = new StringWriter();
210         BUILDER.marshalSamlXmlObject(response, writer);
211 
212         final String result = writer.toString();
213         logger.debug("Generated Google SAML response: {}", result);
214         return result;
215     }
