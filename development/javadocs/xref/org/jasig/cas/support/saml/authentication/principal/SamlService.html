29  import javax.servlet.http.HttpServletRequest;
30  import java.io.BufferedReader;

34  /**
35   * Class to represent that this service wants to use SAML. We use this in
36   * combination with the CentralAuthenticationServiceImpl to choose the right
37   * UniqueTicketIdGenerator.
38   *
39   * @author Scott Battaglia
40   * @since 3.1
41   */
42  public final class SamlService extends AbstractWebApplicationService {
43  
44      private static final Logger LOGGER = LoggerFactory.getLogger(SamlService.class);
45  
46      /**
47       * Unique Id for serialization.
48       */
49      private static final long serialVersionUID = -6867572626767140223L;
50  
51      private static final String CONST_START_ARTIFACT_XML_TAG_NO_NAMESPACE = "<AssertionArtifact>";
52  
53      private static final String CONST_END_ARTIFACT_XML_TAG_NO_NAMESPACE = "</AssertionArtifact>";
54  
55      private static final String CONST_START_ARTIFACT_XML_TAG = "<samlp:AssertionArtifact>";
56  
57      private static final String CONST_END_ARTIFACT_XML_TAG = "</samlp:AssertionArtifact>";
58  
59      private static final int CONST_REQUEST_ID_LENGTH = 11;
60  
61      private String requestId;
62  
63      /**
64       * Instantiates a new SAML service.
65       *
66       * @param id the service id
67       */
68      protected SamlService(final String id) {
69          super(id, id, null);
70      }
71  
72      /**
73       * Instantiates a new SAML service.
74       *
75       * @param id the service id
76       * @param originalUrl the original url
77       * @param artifactId the artifact id
78       * @param requestId the request id
79       */
80      protected SamlService(final String id, final String originalUrl,
81              final String artifactId, final String requestId) {
82          super(id, originalUrl, artifactId);
83          this.requestId = requestId;
84      }
85  
86      public String getRequestID() {
87          return this.requestId;
88      }
89  
90      /**
91       * Creates the SAML service from the request.
92       *
93       * @param request the request
94       * @return the SAML service
95       */
96      public static SamlService createServiceFrom(
97              final HttpServletRequest request) {
98          final String service = request.getParameter(SamlProtocolConstants.CONST_PARAM_TARGET);
99          final String artifactId;
100         final String requestBody = getRequestBody(request);
101         final String requestId;
102 
103         if (!StringUtils.hasText(service) && !StringUtils.hasText(requestBody)) {
104             LOGGER.debug("Request does not specify a {} or request body is empty",
105                     SamlProtocolConstants.CONST_PARAM_TARGET);
106             return null;
107         }
108 
109         final String id = cleanupUrl(service);
110 
111         if (StringUtils.hasText(requestBody)) {
112 
113             final String tagStart;
114             final String tagEnd;
115             if (requestBody.contains(CONST_START_ARTIFACT_XML_TAG)) {
116                 tagStart = CONST_START_ARTIFACT_XML_TAG;
117                 tagEnd = CONST_END_ARTIFACT_XML_TAG;
118             } else {
119                 tagStart = CONST_START_ARTIFACT_XML_TAG_NO_NAMESPACE;
120                 tagEnd = CONST_END_ARTIFACT_XML_TAG_NO_NAMESPACE;
121             }
122             final int startTagLocation = requestBody.indexOf(tagStart);
123             final int artifactStartLocation = startTagLocation + tagStart.length();
124             final int endTagLocation = requestBody.indexOf(tagEnd);
125 
126             artifactId = requestBody.substring(artifactStartLocation, endTagLocation).trim();
127 
128             // is there a request id?
129             requestId = extractRequestId(requestBody);
130         } else {
131             artifactId = null;
132             requestId = null;
133         }
134 
135         LOGGER.debug("Attempted to extract Request from HttpServletRequest. Results:");
136         LOGGER.debug("Request Body: {}", requestBody);
137         LOGGER.debug("Extracted ArtifactId: {}", artifactId);
138         LOGGER.debug("Extracted Request Id: {}", requestId);
139 
140         return new SamlService(id, service, artifactId, requestId);
141     }
142 
143     @Override
144     public Response getResponse(final String ticketId) {
145         final Map<String, String> parameters = new HashMap<>();
146         parameters.put(SamlProtocolConstants.CONST_PARAM_ARTIFACT, ticketId);
147         return DefaultResponse.getRedirectResponse(getOriginalUrl(), parameters);
148     }
149 
150     /**
151      * Extract request id from the body.
152      *
153      * @param requestBody the request body
154      * @return the string
155      */
156     protected static String extractRequestId(final String requestBody) {
157         if (!requestBody.contains("RequestID")) {
158             LOGGER.debug("Request body does not contain a request id");
159             return null;
160         }
161 
162         try {
163             final int position = requestBody.indexOf("RequestID=\"") + CONST_REQUEST_ID_LENGTH;
164             final int nextPosition = requestBody.indexOf('"', position);
165 
166             return requestBody.substring(position,  nextPosition);
167         } catch (final Exception e) {
168             LOGGER.debug("Exception parsing RequestID from request.", e);
169             return null;
170         }
171     }
172 
173     /**
174      * Gets the request body from the request.
175      *
176      * @param request the request
177      * @return the request body
178      */
179     protected static String getRequestBody(final HttpServletRequest request) {
180         final StringBuilder builder = new StringBuilder();
181         try (final BufferedReader reader = request.getReader()) {
182 
183             if (reader == null) {
184                 LOGGER.debug("Request body could not be read because it's empty.");
185                 return null;
186             }
187             String line;
188             while ((line = reader.readLine()) != null) {
189                 builder.append(line);
190             }
191             return builder.toString();
192         } catch (final Exception e) {
193             LOGGER.trace("Could not obtain the saml request body from the http request", e);
194             return null;
195         }
196     }
197 }
