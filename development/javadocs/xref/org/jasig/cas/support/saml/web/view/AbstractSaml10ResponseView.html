50  /**
51   * Base class for all views that render SAML1 SOAP messages directly to the HTTP response stream.
52   *
53   * @author Marvin S. Addison
54   * @since 3.5.1
55   */
56  public abstract class AbstractSaml10ResponseView extends AbstractCasView {
57  
58      private static final String DEFAULT_ELEMENT_NAME_FIELD = "DEFAULT_ELEMENT_NAME";
59  
60      private static final String DEFAULT_ENCODING = "UTF-8";
61  
62      private final SamlArgumentExtractor samlArgumentExtractor = new SamlArgumentExtractor();
63  
64      private final HTTPSOAP11Encoder encoder = new CasHTTPSOAP11Encoder();
65  
66      private final SecureRandomIdentifierGenerator idGenerator;
67  
68      @NotNull
69      private String encoding = DEFAULT_ENCODING;
70  
71      private int skewAllowance = 0;
72  
73      /**
74       * Sets the character encoding in the HTTP response.
75       *
76       * @param encoding Response character encoding.
77       */
78      public void setEncoding(final String encoding) {
79          this.encoding = encoding;
80      }
81  
82  
83  
84      /**
85      * Sets the allowance for time skew in seconds
86      * between CAS and the client server.  Default 0s.
87      * This value will be subtracted from the current time when setting the SAML
88      * <code>NotBeforeDate</code> attribute, thereby allowing for the
89      * CAS server to be ahead of the client by as much as the value defined here.
90      *
91      * <p><strong>Note:</strong> Skewing of the issue instant via setting this property
92      * applies to all saml assertions that are issued by CAS and it
93      * currently cannot be controlled on a per relying party basis.
94      * Before configuring this, it is recommended that each service provider
95      * attempt to correctly sync their system time with an NTP server
96      * so as to match the CAS server's issue instant config and to
97      * avoid applying this setting globally. This should only
98      * be used in situations where the NTP server is unresponsive to
99      * sync time on the client, or the client is simply unable
100     * to adjust their server time configuration.</p>
101     *
102     * @param skewAllowance Number of seconds to allow for variance.
103     */
104     public void setSkewAllowance(final int skewAllowance) {
105         logger.debug("Using {} seconds as skew allowance.", skewAllowance);
106         this.skewAllowance = skewAllowance;
107     }
108 
109     static {
110         try {
111             // Initialize OpenSAML default configuration
112             // (only needed once per classloader)
113             DefaultBootstrap.bootstrap();
114         } catch (final ConfigurationException e) {
115             throw new IllegalStateException("Error initializing OpenSAML library.", e);
116         }
117     }
118 
119     /**
120      * Instantiates a new abstract saml10 response view.
121      */
122     protected AbstractSaml10ResponseView() {
123         try {
124             this.idGenerator = new SecureRandomIdentifierGenerator();
125         } catch (final NoSuchAlgorithmException e) {
126             throw new IllegalStateException("Cannot create secure random ID generator for SAML message IDs.");
127         }
128     }
129 
130     @Override
131     protected void renderMergedOutputModel(
132             final Map<String, Object> model, final HttpServletRequest request, final HttpServletResponse response) throws Exception {
133 
134         response.setCharacterEncoding(this.encoding);
135 
136         final WebApplicationService service = this.samlArgumentExtractor.extractService(request);
137         final String serviceId = service != null ? service.getId() : "UNKNOWN";
138 
139         try {
140             final Response samlResponse = newSamlObject(Response.class);
141             samlResponse.setID(generateId());
142             samlResponse.setIssueInstant(DateTime.now().minusSeconds(skewAllowance));
143             samlResponse.setVersion(SAMLVersion.VERSION_11);
144             samlResponse.setRecipient(serviceId);
145             if (service instanceof SamlService) {
146                 final SamlService samlService = (SamlService) service;
147 
148                 if (samlService.getRequestID() != null) {
149                     samlResponse.setInResponseTo(samlService.getRequestID());
150                 }
151             }
152             prepareResponse(samlResponse, model);
153 
154             final BasicSAMLMessageContext messageContext = new BasicSAMLMessageContext();
155             messageContext.setOutboundMessageTransport(new HttpServletResponseAdapter(response, request.isSecure()));
156             messageContext.setOutboundSAMLMessage(samlResponse);
157             this.encoder.encode(messageContext);
158         } catch (final Exception e) {
159             logger.error("Error generating SAML response for service {}.", serviceId);
160             throw e;
161         }
162     }
163 
164     /**
165      * Subclasses must implement this method by adding child elements (status, assertion, etc) to
166      * the given empty SAML 1 response message.  Impelmenters need not be concerned with error handling.
167      *
168      * @param response SAML 1 response message to be filled.
169      * @param model Spring MVC model map containing data needed to prepare response.
170      */
171     protected abstract void prepareResponse(Response response, Map<String, Object> model);
172 
173 
174     /**
175      * Generate id.
176      *
177      * @return the string
178      */
179     protected final String generateId() {
180         return this.idGenerator.generateIdentifier();
181     }
182 
183     /**
184      * Create a new SAML object.
185      *
186      * @param <T> the generic type
187      * @param objectType the object type
188      * @return the t
189      */
190     protected final <T extends SAMLObject> T newSamlObject(final Class<T> objectType) {
191         final QName qName;
192         try {
193             final Field f = objectType.getField(DEFAULT_ELEMENT_NAME_FIELD);
194             qName = (QName) f.get(null);
195         } catch (final NoSuchFieldException e) {
196             throw new IllegalStateException("Cannot find field " + objectType.getName() + "." + DEFAULT_ELEMENT_NAME_FIELD);
197         } catch (final IllegalAccessException e) {
198             throw new IllegalStateException("Cannot access field " + objectType.getName() + "." + DEFAULT_ELEMENT_NAME_FIELD);
199         }
200         final SAMLObjectBuilder<T> builder = (SAMLObjectBuilder<T>) Configuration.getBuilderFactory().getBuilder(qName);
201         if (builder == null) {
202             throw new IllegalStateException("No SAMLObjectBuilder registered for class " + objectType.getName());
203         }
204         return objectType.cast(builder.buildObject());
205     }
206 
207     /**
208      * Create a new SAML status object.
209      *
210      * @param codeValue the code value
211      * @param statusMessage the status message
212      * @return the status
213      */
214     protected final Status newStatus(final QName codeValue, final String statusMessage) {
215         final Status status = newSamlObject(Status.class);
216         final StatusCode code = newSamlObject(StatusCode.class);
217         code.setValue(codeValue);
218         status.setStatusCode(code);
219         if (statusMessage != null) {
220             final StatusMessage message = newSamlObject(StatusMessage.class);
221             message.setMessage(statusMessage);
222             status.setStatusMessage(message);
223         }
224         return status;
225     }
