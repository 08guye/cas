19  package org.jasig.cas.support.spnego.web.flow;
20  
21  import java.util.ArrayList;
22  import java.util.List;
23  
24  import javax.servlet.http.HttpServletRequest;
25  import javax.servlet.http.HttpServletResponse;
26  
27  import org.jasig.cas.support.spnego.util.SpnegoConstants;
28  import org.jasig.cas.web.support.WebUtils;
29  
30  import org.slf4j.Logger;
31  import org.slf4j.LoggerFactory;
32  import org.springframework.util.StringUtils;
33  import org.springframework.webflow.action.AbstractAction;
34  import org.springframework.webflow.execution.Event;
35  import org.springframework.webflow.execution.RequestContext;
36  
37  /**
38   * First action of a SPNEGO flow : negotiation.
39   * <p>The server checks if the
40   * negotiation string is in the request header and this is a supported browser:
41   * <ul>
42   * <li>If found do nothing and return <code>success()</code></li>
43   * <li>else add a WWW-Authenticate response header and a 401 response status,
44   * then return <code>success()</code></li>
45   * </ul>
46   *
47   * @author Arnaud Lesueur
48   * @author Marc-Antoine Garrigue
49   * @author Scott Battaglia
50   * @author John Gasper
51   * @see <a href="http://ietfreport.isoc.org/idref/rfc4559/#page-2">RFC 4559</a>
52   * @since 3.1
53   */
54  public final class SpnegoNegociateCredentialsAction extends AbstractAction {
55  
56      private static final Logger LOGGER = LoggerFactory.getLogger(SpnegoNegociateCredentialsAction.class);
57  
58      /** Whether this is using the NTLM protocol or not. */
59      private boolean ntlm;
60  
61      private boolean mixedModeAuthentication;
62  
63      private List<String> supportedBrowser;
64  
65      private String messageBeginPrefix = constructMessagePrefix();
66  
67      /**
68       * Instantiates a new Spnego negociate credentials action.
69       * Also initializes the list of supported browser user agents with the following:
70       * <ul>
71       *     <li><code>MSIE</code></li>
72       *     <li><code>Trident</code></li>
73       *     <li><code>Firefox</code></li>
74       *     <li><code>AppleWebKit</code></li>
75       * </ul>
76       * @see #setSupportedBrowser(List)
77       * @since 4.1
78       */
79      public SpnegoNegociateCredentialsAction() {
80          super();
81  
82          this.supportedBrowser = new ArrayList<>();
83          this.supportedBrowser.add("MSIE");
84          this.supportedBrowser.add("Trident");
85          this.supportedBrowser.add("Firefox");
86          this.supportedBrowser.add("AppleWebKit");
87  
88      }
89  
90      @Override
91      protected Event doExecute(final RequestContext context) {
92          final HttpServletRequest request = WebUtils.getHttpServletRequest(context);
93          final HttpServletResponse response = WebUtils.getHttpServletResponse(context);
94  
95          final String authorizationHeader = request.getHeader(SpnegoConstants.HEADER_AUTHORIZATION);
96          final String userAgent = request.getHeader(SpnegoConstants.HEADER_USER_AGENT);
97  
98          LOGGER.debug("Authorization header [{}], User Agent header [{}]", authorizationHeader, userAgent);
99  
100         if (!StringUtils.hasText(userAgent) || this.supportedBrowser.isEmpty()) {
101             LOGGER.debug("User Agent header [{}] is empty, or no browsers are supported", userAgent);
102             return success();
103         }
104 
105         if (!isSupportedBrowser(userAgent)) {
106             LOGGER.debug("User Agent header [{}] is not supported in the list of supported browsers [{}]",
107                     userAgent, this.supportedBrowser);
108             return success();
109         }
110 
111         if (!StringUtils.hasText(authorizationHeader)
112                 || !authorizationHeader.startsWith(this.messageBeginPrefix)
113                 || authorizationHeader.length() <= this.messageBeginPrefix
114                 .length()) {
115 
116             final String wwwHeader = this.ntlm ? SpnegoConstants.NTLM : SpnegoConstants.NEGOTIATE;
117             LOGGER.debug("Authorization header not found or does not match the message prefix [{}]. Sending [{}] header [{}]",
118                     this.messageBeginPrefix, SpnegoConstants.HEADER_AUTHENTICATE, wwwHeader);
119             response.setHeader(SpnegoConstants.HEADER_AUTHENTICATE, wwwHeader);
120 
121             response.setStatus(HttpServletResponse.SC_UNAUTHORIZED);
122             // The responseComplete flag tells the pausing view-state not to render the response
123             // because another object has taken care of it. If mixed mode authentication is allowed
124             // then responseComplete should not be called so that webflow will display the login page.
125             if (!this.mixedModeAuthentication) {
126                 LOGGER.debug("Mixed-mode authentication is disabled. Executing completion of response");
127                 context.getExternalContext().recordResponseComplete();
128             }
129         }
130         return success();
131     }
132 
133     /**
134      * Sets the ntlm. Generates the message prefix as well.
135      *
136      * @param ntlm the new ntlm
137      */
138     public void setNtlm(final boolean ntlm) {
139         this.ntlm = ntlm;
140         this.messageBeginPrefix = constructMessagePrefix();
141     }
142 
143     /**
144      * Sets supported browsers by their user agent. The user agent
145      * header defined by {@link SpnegoConstants#HEADER_USER_AGENT}
146      * will be compared against this list. The user agents configured
147      * here need not be an exact match. So longer is the user agent identifier
148      * configured in this list is "found" in the user agent header retrieved,
149      * the check will pass.
150      *
151      * @param supportedBrowser the supported browsers list
152      */
153     public void setSupportedBrowser(final List<String> supportedBrowser) {
154         this.supportedBrowser = supportedBrowser;
155     }
156 
157     /**
158      * Sets whether mixed mode authentication should be enabled. If it is
159      * enabled then control is allowed to pass back to the Spring Webflow
160      * instead of immediately terminating the page after issuing the
161      * unauthorized (401) header. This has the effect of displaying the login
162      * page on unsupported/configured browsers.
163      * <p>
164      * If this is set to false then the page is immediately closed after the
165      * unauthorized header is sent. This is ideal in environments that only
166      * want to use Windows Integrated Auth/SPNEGO and not forms auth.
167      *
168      * @param enabled should mixed mode authentication be allowed. Default is false.
169      */
170     public void setMixedModeAuthentication(final boolean enabled) {
171         this.mixedModeAuthentication = enabled;
172     }
173 
174     /**
175      * Construct message prefix.
176      *
177      * @return if {@link #ntlm} is enabled, {@link SpnegoConstants#NTLM}, otherwise
178      * {@link SpnegoConstants#NEGOTIATE}. An extra space is appended to the end.
179      */
180     protected String constructMessagePrefix() {
181         return (this.ntlm ? SpnegoConstants.NTLM : SpnegoConstants.NEGOTIATE)
182                 + ' ';
183     }
184 
185     /**
186      * Checks if is supported browser.
187      *
188      * @param userAgent the user agent
189      * @return true, if  supported browser
190      */
191     protected boolean isSupportedBrowser(final String userAgent) {
192         for (final String supportedBrowser : this.supportedBrowser) {
193             if (userAgent.contains(supportedBrowser)) {
194                 return true;
195             }
196         }
197         return false;
198     }
199 }
