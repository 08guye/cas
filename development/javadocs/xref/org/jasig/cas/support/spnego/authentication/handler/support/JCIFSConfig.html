84      public JCIFSConfig() {
85          Config.setProperty("jcifs.smb.client.soTimeout", "300000");
86          Config.setProperty("jcifs.netbios.cachePolicy", "600");
87      }

89      @Override
90      public void afterPropertiesSet() throws Exception {
91          final String propValue = System.getProperty(SYS_PROP_LOGIN_CONF);
92          if (propValue != null) {
93              logger.warn("found login config in system property, may overide : {}", propValue);
94          }
95  
96          URL url = getClass().getResource(
97                  this.loginConf == null ? DEFAULT_LOGIN_CONFIG : this.loginConf);
98          if (url != null) {
99              this.loginConf = url.toExternalForm();
100         }
101         if (this.loginConf != null) {
102             System.setProperty(SYS_PROP_LOGIN_CONF, this.loginConf);
103         } else {
104             url = getClass().getResource("/jcifs/http/login.conf");
105             if (url != null) {
106                 System.setProperty(SYS_PROP_LOGIN_CONF, url.toExternalForm());
107             }
108         }
109         logger.debug("configured login configuration path : {}", propValue);
110     }
