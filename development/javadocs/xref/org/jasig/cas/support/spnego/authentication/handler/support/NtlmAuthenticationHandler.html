45  /**
46   * Implementation of an AuthenticationHandler for NTLM supports.
47   *
48   * @author <a href="mailto:julien.henry@capgemini.com">Julien Henry</a>
49   * @author Scott Battaglia
50   * @author Arnaud Lesueur
51   * @since 3.1
52   */
53  
54  public class NtlmAuthenticationHandler extends AbstractPreAndPostProcessingAuthenticationHandler {
55  
56      private boolean loadBalance = true;
57  
58      @NotNull
59      private String domainController = Config.getProperty("jcifs.smb.client.domain");
60  
61      private String includePattern = null;
62  
63      @Override
64      protected final HandlerResult doAuthentication(
65              final Credential credential) throws GeneralSecurityException, PreventedException {
66  
67          final SpnegoCredential ntlmCredential = (SpnegoCredential) credential;
68          final byte[] src = ntlmCredential.getInitToken();
69  
70          UniAddress dc = null;
71  
72          boolean success = false;
73          try {
74              if (this.loadBalance) {
75                  // find the first dc that matches the includepattern
76                  if(this.includePattern != null){
77                      final NbtAddress [] dcs  = NbtAddress.getAllByName(this.domainController, 0x1C, null, null);
78                      for (NbtAddress dc2 : dcs) {
79                          if(dc2.getHostAddress().matches(this.includePattern)){
80                              dc = new UniAddress(dc2);
81                              break;
82                          }
83                      }
84                  } else {
85                      dc = new UniAddress(NbtAddress.getByName(this.domainController,
86                              0x1C, null));
87                  }
88              } else {
89                  dc = UniAddress.getByName(this.domainController, true);
90              }
91              final byte[] challenge = SmbSession.getChallenge(dc);
92  
93              switch (src[8]) {
94                  case 1:
95                      logger.debug("Type 1 received");
96                      final Type1Message type1 = new Type1Message(src);
97                      final Type2Message type2 = new Type2Message(type1,
98                              challenge, null);
99                      logger.debug("Type 2 returned. Setting next token.");
100                     ntlmCredential.setNextToken(type2.toByteArray());
101                 case 3:
102                     logger.debug("Type 3 received");
103                     final Type3Message type3 = new Type3Message(src);
104                     final byte[] lmResponse = type3.getLMResponse() == null ? new byte[0] : type3.getLMResponse();
105                     final byte[] ntResponse = type3.getNTResponse() == null ? new byte[0] : type3.getNTResponse();
106                     final NtlmPasswordAuthentication ntlm = new NtlmPasswordAuthentication(
107                             type3.getDomain(), type3.getUser(), challenge,
108                             lmResponse, ntResponse);
109                     logger.debug("Trying to authenticate {} with domain controller", type3.getUser());
110                     try {
111                         SmbSession.logon(dc, ntlm);
112                         ntlmCredential.setPrincipal(new SimplePrincipal(type3.getUser()));
113                         success = true;
114                     } catch (final SmbAuthException sae) {
115                         throw new FailedLoginException(sae.getMessage());
116                     }
117                 default:
118                     logger.debug("Unknown type: {}", src[8]);
119             }
120         } catch (final Exception e) {
121             throw new FailedLoginException(e.getMessage());
122         }
123 
124         if (!success) {
125             throw new FailedLoginException();
126         }
127         return new HandlerResult(this, new BasicCredentialMetaData(ntlmCredential), ntlmCredential.getPrincipal());
128     }
129 
130     @Override
131     public boolean supports(final Credential credential) {
132         return credential instanceof SpnegoCredential;
133     }
134 
135     public void setLoadBalance(final boolean loadBalance) {
136         this.loadBalance = loadBalance;
137     }
138 
139     public void setDomainController(final String domainController) {
140         this.domainController = domainController;
141     }
142 
143     public void setIncludePattern(final String includePattern) {
144         this.includePattern = includePattern;
145     }
