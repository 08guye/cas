41  /**
42   * This controller returns a profile for the authenticated user
43   * (identifier + attributes), found with the access token (CAS granting
44   * ticket).
45   *
46   * @author Jerome Leleu
47   * @since 3.5.0
48   */
49  public final class OAuth20ProfileController extends AbstractController {
50  
51      private static Logger LOGGER = LoggerFactory.getLogger(OAuth20ProfileController.class);
52  
53      private static final String ID = "id";
54  
55      private static final String ATTRIBUTES = "attributes";
56  
57      private final TicketRegistry ticketRegistry;
58  
59      private final JsonFactory jsonFactory = new JsonFactory(new ObjectMapper());
60  
61      /**
62       * Instantiates a new o auth20 profile controller.
63       *
64       * @param ticketRegistry the ticket registry
65       */
66      public OAuth20ProfileController(final TicketRegistry ticketRegistry) {
67          this.ticketRegistry = ticketRegistry;
68      }
69  
70      @Override
71      protected ModelAndView handleRequestInternal(final HttpServletRequest request, final HttpServletResponse response)
72              throws Exception {
73  
74          String accessToken = request.getParameter(OAuthConstants.ACCESS_TOKEN);
75          if (StringUtils.isBlank(accessToken)) {
76              final String authHeader = request.getHeader("Authorization");
77              if (StringUtils.isNotBlank(authHeader) && authHeader.startsWith(OAuthConstants.BEARER_TOKEN + " ")) {
78                  accessToken = authHeader.substring(OAuthConstants.BEARER_TOKEN.length() + 1);
79              }
80          }
81          LOGGER.debug("{} : {}", OAuthConstants.ACCESS_TOKEN, accessToken);
82  
83          final JsonGenerator jsonGenerator = this.jsonFactory.createJsonGenerator(response.getWriter());
84  
85          try {
86              response.setContentType("application/json");
87  
88              // accessToken is required
89              if (StringUtils.isBlank(accessToken)) {
90                  LOGGER.error("Missing {}", OAuthConstants.ACCESS_TOKEN);
91                  jsonGenerator.writeStartObject();
92                  jsonGenerator.writeStringField("error", OAuthConstants.MISSING_ACCESS_TOKEN);
93                  jsonGenerator.writeEndObject();
94                  return null;
95              }
96              // get ticket granting ticket
97              final TicketGrantingTicket ticketGrantingTicket = (TicketGrantingTicket) this.ticketRegistry
98                      .getTicket(accessToken);
99              if (ticketGrantingTicket == null || ticketGrantingTicket.isExpired()) {
100                 LOGGER.error("expired accessToken : {}", accessToken);
101                 jsonGenerator.writeStartObject();
102                 jsonGenerator.writeStringField("error", OAuthConstants.EXPIRED_ACCESS_TOKEN);
103                 jsonGenerator.writeEndObject();
104                 return null;
105             }
106             // generate profile : identifier + attributes
107             final Principal principal = ticketGrantingTicket.getAuthentication().getPrincipal();
108             jsonGenerator.writeStartObject();
109             jsonGenerator.writeStringField(ID, principal.getId());
110             jsonGenerator.writeArrayFieldStart(ATTRIBUTES);
111             final Map<String, Object> attributes = principal.getAttributes();
112             for (final String key : attributes.keySet()) {
113                 jsonGenerator.writeStartObject();
114                 jsonGenerator.writeObjectField(key, attributes.get(key));
115                 jsonGenerator.writeEndObject();
116             }
117             jsonGenerator.writeEndArray();
118             jsonGenerator.writeEndObject();
119             return null;
120         } finally {
121             IOUtils.closeQuietly(jsonGenerator);
122             response.flushBuffer();
123         }
124     }
