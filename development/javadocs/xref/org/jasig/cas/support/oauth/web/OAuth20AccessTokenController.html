19  package org.jasig.cas.support.oauth.web;

22  import javax.servlet.http.HttpServletResponse;
23  
24  import org.apache.commons.lang.StringUtils;
25  import org.jasig.cas.services.ServicesManager;
26  import org.jasig.cas.support.oauth.OAuthConstants;
27  import org.jasig.cas.support.oauth.OAuthUtils;
28  import org.jasig.cas.support.oauth.services.OAuthRegisteredService;
29  import org.jasig.cas.ticket.ServiceTicket;
30  import org.jasig.cas.ticket.TicketGrantingTicket;
31  import org.jasig.cas.ticket.registry.TicketRegistry;
32  import org.slf4j.Logger;
33  import org.slf4j.LoggerFactory;
34  import org.springframework.web.servlet.ModelAndView;
35  import org.springframework.web.servlet.mvc.AbstractController;

37  /**
38   * This controller returns an access token which is the CAS
39   * granting ticket according to the service and code (service ticket) given.
40   *
41   * @author Jerome Leleu
42   * @since 3.5.0
43   */
44  public final class OAuth20AccessTokenController extends AbstractController {
45  
46      private static Logger LOGGER = LoggerFactory.getLogger(OAuth20AccessTokenController.class);
47  
48      private final ServicesManager servicesManager;
49  
50      private final TicketRegistry ticketRegistry;
51  
52      private final long timeout;
53  
54      /**
55       * Instantiates a new o auth20 access token controller.
56       *
57       * @param servicesManager the services manager
58       * @param ticketRegistry the ticket registry
59       * @param timeout the timeout
60       */
61      public OAuth20AccessTokenController(final ServicesManager servicesManager, final TicketRegistry ticketRegistry,
62              final long timeout) {
63          this.servicesManager = servicesManager;
64          this.ticketRegistry = ticketRegistry;
65          this.timeout = timeout;
66      }
67  
68      @Override
69      protected ModelAndView handleRequestInternal(final HttpServletRequest request, final HttpServletResponse response)
70              throws Exception {
71  
72          final String redirectUri = request.getParameter(OAuthConstants.REDIRECT_URI);
73          LOGGER.debug("{} : {}", OAuthConstants.REDIRECT_URI, redirectUri);
74  
75          final String clientId = request.getParameter(OAuthConstants.CLIENT_ID);
76          LOGGER.debug("{} : {}", OAuthConstants.CLIENT_ID, clientId);
77  
78          final String clientSecret = request.getParameter(OAuthConstants.CLIENT_SECRET);
79  
80          final String code = request.getParameter(OAuthConstants.CODE);
81          LOGGER.debug("{} : {}", OAuthConstants.CODE, code);
82  
83          final boolean isVerified = verifyAccessTokenRequest(response, redirectUri, clientId, clientSecret, code);
84          if (!isVerified) {
85              return OAuthUtils.writeTextError(response, OAuthConstants.INVALID_REQUEST, 400);
86          }
87  
88          final ServiceTicket serviceTicket = (ServiceTicket) ticketRegistry.getTicket(code);
89          // service ticket should be valid
90          if (serviceTicket == null || serviceTicket.isExpired()) {
91              LOGGER.error("Code expired : {}", code);
92              return OAuthUtils.writeTextError(response, OAuthConstants.INVALID_GRANT, 400);
93          }
94          final TicketGrantingTicket ticketGrantingTicket = serviceTicket.getGrantingTicket();
95          // remove service ticket
96          ticketRegistry.deleteTicket(serviceTicket.getId());
97  
98          response.setContentType("text/plain");
99          final int expires = (int) (timeout - (System.currentTimeMillis()
100                 - ticketGrantingTicket.getCreationTime()) / 1000);
101 
102         final String text = String.format("%s=%s&%s=%s", OAuthConstants.ACCESS_TOKEN, ticketGrantingTicket.getId(),
103                                                     OAuthConstants.EXPIRES, expires);
104         LOGGER.debug("text : {}", text);
105         return OAuthUtils.writeText(response, text, 200);
106     }
107 
108     /**
109      * Verify access token request by reviewing the values of
110      * client id, redirect uri, client secret, code, etc.
111      *
112      * @param response the response
113      * @param redirectUri the redirect uri
114      * @param clientId the client id
115      * @param clientSecret the client secret
116      * @param code the code
117      * @return true, if successful
118      */
119     private boolean verifyAccessTokenRequest(final HttpServletResponse response, final String redirectUri,
120                                              final String clientId, final String clientSecret, final String code) {
121 
122         // clientId is required
123         if (StringUtils.isBlank(clientId)) {
124             LOGGER.error("Missing {}", OAuthConstants.CLIENT_ID);
125             return false;
126         }
127         // redirectUri is required
128         if (StringUtils.isBlank(redirectUri)) {
129             LOGGER.error("Missing {}", OAuthConstants.REDIRECT_URI);
130             return false;
131         }
132         // clientSecret is required
133         if (StringUtils.isBlank(clientSecret)) {
134             LOGGER.error("Missing {}", OAuthConstants.CLIENT_SECRET);
135             return false;
136         }
137         // code is required
138         if (StringUtils.isBlank(code)) {
139             LOGGER.error("Missing {}", OAuthConstants.CODE);
140             return false;
141         }
142 
143         final OAuthRegisteredService service = OAuthUtils.getRegisteredOAuthService(this.servicesManager, clientId);
144         if (service == null) {
145             LOGGER.error("Unknown {} : {}", OAuthConstants.CLIENT_ID, clientId);
146             return false;
147         }
148 
149         final String serviceId = service.getServiceId();
150         if (!redirectUri.matches(serviceId)) {
151             LOGGER.error("Unsupported {} : {} for serviceId : {}", OAuthConstants.REDIRECT_URI, redirectUri, serviceId);
152             return false;
153         }
154 
155         if (!StringUtils.equals(service.getClientSecret(), clientSecret)) {
156             LOGGER.error("Wrong client secret for service {}", service);
157             return false;
158         }
159         return true;
160     }
