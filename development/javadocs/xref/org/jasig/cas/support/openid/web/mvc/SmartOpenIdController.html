30  import javax.servlet.http.HttpServletRequest;
31  import javax.servlet.http.HttpServletResponse;
32  import javax.validation.constraints.NotNull;
33  import java.io.Serializable;

37  /**
38   * Generates an association to an openid association request.
39   * @author Frederic Esnault
40   * @since 3.5
41   */
42  public class SmartOpenIdController extends DelegateController implements Serializable {
43  
44      private static final long serialVersionUID = -594058549445950430L;
45  
46      private final Logger logger = LoggerFactory.getLogger(SmartOpenIdController.class);
47  
48      private ServerManager serverManager;
49  
50      /** View if association Fails. */
51      private static final String DEFAULT_ASSOCIATION_FAILURE_VIEW_NAME = "casOpenIdAssociationFailureView";
52  
53      /** View if association Succeeds. */
54      private static final String DEFAULT_ASSOCIATION_SUCCESS_VIEW_NAME = "casOpenIdAssociationSuccessView";
55  
56      /** The view to redirect to on a successful validation. */
57      @NotNull
58      private String successView = DEFAULT_ASSOCIATION_SUCCESS_VIEW_NAME;
59  
60      /** The view to redirect to on a validation failure. Not used for now,
61       *  the succes view may return failed association attemps. No need for another view. */
62      @NotNull
63      private String failureView = DEFAULT_ASSOCIATION_FAILURE_VIEW_NAME;
64  
65      /**
66       * Gets the association response. Determines the mode first.
67       * If mode is set to associate, will set the response. Then
68       * builds the response parameters next and returns.
69       *
70       * @param request the request
71       * @return the association response
72       */
73      public Map<String, String> getAssociationResponse(final HttpServletRequest request) {
74          final ParameterList parameters = new ParameterList(request.getParameterMap());
75  
76          final String mode = parameters.hasParameter("openid.mode")
77                  ? parameters.getParameterValue("openid.mode")
78                  : null;
79  
80          Message response = null;
81          if (mode != null && mode.equals("associate")) {
82              response = serverManager.associationResponse(parameters);
83          }
84          final Map<String, String> responseParams = new HashMap<String, String>();
85          if (response != null) {
86              responseParams.putAll(response.getParameterMap());
87          }
88  
89          return responseParams;
90  
91      }
92  
93      @Override
94      protected ModelAndView handleRequestInternal(final HttpServletRequest request, final HttpServletResponse response)
95              throws Exception {
96          final Map<String, String> parameters = new HashMap<String, String>();
97          parameters.putAll(getAssociationResponse(request));
98          return new ModelAndView(successView, "parameters", parameters);
99      }
100 
101     @Override
102     public boolean canHandle(final HttpServletRequest request, final HttpServletResponse response) {
103         final String openIdMode = request.getParameter("openid.mode");
104         if (openIdMode != null && openIdMode.equals("associate")) {
105             logger.info("Handling request. openid.mode : {}", openIdMode);
106             return true;
107         }
108         logger.info("Cannot handle request. openid.mode : {}", openIdMode);
109         return false;
110     }
111 
112     public void setSuccessView(final String successView) {
113         this.successView = successView;
114     }
115 
116     public void setFailureView(final String failureView) {
117         this.failureView = failureView;
118     }
119 
120     @NotNull
121     public void setServerManager(final ServerManager serverManager) {
122         this.serverManager = serverManager;
123     }
124 }
