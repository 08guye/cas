19  package org.jasig.cas.services;
20  
21  import java.io.Serializable;
22  import java.util.Set;
23  
24  import org.jasig.cas.authentication.principal.Service;
25  
26  /**
27   * Interface for a service that can be registered by the Services Management
28   * interface.
29   *
30   * @author Scott Battaglia
31   * @since 3.1
32   */
33  public interface RegisteredService extends Cloneable, Serializable {
34  
35      /** Initial ID value of newly created (but not persisted) registered service. **/
36      long INITIAL_IDENTIFIER_VALUE = Long.MAX_VALUE;
37  
38      /**
39       * Is this application currently allowed to use CAS?
40       *
41       * @return true if it can use CAS, false otherwise.
42       */
43      boolean isEnabled();
44  
45      /**
46       * Determines whether the service is allowed anonymous or privileged access
47       * to user information. Anonymous access should not return any identifying
48       * information such as user id.
49       *
50       * @return if we should use a pseudo random identifier instead of their real id
51       */
52      boolean isAnonymousAccess();
53  
54      /**
55       * Get the proxy policy rules for this service.
56       *
57       * @return the proxy policy
58       */
59      RegisteredServiceProxyPolicy getProxyPolicy();
60  
61      /**
62       * The unique identifier for this service.
63       *
64       * @return the unique identifier for this service.
65       */
66      String getServiceId();
67  
68      /**
69       * The numeric identifier for this service. Implementations
70       * are expected to initialize the id with the value of {@link #INITIAL_IDENTIFIER_VALUE}.
71       * @return the numeric identifier for this service.
72       */
73      long getId();
74  
75      /**
76       * Returns the name of the service.
77       *
78       * @return the name of the service.
79       */
80      String getName();
81  
82      /**
83       * Returns a short theme name. Services do not need to have unique theme
84       * names.
85       *
86       * @return the theme name associated with this service.
87       */
88      String getTheme();
89  
90      /**
91       * Does this application participate in the SSO session?
92       *
93       * @return true if it does, false otherwise.
94       */
95      boolean isSsoEnabled();
96  
97      /**
98       * Returns the description of the service.
99       *
100      * @return the description of the service.
101      */
102     String getDescription();
103 
104     /**
105      * Gets the relative evaluation order of this service when determining
106      * matches.
107      * @return Evaluation order relative to other registered services.
108      * Services with lower values will be evaluated for a match before others.
109      */
110     int getEvaluationOrder();
111 
112     /**
113      * Sets the relative evaluation order of this service when determining
114      * matches.
115      * @param evaluationOrder the service evaluation order
116      */
117     void setEvaluationOrder(final int evaluationOrder);
118 
119     /**
120      * Get the name of the attribute this service prefers to consume as username.
121      *
122      * @return Either of the following values:
123      * <ul>
124      *  <li><code>String</code> representing the name of the attribute to consume as username</li>
125      *  <li><code>null</code> indicating the default username</li>
126      * </ul>
127      */
128     String getUsernameAttribute();
129 
130     /**
131      * Gets the set of handler names that must successfully authenticate credentials in order to access the service.
132      * An empty set indicates that there are no requirements on particular authentication handlers; any will suffice.
133      *
134      * @return Non-null set of required handler names.
135      */
136     Set<String> getRequiredHandlers();
137 
138     /**
139      * Returns whether the service matches the registered service.
140      * <p>Note, as of 3.1.2, matches are case insensitive.
141      *
142      * @param service the service to match.
143      * @return true if they match, false otherwise.
144      */
145     boolean matches(final Service service);
146 
147     
148     /**
149      * Clone this service.
150      *
151      * @return the registered service
152      * @throws CloneNotSupportedException the clone not supported exception
153      */
154     RegisteredService clone() throws CloneNotSupportedException;
155 
156     /**
157      * Returns the logout type of the service.
158      *
159      * @return the logout type of the service.
160      */
161     LogoutType getLogoutType();
162     
163     /**
164      * Gets the attribute filtering policy to determine
165      * how attributes are to be filtered and released for
166      * this service.
167      *
168      * @return the attribute release policy
169      */
170     AttributeReleasePolicy getAttributeReleasePolicy();
171 }
