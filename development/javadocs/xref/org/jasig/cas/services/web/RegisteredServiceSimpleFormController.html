43  /**
44   * Handle adding/editing of RegisteredServices.
45   *
46   * @author Scott Battaglia
47   * @since 3.1
48   */
49  @Controller
50  public final class RegisteredServiceSimpleFormController extends AbstractManagementController {
51  
52      /**
53       * Instance of AttributeRegistry.
54       */
55      @NotNull
56      private final IPersonAttributeDao personAttributeDao;
57  
58      /**
59       * Instantiates a new registered service simple form controller.
60       *
61       * @param servicesManager    the services manager
62       * @param personAttributeDao the attribute repository
63       */
64      @Autowired
65      public RegisteredServiceSimpleFormController(final ServicesManager servicesManager,
66                                                   final IPersonAttributeDao personAttributeDao) {
67          super(servicesManager);
68          this.personAttributeDao = personAttributeDao;
69      }
70  
71      /**
72       * Adds the service to the Service Registry.
73       * @param request the request
74       * @param response the response
75       * @param result the result
76       * @param service the edit bean
77       */
78      @RequestMapping(method = RequestMethod.POST, value = {"saveService.html"})
79      public void saveService(final HttpServletRequest request,
80                              final HttpServletResponse response,
81                              @RequestBody final RegisteredServiceEditBean.ServiceData service,
82                              final BindingResult result) {
83          try {
84  
85              final RegisteredService svcToUse = service.toRegisteredService(this.personAttributeDao);
86              if (svcToUse.getId() != RegisteredService.INITIAL_IDENTIFIER_VALUE) {
87                  final RegisteredService curService = this.servicesManager.findServiceBy(svcToUse.getId());
88                  if (curService != null && this.servicesManager.delete(curService.getId()) == null) {
89                      throw new IllegalArgumentException("Service " + curService.getId() + " cannot be updated");
90                  }
91              }
92  
93              final RegisteredService newSvc = this.servicesManager.save(svcToUse);
94              logger.info("Saved changes to service {}", svcToUse.getId());
95  
96              final Map<String, Object> model = new HashMap<>();
97              model.put("id", newSvc.getId());
98              model.put("status", HttpServletResponse.SC_OK);
99              JsonViewUtils.render(model, response);
100 
101         } catch (final Exception e) {
102             throw new RuntimeException(e);
103         }
104     }
105 
106     /**
107      * Gets service by id.
108      *
109      * @param id       the id
110      * @param request  the request
111      * @param response the response
112      */
113     @RequestMapping(method = RequestMethod.GET, value = {"getService.html"})
114     public void getServiceById(@RequestParam(value = "id", required = false) final Long id,
115                                final HttpServletRequest request, final HttpServletResponse response) {
116 
117         try {
118             RegisteredServiceEditBean bean = null;
119             if (id == -1) {
120                 bean = new RegisteredServiceEditBean();
121                 bean.setServiceData(null);
122             } else {
123                 final RegisteredService service = this.servicesManager.findServiceBy(id);
124 
125                 if (service == null) {
126                     logger.warn("Invalid service id specified [{}]. Cannot find service in the registry", id);
127                     throw new IllegalArgumentException("Service id cannot be found");
128                 }
129                 bean = RegisteredServiceEditBean.fromRegisteredService(service);
130             }
131             final RegisteredServiceEditBean.FormData formData = bean.getFormData();
132             final List<String> possibleAttributeNames = new ArrayList<>();
133             possibleAttributeNames.addAll(this.personAttributeDao.getPossibleUserAttributeNames());
134             Collections.sort(possibleAttributeNames);
135             formData.setAvailableAttributes(possibleAttributeNames);
136 
137             bean.setStatus(HttpServletResponse.SC_OK);
138             JsonViewUtils.render(bean, response);
139         } catch (final Exception e) {
140             throw new RuntimeException(e);
141         }
142     }
143 }
