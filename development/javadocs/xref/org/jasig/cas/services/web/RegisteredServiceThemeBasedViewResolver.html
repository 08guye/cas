19  package org.jasig.cas.services.web;
20  
21  import org.jasig.cas.authentication.principal.WebApplicationService;
22  import org.jasig.cas.services.RegisteredService;
23  import org.jasig.cas.services.ServicesManager;
24  import org.jasig.cas.web.support.WebUtils;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  import org.springframework.beans.BeanUtils;
28  import org.springframework.util.StringUtils;
29  import org.springframework.web.servlet.view.AbstractUrlBasedView;
30  import org.springframework.web.servlet.view.InternalResourceView;
31  import org.springframework.web.servlet.view.InternalResourceViewResolver;
32  import org.springframework.webflow.execution.RequestContext;
33  import org.springframework.webflow.execution.RequestContextHolder;
34  
35  /**
36   * {@link RegisteredServiceThemeBasedViewResolver} is an alternate Spring View Resolver that utilizes a service's
37   * associated theme to selectively choose which set of UI views will be used to generate
38   * the standard views (casLoginView.jsp, casLogoutView.jsp etc).
39   *
40   * <p>Views associated with a particular theme by default are expected to be found at:
41   * {@link #DEFAULT_PATH_PREFIX}/<code>themeId/ui</code>. A starting point may be to
42   * clone the default set of view pages into a new directory based on the theme id.</p>
43   *
44   * <p>Note: There also exists a {@link org.jasig.cas.services.web.ServiceThemeResolver}
45   * that attempts to resolve the view name based on the service theme id. The difference
46   * however is that {@link org.jasig.cas.services.web.ServiceThemeResolver} only decorates
47   * the default view pages with additional tags and coloring, such as CSS and JS. The
48   * component presented here on the other hand has the ability to load an entirely new
49   * set of pages that are may be totally different from that of the default's. This
50   * is specially useful in cases where the set of pages for a theme that are targetted
51   * for a different type of audience are entirely different structurally that simply
52   * using the {@link org.jasig.cas.services.web.ServiceThemeResolver} is not practical
53   * to augment the default views. In such cases, new view pages may be required.</p>
54   *
55   * @author John Gasper
56   * @author Misagh Moayyed
57   * @since 4.1
58   */
59  public final class RegisteredServiceThemeBasedViewResolver extends InternalResourceViewResolver {
60      private static final Logger LOGGER = LoggerFactory.getLogger(RegisteredServiceThemeBasedViewResolver.class);
61      private static final String DEFAULT_PATH_PREFIX = "/WEB-INF/view/jsp";
62  
63      /** The ServiceRegistry to look up the service. */
64      private final ServicesManager servicesManager;
65  
66      private final String defaultThemeId;
67  
68      private String pathPrefix = DEFAULT_PATH_PREFIX;
69  
70      /**
71       * The {@link RegisteredServiceThemeBasedViewResolver} constructor.
72       * @param defaultThemeId the theme to apply if the service doesn't specific one or a service is not provided
73       * @param servicesManager the serviceManager implementation
74       * @see #setCache(boolean)
75       */
76      public RegisteredServiceThemeBasedViewResolver(final String defaultThemeId, final ServicesManager servicesManager) {
77          super();
78          super.setCache(false);
79  
80          this.defaultThemeId = defaultThemeId;
81          this.servicesManager = servicesManager;
82      }
83  
84      /**
85       * Uses the viewName and the theme associated with the service.
86       * being requested and returns the appropriate view.
87       * @param viewName the name of the view to be resolved
88       * @return a theme-based UrlBasedView
89       * @throws Exception an exception
90       */
91      @Override
92      protected AbstractUrlBasedView buildView(final String viewName) throws Exception {
93          final RequestContext requestContext = RequestContextHolder.getRequestContext();
94          final WebApplicationService service = (WebApplicationService) WebUtils.getService(requestContext);
95          final RegisteredService registeredService = this.servicesManager.findServiceBy(service);
96  
97          final String themeId = service != null && registeredService != null && registeredService.isEnabled()
98                  && StringUtils.hasText(registeredService.getTheme()) ? registeredService.getTheme() : defaultThemeId;
99  
100         final String themePrefix = String.format("%s/%s/ui/", pathPrefix, themeId);
101         LOGGER.debug("Prefix {} set for service {} with theme {}", themePrefix, service, themeId);
102 
103         //Build up the view like the base classes do, but we need to forcefully set the prefix for each request.
104         //From UrlBasedViewResolver.buildView
105         final InternalResourceView view = (InternalResourceView) BeanUtils.instantiateClass(getViewClass());
106         view.setUrl(themePrefix + viewName + getSuffix());
107         final String contentType = getContentType();
108         if (contentType != null) {
109             view.setContentType(contentType);
110         }
111         view.setRequestContextAttribute(getRequestContextAttribute());
112         view.setAttributesMap(getAttributesMap());
113 
114         //From InternalResourceViewResolver.buildView
115         view.setAlwaysInclude(false);
116         view.setExposeContextBeansAsAttributes(false);
117         view.setPreventDispatchLoop(true);
118 
119         LOGGER.debug("View resolved: {}", view.getUrl());
120 
121         return view;
122     }
123 
124     /**
125      * setCache is not supported in the {@link RegisteredServiceThemeBasedViewResolver} because each
126      * request must be independently evaluated.
127      * @param cache a value indicating whether the view should cache results.
128      */
129     @Override
130     public void setCache(final boolean cache) {
131         LOGGER.warn("The {} does not support caching. Turned off caching forcefully.", this.getClass().getSimpleName());
132         super.setCache(false);
133     }
134 
135     /**
136      * Sets path prefix. This is the location where
137      * views are expected to be found. The default
138      * prefix is {@link #DEFAULT_PATH_PREFIX}.
139      *
140      * @param pathPrefix the path prefix
141      */
142     public void setPathPrefix(final String pathPrefix) {
143         this.pathPrefix = pathPrefix;
144     }
145 }
