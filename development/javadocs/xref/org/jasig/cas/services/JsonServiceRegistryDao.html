42  /**
43   * Implementation of <code>ServiceRegistryDao</code> that reads services definition from JSON
44   * configuration file at the Spring Application Context initialization time. JSON files are
45   * expected to be found inside a directory location and this DAO will recursively look through
46   * the directory structure to find relevant JSON files. Files are expected to have the
47   * {@value #FILE_EXTENSION} extension. An example of the JSON file is included here:
48   *
49   * <pre>
50   {
51       "@class" : "org.jasig.cas.services.RegexRegisteredService",
52       "id" : 103935657744185,
53       "description" : "This is the application description",
54       "serviceId" : "https://app.school.edu",
55       "name" : "testSaveAttributeReleasePolicyAllowedAttrRulesAndFilter",
56       "theme" : "testtheme",
57       "proxyPolicy" : {
58          "@class" : "org.jasig.cas.services.RegexMatchingRegisteredServiceProxyPolicy",
59          "pattern" : "https://.+"
60       },
61       "enabled" : true,
62       "ssoEnabled" : false,
63       "evaluationOrder" : 1000,
64       "usernameAttributeProvider" : {
65          "@class" : "org.jasig.cas.services.DefaultRegisteredServiceUsernameProvider"
66       },
67       "logoutType" : "BACK_CHANNEL",
68       "requiredHandlers" : [ "java.util.HashSet", [ "handler1", "handler2" ] ],
69       "attributeReleasePolicy" : {
70          "@class" : "org.jasig.cas.services.ReturnAllowedAttributeReleasePolicy",
71          "attributeFilter" : {
72              "@class" : "org.jasig.cas.services.support.RegisteredServiceRegexAttributeFilter",
73              "pattern" : "\\w+"
74          },
75          "allowedAttributes" : [ "java.util.ArrayList", [ "uid", "sn", "cn" ] ]
76       }
77   }
78   * </pre>
79   *
80   * @author Dmitriy Kopylenko
81   * @author Marvin S. Addison
82   * @author Misagh Moayyed
83   * @since 4.1
84   */
85  public class JsonServiceRegistryDao implements ServiceRegistryDao {
86      private static final Logger LOGGER = LoggerFactory.getLogger(JsonServiceRegistryDao.class);
87  
88      /**
89       * File extension of registered service JSON files.
90       */
91      private static final String FILE_EXTENSION = "json";
92  
93      /**
94       * Map of service ID to registered service.
95       */
96      private Map<Long, RegisteredService> serviceMap = new ConcurrentHashMap<Long, RegisteredService>();
97  
98      /**
99       * The Service registry directory.
100      */
101     private final File serviceRegistryDirectory;
102 
103     /**
104      * The Registered service json serializer.
105      */
106     private final JsonSerializer<RegisteredService> registeredServiceJsonSerializer;
107 
108     /**
109      * Instantiates a new Json service registry dao.
110      *
111      * @param configDirectory                 the config directory
112      * @param registeredServiceJsonSerializer the registered service json serializer
113      */
114     public JsonServiceRegistryDao(final File configDirectory, final JsonSerializer<RegisteredService> registeredServiceJsonSerializer) {
115         this.serviceRegistryDirectory = configDirectory;
116         Assert.isTrue(this.serviceRegistryDirectory.exists(), serviceRegistryDirectory + " does not exist");
117         Assert.isTrue(this.serviceRegistryDirectory.isDirectory(), serviceRegistryDirectory + " is not a directory");
118         this.registeredServiceJsonSerializer = registeredServiceJsonSerializer;
119     }
120 
121     /**
122      * Instantiates a new Json service registry dao.
123      * Sets the path to the directory where JSON service registry entries are
124      * stored. Uses the {@link RegisteredServiceJsonSerializer} by default.
125      *
126      * @param configDirectory the config directory where service registry files can be found.
127      */
128     public JsonServiceRegistryDao(final File configDirectory) {
129         this(configDirectory, new RegisteredServiceJsonSerializer());
130     }
131 
132     @Override
133     public final RegisteredService save(final RegisteredService service) {
134         if (service.getId() == RegisteredService.INITIAL_IDENTIFIER_VALUE && service instanceof AbstractRegisteredService) {
135             LOGGER.debug("Service id not set. Calculating id based on system time...");
136             ((AbstractRegisteredService) service).setId(System.nanoTime());
137         }
138         LockedOutputStream out = null;
139         try {
140             final File f = makeFile(service);
141             out = new LockedOutputStream(new FileOutputStream(f));
142             this.registeredServiceJsonSerializer.toJson(out, service);
143 
144             this.serviceMap.put(service.getId(), service);
145             LOGGER.debug("Saved service to [{}]", f.getCanonicalPath());
146         } catch (final IOException e) {
147             throw new RuntimeException("IO error opening file stream.", e);
148         } finally {
149             IOUtils.closeQuietly(out);
150         }
151 
152         return findServiceById(service.getId());
153     }
154 
155     @Override
156     public final synchronized boolean delete(final RegisteredService service) {
157         try {
158             final File f = makeFile(service);
159             final boolean result = f.delete();
160             if (!result) {
161                 LOGGER.warn("Failed to delete service definition file [{}]", f.getCanonicalPath());
162             } else {
163                 serviceMap.remove(service.getId());
164                 LOGGER.debug("Successfully deleted service definition file [{}]", f.getCanonicalPath());
165             }
166             return result;
167         } catch (final IOException e) {
168             throw new RuntimeException(e);
169         }
170     }
171 
172     @Override
173     public final synchronized List<RegisteredService> load() {
174         final Map<Long, RegisteredService> temp = new ConcurrentHashMap<Long, RegisteredService>();
175 
176         int errorCount = 0;
177 
178         final Collection<File> c = FileUtils.listFiles(this.serviceRegistryDirectory, new String[]{FILE_EXTENSION}, true);
179         for (final File file : c) {
180             BufferedInputStream in = null;
181             try {
182                 if (file.length() > 0) {
183                     in = new BufferedInputStream(new FileInputStream(file));
184                     final RegisteredService service = this.registeredServiceJsonSerializer.fromJson(in);
185 
186                     temp.put(service.getId(), service);
187                 }
188             } catch (final Exception e) {
189                 errorCount++;
190                 LOGGER.error("Error reading configuration file", e);
191             } finally {
192                 IOUtils.closeQuietly(in);
193             }
194         }
195         if (errorCount == 0) {
196             this.serviceMap = temp;
197         }
198 
199         return new ArrayList<RegisteredService>(this.serviceMap.values());
200     }
201 
202     @Override
203     public final RegisteredService findServiceById(final long id) {
204         return serviceMap.get(id);
205     }
206 
207     /**
208      * Creates a JSON file for a registered service.
209      * The file is named as <code>[SERVICE-NAME]-[SERVICE-ID]-.{@value #FILE_EXTENSION}</code>
210      *
211      * @param service Registered service.
212      * @return JSON file in service registry directory.
213      * @throws IllegalArgumentException if file name is invalid
214      */
215     protected File makeFile(final RegisteredService service) {
216         final String fileName = service.getName() + "-" + service.getId() + "." + FILE_EXTENSION;
217         final File svcFile = new File(serviceRegistryDirectory, fileName);
218         try {
219             final String path = svcFile.getCanonicalPath();
220             return svcFile;
221         } catch (final IOException e) {
222             LOGGER.warn("Service file name " + fileName + " is invalid; Examine for illegal characters in the name.", e);
223             throw new IllegalArgumentException(e);
224         }
225     }
