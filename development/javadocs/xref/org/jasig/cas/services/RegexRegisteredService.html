19  package org.jasig.cas.services;
20  
21  import org.jasig.cas.authentication.principal.Service;
22  
23  import javax.persistence.DiscriminatorValue;
24  import javax.persistence.Entity;
25  
26  import java.util.regex.Pattern;
27  
28  /**
29   * Mutable registered service that uses Java regular expressions for service matching.
30   * Matching is case insensitive, and is successful, if, and only if, the entire region
31   * sequence matches the pattern.
32   *
33   * @author Marvin S. Addison
34   * @author Misagh Moayyed
35   * @since 3.4
36   */
37  @Entity
38  @DiscriminatorValue("regex")
39  public class RegexRegisteredService extends AbstractRegisteredService {
40  
41      private static final long serialVersionUID = -8258660210826975771L;
42  
43      private transient Pattern servicePattern;
44  
45      @Override
46      public void setServiceId(final String id) {
47          serviceId = id;
48      }
49      
50      @Override
51      public boolean matches(final Service service) {
52          if (servicePattern == null) {
53              servicePattern = createPattern(serviceId);
54          }
55          return service != null && servicePattern.matcher(service.getId()).matches();
56      }
57  
58      @Override
59      protected AbstractRegisteredService newInstance() {
60          return new RegexRegisteredService();
61      }
62  
63      /**
64       * Creates the pattern. Matching is by default
65       * case insensitive.
66       *
67       * @param pattern the pattern, may not be null.
68       * @return the pattern
69       */
70      private Pattern createPattern(final String pattern) {
71          if (pattern == null) {
72              throw new IllegalArgumentException("Pattern cannot be null.");
73          }
74          
75          return Pattern.compile(pattern, Pattern.CASE_INSENSITIVE);
76      }
77  }
