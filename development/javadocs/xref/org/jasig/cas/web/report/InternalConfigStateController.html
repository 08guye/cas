20  package org.jasig.cas.web.report;
21  
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  import org.springframework.beans.factory.annotation.Autowired;
25  import org.springframework.beans.factory.annotation.Qualifier;
26  import org.springframework.context.ApplicationContext;
27  import org.springframework.stereotype.Controller;
28  import org.springframework.web.bind.annotation.RequestMapping;
29  import org.springframework.web.bind.annotation.RequestMethod;
30  import org.springframework.web.bind.annotation.ResponseBody;
31  import org.springframework.web.servlet.ModelAndView;
32  
33  import javax.servlet.http.HttpServletRequest;
34  import javax.servlet.http.HttpServletResponse;
35  import java.util.Map;
36  import java.util.Properties;
37  import java.util.Set;
38  
39  /**
40   * Controller that exposes the CAS internal state and beans
41   * as JSON. The report is available at <code>/status/config</code>.
42   * @author Misagh Moayyed
43   * @since 4.1
44   */
45  @Controller("internalConfigController")
46  public final class InternalConfigStateController {
47  
48      private static final Logger LOGGER = LoggerFactory.getLogger(InternalConfigStateController.class);
49  
50      private static final String VIEW_CONFIG = "monitoring/viewConfig";
51  
52      @Autowired
53      private ApplicationContext applicationContext;
54  
55      @Autowired(required = true)
56      @Qualifier("casProperties")
57      private Properties casProperties;
58  
59      /**
60       * Handle request.
61       *
62       * @param request the request
63       * @param response the response
64       * @return the model and view
65       * @throws Exception the exception
66       */
67      @RequestMapping(method = RequestMethod.GET, value="/status/config")
68      protected ModelAndView handleRequestInternal(
69              final HttpServletRequest request, final HttpServletResponse response)
70              throws Exception {
71  
72          return new ModelAndView(VIEW_CONFIG);
73      }
74  
75      /**
76       * Returns the current state of CAS properties.
77       * @return properties configured in CAS.
78       */
79      @RequestMapping(value = "/getProperties", method = RequestMethod.GET)
80      @ResponseBody
81      protected Set<Map.Entry<Object, Object>> getProperties() {
82          return casProperties.entrySet();
83      }
84  
85  }
