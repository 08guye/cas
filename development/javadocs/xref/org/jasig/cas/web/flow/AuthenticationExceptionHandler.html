19  package org.jasig.cas.web.flow;
20  
21  import org.jasig.cas.authentication.AuthenticationException;
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  import org.springframework.binding.message.MessageBuilder;
25  import org.springframework.binding.message.MessageContext;
26  import org.springframework.stereotype.Component;
27  
28  import javax.validation.constraints.NotNull;
29  import java.util.ArrayList;
30  import java.util.Collections;
31  import java.util.List;
32  
33  /**
34   * Performs two important error handling functions on an {@link AuthenticationException} raised from the authentication
35   * layer:
36   *
37   * <ol>
38   *     <li>Maps handler errors onto message bundle strings for display to user.</li>
39   *     <li>Determines the next webflow state by comparing handler erors against {@link #errors}
40   *     in list order. The first entry that matches determines the outcome state, which
41   *     is the simple class name of the exception.</li>
42   * </ol>
43   *
44   * @author Marvin S. Addison
45   * @since 4.0.0
46   */
47  @Component("authenticationExceptionHandler")
48  public class AuthenticationExceptionHandler {
49  
50      /** State name when no matching exception is found. */
51      private static final String UNKNOWN = "UNKNOWN";
52  
53      /** Default message bundle prefix. */
54      private static final String DEFAULT_MESSAGE_BUNDLE_PREFIX = "authenticationFailure.";
55  
56      /** Default list of errors this class knows how to handle. */
57      private static final List<Class<? extends Exception>> DEFAULT_ERROR_LIST =
58              new ArrayList<>();
59  
60      private final Logger logger = LoggerFactory.getLogger(this.getClass());
61      
62      static {
63          DEFAULT_ERROR_LIST.add(javax.security.auth.login.AccountLockedException.class);
64          DEFAULT_ERROR_LIST.add(javax.security.auth.login.FailedLoginException.class);
65          DEFAULT_ERROR_LIST.add(javax.security.auth.login.CredentialExpiredException.class);
66          DEFAULT_ERROR_LIST.add(javax.security.auth.login.AccountNotFoundException.class);
67          DEFAULT_ERROR_LIST.add(org.jasig.cas.authentication.AccountDisabledException.class);
68          DEFAULT_ERROR_LIST.add(org.jasig.cas.authentication.InvalidLoginLocationException.class);
69          DEFAULT_ERROR_LIST.add(org.jasig.cas.authentication.AccountPasswordMustChangeException.class);
70          DEFAULT_ERROR_LIST.add(org.jasig.cas.authentication.InvalidLoginTimeException.class);
71      }
72  
73      /** Ordered list of error classes that this class knows how to handle. */
74      @NotNull
75      private List<Class<? extends Exception>> errors = DEFAULT_ERROR_LIST;
76  
77      /** String appended to exception class name to create a message bundle key for that particular error. */
78      private String messageBundlePrefix = DEFAULT_MESSAGE_BUNDLE_PREFIX;
79  
80      /**
81       * Sets the list of errors that this class knows how to handle.
82       *
83       * @param errors List of errors in order of descending precedence.
84       */
85      public void setErrors(final List<Class<? extends Exception>> errors) {
86          this.errors = errors;
87      }
88  
89      public final List<Class<? extends Exception>> getErrors() {
90          return Collections.unmodifiableList(this.errors);
91      }
92      
93      /**
94       * Sets the message bundle prefix appended to exception class names to create a message bundle key for that
95       * particular error.
96       *
97       * @param prefix Prefix appended to exception names.
98       */
99      public void setMessageBundlePrefix(final String prefix) {
100         this.messageBundlePrefix = prefix;
101     }
102 
103     /**
104      * Maps an authentication exception onto a state name equal to the simple class name of the.
105      *
106      * @param e Authentication error to handle.
107      * @param messageContext the spring message context
108      * @return Name of next flow state to transition to or {@value #UNKNOWN}
109      * {@link org.jasig.cas.authentication.AuthenticationException#getHandlerErrors()} with highest precedence.
110      * Also sets an ERROR severity message in the message context of the form
111      * <code>[messageBundlePrefix][exceptionClassSimpleName]</code> for each handler error that is
112      * configured. If not match is found, {@value #UNKNOWN} is returned.
113      */
114     public String handle(final AuthenticationException e, final MessageContext messageContext) {
115         if (e != null) {
116             final MessageBuilder builder = new MessageBuilder();
117             for (final Class<? extends Exception> kind : this.errors) {
118                 for (final Class<? extends Exception> handlerError : e.getHandlerErrors().values()) {
119                     if (handlerError != null && handlerError.equals(kind)) {
120                         final String handlerErrorName = handlerError.getSimpleName();
121                         final String messageCode = this.messageBundlePrefix + handlerErrorName;
122                         messageContext.addMessage(builder.error().code(messageCode).build());
123                         return handlerErrorName;
124                     }
125                 }
126 
127             }
128         }
129         final String messageCode = this.messageBundlePrefix + UNKNOWN;
130         logger.trace("Unable to translate handler errors of the authentication exception {}. Returning {} by default...", e, messageCode);
131         messageContext.addMessage(new MessageBuilder().error().code(messageCode).build());
132         return UNKNOWN;
133     }
134 }
