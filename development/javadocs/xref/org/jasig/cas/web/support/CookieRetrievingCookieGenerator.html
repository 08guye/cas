19  package org.jasig.cas.web.support;
20  
21  import java.lang.reflect.Method;
22  
23  import javax.servlet.http.Cookie;
24  import javax.servlet.http.HttpServletRequest;
25  import javax.servlet.http.HttpServletResponse;
26  
27  import org.jasig.cas.authentication.RememberMeCredential;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  import org.springframework.util.ReflectionUtils;
31  import org.springframework.util.StringUtils;
32  import org.springframework.web.util.CookieGenerator;
33  
34  /**
35   * Extends CookieGenerator to allow you to retrieve a value from a request.
36   * The cookie is automatically marked as httpOnly, if the servlet container has support for it.
37   * 
38   * <p>
39   * Also has support for RememberMe Services
40   *
41   * @author Scott Battaglia
42   * @author Misagh Moayyed
43   * @since 3.1
44   *
45   */
46  public class CookieRetrievingCookieGenerator extends CookieGenerator {
47  
48      private final Logger logger = LoggerFactory.getLogger(this.getClass());
49      
50      /** The maximum age the cookie should be remembered for.
51       * The default is three months (7889231 in seconds, according to Google) */
52      private int rememberMeMaxAge = 7889231;
53  
54      public CookieRetrievingCookieGenerator() {
55          super();
56          final Method setHttpOnlyMethod = ReflectionUtils.findMethod(Cookie.class, "setHttpOnly", boolean.class);
57          if(setHttpOnlyMethod != null) {
58              super.setCookieHttpOnly(true);
59          } else {
60              logger.debug("Cookie cannot be marked as HttpOnly; container is not using servlet 3.0.");
61          }
62      }
63      
64      public void addCookie(final HttpServletRequest request, final HttpServletResponse response, final String cookieValue) {
65  
66          if (!StringUtils.hasText(request.getParameter(RememberMeCredential.REQUEST_PARAMETER_REMEMBER_ME))) {
67              super.addCookie(response, cookieValue);
68          } else {
69              final Cookie cookie = createCookie(cookieValue);
70              cookie.setMaxAge(this.rememberMeMaxAge);
71              if (isCookieSecure()) {
72                  cookie.setSecure(true);
73              }
74              response.addCookie(cookie);
75          }
76      }
77  
78      public String retrieveCookieValue(final HttpServletRequest request) {
79          final Cookie cookie = org.springframework.web.util.WebUtils.getCookie(
80                  request, getCookieName());
81  
82          return cookie == null ? null : cookie.getValue();
83      }
84  
85      public void setRememberMeMaxAge(final int maxAge) {
86          this.rememberMeMaxAge = maxAge;
87      }
88  }
