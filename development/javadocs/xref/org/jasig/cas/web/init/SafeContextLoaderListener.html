19  package org.jasig.cas.web.init;
20  
21  import javax.servlet.ServletContext;
22  import javax.servlet.ServletContextEvent;
23  import javax.servlet.ServletContextListener;
24  
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  import org.springframework.web.context.ContextLoaderListener;
28  
29  /**
30   * A {@link ServletContextListener} which wraps Spring's
31   * {@link ContextLoaderListener} and catches anything that delegate throws so as
32   * to prevent its having thrown a Throwable from aborting the initialization of
33   * our entire web application context. Use of this ContextListener will not be
34   * appropriate for all deployments of a web application. Sometimes, a context
35   * listener's aborting context initialization is exactly the desired behavior.
36   * This might be because the resulting application inavailability is acceptable
37   * or because another layer (Apache, a proxy, a balancer, etc.) detects the
38   * inavailability of the context and provides an appropriate user experience. Or
39   * because a root context handles all requests that aren't otherwise handled.
40   * There are many fine alternatives to this approach. However, when using the
41   * bare Tomcat container with CAS as the root context, if your desired behavior
42   * is that a failure at context initialization results in a dummy CAS context
43   * that presents a user-friendly "CAS is unavailable at this time" message,
44   * using this ContextListener in place of Spring's {@link ContextLoaderListener}
45   * should do the trick. The error page associated with this deployment failure
46   * is configured in the web.xml via the standard error handling mechanism.
47   * Rather than being a generic ContextListener wrapper that will make safe any
48   * ContextListener, instead this implementation specifically wraps and delegates
49   * to a Spring {@link ContextLoaderListener}. As such, mapping this listener in
50   * web.xml is a one for one replacement for {@link ContextLoaderListener}.
51   * <p>
52   * The exception thrown is exposed in the Servlet Context under the key
53   * "exceptionCaughtByListener".
54   *
55   * @author Andrew Petro
56  
57   * @see ContextLoaderListener
58   */
59  public final class SafeContextLoaderListener implements ServletContextListener {
60  
61      /** Instance of Commons Logging. */
62      private final Logger logger = LoggerFactory.getLogger(getClass());
63  
64      /**
65       * The name of the ServletContext attribute whereat we will place a List of
66       * Throwables that we caught from our delegate context listeners.
67       */
68      public static final String CAUGHT_THROWABLE_KEY = "exceptionCaughtByListener";
69  
70      /** The actual ContextLoaderListener to which we will delegate to. */
71      private final ContextLoaderListener delegate = new ContextLoaderListener();
72  
73      public void contextInitialized(final ServletContextEvent sce) {
74          try {
75              this.delegate.contextInitialized(sce);
76          } catch (final Throwable t) {
77              /*
78               * no matter what went wrong, our role is to capture this error and
79               * prevent it from blocking initialization of the context. logging
80               * overkill so that our deployer will find a record of this problem
81               * even if unfamiliar with Commons Logging and properly configuring
82               * it.
83               */
84  
85              final String message = "SafeContextLoaderListener: \n"
86                  + "The Spring ContextLoaderListener we wrap threw on contextInitialized.\n"
87                  + "But for our having caught this error, the web application context would not have initialized.";
88  
89              // logger it via Commons Logging
90              logger.error(message, t);
91  
92              // logger it to the ServletContext
93              final ServletContext context = sce.getServletContext();
94              context.log(message, t);
95  
96              /*
97               * record the error so that the application has access to later
98               * display a proper error message based on the exception.
99               */
100             context.setAttribute(CAUGHT_THROWABLE_KEY, t);
101         }
102     }
103 
104     public void contextDestroyed(final ServletContextEvent sce) {
105         this.delegate.contextDestroyed(sce);
106     }
107 
108 }
