19  package org.jasig.cas.web.view;
20  
21  import javax.servlet.http.HttpServletRequest;
22  import javax.servlet.http.HttpServletResponse;
23  import java.util.Map;
24  
25  /**
26   * Custom View to Return the CAS 1.0 Protocol Response. Implemented as a view
27   * class rather than a JSP (like CAS 2.0 spec) because of the requirement of the
28   * line feeds to be "\n".
29   *
30   * @author Scott Battaglia
31   * @since 3.0
32   */
33  public final class Cas10ResponseView extends AbstractCasView {
34  
35  
36  
37      @Override
38      protected void renderMergedOutputModel(final Map model,
39              final HttpServletRequest request, final HttpServletResponse response)
40                      throws Exception {
41  
42          if (this.successResponse) {
43              response.getWriter().print(
44                      "yes\n" + getPrimaryAuthenticationFrom(model).getPrincipal().getId() + "\n");
45          } else {
46              response.getWriter().print("no\n\n");
47          }
48      }
49  
50  
51  }
