102     /**
103      * Overrideable method to determine which credentials to use to grant a
104      * proxy granting ticket. Default is to use the pgtUrl.
105      *
106      * @param service the webapp service requesting proxy
107      * @param request the HttpServletRequest object.
108      * @return the credentials or null if there was an error or no credentials
109      * provided.
110      */
111     protected Credential getServiceCredentialsFromRequest(final WebApplicationService service, final HttpServletRequest request) {
112         final String pgtUrl = request.getParameter(CasProtocolConstants.PARAMETER_PROXY_CALLBACK_URL);
113         if (StringUtils.hasText(pgtUrl)) {
114             try {
115                 final RegisteredService registeredService = this.servicesManager.findServiceBy(service);
116                 verifyRegisteredServiceProperties(registeredService, service);
117                 return new HttpBasedServiceCredential(new URL(pgtUrl), registeredService);
118             } catch (final Exception e) {
119                 logger.error("Error constructing pgtUrl", e);
120             }
121         }
122 
123         return null;
124     }

136     @Override
137     protected final ModelAndView handleRequestInternal(final HttpServletRequest request, final HttpServletResponse response)
138             throws Exception {
139         final WebApplicationService service = this.argumentExtractor.extractService(request);
140         final String serviceTicketId = service != null ? service.getArtifactId() : null;
141 
142         if (service == null || serviceTicketId == null) {
143             logger.debug("Could not identify service and/or service ticket. Service: {}, Service ticket id: {}", service, serviceTicketId);
144             return generateErrorView(CasProtocolConstants.ERROR_CODE_INVALID_REQUEST,
145                     CasProtocolConstants.ERROR_CODE_INVALID_REQUEST, null);
146         }
147 
148         try {
149             final Credential serviceCredential = getServiceCredentialsFromRequest(service, request);
150             String proxyGrantingTicketId = null;
151             
152             if (serviceCredential != null) {
153                 try {
154                     proxyGrantingTicketId = this.centralAuthenticationService.delegateTicketGrantingTicket(serviceTicketId,
155                                 serviceCredential);
156                     logger.debug("Generated PGT [{}] off of service ticket [{}] and credential [{}]",
157                             proxyGrantingTicketId, serviceTicketId, serviceCredential);
158                 } catch (final AuthenticationException e) {
159                     logger.info("Failed to authenticate service credential {}", serviceCredential);
160                 } catch (final TicketException e) {
161                     logger.error("Failed to create proxy granting ticket for {}", serviceCredential, e);
162                 }
163                 
164                 if (StringUtils.isEmpty(proxyGrantingTicketId)) {
165                     return generateErrorView(CasProtocolConstants.ERROR_CODE_INVALID_PROXY_CALLBACK,
166                             CasProtocolConstants.ERROR_CODE_INVALID_PROXY_CALLBACK,
167                             new Object[] {serviceCredential.getId()});
168                 }
169             }
170 
171             final Assertion assertion = this.centralAuthenticationService.validateServiceTicket(serviceTicketId, service);
172 
173             final ValidationSpecification validationSpecification = this.getCommandClass();
174             final ServletRequestDataBinder binder = new ServletRequestDataBinder(validationSpecification, "validationSpecification");
175             initBinder(request, binder);
176             binder.bind(request);
177 
178             if (!validationSpecification.isSatisfiedBy(assertion)) {
179                 logger.debug("Service ticket [{}] does not satisfy validation specification.", serviceTicketId);
180                 return generateErrorView(CasProtocolConstants.ERROR_CODE_INVALID_TICKET,
181                         CasProtocolConstants.ERROR_CODE_INVALID_TICKET, null);
182             }
183 
184             String proxyIou = null;
185             if (serviceCredential != null && proxyGrantingTicketId != null && this.proxyHandler.canHandle(serviceCredential)) {
186                 proxyIou = this.proxyHandler.handle(serviceCredential, proxyGrantingTicketId);
187                 if (StringUtils.isEmpty(proxyIou)) {
188                     return generateErrorView(CasProtocolConstants.ERROR_CODE_INVALID_PROXY_CALLBACK,
189                             CasProtocolConstants.ERROR_CODE_INVALID_PROXY_CALLBACK,
190                             new Object[] {serviceCredential.getId()});
191                 }
192             }
193 
194             onSuccessfulValidation(serviceTicketId, assertion);
195             logger.debug("Successfully validated service ticket {} for service [{}]", serviceTicketId, service.getId());
196             return generateSuccessView(assertion, proxyIou);
197         } catch (final TicketValidationException e) {
198             return generateErrorView(e.getCode(), e.getCode(),
199                     new Object[] {serviceTicketId, e.getOriginalService().getId(), service.getId()});
200         } catch (final TicketException te) {
201             return generateErrorView(te.getCode(), te.getCode(),
202                 new Object[] {serviceTicketId});
203         } catch (final UnauthorizedProxyingException e) {
204             return generateErrorView(e.getMessage(), e.getMessage(), new Object[] {service.getId()});
205         } catch (final UnauthorizedServiceException e) {
206             return generateErrorView(e.getMessage(), e.getMessage(), null);
207         }
208     }

229     private ModelAndView generateErrorView(final String code, final String description, final Object[] args) {

238     /**
239      * Generate the success view. The result will contain the assertion and the proxy iou.
240      *
241      * @param assertion the assertion
242      * @param proxyIou the proxy iou
243      * @return the model and view, pointed to the view name set by {@link #setSuccessView(String)}
244      */
245     private ModelAndView generateSuccessView(final Assertion assertion, final String proxyIou) {
246         final Map<String, Object> attributes = assertion.getPrimaryAuthentication().getAttributes();
247         final Object o = attributes.get(RememberMeCredential.AUTHENTICATION_ATTRIBUTE_REMEMBER_ME);
248         final boolean isRemembered = (o == Boolean.TRUE && !assertion.isFromNewLogin());
249         
250         final ModelAndView success = new ModelAndView(this.successView);
251         success.addObject(VALIDATION_CAS_MODEL_ASSERTION, assertion);
252         success.addObject(CasProtocolConstants.VALIDATION_CAS_MODEL_PROXY_GRANTING_TICKET_IOU, proxyIou);
253         success.addObject(CasProtocolConstants.VALIDATION_REMEMBER_ME_ATTRIBUTE_NAME, isRemembered);
254         
255         final Map<String, ?> augmentedModelObjects = augmentSuccessViewModelObjects(assertion);
256         if (augmentedModelObjects != null) {
257             success.addAllObjects(augmentedModelObjects);
258         }
259         return success;
260     }

347     /**
348      * Ensure that the service is found and enabled in the service registry.
349      * @param registeredService the located entry in the registry
350      * @param service authenticating service
351      * @throws UnauthorizedServiceException
352      */
353     private void verifyRegisteredServiceProperties(final RegisteredService registeredService, final Service service) {
354         if (registeredService == null) {
355             final String msg = String.format("ServiceManagement: Unauthorized Service Access. "
356                     + "Service [%s] is not found in service registry.", service.getId());
357             logger.warn(msg);
358             throw new UnauthorizedServiceException(UnauthorizedServiceException.CODE_UNAUTHZ_SERVICE, msg);
359         }
360         if (!registeredService.isEnabled()) {
361             final String msg = String.format("ServiceManagement: Unauthorized Service Access. "
362                     + "Service %s] is not enabled in service registry.", service.getId());
363             
364             logger.warn(msg);
365             throw new UnauthorizedServiceException(UnauthorizedServiceException.CODE_UNAUTHZ_SERVICE, msg);
366         }
