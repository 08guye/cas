45  /**
46   * Default implementation of {@link LdapRegisteredServiceMapper} that is able
47   * to map ldap entries to {@link RegisteredService} instances based on
48   * certain attributes names. This implementation also respects the object class
49   * attribute of LDAP entries via {@link LdapUtils#OBJECTCLASS_ATTRIBUTE}.
50   * @author Misagh Moayyed
51   * @since 3.0.0
52   */

96      @Override
97      public LdapEntry mapFromRegisteredService(final String dn, final RegisteredService svc) {
98  
99          try {
100             if (svc.getId() == RegisteredService.INITIAL_IDENTIFIER_VALUE) {
101                 ((AbstractRegisteredService) svc).setId(System.nanoTime());
102             }
103             final String newDn = getDnForRegisteredService(dn, svc);
104             LOGGER.debug("Creating entry {}", newDn);
105 
106             final Collection<LdapAttribute> attrs = new ArrayList<LdapAttribute>();
107             attrs.add(new LdapAttribute(this.idAttribute, String.valueOf(svc.getId())));
108             attrs.add(new LdapAttribute(this.serviceIdAttribute, svc.getServiceId()));
109             attrs.add(new LdapAttribute(this.serviceNameAttribute, svc.getName()));
110             attrs.add(new LdapAttribute(this.serviceDescriptionAttribute, svc.getDescription()));
111             attrs.add(new LdapAttribute(this.serviceEnabledAttribute, Boolean.toString(svc.isEnabled()).toUpperCase()));
112             attrs.add(new LdapAttribute(this.serviceSsoEnabledAttribute, Boolean.toString(svc.isSsoEnabled()).toUpperCase()));
113             attrs.add(new LdapAttribute(this.evaluationOrderAttribute, String.valueOf(svc.getEvaluationOrder())));
114             attrs.add(new LdapAttribute(this.serviceThemeAttribute, svc.getTheme()));
115 
116             if (svc.getUsernameAttributeProvider() != null) {
117                 final byte[] data = SerializationUtils.serialize(svc.getUsernameAttributeProvider());
118                 final LdapAttribute attr = new LdapAttribute(this.usernameAttributeProvider, data);
119                 attrs.add(attr);
120             }
121             if (svc.getProxyPolicy() != null) {
122                 final byte[] data = SerializationUtils.serialize(svc.getProxyPolicy());
123                 final LdapAttribute attr = new LdapAttribute(this.serviceProxyPolicyAttribute, data);
124                 attrs.add(attr);
125             }
126             if (svc.getAttributeReleasePolicy() != null) {
127                 final byte[] data = SerializationUtils.serialize(svc.getAttributeReleasePolicy());
128                 final LdapAttribute attr = new LdapAttribute(this.attributeReleasePolicyAttribute, data);
129                 attrs.add(attr);
130             }
131 
132             if (svc.getRequiredHandlers().size() > 0) {
133                 attrs.add(new LdapAttribute(this.requiredHandlersAttribute, svc.getRequiredHandlers().toArray(new String[]{})));
134             }
135 
136 
137             attrs.add(new LdapAttribute(LdapUtils.OBJECTCLASS_ATTRIBUTE, "top", this.objectClass));
138 
139             return new LdapEntry(newDn, attrs);
140         } catch (final Exception e) {
141             throw new RuntimeException(e);
142         }
143     }

145     @Override
146     public RegisteredService mapToRegisteredService(final LdapEntry entry) {
147 
148         final LdapAttribute attr = entry.getAttribute(this.serviceIdAttribute);
149 
150         if (attr != null) {
151             final AbstractRegisteredService s = getRegisteredService(attr.getStringValue());
152 
153             if (s != null) {
154                 s.setId(LdapUtils.getLong(entry, this.idAttribute, Long.valueOf(entry.getDn().hashCode())));
155 
156                 s.setServiceId(LdapUtils.getString(entry, this.serviceIdAttribute));
157                 s.setName(LdapUtils.getString(entry, this.serviceNameAttribute));
158                 s.setDescription(LdapUtils.getString(entry, this.serviceDescriptionAttribute));
159                 s.setEnabled(LdapUtils.getBoolean(entry, this.serviceEnabledAttribute));
160                 s.setTheme(LdapUtils.getString(entry, this.serviceThemeAttribute));
161                 s.setEvaluationOrder(LdapUtils.getLong(entry, this.evaluationOrderAttribute).intValue());
162                 s.setSsoEnabled(LdapUtils.getBoolean(entry, this.serviceSsoEnabledAttribute));
163                 s.setRequiredHandlers(new HashSet<String>(getMultiValuedAttributeValues(entry, this.requiredHandlersAttribute)));
164                 
165                 final byte[] usernameAttrData = LdapUtils.getBinary(entry, this.usernameAttributeProvider);
166                 if (usernameAttrData != null && usernameAttrData.length > 0) {
167                     final RegisteredServiceUsernameAttributeProvider provider =
168                             (RegisteredServiceUsernameAttributeProvider) SerializationUtils.deserialize(usernameAttrData);
169                     s.setUsernameAttributeProvider(provider);
170                 }
171                 
172                 final byte[] data = LdapUtils.getBinary(entry, this.attributeReleasePolicyAttribute);
173                 if (data != null && data.length > 0) {
174                     final AttributeReleasePolicy policy = (AttributeReleasePolicy) SerializationUtils.deserialize(data);
175                     s.setAttributeReleasePolicy(policy);
176                 }
177                 
178                 final byte[] proxyData = LdapUtils.getBinary(entry, this.serviceProxyPolicyAttribute);
179                 if (proxyData != null && proxyData.length > 0) {
180                     final RegisteredServiceProxyPolicy policy = (RegisteredServiceProxyPolicy) SerializationUtils.deserialize(proxyData);
181                     s.setProxyPolicy(policy);
182                 }
183             }
184             return s;
185         }
186         return null;
187     }

258     @Override
259     public String getDnForRegisteredService(final String parentDn, final RegisteredService svc) {
260         return String.format("%s=%s,%s", this.idAttribute, svc.getId(), parentDn);
261     }

263     /**
264      * Checks if is valid regex pattern.
265      *
266      * @param pattern the pattern
267      * @return true, if  valid regex pattern
268      */
269     private boolean isValidRegexPattern(final String pattern) {
270         try {
271             Pattern.compile(pattern);
272         } catch (final PatternSyntaxException e) {
273             LOGGER.debug("Failed to identify [{}] as a regular expression", pattern);
274             return false;
275         }
276         return true;
277     }

279     /**
280      * Gets the attribute values if more than one, otherwise an empty list.
281      *
282      * @param entry the entry
283      * @param attrName the attr name
284      * @return the collection of attribute values
285      */
286     private Collection<String> getMultiValuedAttributeValues(@NotNull final LdapEntry entry, @NotNull final String attrName) {
287         final LdapAttribute attrs = entry.getAttribute(attrName);
288         if (attrs != null) {
289             return attrs.getStringValues();
290         }
291         return Collections.emptyList();
292     }
293 
294     /**
295      * Gets the registered service by id that would either match an ant or regex pattern.
296      *
297      * @param id the id
298      * @return the registered service
299      */
300     private AbstractRegisteredService getRegisteredService(@NotNull final String id) {
301         if (isValidRegexPattern(id)) {
302             return new RegexRegisteredService();
303         }
304 
305         if (new AntPathMatcher().isPattern(id)) {
306             return new RegisteredServiceImpl();
307         }
308         return null;
309     }
