39  /**
40   * Credential to principal resolver that extracts one or more attribute values
41   * from the certificate subject DN and combines them with intervening delimiters.
42   *
43   * @author Marvin S. Addison
44   * @since 3.4.4
45   *
46   */
47  public class X509SubjectPrincipalResolver extends AbstractX509PrincipalResolver {
48  
49      /** Pattern used to extract attribute names from descriptor. */
50      private static final Pattern ATTR_PATTERN = Pattern.compile("\\$(\\w+)");
51  
52      /** Descriptor representing an abstract format of the principal to be resolved.*/
53      @NotNull
54      private String descriptor;
55  
56      /**
57       * Sets the descriptor that describes for format of the principal ID to
58       * create from X.509 subject DN attributes.  The descriptor is made up of
59       * common X.509 attribute names prefixed by "$", which are replaced by
60       * attribute values extracted from DN attribute values.
61       * <p>
62       * EXAMPLE:
63       * </p>
64       * {@code
65       * <code>
66       * &lt;bean class="org.jasig.cas.adaptors.x509.authentication.principal.X509SubjectPrincipalResolver"
67       *   p:descriptor="$UID@$DC.$DC" /&gt;
68       * </code>
69       * }
70       *
71       * The above bean when applied to a certificate with the DN
72       * <p>
73       * <b>DC=edu, DC=vt/UID=jacky, CN=Jascarnella Ellagwonto</b></p>
74       * <p>
75       * produces the principal <strong>jacky@vt.edu</strong>.</p>
76       *
77       * @param s Descriptor string where attribute names are prefixed with "$"
78       * to identify replacement by real attribute values from the subject DN.
79       * Valid attributes include common X.509 DN attributes such as the following:
80       * <ul>
81       *  <li>C</li>
82       *  <li>CN</li>
83       *  <li>DC</li>
84       *  <li>EMAILADDRESS</li>
85       *  <li>L</li>
86       *  <li>O</li>
87       *  <li>OU</li>
88       *  <li>SERIALNUMBER</li>
89       *  <li>ST</li>
90       *  <li>UID</li>
91       *  <li>UNIQUEIDENTIFIER</li>
92       * </ul>
93       * For a complete list of supported attributes, see
94       * {@link org.cryptacular.x509.dn.StandardAttributeType}.
95       *
96       */
97      public void setDescriptor(final String s) {
98          this.descriptor = s;
99      }
100 
101     /**
102      * Replaces placeholders in the descriptor with values extracted from attribute
103      * values in relative distinguished name components of the DN.
104      *
105      * @param certificate X.509 certificate credential.
106      * @return Resolved principal ID.
107      * @see AbstractX509PrincipalResolver#resolvePrincipalInternal(java.security.cert.X509Certificate)
108      */
109     @Override
110     protected String resolvePrincipalInternal(final X509Certificate certificate) {
111         logger.debug("Resolving principal for {}", certificate);
112         final StringBuffer sb = new StringBuffer();
113         final Matcher m = ATTR_PATTERN.matcher(this.descriptor);
114         final Map<String, AttributeContext> attrMap = new HashMap<>();
115         final RDNSequence rdnSequence = new NameReader(certificate).readSubject();
116         String name;
117         String[] values;
118         AttributeContext context;
119         while (m.find()) {
120             name = m.group(1);
121             if (!attrMap.containsKey(name)) {
122                 values = getAttributeValues(rdnSequence,
123                         StandardAttributeType.fromName(name));
124                 attrMap.put(name, new AttributeContext(name, values));
125             }
126             context = attrMap.get(name);
127             m.appendReplacement(sb, context.nextValue());
128         }
129         m.appendTail(sb);
130         return sb.toString();
131     }    
132     
133     /**
134      * Gets the values of the given attribute contained in the DN.
135      * 
136      * <p><strong>NOTE:</strong> no escaping is done on special characters in the
137      * values, which could be different from what would appear in the string
138      * representation of the DN.</p>
139      *
140      * @param rdnSequence list of relative distinguished names
141      * that contains the attributes comprising the DN.
142      * @param attribute Attribute whose values will be retrieved.
143      * @return The attribute values for the given attribute in the order they
144      * appear would appear in the string representation of the DN or an empty
145      * array if the given attribute does not exist.
146      */
147     private static String[] getAttributeValues(final RDNSequence rdnSequence,
148             final AttributeType attribute) {
149         // Iterates sequence in reverse order as specified in section 2.1 of RFC 2253
150         final List<String> values = new ArrayList<String>();
151         for (final RDN rdn : rdnSequence.backward()) {
152             for (final Attribute attr : rdn.getAttributes()) {
153                 if (attr.getType().equals(attribute)) {
154                     values.add(attr.getValue());
155                 }
156             }
157         }
158         return values.toArray(new String[values.size()]);
159     }
160 
161 
162     private static final class AttributeContext {
163         private int currentIndex;
164         private String name;
165         private final String[] values;
166 
167         /**
168          * Instantiates a new attribute context.
169          *
170          * @param name the name
171          * @param values the values
172          */
173         AttributeContext(final String name, final String[] values) {
174             this.values = values;
175         }
176 
177   
178         /**
179          * Retrieve the next value, by incrementing the current index.
180          *
181          * @return the string
182          * @throws IllegalStateException if no values are remaining.
183          */
184         public String nextValue() {
185             if (this.currentIndex == this.values.length) {
186                 throw new IllegalStateException("No values remaining for attribute " + this.name);
187             }
188             return this.values[this.currentIndex++];
189         }
190     }
191 }
