20  package org.jasig.cas.adaptors.x509.authentication.handler.support.ldap;
21  
22  import java.util.HashMap;
23  import java.util.Map;
24  import javax.annotation.PreDestroy;
25  import javax.validation.constraints.NotNull;
26  import org.ldaptive.ConnectionConfig;
27  import org.ldaptive.ConnectionFactory;
28  import org.ldaptive.DefaultConnectionFactory;
29  import org.ldaptive.SearchExecutor;
30  import org.ldaptive.pool.BlockingConnectionPool;
31  import org.ldaptive.pool.ConnectionPool;
32  import org.ldaptive.pool.PoolConfig;
33  import org.ldaptive.pool.PooledConnectionFactory;
34  
35  /**
36   * Fetches a CRL from an LDAP instance.
37   * @author Daniel Fisher
38   * @since 4.1
39   */
40  public class PoolingLdaptiveResourceCRLFetcher extends LdaptiveResourceCRLFetcher {
41  
42      /** Connection pool template. */
43      protected final BlockingConnectionPool connectionPool;
44  
45      /** Map of connection pools. */
46      private final Map<String, PooledConnectionFactory> connectionPoolMap = new HashMap<>();
47  
48      /**
49       * Instantiates a new Ldap resource cRL fetcher.
50  
51       * @param connectionConfig the connection configuration
52       * @param searchExecutor the search executor
53       * @param connectionPool pooling configuration
54       */
55      public PoolingLdaptiveResourceCRLFetcher(
56              @NotNull final ConnectionConfig connectionConfig,
57              @NotNull final SearchExecutor searchExecutor,
58              @NotNull final BlockingConnectionPool connectionPool) {
59          super(connectionConfig, searchExecutor);
60          this.connectionPool = connectionPool;
61      }
62  
63      /**
64       * Close connection pull and shut down the executor.
65       */
66      @PreDestroy
67      public void destroy() {
68          logger.debug("Shutting down connection pools...");
69          for (final PooledConnectionFactory factory : connectionPoolMap.values()) {
70              factory.getConnectionPool().close();
71          }
72      }
73  
74      @Override
75      protected ConnectionFactory prepareConnectionFactory(final String ldapURL) {
76          final PooledConnectionFactory connectionFactory;
77          synchronized (connectionPoolMap) {
78              if (connectionPoolMap.containsKey(ldapURL)) {
79                  connectionFactory = connectionPoolMap.get(ldapURL);
80              } else {
81                  connectionFactory = new PooledConnectionFactory(newConnectionPool(ldapURL));
82                  connectionPoolMap.put(ldapURL, connectionFactory);
83              }
84          }
85          return connectionFactory;
86      }
87  
88  
89      /**
90       * Creates a new instance of a connection pool. Copied from {@link #connectionPool}.
91       *
92       * @param ldapURL to connect to
93       * @return connection pool
94       */
95      private ConnectionPool newConnectionPool(final String ldapURL) {
96          final BlockingConnectionPool pool = new BlockingConnectionPool(
97                  newPoolConfig(this.connectionPool.getPoolConfig()),
98                  (DefaultConnectionFactory) super.prepareConnectionFactory(ldapURL));
99          pool.setBlockWaitTime(this.connectionPool.getBlockWaitTime());
100         pool.setActivator(this.connectionPool.getActivator());
101         pool.setPassivator(this.connectionPool.getPassivator());
102         pool.setValidator(this.connectionPool.getValidator());
103         pool.setConnectOnCreate(this.connectionPool.getConnectOnCreate());
104         pool.setFailFastInitialize(this.connectionPool.getFailFastInitialize());
105         pool.setName(String.format("x509-crl-%s", ldapURL));
106         pool.setPruneStrategy(this.connectionPool.getPruneStrategy());
107         pool.initialize();
108         return pool;
109     }
110 
111     /**
112      * Creates a new instance of pool config.
113      *
114      * @param config to copy properties from
115      * @return pool config
116      */
117     private PoolConfig newPoolConfig(final PoolConfig config) {
118         final PoolConfig pc = new PoolConfig();
119         pc.setMinPoolSize(config.getMinPoolSize());
120         pc.setMaxPoolSize(config.getMaxPoolSize());
121         pc.setValidateOnCheckIn(config.isValidateOnCheckIn());
122         pc.setValidateOnCheckOut(config.isValidateOnCheckOut());
123         pc.setValidatePeriodically(config.isValidatePeriodically());
124         pc.setValidatePeriod(config.getValidatePeriod());
125         return pc;
126     }
127 }
