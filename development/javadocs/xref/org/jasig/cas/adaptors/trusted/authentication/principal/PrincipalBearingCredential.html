19  package org.jasig.cas.adaptors.trusted.authentication.principal;
20  
21  import org.jasig.cas.authentication.AbstractCredential;
22  import org.jasig.cas.authentication.principal.Principal;
23  import org.springframework.util.Assert;
24  
25  /**
26   * Credential that bear the fully resolved and authenticated Principal, or an
27   * indication that there is no such Principal. These Credential are a mechanism
28   * to pass into CAS information about an authentication and Principal resolution
29   * that has already happened in layers in front of CAS, e.g. by means of a Java
30   * Servlet Filter or by means of container authentication in the servlet
31   * container or Apache layers. DO NOT accept these Credential from arbitrary
32   * web-servicey calls to CAS. Rather, the code constructing these Credential
33   * must be trusted to perform appropriate authentication before issuing these
34   * credentials.
35   *
36   * @author Andrew Petro
37   * @since 3.0.5
38   */
39  public final class PrincipalBearingCredential extends AbstractCredential {
40  
41      /** Serialization version marker. */
42      private static final long serialVersionUID = 8866786438439775669L;
43  
44      /** The trusted principal. */
45      private final Principal principal;
46  
47      /**
48       * Instantiates a new principal bearing credential.
49       *
50       * @param principal the principal
51       */
52      public PrincipalBearingCredential(final Principal principal) {
53          Assert.notNull(principal, "principal cannot be null");
54          this.principal = principal;
55      }
56  
57      /**
58       * Get the previously authenticated Principal.
59       *
60       * @return authenticated Principal
61       */
62      public Principal getPrincipal() {
63          return this.principal;
64      }
65  
66      @Override
67      public String getId() {
68          return this.principal.getId();
69      }
70  
71  }
