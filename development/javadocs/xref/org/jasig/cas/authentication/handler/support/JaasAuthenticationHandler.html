40  /**
41   * JAAS Authentication Handler for CAAS. This is a simple bridge from CAS'
42   * authentication to JAAS.
43   *
44   * <p>
45   * Using the JAAS Authentication Handler requires you to configure the
46   * appropriate JAAS modules. You can specify the location of a jass.conf file
47   * using the following VM parameter:
48   * <pre>
49   * -Djava.security.auth.login.config=$PATH_TO_JAAS_CONF/jaas.conf
50   * </pre>
51   *
52   * <p>
53   * This example jaas.conf would try Kerberos based authentication, then try LDAP
54   * authentication:
55   * <pre>
56   * CAS {
57   *   com.sun.security.auth.module.Krb5LoginModule sufficient
58   *     client=TRUE
59   *     debug=FALSE
60   *     useTicketCache=FALSE;
61   *   edu.uconn.netid.jaas.LDAPLoginModule sufficient
62   *     java.naming.provider.url="ldap://ldapserver.my.edu:389/dc=my,dc=edu"
63   *     java.naming.security.principal="uid=jaasauth,dc=my,dc=edu"
64   *     java.naming.security.credentials="password"
65   *     Attribute="uid"
66   *     startTLS="true";
67   * };
68   * </pre>
69   *
70   * @author <a href="mailto:dotmatt@uconn.edu">Matthew J. Smith</a>
71   * @author Marvin S. Addison
72   * @author Misagh Moayyed
73   *
74   * @since 3.0.5
75   * @see javax.security.auth.callback.CallbackHandler
76   * @see javax.security.auth.callback.PasswordCallback
77   * @see javax.security.auth.callback.NameCallback
78   */

113     /** {@inheritDoc} */
114     @Override
115     protected final HandlerResult authenticateUsernamePasswordInternal(final UsernamePasswordCredential credential)
116             throws GeneralSecurityException, PreventedException {
117 
118         if (this.kerberosKdcSystemProperty != null) {
119             logger.debug("Setting kerberos system property {} to {}", SYS_PROP_KERB5_KDC, this.kerberosKdcSystemProperty);
120             System.setProperty(SYS_PROP_KERB5_KDC, this.kerberosKdcSystemProperty);
121         }
122         if (this.kerberosRealmSystemProperty != null) {
123             logger.debug("Setting kerberos system property {} to {}", SYS_PROP_KRB5_REALM, this.kerberosRealmSystemProperty);
124             System.setProperty(SYS_PROP_KRB5_REALM, this.kerberosRealmSystemProperty);
125         }
126         
127         final String username = credential.getUsername();
128         final String password = getPasswordEncoder().encode(credential.getPassword());
129         final LoginContext lc = new LoginContext(
130                 this.realm,
131                 new UsernamePasswordCallbackHandler(username, password));
132         try {
133             logger.debug("Attempting authentication for: {}", username);
134             lc.login();
135         } finally {
136             lc.logout();
137         }
138 
139         Principal principal = null;
140         final Set<java.security.Principal> principals = lc.getSubject().getPrincipals();
141         if (principals != null && principals.size() > 0) {
142             principal = new SimplePrincipal(principals.iterator().next().getName());
143         }
144         return createHandlerResult(credential, principal, null);
145     }

191     /**
192      * A simple JAAS CallbackHandler which accepts a Name String and Password
193      * String in the constructor. Only NameCallbacks and PasswordCallbacks are
194      * accepted in the callback array. This code based loosely on example given
195      * in Sun's javadoc for CallbackHandler interface.
196      */
197     protected static final class UsernamePasswordCallbackHandler implements CallbackHandler {
198 
199         /** The username of the principal we are trying to authenticate. */
200         private final String userName;
201 
202         /** The password of the principal we are trying to authenticate. */
203         private final String password;
204 
205         /**
206          * Constructor accepts name and password to be used for authentication.
207          *
208          * @param userName name to be used for authentication
209          * @param password Password to be used for authentication
210          */
211         protected UsernamePasswordCallbackHandler(final String userName,
212             final String password) {
213             this.userName = userName;
214             this.password = password;
215         }
216 
217         @Override
218         public void handle(final Callback[] callbacks)
219             throws UnsupportedCallbackException {
220             for (final Callback callback : callbacks) {
221                 if (callback.getClass().equals(NameCallback.class)) {
222                     ((NameCallback) callback).setName(this.userName);
223                 } else if (callback.getClass().equals(PasswordCallback.class)) {
224                     ((PasswordCallback) callback).setPassword(this.password
225                         .toCharArray());
226                 } else {
227                     throw new UnsupportedCallbackException(callback,
228                         "Unrecognized Callback");
229                 }
230             }
231         }
232     }
