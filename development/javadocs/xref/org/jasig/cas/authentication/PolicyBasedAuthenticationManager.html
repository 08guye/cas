40  /**
41   * Provides an authenticaiton manager that is inherently aware of multiple credentials and supports pluggable
42   * security policy via the {@link AuthenticationPolicy} component. The authentication process is as follows:
43   *
44   * <ul>
45   *   <li>For each given credential do the following:
46   *     <ul>
47   *       <li>Iterate over all configured authentication handlers.</li>
48   *       <li>Attempt to authenticate a credential if a handler supports it.</li>
49   *       <li>On success attempt to resolve a principal by doing the following:
50   *         <ul>
51   *           <li>Check whether a resolver is configured for the handler that authenticated the credential.</li>
52   *           <li>If a suitable resolver is found, attempt to resolve the principal.</li>
53   *           <li>If a suitable resolver is not found, use the principal resolved by the authentication handler.</li>
54   *         </ul>
55   *       </li>
56   *       <li>Check whether the security policy (e.g. any, all) is satisfied.
57   *         <ul>
58   *           <li>If security policy is met return immediately.</li>
59   *           <li>Continue if security policy is not met.</li>
60   *         </ul>
61   *       </li>
62   *     </ul>
63   *   </li>
64   *   <li>
65   *     After all credentials have been attempted check security policy again.
66   *     Note there is an implicit security policy that requires at least one credential to be authenticated.
67   *     Then the security policy given by {@link #setAuthenticationPolicy(AuthenticationPolicy)} is applied.
68   *     In all cases {@link AuthenticationException} is raised if security policy is not met.
69   *   </li>
70   * </ul>
71   *
72   * It is an error condition to fail to resolve a principal.
73   *
74   * @author Marvin S. Addison
75   * @since 4.0.0
76   */
77  public class PolicyBasedAuthenticationManager implements AuthenticationManager {
78  
79      /** Log instance for logging events, errors, warnings, etc. */
80      protected final Logger logger = LoggerFactory.getLogger(getClass());
81  
82      /** An array of AuthenticationAttributesPopulators. */
83      @NotNull
84      private List<AuthenticationMetaDataPopulator> authenticationMetaDataPopulators =
85              new ArrayList<>();
86  
87      /** Authentication security policy. */
88      @NotNull
89      private AuthenticationPolicy authenticationPolicy = new AnyAuthenticationPolicy();
90  
91      /** Map of authentication handlers to resolvers to be used when handler does not resolve a principal. */
92      @NotNull
93      private final Map<AuthenticationHandler, PrincipalResolver> handlerResolverMap;
94  
95  
96      /**
97       * Creates a new authentication manager with a varargs array of authentication handlers that are attempted in the
98       * listed order for supported credentials. This form may only be used by authentication handlers that
99       * resolve principals during the authentication process.
100      *
101      * @param handlers One or more authentication handlers.
102      */
103     public PolicyBasedAuthenticationManager(final AuthenticationHandler ... handlers) {
104         this(Arrays.asList(handlers));
105     }
106 
107     /**
108      * Creates a new authentication manager with a list of authentication handlers that are attempted in the
109      * listed order for supported credentials. This form may only be used by authentication handlers that
110      * resolve principals during the authentication process.
111      *
112      * @param handlers Non-null list of authentication handlers containing at least one entry.
113      */
114     public PolicyBasedAuthenticationManager(final List<AuthenticationHandler> handlers) {
115         Assert.notEmpty(handlers, "At least one authentication handler is required");
116         this.handlerResolverMap = new LinkedHashMap<>(
117                 handlers.size());
118         for (final AuthenticationHandler handler : handlers) {
119             this.handlerResolverMap.put(handler, null);
120         }
121     }
122 
123     /**
124      * Creates a new authentication manager with a map of authentication handlers to the principal resolvers that
125      * should be used upon successful authentication if no principal is resolved by the authentication handler. If
126      * the order of evaluation of authentication handlers is important, a map that preserves insertion order
127      * (e.g. {@link LinkedHashMap}) should be used.
128      *
129      * @param map Non-null map of authentication handler to principal resolver containing at least one entry.
130      */
131     public PolicyBasedAuthenticationManager(final Map<AuthenticationHandler, PrincipalResolver> map) {
132         Assert.notEmpty(map, "At least one authentication handler is required");
133         this.handlerResolverMap = map;
134     }
135 
136     /**
137      * {@inheritDoc}
138      */
139     @Override
140     @Audit(
141         action="AUTHENTICATION",
142         actionResolverName="AUTHENTICATION_RESOLVER",
143         resourceResolverName="AUTHENTICATION_RESOURCE_RESOLVER")
144     @Timed(name="AUTHENTICATE")
145     @Metered(name="AUTHENTICATE")
146     @Counted(name="AUTHENTICATE", monotonic=true)
147     public final Authentication authenticate(final Credential... credentials) throws AuthenticationException {
148 
149         final AuthenticationBuilder builder = authenticateInternal(credentials);
150         final Authentication authentication = builder.build();
151         final Principal principal = authentication.getPrincipal();
152         if (principal instanceof NullPrincipal) {
153             throw new UnresolvedPrincipalException(authentication);
154         }
155 
156         addAuthenticationMethodAttribute(builder, authentication);
157 
158         logger.info("Authenticated {} with credentials {}.", principal, Arrays.asList(credentials));
159         logger.debug("Attribute map for {}: {}", principal.getId(), principal.getAttributes());
160 
161         populateAuthenticationMetadataAttributes(builder, credentials);
162 
163         return builder.build();
164     }
165 
166     /**
167      * Populate authentication metadata attributes.
168      *
169      * @param builder the builder
170      * @param credentials the credentials
171      */
172     private void populateAuthenticationMetadataAttributes(final AuthenticationBuilder builder, final Credential[] credentials) {
173         for (final AuthenticationMetaDataPopulator populator : this.authenticationMetaDataPopulators) {
174             for (final Credential credential : credentials) {
175                 if (populator.supports(credential)) {
176                     populator.populateAttributes(builder, credential);
177                 }
178             }
179         }
180     }
181 
182     /**
183      * Add authentication method attribute.
184      *
185      * @param builder the builder
186      * @param authentication the authentication
187      */
188     private void addAuthenticationMethodAttribute(final AuthenticationBuilder builder, final Authentication authentication) {
189         for (final HandlerResult result : authentication.getSuccesses().values()) {
190             builder.addAttribute(AUTHENTICATION_METHOD_ATTRIBUTE, result.getHandlerName());
191         }
192     }

195      * Sets the authentication metadata populators that will be applied to every successful authentication event.
196      *
197      * @param populators Non-null list of metadata populators.
198      */
199     public final void setAuthenticationMetaDataPopulators(final List<AuthenticationMetaDataPopulator> populators) {
200         this.authenticationMetaDataPopulators = populators;
201     }

203     /**
204      * Sets the authentication policy used by this component.
205      *
206      * @param policy Non-null authentication policy. The default policy is {@link AnyAuthenticationPolicy}.
207      */
208     public void setAuthenticationPolicy(final AuthenticationPolicy policy) {
209         this.authenticationPolicy = policy;
210     }

212     /**
213      * Follows the same contract as {@link AuthenticationManager#authenticate(Credential...)}.
214      *
215      * @param credentials One or more credentials to authenticate.
216      *
217      * @return An authentication containing a resolved principal and metadata about successful and failed
218      * authentications. There SHOULD be a record of each attempted authentication, whether success or failure.
219      *
220      * @throws AuthenticationException When one or more credentials failed authentication such that security policy
221      * was not satisfied.
222      */
223     protected AuthenticationBuilder authenticateInternal(final Credential... credentials)
224             throws AuthenticationException {
225 
226         final AuthenticationBuilder builder = new DefaultAuthenticationBuilder(NullPrincipal.getInstance());
227         for (final Credential c : credentials) {
228             builder.addCredential(new BasicCredentialMetaData(c));
229         }
230         boolean found;
231         for (final Credential credential : credentials) {
232             found = false;
233             for (final Map.Entry<AuthenticationHandler, PrincipalResolver> entry : this.handlerResolverMap.entrySet()) {
234                 final AuthenticationHandler handler = entry.getKey();
235                 if (handler.supports(credential)) {
236                     found = true;
237                     try {
238                         authenticateAndResolvePrincipal(builder, credential, entry.getValue(), handler);
239                         if (this.authenticationPolicy.isSatisfiedBy(builder.build())) {
240                             return builder;
241                         }
242                     } catch (final GeneralSecurityException e) {
243                         logger.info("{} failed authenticating {}", handler.getName(), credential);
244                         logger.debug("{} exception details: {}", handler.getName(), e.getMessage());
245                         builder.addFailure(handler.getName(), e.getClass());
246                     } catch (final PreventedException e) {
247                         logger.error("{}: {}  (Details: {})", handler.getName(), e.getMessage(), e.getCause().getMessage());
248                         builder.addFailure(handler.getName(), e.getClass());
249                     }
250                 }
251             }
252             if (!found) {
253                 logger.warn(
254                         "Cannot find authentication handler that supports {}, which suggests a configuration problem.",
255                         credential);
256             }
257         }
258         evaluateProducedAuthenticationContext(builder);
259 
260         return builder;
261     }
262 
263     /**
264      * Evaluate produced authentication context.
265      *
266      * @param builder the builder
267      * @throws AuthenticationException the authentication exception
268      */
269     private void evaluateProducedAuthenticationContext(final AuthenticationBuilder builder) throws AuthenticationException {
270         // We apply an implicit security policy of at least one successful authentication
271         if (builder.getSuccesses().isEmpty()) {
272             throw new AuthenticationException(builder.getFailures(), builder.getSuccesses());
273         }
274         // Apply the configured security policy
275         if (!this.authenticationPolicy.isSatisfiedBy(builder.build())) {
276             throw new AuthenticationException(builder.getFailures(), builder.getSuccesses());
277         }
278     }
279 
280     /**
281      * Authenticate and resolve principal.
282      *
283      * @param builder the builder
284      * @param credential the credential
285      * @param resolver the resolver
286      * @param handler the handler
287      * @throws GeneralSecurityException the general security exception
288      * @throws PreventedException the prevented exception
289      */
290     private void authenticateAndResolvePrincipal(final AuthenticationBuilder builder, final Credential credential,
291                                                  final PrincipalResolver resolver, final AuthenticationHandler handler)
292             throws GeneralSecurityException, PreventedException {
293 
294         final Principal principal;
295         final HandlerResult result = handler.authenticate(credential);
296         builder.addSuccess(handler.getName(), result);
297         logger.info("{} successfully authenticated {}", handler.getName(), credential);
298         if (resolver == null) {
299             principal = result.getPrincipal();
300             logger.debug(
301                     "No resolver configured for {}. Falling back to handler principal {}",
302                     handler.getName(),
303                     principal);
304         } else {
305             principal = resolvePrincipal(handler.getName(), resolver, credential);
306         }
307         // Must avoid null principal since AuthenticationBuilder/ImmutableAuthentication
308         // require principal to be non-null
309         if (principal != null) {
310             builder.setPrincipal(principal);
311         }
312     }

315     /**
316      * Resolve principal.
317      *
318      * @param handlerName the handler name
319      * @param resolver the resolver
320      * @param credential the credential
321      * @return the principal
322      */
323     protected Principal resolvePrincipal(
324             final String handlerName, final PrincipalResolver resolver, final Credential credential) {
325         if (resolver.supports(credential)) {
326             try {
327                 final Principal p = resolver.resolve(credential);
328                 logger.debug("{} resolved {} from {}", resolver, p, credential);
329                 return p;
330             } catch (final Exception e) {
331                 logger.error("{} failed to resolve principal from {}", resolver, credential, e);
332             }
333         } else {
334             logger.warn(
335                     "{} is configured to use {} but it does not support {}, which suggests a configuration problem.",
336                     handlerName,
337                     resolver,
338                     credential);
339         }
340         return null;
341     }
