38  /**
39   * Provides an authenticaiton manager that is inherently aware of multiple credentials and supports pluggable
40   * security policy via the {@link AuthenticationPolicy} component. The authentication process is as follows:
41   *
42   * <ul>
43   *   <li>For each given credential do the following:</li>
44   *   <ul>
45   *     <li>Iterate over all configured authentication handlers.</li>
46   *     <li>Attempt to authenticate a credential if a handler supports it.</li>
47   *     <li>On success attempt to resolve a principal by doing the following:</li>
48   *     <ul>
49   *       <li>Check whether a resolver is configured for the handler that authenticated the credential.</li>
50   *       <li>If a suitable resolver is found, attempt to resolve the principal.</li>
51   *       <li>If a suitable resolver is not found, use the principal resolved by the authentication handler.</li>
52   *     </ul>
53   *     <li>Check whether the security policy (e.g. any, all) is satisfied.</li>
54   *     <ul>
55   *       <li>If security policy is met return immediately.</li>
56   *       <li>Continue if security policy is not met.</li>
57   *     </ul>
58   *   </ul>
59   *   <li>
60   *     After all credentials have been attempted check security policy again.
61   *     Note there is an implicit security policy that requires at least one credential to be authenticated.
62   *     Then the security policy given by {@link #setAuthenticationPolicy(AuthenticationPolicy)} is applied.
63   *     In all cases {@link AuthenticationException} is raised if security policy is not met.
64   *   </li>
65   * </ul>
66   *
67   * It is an error condition to fail to resolve a principal.
68   *
69   * @author Marvin S. Addison
70   * @since 4.0.0
71   */
72  public class PolicyBasedAuthenticationManager implements AuthenticationManager {
73  
74      /** Log instance for logging events, errors, warnings, etc. */
75      protected final Logger logger = LoggerFactory.getLogger(getClass());
76  
77      /** An array of AuthenticationAttributesPopulators. */
78      @NotNull
79      private List<AuthenticationMetaDataPopulator> authenticationMetaDataPopulators =
80              new ArrayList<AuthenticationMetaDataPopulator>();
81  
82      /** Authentication security policy. */
83      @NotNull
84      private AuthenticationPolicy authenticationPolicy = new AnyAuthenticationPolicy();
85  
86      /** Map of authentication handlers to resolvers to be used when handler does not resolve a principal. */
87      @NotNull
88      private final Map<AuthenticationHandler, PrincipalResolver> handlerResolverMap;
89  
90  
91      /**
92       * Creates a new authentication manager with a varargs array of authentication handlers that are attempted in the
93       * listed order for supported credentials. This form may only be used by authentication handlers that
94       * resolve principals during the authentication process.
95       *
96       * @param handlers One or more authentication handlers.
97       */
98      public PolicyBasedAuthenticationManager(final AuthenticationHandler ... handlers) {
99          this(Arrays.asList(handlers));
100     }
101 
102     /**
103      * Creates a new authentication manager with a list of authentication handlers that are attempted in the
104      * listed order for supported credentials. This form may only be used by authentication handlers that
105      * resolve principals during the authentication process.
106      *
107      * @param handlers Non-null list of authentication handlers containing at least one entry.
108      */
109     public PolicyBasedAuthenticationManager(final List<AuthenticationHandler> handlers) {
110         Assert.notEmpty(handlers, "At least one authentication handler is required");
111         this.handlerResolverMap = new LinkedHashMap<AuthenticationHandler, PrincipalResolver>(
112                 handlers.size());
113         for (final AuthenticationHandler handler : handlers) {
114             this.handlerResolverMap.put(handler, null);
115         }
116     }
117 
118     /**
119      * Creates a new authentication manager with a map of authentication handlers to the principal resolvers that
120      * should be used upon successful authentication if no principal is resolved by the authentication handler. If
121      * the order of evaluation of authentication handlers is important, a map that preserves insertion order
122      * (e.g. {@link LinkedHashMap}) should be used.
123      *
124      * @param map Non-null map of authentication handler to principal resolver containing at least one entry.
125      */
126     public PolicyBasedAuthenticationManager(final Map<AuthenticationHandler, PrincipalResolver> map) {
127         Assert.notEmpty(map, "At least one authentication handler is required");
128         this.handlerResolverMap = map;
129     }
130 
131     /**
132      * {@inheritDoc}
133      */
134     @Override
135     @Audit(
136         action="AUTHENTICATION",
137         actionResolverName="AUTHENTICATION_RESOLVER",
138         resourceResolverName="AUTHENTICATION_RESOURCE_RESOLVER")
139     @Profiled(tag = "AUTHENTICATE", logFailuresSeparately = false)
140     public final Authentication authenticate(final Credential... credentials) throws AuthenticationException {
141 
142         final AuthenticationBuilder builder = authenticateInternal(credentials);
143         final Authentication authentication = builder.build();
144         final Principal principal = authentication.getPrincipal();
145         if (principal  instanceof NullPrincipal) {
146             throw new UnresolvedPrincipalException(authentication);
147         }
148 
149         for (final HandlerResult result : authentication.getSuccesses().values()) {
150             builder.addAttribute(AUTHENTICATION_METHOD_ATTRIBUTE, result.getHandlerName());
151         }
152 
153         logger.info("Authenticated {} with credentials {}.", principal, Arrays.asList(credentials));
154         logger.debug("Attribute map for {}: {}", principal.getId(), principal.getAttributes());
155 
156         for (final AuthenticationMetaDataPopulator populator : this.authenticationMetaDataPopulators) {
157             for (final Credential credential : credentials) {
158                 if (populator.supports(credential)) {
159                     populator.populateAttributes(builder, credential);
160                 }
161             }
162         }
163 
164         return builder.build();
165     }
166 
167     /**
168      * Sets the authentication metadata populators that will be applied to every successful authentication event.
169      *
170      * @param populators Non-null list of metadata populators.
171      */
172     public final void setAuthenticationMetaDataPopulators(final List<AuthenticationMetaDataPopulator> populators) {
173         this.authenticationMetaDataPopulators = populators;
174     }
175 
176     /**
177      * Sets the authentication policy used by this component.
178      *
179      * @param policy Non-null authentication policy. The default policy is {@link AnyAuthenticationPolicy}.
180      */
181     public void setAuthenticationPolicy(final AuthenticationPolicy policy) {
182         this.authenticationPolicy = policy;
183     }
184 
185     /**
186      * Follows the same contract as {@link AuthenticationManager#authenticate(Credential...)}.
187      *
188      * @param credentials One or more credentials to authenticate.
189      *
190      * @return An authentication containing a resolved principal and metadata about successful and failed
191      * authentications. There SHOULD be a record of each attempted authentication, whether success or failure.
192      *
193      * @throws AuthenticationException When one or more credentials failed authentication such that security policy
194      * was not satisfied.
195      */
196     protected AuthenticationBuilder authenticateInternal(final Credential... credentials)
197             throws AuthenticationException {
198 
199         final AuthenticationBuilder builder = new AuthenticationBuilder(NullPrincipal.getInstance());
200         for (final Credential c : credentials) {
201             builder.addCredential(new BasicCredentialMetaData(c));
202         }
203         boolean found;
204         Principal principal;
205         PrincipalResolver resolver;
206         for (final Credential credential : credentials) {
207             found = false;
208             for (final Map.Entry<AuthenticationHandler, PrincipalResolver> entry : this.handlerResolverMap.entrySet()) {
209                 final AuthenticationHandler handler = entry.getKey();
210                 if (handler.supports(credential)) {
211                     found = true;
212                     try {
213                         final HandlerResult result = handler.authenticate(credential);
214                         builder.addSuccess(handler.getName(), result);
215                         logger.info("{} successfully authenticated {}", handler.getName(), credential);
216                         resolver = entry.getValue();
217                         if (resolver == null) {
218                             principal = result.getPrincipal();
219                             logger.debug(
220                                     "No resolver configured for {}. Falling back to handler principal {}",
221                                     handler.getName(),
222                                     principal);
223                         } else {
224                             principal = resolvePrincipal(handler.getName(), resolver, credential);
225                         }
226                         // Must avoid null principal since AuthenticationBuilder/ImmutableAuthentication
227                         // require principal to be non-null
228                         if (principal != null) {
229                             builder.setPrincipal(principal);
230                         }
231                         if (this.authenticationPolicy.isSatisfiedBy(builder.build())) {
232                             return builder;
233                         }
234                     } catch (final GeneralSecurityException e) {
235                         logger.info("{} failed authenticating {}", handler.getName(), credential);
236                         logger.debug("{} exception details: {}", handler.getName(), e.getMessage());
237                         builder.addFailure(handler.getName(), e.getClass());
238                     } catch (final PreventedException e) {
239                         logger.error("{}: {}  (Details: {})", handler.getName(), e.getMessage(), e.getCause().getMessage());
240                         builder.addFailure(handler.getName(), e.getClass());
241                     }
242                 }
243             }
244             if (!found) {
245                 logger.warn(
246                         "Cannot find authentication handler that supports {}, which suggests a configuration problem.",
247                         credential);
248             }
249         }
250         // We apply an implicit security policy of at least one successful authentication
251         if (builder.getSuccesses().isEmpty()) {
252             throw new AuthenticationException(builder.getFailures(), builder.getSuccesses());
253         }
254         // Apply the configured security policy
255         if (!this.authenticationPolicy.isSatisfiedBy(builder.build())) {
256             throw new AuthenticationException(builder.getFailures(), builder.getSuccesses());
257         }
258         return builder;
259     }
260 
261       
262     /**
263      * Resolve principal.
264      *
265      * @param handlerName the handler name
266      * @param resolver the resolver
267      * @param credential the credential
268      * @return the principal
269      */
270     protected Principal resolvePrincipal(
271             final String handlerName, final PrincipalResolver resolver, final Credential credential) {
272         if (resolver.supports(credential)) {
273             try {
274                 final Principal p = resolver.resolve(credential);
275                 logger.debug("{} resolved {} from {}", resolver, p, credential);
276                 return p;
277             } catch (final Exception e) {
278                 logger.error("{} failed to resolve principal from {}", resolver, credential, e);
279             }
280         } else {
281             logger.warn(
282                     "{} is configured to use {} but it does not support {}, which suggests a configuration problem.",
283                     handlerName,
284                     resolver,
285                     credential);
286         }
287         return null;
288     }
