31  /**
32   * Immutable authentication event whose attributes may not change after creation.
33   * This class is designed for serialization and is suitable for long-term storage.

41  public final class ImmutableAuthentication implements Authentication {

43      /** UID for serializing. */

46      /** Authentication date stamp. */
47      private final long authenticationDate;

49      /** List of metadata about credentials presented at authentication. */
50      private final List<CredentialMetaData> credentials;
51  
52      /** Authenticated principal. */
53      private final Principal principal;
54  
55      /** Authentication metadata attributes. */

65      private ImmutableAuthentication() {
66          this.authenticationDate = 0;
67          this.credentials = null;

74      /**
75       * Creates a new instance with the given data.
76       *
77       * @param date Non-null authentication date.
78       * @param credentials Non-null list of credential metadata containing at least one entry.
79       * @param principal Non-null authenticated principal.
80       * @param attributes Nullable map of authentication metadata.
81       * @param successes Non-null map of authentication successes containing at least one entry.
82       * @param failures Nullable map of authentication failures.
83       */
84      public ImmutableAuthentication(
85              final Date date,
86              final List<CredentialMetaData> credentials,
87              final Principal principal,
88              final Map<String, Object> attributes,
89              final Map<String, HandlerResult> successes,
90              final Map<String, Class<? extends Exception>> failures) {
91  
92          Assert.notNull(date, "Date cannot be null");
93          Assert.notNull(credentials, "Credential cannot be null");
94          Assert.notNull(principal, "Principal cannot be null");
95          Assert.notNull(successes, "Successes cannot be null");
96          Assert.notEmpty(credentials, "Credential cannot be empty");
97          Assert.notEmpty(successes, "Successes cannot be empty");
98  
99          this.authenticationDate = date.getTime();
100         this.credentials = credentials;
101         this.principal = principal;
102         this.attributes = attributes.isEmpty() ? null : attributes;
103         this.successes = successes;
104         this.failures = failures.isEmpty() ? null : failures;
105     }

167     /**
168      * Wraps a possibly null map in an immutable wrapper.
169      *
170      * @param <K> the key type
171      * @param <V> the value type
172      * @param source Nullable map to wrap.
173      * @return {@link Collections#unmodifiableMap(java.util.Map)} if given map is not null, otherwise
174      * {@link java.util.Collections#emptyMap()}.
175      */
176     private static <K, V> Map<K, V> wrap(final Map<K, V> source) {
177         if (source != null) {
178             return Collections.unmodifiableMap(source);
179         }
180         return Collections.emptyMap();
181     }
