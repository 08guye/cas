3   import java.security.InvalidKeyException;
4   import java.security.NoSuchAlgorithmException;

6   import javax.crypto.Mac;
7   import javax.crypto.spec.SecretKeySpec;
8   
9   public class Util {
10      public static String hmacSign(String skey, String data)
11              throws NoSuchAlgorithmException, InvalidKeyException {
12          SecretKeySpec key = new SecretKeySpec(skey.getBytes(), "HmacSHA1");
13          Mac mac = Mac.getInstance("HmacSHA1");
14          mac.init(key);
15          byte[] raw = mac.doFinal(data.getBytes());
16          return bytesToHex(raw);
17      }
18  
19      public static String bytesToHex(byte[] b) {
20          String result = "";
21          for (int i = 0; i < b.length; i++) {
22              result += Integer.toString((b[i] & 0xff) + 0x100, 16).substring(1);
23          }
24          return result;
25      }
26  }
