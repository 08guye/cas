297     /**
298      * Returns one of the _SOMETHING_ALPHABET byte arrays depending on the
299      * options specified. It's possible, though silly, to specify ORDERED
300      * <b>and</b> URLSAFE in which case one of them will be picked, though there

305             return _URL_SAFE_ALPHABET;
306         } else if ((options & ORDERED) == ORDERED) {

358     /**
359      * <p>
360      * Encodes up to three bytes of the array <var>source</var> and writes the
361      * resulting four Base64 bytes to <var>destination</var>. The source and
362      * destination arrays can be manipulated anywhere along their length by
363      * specifying <var>srcOffset</var> and <var>destOffset</var>. This method
364      * does not check to make sure your arrays are large enough to accomodate
365      * <var>srcOffset</var> + 3 for the <var>source</var> array or
366      * <var>destOffset</var> + 4 for the <var>destination</var> array. The
367      * actual number of significant bytes in your array is given by
368      * <var>numSigBytes</var>.
369      * </p>
370      * <p>
371      * This is the lowest level of the encoding methods with all possible
372      * parameters.
373      * </p>
374      *
375      * @param source
376      *            the array to convert
377      * @param srcOffset
378      *            the index where conversion begins
379      * @param numSigBytes
380      *            the number of significant bytes in your array
381      * @param destination
382      *            the array to hold the conversion
383      * @param destOffset
384      *            the index where output will be put
385      * @return the <var>destination</var> array
386      * @since 1.3
387      */
388     private static byte[] encode3to4(byte[] source, int srcOffset,
389             int numSigBytes, byte[] destination, int destOffset, int options) {
390 
391         byte[] ALPHABET = getAlphabet(options);
392 
393         // 1 2 3
394         // 01234567890123456789012345678901 Bit position
395         // --------000000001111111122222222 Array position from threeBytes
396         // --------| || || || | Six bit groups to index ALPHABET
397         // >>18 >>12 >> 6 >> 0 Right shift necessary
398         // 0x3f 0x3f 0x3f Additional AND
399 
400         // Create buffer with zero-padding if there are only one or two
401         // significant bytes passed in the array.
402         // We have to shift left 24 in order to flush out the 1's that appear
403         // when Java treats a value as negative that is cast from a byte to an
404         // int.
405         int inBuff = (numSigBytes > 0 ? ((source[srcOffset] << 24) >>> 8) : 0)
406                 | (numSigBytes > 1 ? ((source[srcOffset + 1] << 24) >>> 16) : 0)
407                 | (numSigBytes > 2 ? ((source[srcOffset + 2] << 24) >>> 24) : 0);
408 
409         switch (numSigBytes) {
410         case 3:
411             destination[destOffset] = ALPHABET[(inBuff >>> 18)];
412             destination[destOffset + 1] = ALPHABET[(inBuff >>> 12) & 0x3f];
413             destination[destOffset + 2] = ALPHABET[(inBuff >>> 6) & 0x3f];
414             destination[destOffset + 3] = ALPHABET[(inBuff) & 0x3f];
415             return destination;
416 
417         case 2:
418             destination[destOffset] = ALPHABET[(inBuff >>> 18)];
419             destination[destOffset + 1] = ALPHABET[(inBuff >>> 12) & 0x3f];
420             destination[destOffset + 2] = ALPHABET[(inBuff >>> 6) & 0x3f];
421             destination[destOffset + 3] = EQUALS_SIGN;
422             return destination;
423 
424         case 1:
425             destination[destOffset] = ALPHABET[(inBuff >>> 18)];
426             destination[destOffset + 1] = ALPHABET[(inBuff >>> 12) & 0x3f];
427             destination[destOffset + 2] = EQUALS_SIGN;
428             destination[destOffset + 3] = EQUALS_SIGN;
429             return destination;
430 
431         default:
432             return destination;
433         } // end switch
434     } // end encode3to4

854             java.util.zip.GZIPOutputStream gzos = null;

860                 b64os = new Base64.OutputStream(baos, ENCODE | options);

882                 } catch (Exception e) {

890         else {
891             boolean breakLines = (options & DO_BREAK_LINES) != 0;
892

1090      * array. <strong>Ignores GUNZIP option, if it's set.</strong> This is not
1091      * generally a recommended method, although it is used internally as part of
1092      * the decoding process. Special case: if len = 0, an empty array is
1093      * returned. Still, if you need more speed and reduced memory footprint (and
1094      * aren't gzipping), consider this method.
1095      *
1096      * @param source
1097      *            The Base64 encoded data
1098      * @param off
1099      *            The offset of where to begin decoding
1100      * @param len
1101      *            The length of characters to decode
1102      * @param options
1103      *            Can specify options such as alphabet type to use
1104      * @return decoded data
1105      * @throws java.io.IOException
1106      *             If bogus characters exist in source data
1107      * @since 1.3
1108      */
1109     public static byte[] decode(byte[] source, int off, int len, int options)
1110             throws java.io.IOException {
1111 
1112         // Lots of error checking and exception throwing
1113         if (source == null) {
1114             throw new NullPointerException("Cannot decode null source array.");
1115         } // end if
1116         if (off < 0 || off + len > source.length) {
1117             throw new IllegalArgumentException(
1118                     String
1119                             .format(
1120                                     "Source array with length %d cannot have offset of %d and process %d bytes.",
1121                                     source.length, off, len));
1122         } // end if
1123 
1124         if (len == 0) {
1125             return new byte[0];
1126         } else if (len < 4) {
1127             throw new IllegalArgumentException(
1128                     "Base64-encoded string must have at least four characters, but length specified was "
1129                             + len);
1130         } // end if
1131 
1132         byte[] DECODABET = getDecodabet(options);
1133 
1134         int len34 = len * 3 / 4; // Estimate on array size
1135         byte[] outBuff = new byte[len34]; // Upper limit on size of output
1136         int outBuffPosn = 0; // Keep track of where we're writing
1137 
1138         byte[] b4 = new byte[4]; // Four byte buffer from source, eliminating
1139                                     // white space
1140         int b4Posn = 0; // Keep track of four byte input buffer
1141         int i = 0; // Source array counter
1142         byte sbiDecode = 0; // Special value from DECODABET
1143 
1144         for (i = off; i < off + len; i++) { // Loop through source
1145 
1146             sbiDecode = DECODABET[source[i] & 0xFF];
1147 
1148             // White space, Equals sign, or legit Base64 character
1149             // Note the values such as -5 and -9 in the
1150             // DECODABETs at the top of the file.
1151             if (sbiDecode >= WHITE_SPACE_ENC) {
1152                 if (sbiDecode >= EQUALS_SIGN_ENC) {
1153                     b4[b4Posn++] = source[i]; // Save non-whitespace
1154                     if (b4Posn > 3) { // Time to decode?
1155                         outBuffPosn += decode4to3(b4, 0, outBuff, outBuffPosn,
1156                                 options);
1157                         b4Posn = 0;
1158 
1159                         // If that was the equals sign, break out of 'for' loop
1160                         if (source[i] == EQUALS_SIGN) {
1161                             break;
1162                         } // end if: equals sign
1163                     } // end if: quartet built
1164                 } // end if: equals sign or better
1165             } // end if: white space, equals sign or better
1166             else {
1167                 // There's a bad input character in the Base64 stream.
1168                 throw new java.io.IOException(
1169                         String
1170                                 .format(
1171                                         "Bad Base64 input character decimal %d in array position %d",
1172                                         ((int) source[i]) & 0xFF, i));
1173             } // end else:
1174         } // each input character
1175 
1176         byte[] out = new byte[outBuffPosn];
1177         System.arraycopy(outBuff, 0, out, 0, outBuffPosn);
1178         return out;
1179     } // end decode

1196     /**
1197      * Decodes data from Base64 notation, automatically detecting
1198      * gzip-compressed data and decompressing it.
1199      *
1200      * @param s
1201      *            the string to decode
1202      * @param options
1203      *            encode options such as URL_SAFE
1204      * @return the decoded data
1205      * @throws java.io.IOException
1206      *             if there is an error
1207      * @throws NullPointerException
1208      *             if <tt>s</tt> is null
1209      * @since 1.4
1210      */
1211     public static byte[] decode(String s, int options)
1212             throws java.io.IOException {
1213 
1214         if (s == null) {
1215             throw new NullPointerException("Input string was null.");
1216         } // end if
1217 
1218         byte[] bytes;
1219         try {
1220             bytes = s.getBytes(PREFERRED_ENCODING);
1221         } // end try
1222         catch (java.io.UnsupportedEncodingException uee) {
1223             bytes = s.getBytes();
1224         } // end catch
1225         // </change>
1226 
1227         // Decode
1228         bytes = decode(bytes, 0, bytes.length, options);
1229 
1230         // Check to see if it's gzip-compressed
1231         // GZIP Magic Two-Byte Number: 0x8b1f (35615)
1232         boolean dontGunzip = (options & DONT_GUNZIP) != 0;
1233         if ((bytes != null) && (bytes.length >= 4) && (!dontGunzip)) {
1234 
1235             int head = ((int) bytes[0] & 0xff) | ((bytes[1] << 8) & 0xff00);
1236             if (java.util.zip.GZIPInputStream.GZIP_MAGIC == head) {
1237                 java.io.ByteArrayInputStream bais = null;
1238                 java.util.zip.GZIPInputStream gzis = null;
1239                 java.io.ByteArrayOutputStream baos = null;
1240                 byte[] buffer = new byte[2048];
1241                 int length = 0;
1242 
1243                 try {
1244                     baos = new java.io.ByteArrayOutputStream();
1245                     bais = new java.io.ByteArrayInputStream(bytes);
1246                     gzis = new java.util.zip.GZIPInputStream(bais);
1247 
1248                     while ((length = gzis.read(buffer)) >= 0) {
1249                         baos.write(buffer, 0, length);
1250                     } // end while: reading input
1251 
1252                     // No error? Get new bytes.
1253                     bytes = baos.toByteArray();
1254 
1255                 } // end try
1256                 catch (java.io.IOException e) {
1257                     e.printStackTrace();
1258                     // Just return originally-decoded bytes
1259                 } // end catch
1260                 finally {
1261                     try {
1262                         baos.close();
1263                     } catch (Exception e) {
1264                     }
1265                     try {
1266                         gzis.close();
1267                     } catch (Exception e) {
1268                     }
1269                     try {
1270                         bais.close();
1271                     } catch (Exception e) {
1272                     }
1273                 } // end finally
1274 
1275             } // end if: gzipped
1276         } // end if: bytes.length >= 2
1277 
1278         return bytes;
1279     } // end decode

1318          * mode.

1321          *
1322          * <pre>

1354         /**
1355          * Writes the byte to the output stream after converting to/from Base64
1356          * notation. When encoding, bytes are buffered three at a time before
1357          * the output stream actually gets a write() call. When decoding, bytes
1358          * are buffered four at a time.
1359          *
1360          * @param theByte
1361          *            the byte to write
1362          * @since 1.3
1363          */
1364         @Override
1365         public void write(int theByte) throws java.io.IOException {
1366             // Encoding suspended?
1367             if (suspendEncoding) {
1368                 this.out.write(theByte);
1369                 return;
1370             } // end if: supsended
1371 
1372             // Encode?
1373             if (encode) {
1374                 buffer[position++] = (byte) theByte;
1375                 if (position >= bufferLength) { // Enough to encode.
1376 
1377                     this.out
1378                             .write(encode3to4(b4, buffer, bufferLength, options));
1379 
1380                     lineLength += 4;
1381                     if (breakLines && lineLength >= MAX_LINE_LENGTH) {
1382                         this.out.write(NEW_LINE);
1383                         lineLength = 0;
1384                     } // end if: end of line
1385 
1386                     position = 0;
1387                 } // end if: enough to output
1388             } // end if: encoding
1389 
1390             // Else, Decoding
1391             else {
1392                 // Meaningful Base64 character?
1393                 if (decodabet[theByte & 0x7f] > WHITE_SPACE_ENC) {
1394                     buffer[position++] = (byte) theByte;
1395                     if (position >= bufferLength) { // Enough to output.
1396 
1397                         int len = Base64.decode4to3(buffer, 0, b4, 0, options);
1398                         out.write(b4, 0, len);
1399                         position = 0;
1400                     } // end if: enough to output
1401                 } // end if: meaningful base64 character
1402                 else if (decodabet[theByte & 0x7f] != WHITE_SPACE_ENC) {
1403                     throw new java.io.IOException(
1404                             "Invalid character in Base64 data.");
1405                 } // end else: not white space either
1406             } // end else: decoding
1407         } // end write
