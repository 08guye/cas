19  package org.jasig.cas.support.openid.web.mvc;
20  
21  import org.jasig.cas.web.DelegateController;
22  import org.openid4java.message.Message;
23  import org.openid4java.message.ParameterList;
24  import org.openid4java.server.ServerManager;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  import org.springframework.web.servlet.ModelAndView;
28  
29  import javax.servlet.http.HttpServletRequest;
30  import javax.servlet.http.HttpServletResponse;
31  import javax.validation.constraints.NotNull;
32  import java.io.Serializable;
33  import java.util.HashMap;
34  import java.util.Map;
35  
36  /**
37   * Generates an association to an openid association request.
38   * @author Frederic Esnault
39   * @since 3.5
40   */
41  public class SmartOpenIdController extends DelegateController implements Serializable {
42  
43      private static final long serialVersionUID = -594058549445950430L;
44  
45      private final Logger logger = LoggerFactory.getLogger(SmartOpenIdController.class);
46  
47      private ServerManager serverManager;
48  
49      /** View if association Fails. */
50      private static final String DEFAULT_ASSOCIATION_FAILURE_VIEW_NAME = "casOpenIdAssociationFailureView";
51  
52      /** View if association Succeeds. */
53      private static final String DEFAULT_ASSOCIATION_SUCCESS_VIEW_NAME = "casOpenIdAssociationSuccessView";
54  
55      /** The view to redirect to on a successful validation. */
56      @NotNull
57      private String successView = DEFAULT_ASSOCIATION_SUCCESS_VIEW_NAME;
58  
59      /** The view to redirect to on a validation failure. Not used for now,
60       *  the succes view may return failed association attemps. No need for another view. */
61      @NotNull
62      private String failureView = DEFAULT_ASSOCIATION_FAILURE_VIEW_NAME;
63  
64      /**
65       * Gets the association response. Determines the mode first.
66       * If mode is set to associate, will set the response. Then
67       * builds the response parameters next and returns.
68       *
69       * @param request the request
70       * @return the association response
71       */
72      public Map<String, String> getAssociationResponse(final HttpServletRequest request) {
73          final ParameterList parameters = new ParameterList(request.getParameterMap());
74  
75          final String mode = parameters.hasParameter("openid.mode")
76                  ? parameters.getParameterValue("openid.mode")
77                  : null;
78  
79          Message response = null;
80          if (mode != null && mode.equals("associate")) {
81              response = serverManager.associationResponse(parameters);
82          }
83          final Map<String, String> responseParams = new HashMap<String, String>();
84          if (response != null) {
85              responseParams.putAll(response.getParameterMap());
86          }
87  
88          return responseParams;
89  
90      }
91  
92      @Override
93      protected ModelAndView handleRequestInternal(final HttpServletRequest request, final HttpServletResponse response)
94              throws Exception {
95          final Map<String, String> parameters = new HashMap<String, String>();
96          parameters.putAll(getAssociationResponse(request));
97          return new ModelAndView(successView, "parameters", parameters);
98      }
99  
100     @Override
101     public boolean canHandle(final HttpServletRequest request, final HttpServletResponse response) {
102         final String openIdMode = request.getParameter("openid.mode");
103         if (openIdMode != null && openIdMode.equals("associate")) {
104             logger.info("Handling request. openid.mode : {}", openIdMode);
105             return true;
106         }
107         logger.info("Cannot handle request. openid.mode : {}", openIdMode);
108         return false;
109     }
110 
111     public void setSuccessView(final String successView) {
112         this.successView = successView;
113     }
114 
115     public void setFailureView(final String failureView) {
116         this.failureView = failureView;
117     }
118 
119     @NotNull
120     public void setServerManager(final ServerManager serverManager) {
121         this.serverManager = serverManager;
122     }
123 }
