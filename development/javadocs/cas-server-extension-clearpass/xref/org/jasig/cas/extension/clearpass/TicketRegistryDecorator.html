19  package org.jasig.cas.extension.clearpass;
20  
21  import java.util.Collection;
22  import java.util.Map;
23  
24  import javax.validation.constraints.NotNull;
25  
26  import org.jasig.cas.monitor.TicketRegistryState;
27  import org.jasig.cas.ticket.Ticket;
28  import org.jasig.cas.ticket.TicketGrantingTicket;
29  import org.jasig.cas.ticket.registry.AbstractTicketRegistry;
30  import org.jasig.cas.ticket.registry.TicketRegistry;
31  
32  /**
33   * Decorator that captures tickets and attempts to map them.
34   *
35   * @author Scott Battaglia
36   * @since 1.0.7
37   */
38  public final class TicketRegistryDecorator extends AbstractTicketRegistry {
39  
40      /** The real instance of the ticket registry that is to be decorated. */
41      @NotNull
42      private final TicketRegistry ticketRegistry;
43  
44      /** Map instance where credentials are stored. */
45      @NotNull
46      private final Map<String, String> cache;
47  
48      /**
49       * Constructs an instance of the decorator wrapping the real ticket registry instance inside.
50       *
51       * @param actualTicketRegistry The real instance of the ticket registry that is to be decorated
52       * @param cache Map instance where credentials are stored.
53       *
54       * @see EhcacheBackedMap
55       */
56      public TicketRegistryDecorator(final TicketRegistry actualTicketRegistry, final Map<String, String> cache) {
57          this.ticketRegistry = actualTicketRegistry;
58          this.cache = cache;
59      }
60  
61      @Override
62      public void addTicket(final Ticket ticket) {
63          if (ticket instanceof TicketGrantingTicket) {
64              final TicketGrantingTicket ticketGrantingTicket = (TicketGrantingTicket) ticket;
65              final String ticketId = ticketGrantingTicket.getId();
66              final String userName = ticketGrantingTicket.getAuthentication().getPrincipal().getId().toLowerCase();
67  
68              logger.debug("Creating mapping ticket {} to user name {}", ticketId, userName);
69  
70              this.cache.put(ticketId, userName);
71          }
72  
73          this.ticketRegistry.addTicket(ticket);
74      }
75  
76      @Override
77      public Ticket getTicket(final String ticketId) {
78          return this.ticketRegistry.getTicket(ticketId);
79      }
80  
81      @Override
82      public boolean deleteTicket(final String ticketId) {
83          final String userName = this.cache.get(ticketId);
84  
85          if (userName != null) {
86              logger.debug("Removing mapping ticket {} for user name {}", ticketId, userName);
87              this.cache.remove(userName);
88          }
89  
90          return this.ticketRegistry.deleteTicket(ticketId);
91      }
92  
93      @Override
94      public Collection<Ticket> getTickets() {
95          return this.ticketRegistry.getTickets();
96      }
97  
98      @Override
99      public int sessionCount() {
100         if (this.ticketRegistry instanceof TicketRegistryState) {
101             return ((TicketRegistryState) this.ticketRegistry).sessionCount();
102         }
103         logger.debug("Ticket registry {} does not report the sessionCount() operation of the registry state.",
104                 this.ticketRegistry.getClass().getName());
105         return super.sessionCount();
106     }
107 
108     @Override
109     public int serviceTicketCount() {
110         if (this.ticketRegistry instanceof TicketRegistryState) {
111             return ((TicketRegistryState) this.ticketRegistry).serviceTicketCount();
112         }
113         logger.debug("Ticket registry {} does not report the serviceTicketCount() operation of the registry state.",
114                 this.ticketRegistry.getClass().getName());
115         return super.serviceTicketCount();
116     }
117 }
