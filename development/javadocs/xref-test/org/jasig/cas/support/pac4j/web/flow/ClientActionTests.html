56  @SuppressWarnings("rawtypes")
57  public final class ClientActionTests {

59      private static final String TGT_NAME = "ticketGrantingTicketId";
60      private static final String TGT_ID = "TGT-00-xxxxxxxxxxxxxxxxxxxxxxxxxx.cas0";
61  
62      private static final String MY_KEY = "my_key";
63  
64      private static final String MY_SECRET = "my_secret";
65  
66      private static final String MY_LOGIN_URL = "http://casserver/login";
67  
68      private static final String MY_SERVICE = "http://myservice";
69  
70      private static final String MY_THEME = "my_theme";
71  
72      private static final String MY_LOCALE = "fr";
73  
74      private static final String MY_METHOD = "POST";
75  
76      @Test
77      public void verifyStartAuthentication() throws Exception {
78          final MockHttpServletRequest mockRequest = new MockHttpServletRequest();
79          mockRequest.setParameter(ClientAction.THEME, MY_THEME);
80          mockRequest.setParameter(ClientAction.LOCALE, MY_LOCALE);
81          mockRequest.setParameter(ClientAction.METHOD, MY_METHOD);
82  
83          final MockHttpSession mockSession = new MockHttpSession();
84          mockRequest.setSession(mockSession);
85  
86          final ServletExternalContext servletExternalContext = mock(ServletExternalContext.class);
87          when(servletExternalContext.getNativeRequest()).thenReturn(mockRequest);
88  
89          final MockRequestContext mockRequestContext = new MockRequestContext();
90          mockRequestContext.setExternalContext(servletExternalContext);
91          mockRequestContext.getFlowScope().put(ClientAction.SERVICE, new SimpleWebApplicationServiceImpl(MY_SERVICE));
92  
93          final FacebookClient facebookClient = new FacebookClient(MY_KEY, MY_SECRET);
94          final TwitterClient twitterClient = new TwitterClient(MY_KEY, MY_SECRET);
95          final Clients clients = new Clients(MY_LOGIN_URL, facebookClient, twitterClient);
96          final ClientAction action = new ClientAction(mock(CentralAuthenticationService.class), clients);
97  
98          final Event event = action.execute(mockRequestContext);
99          assertEquals("error", event.getId());
100         assertEquals(MY_THEME, mockSession.getAttribute(ClientAction.THEME));
101         assertEquals(MY_LOCALE, mockSession.getAttribute(ClientAction.LOCALE));
102         assertEquals(MY_METHOD, mockSession.getAttribute(ClientAction.METHOD));
103         final MutableAttributeMap flowScope = mockRequestContext.getFlowScope();
104         assertTrue(((String) flowScope.get("FacebookClientUrl"))
105                 .startsWith("https://www.facebook.com/v2.2/dialog/oauth?client_id=my_key&redirect_uri=http%3A%2F%2Fcasserver%2Flogin%3F"
106                         + Clients.DEFAULT_CLIENT_NAME_PARAMETER + "%3DFacebookClient&state="));
107         assertEquals(MY_LOGIN_URL + "?" + Clients.DEFAULT_CLIENT_NAME_PARAMETER
108                 + "=TwitterClient&needs_client_redirection=true", flowScope.get("TwitterClientUrl"));
109     }
110 
111     @Test
112     public void verifyFinishAuthentication() throws Exception {
113         final MockHttpServletRequest mockRequest = new MockHttpServletRequest();
114         mockRequest.setParameter(Clients.DEFAULT_CLIENT_NAME_PARAMETER, "FacebookClient");
115 
116         final MockHttpSession mockSession = new MockHttpSession();
117         mockSession.setAttribute(ClientAction.THEME, MY_THEME);
118         mockSession.setAttribute(ClientAction.LOCALE, MY_LOCALE);
119         mockSession.setAttribute(ClientAction.METHOD, MY_METHOD);
120         final Service service = new SimpleWebApplicationServiceImpl(MY_SERVICE);
121         mockSession.setAttribute(ClientAction.SERVICE, service);
122         mockRequest.setSession(mockSession);
123 
124         final ServletExternalContext servletExternalContext = mock(ServletExternalContext.class);
125         when(servletExternalContext.getNativeRequest()).thenReturn(mockRequest);
126 
127         final MockRequestContext mockRequestContext = new MockRequestContext();
128         mockRequestContext.setExternalContext(servletExternalContext);
129 
130         final FacebookClient facebookClient = new MockFacebookClient();
131         final Clients clients = new Clients(MY_LOGIN_URL, facebookClient);
132 
133         final TicketGrantingTicket tgt = new TicketGrantingTicketImpl(TGT_ID, mock(Authentication.class), mock(ExpirationPolicy.class));
134         final CentralAuthenticationService casImpl = mock(CentralAuthenticationService.class);
135         when(casImpl.createTicketGrantingTicket(any(Credential.class))).thenReturn(tgt);
136         final ClientAction action = new ClientAction(casImpl, clients);
137         final Event event = action.execute(mockRequestContext);
138         assertEquals("success", event.getId());
139         assertEquals(MY_THEME, mockRequest.getAttribute(ClientAction.THEME));
140         assertEquals(MY_LOCALE, mockRequest.getAttribute(ClientAction.LOCALE));
141         assertEquals(MY_METHOD, mockRequest.getAttribute(ClientAction.METHOD));
142         assertEquals(MY_SERVICE, mockRequest.getAttribute(ClientAction.SERVICE));
143         final MutableAttributeMap flowScope = mockRequestContext.getFlowScope();
144         final MutableAttributeMap requestScope = mockRequestContext.getRequestScope();
145         assertEquals(service, flowScope.get(ClientAction.SERVICE));
146         assertEquals(TGT_ID, flowScope.get(TGT_NAME));
147         assertEquals(TGT_ID, requestScope.get(TGT_NAME));
148     }

150     @Test
151     public void checkUnautorizedProtocol() throws Exception {
152         final MockHttpServletRequest mockRequest = new MockHttpServletRequest();
153         mockRequest.setParameter(Clients.DEFAULT_CLIENT_NAME_PARAMETER, "BasicAuthClient");
154 
155         final ServletExternalContext servletExternalContext = mock(ServletExternalContext.class);
156         when(servletExternalContext.getNativeRequest()).thenReturn(mockRequest);
157 
158         final MockRequestContext mockRequestContext = new MockRequestContext();
159         mockRequestContext.setExternalContext(servletExternalContext);
160 
161         final BasicAuthClient basicAuthClient = new BasicAuthClient();
162         final Clients clients = new Clients(MY_LOGIN_URL, basicAuthClient);
163         final ClientAction action = new ClientAction(mock(CentralAuthenticationService.class), clients);
164 
165         try {
166             action.execute(mockRequestContext);
167             fail("Should fail as the HTTP protocol is not authorized");
168         } catch (final TechnicalException e) {
169             assertEquals("Only CAS, OAuth, OpenID and SAML protocols are supported: " + basicAuthClient, e.getMessage());
170         }
171     }
