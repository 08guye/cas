57          assertNotNull("testParseTokenString() - Not null", result);

60      @Test
61      public void verifyCreateCredentialFromToken() throws Exception {
62          final String wresult = testTokens.get("goodToken");
63          final Assertion assertion = wsFederationHelper.parseTokenFromString(wresult);
64          
65          final WsFederationCredential expResult = new WsFederationCredential();
66          expResult.setIssuedOn(new DateTime("2014-02-26T22:51:16.504Z").withZone(DateTimeZone.UTC));
67          expResult.setNotBefore(new DateTime("2014-02-26T22:51:16.474Z").withZone(DateTimeZone.UTC));
68          expResult.setNotOnOrAfter(new DateTime("2014-02-26T23:51:16.474Z").withZone(DateTimeZone.UTC));
69          expResult.setIssuer("http://adfs.example.com/adfs/services/trust");
70          expResult.setAudience("urn:federation:cas");
71          expResult.setId("_6257b2bf-7361-4081-ae1f-ec58d4310f61");
72          
73          final WsFederationCredential result = wsFederationHelper.createCredentialFromToken(assertion);
74          
75          assertNotNull("testCreateCredentialFromToken() - Not Null", result);
76          assertEquals("testCreateCredentialFromToken() - IssuedOn", expResult.getIssuedOn(), result.getIssuedOn());
77          assertEquals("testCreateCredentialFromToken() - NotBefore", expResult.getNotBefore(), result.getNotBefore());
78          assertEquals("testCreateCredentialFromToken() - NotOnOrAfter", expResult.getNotOnOrAfter(), result.getNotOnOrAfter());
79          assertEquals("testCreateCredentialFromToken() - Issuer", expResult.getIssuer(), result.getIssuer());
80          assertEquals("testCreateCredentialFromToken() - Audience", expResult.getAudience(), result.getAudience());
81          assertEquals("testCreateCredentialFromToken() - Id", expResult.getId(), result.getId());
82      }

103         final Assertion assertion = wsFederationHelper.parseTokenFromString(wresult);

108     @Test
109     @DirtiesContext
110     public void verifyValidateSignatureBadKey() throws Exception {
111         final List<Credential> signingWallet = new ArrayList<>();
112         final WsFederationConfiguration cfg = new WsFederationConfiguration();
113         cfg.setSigningCertificateFiles(ctx.getResource("classpath:bad-signing.crt"));
114 
115         signingWallet.addAll(cfg.getSigningCertificates());
116         final String wresult = testTokens.get("goodToken");
117         final Assertion assertion = wsFederationHelper.parseTokenFromString(wresult);
118         wsFedConfig.getSigningCertificates().clear();
119         wsFedConfig.getSigningCertificates().addAll(signingWallet);
120         final boolean result = wsFederationHelper.validateSignature(assertion, wsFedConfig);
121         assertFalse("testValidateSignatureModifiedKey() - False", result);
122     }
