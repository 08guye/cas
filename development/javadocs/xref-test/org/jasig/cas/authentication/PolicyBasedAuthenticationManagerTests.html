130     /**
131      * Creates a new mock authentication handler that either successfully validates all credentials or fails to
132      * validate all credentials.
133      *
134      * @param success True to authenticate all credentials, false to fail all credentials.
135      *
136      * @return New mock authentication handler instance.
137      *
138      * @throws Exception On errors.
139      */
140     private static AuthenticationHandler newMockHandler(final boolean success) throws Exception {
141         return newMockHandler("MockAuthenticationHandler" + System.nanoTime(), success);
142     }
143 
144     /**
145      * Creates a new named mock authentication handler that either successfully validates all credentials or fails to
146      * validate all credentials.
147      *
148      * @param name Authentication handler name.
149      * @param success True to authenticate all credentials, false to fail all credentials.
150      *
151      * @return New mock authentication handler instance.
152      *
153      * @throws Exception On errors.
154      */
155     private static AuthenticationHandler newMockHandler(final String name, final boolean success) throws Exception {
156         final AuthenticationHandler mock = mock(AuthenticationHandler.class);
157         when(mock.getName()).thenReturn(name);
158         when(mock.supports(any(Credential.class))).thenReturn(true);
159         if (success) {
160             final Principal p = new DefaultPrincipalFactory().createPrincipal("nobody");
161 
162             final HandlerResult result = new DefaultHandlerResult(
163                     mock,
164                     mock(CredentialMetaData.class),
165                     p);
166             when(mock.authenticate(any(Credential.class))).thenReturn(result);
167         } else {
168             when(mock.authenticate(any(Credential.class))).thenThrow(new FailedLoginException());
169         }
170         return mock;
171     }
