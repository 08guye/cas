43  /**
44   * @author Misagh Moayyed
45   */
46  public final class InMemoryTestLdapDirectoryServer implements Closeable {
47      private static final Logger LOGGER = LoggerFactory.getLogger(InMemoryTestLdapDirectoryServer.class);
48  
49      private final InMemoryDirectoryServer directoryServer;
50  
51      private Collection<LdapEntry> ldapEntries;
52  
53      /**
54       * Instantiates a new Ldap directory server.
55       */
56      public InMemoryTestLdapDirectoryServer(final File properties, final File ldifFile, final File... schemaFile) {
57  
58          try {
59  
60              final Properties p = new Properties();
61              p.load(new FileInputStream(properties));
62  
63              final InMemoryDirectoryServerConfig config =
64                      new InMemoryDirectoryServerConfig(p.getProperty("ldap.rootDn"));
65              config.addAdditionalBindCredentials(p.getProperty("ldap.managerDn"), p.getProperty("ldap.managerPassword"));
66  
67              final String serverKeyStorePath = new ClassPathResource("/ldapServerTrustStore").getFile().getCanonicalPath();
68              final SSLUtil serverSSLUtil = new SSLUtil(
69                      new KeyStoreKeyManager(serverKeyStorePath, "changeit".toCharArray()), new TrustStoreTrustManager(serverKeyStorePath));
70              final SSLUtil clientSSLUtil = new SSLUtil(new TrustStoreTrustManager(serverKeyStorePath));
71              config.setListenerConfigs(
72                      InMemoryListenerConfig.createLDAPConfig("LDAP", // Listener name
73                              null, // Listen address. (null = listen on all interfaces)
74                              1389, // Listen port (0 = automatically choose an available port)
75                              serverSSLUtil.createSSLSocketFactory()), // StartTLS factory
76                      InMemoryListenerConfig.createLDAPSConfig("LDAPS", // Listener name
77                              null, // Listen address. (null = listen on all interfaces)
78                              1636, // Listen port (0 = automatically choose an available port)
79                              serverSSLUtil.createSSLServerSocketFactory(), // Server factory
80                              clientSSLUtil.createSSLSocketFactory())); // Client factory
81  
82              config.setEnforceSingleStructuralObjectClass(false);
83              config.setEnforceAttributeSyntaxCompliance(true);
84  
85              final Schema s = Schema.mergeSchemas(Schema.getSchema(schemaFile));
86              config.setSchema(s);
87  
88              this.directoryServer = new InMemoryDirectoryServer(config);
89  
90              LOGGER.debug("Populating directory with {}", ldifFile);
91              this.directoryServer.importFromLDIF(true, ldifFile.getCanonicalPath());
92              this.directoryServer.startListening();
93  
94              final LDAPConnection c = getConnection();
95              LOGGER.debug("Connected to {}:{}", c.getConnectedAddress(), c.getConnectedPort());
96  
97              populateEntries();
98  
99          } catch (final Exception e) {
100             throw new RuntimeException(e);
101         }
102     }
103 
104     private void populateEntries() throws Exception {
105         this.ldapEntries = LdapTestUtils.readLdif(new ClassPathResource("ldif/users-groups.ldif"), getBaseDn());
106         final LDAPConnection c = getConnection();
107         LdapTestUtils.createLdapEntries(c, ldapEntries);
108         c.close();
109     }
110 
111     public String getBaseDn() {
112         return this.directoryServer.getBaseDNs().get(0).toNormalizedString();
113     }
114 
115     public  Collection<LdapEntry> getLdapEntries() {
116         return this.ldapEntries;
117     }
118 
119     public LDAPConnection getConnection() throws LDAPException {
120         return this.directoryServer.getConnection();
121     }
122 
123     @Override
124     public void close() throws IOException {
125          this.directoryServer.shutDown(true);
126     }
127 }
