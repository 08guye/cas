89      private static class VerifyServiceByIdMatcher extends ArgumentMatcher<Service> {
90          private final String id;
91  
92          VerifyServiceByIdMatcher(final String id) {
93              this.id = id;
94          }
95  
96          @Override
97          public boolean matches(final Object argument) {
98              final Service s = (Service) argument;
99              return s != null && s.getId().equals(this.id);
100         }

102     }

151     private ServicesManager getServicesManager(final Service service1, final Service service2) {
152         final RegisteredService mockRegSvc1 = createMockRegisteredService(service1.getId(), true, getServiceProxyPolicy(false));
153         final RegisteredService mockRegSvc2 = createMockRegisteredService("test", false, getServiceProxyPolicy(true));
154         final RegisteredService mockRegSvc3 = createMockRegisteredService(service2.getId(), true, getServiceProxyPolicy(true));
155 
156         final ServicesManager smMock = mock(ServicesManager.class);
157         when(smMock.findServiceBy(argThat(new VerifyServiceByIdMatcher(service1.getId())))).thenReturn(mockRegSvc1);
158         when(smMock.findServiceBy(argThat(new VerifyServiceByIdMatcher("test")))).thenReturn(mockRegSvc2);
159         when(smMock.findServiceBy(argThat(new VerifyServiceByIdMatcher(service2.getId())))).thenReturn(mockRegSvc3);
160         return smMock;
161     }

204     @Test
205     public void verifyChainedAuthenticationsOnValidation() throws TicketException {
206         final Service svc = TestUtils.getService(SVC2_ID);
207         final ServiceTicket st = this.cas.grantServiceTicket(TGT2_ID, svc);
208         assertNotNull(st);
209         
210         final Assertion assertion = this.cas.validateServiceTicket(st.getId(), svc);
211         assertNotNull(assertion);
212         
213         assertEquals(assertion.getService(), svc);
214         assertEquals(assertion.getPrimaryAuthentication().getPrincipal().getId(), PRINCIPAL);
215         assertTrue(assertion.getChainedAuthentications().size()  == 2);
216         for (int i = 0; i < assertion.getChainedAuthentications().size(); i++) {
217             final Authentication auth = assertion.getChainedAuthentications().get(i);
218             assertEquals(auth, authentication);
219         }
220     }

229     private TicketGrantingTicket createMockTicketGrantingTicket(final String id,
230             final ServiceTicket svcTicket, final boolean isExpired, 
231             final TicketGrantingTicket root, final List<Authentication> chainedAuthnList) {
232         final TicketGrantingTicket tgtMock = mock(TicketGrantingTicket.class);
233         when(tgtMock.isExpired()).thenReturn(isExpired);
234         when(tgtMock.getId()).thenReturn(id);
235 
236         final String svcId = svcTicket.getService().getId();
237         when(tgtMock.grantServiceTicket(anyString(), argThat(new VerifyServiceByIdMatcher(svcId)),
238                 any(ExpirationPolicy.class), anyBoolean())).thenReturn(svcTicket);
239         when(tgtMock.getRoot()).thenReturn(root);
240         when(tgtMock.getChainedAuthentications()).thenReturn(chainedAuthnList);
241         when(svcTicket.getGrantingTicket()).thenReturn(tgtMock);   
242         
243         return tgtMock;
244     }

254     private RegisteredServiceProxyPolicy getServiceProxyPolicy(final boolean canProxy) {
255         if (!canProxy) {
256             return new RefuseRegisteredServiceProxyPolicy();
257         }
258         
259         return new RegexMatchingRegisteredServiceProxyPolicy(".*");
260     }
261 
262     private RegisteredService createMockRegisteredService(final String svcId,
263             final boolean enabled, final RegisteredServiceProxyPolicy proxy) {
264         final RegisteredService mockRegSvc = mock(RegisteredService.class);
265         when(mockRegSvc.getServiceId()).thenReturn(svcId);
266         when(mockRegSvc.getProxyPolicy()).thenReturn(proxy);
267         when(mockRegSvc.getName()).thenReturn(svcId);
268         when(mockRegSvc.matches(argThat(new VerifyServiceByIdMatcher(svcId)))).thenReturn(true);
269         when(mockRegSvc.getAttributeReleasePolicy()).thenReturn(new ReturnAllAttributeReleasePolicy());
270         when(mockRegSvc.getUsernameAttributeProvider()).thenReturn(new DefaultRegisteredServiceUsernameProvider());
271         when(mockRegSvc.getAccessStrategy()).thenReturn(new DefaultRegisteredServiceAccessStrategy(enabled, true));
272         return mockRegSvc;
273     }
