31  import de.flapdoodle.embed.process.config.store.IDownloadConfig;
32  import de.flapdoodle.embed.process.io.progress.StandardConsoleProgressListener;
33  import org.slf4j.Logger;
34  
35  import java.net.Socket;
36  
37  import static org.slf4j.LoggerFactory.getLogger;
38  
39  /**
40   * This is {@link AbstractMemcachedTests}.
41   *
42   * @author Misagh Moayyed
43   * @since 4.2.0
44   */
45  public abstract class AbstractMemcachedTests {
46  
47      private static final int PORT = 11211;
48  
49      private static MemcachedExecutable MEMCACHED_EXECUTABLE;
50      private static MemcachedProcess MEMCACHED;
51  
52      protected final Logger logger = getLogger(this.getClass());
53  
54      public static void bootstrap() {
55          try {
56              final MemcachedStarter runtime = MemcachedStarter.getInstance(
57                      new CasRuntimeConfigBuilder().defaults(Command.MemcacheD).build());
58              MEMCACHED_EXECUTABLE = runtime.prepare(new MemcachedConfig(Version.V1_4_22, PORT));
59              MEMCACHED = MEMCACHED_EXECUTABLE.start();
60          } catch (final Exception e) {
61              getLogger(AbstractMemcachedTests.class).warn("Aborting since no memcached server could be started.", e);
62          }
63      }
64  
65      public static void shutdown() {
66          if (MEMCACHED != null && MEMCACHED.isProcessRunning()) {
67              MEMCACHED.stop();
68          }
69          if (MEMCACHED_EXECUTABLE != null) {
70              MEMCACHED_EXECUTABLE.stop();
71          }
72      }
73  
74      public boolean isMemcachedListening() {
75          try (final Socket socket = new Socket("127.0.0.1", PORT)) {
76              return true;
77          } catch (final Exception e) {
78              return false;
79          }
80      }
81  
82      private static class CasRuntimeConfigBuilder extends RuntimeConfigBuilder {
83          @Override
84          public RuntimeConfigBuilder defaults(final Command command) {
85              final RuntimeConfigBuilder builder = super.defaults(command);
86  
87              final IDownloadConfig downloadConfig = new CasDownloadConfigBuilder()
88                      .defaultsForCommand(command)
89                      .progressListener(new StandardConsoleProgressListener())
90                      .build();
91              this.artifactStore().overwriteDefault(new ArtifactStoreBuilder()
92                      .defaults(command).download(downloadConfig).build());
93              return builder;
94          }
95      }
96  
97      /**
98       * Download an embedded memcached instance based on environment.
99       */
100     private static class CasDownloadConfigBuilder extends DownloadConfigBuilder {
101         @Override
102         public DownloadConfigBuilder defaults() {
103             final DownloadConfigBuilder bldr = super.defaults();
104             bldr.downloadPath("http://heli0s.darktech.org/memcached/");
105             return bldr;
106         }
107     }
108 }
