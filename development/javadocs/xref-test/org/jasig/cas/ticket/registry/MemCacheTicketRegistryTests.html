53  /**
54   * Unit test for MemCacheTicketRegistry class.
55   *
56   * @author Middleware Services
57   * @since 3.0.0
58   */
59  @RunWith(Parameterized.class)
60  public class MemCacheTicketRegistryTests {
61      private static final Logger LOGGER = getLogger(MemCacheTicketRegistryTests.class);
62      private static final int PORT = 11211;
63  
64      private static MemcachedExecutable MEMCACHED_EXECUTABLE;
65      private static MemcachedProcess MEMCACHED;
66  
67      private MemCacheTicketRegistry registry;
68  
69      private final String registryBean;
70  
71      private final boolean binaryProtocol;
72  
73      public MemCacheTicketRegistryTests(final String beanName, final boolean binary) {
74          registryBean = beanName;
75          binaryProtocol = binary;
76      }
77  
78      @Parameterized.Parameters
79      public static Collection<Object[]> getTestParameters() throws Exception {
80          return Arrays.asList(new Object[] {"testCase1", false}, new Object[] {"testCase2", true});
81      }
82  
83      @BeforeClass
84      public static void beforeClass() throws IOException {
85          try {
86              final MemcachedStarter runtime = MemcachedStarter.getInstance(
87                      new CasRuntimeConfigBuilder().defaults(Command.MemcacheD).build());
88              MEMCACHED_EXECUTABLE = runtime.prepare(new MemcachedConfig(Version.V1_4_22, PORT));
89              MEMCACHED = MEMCACHED_EXECUTABLE.start();
90          } catch (final Exception e) {
91              LOGGER.warn("Aborting since no memcached server could be started.", e);
92          }
93      }
94  
95      @AfterClass
96      public static void afterClass() throws Exception {
97         if (MEMCACHED != null && MEMCACHED.isProcessRunning()) {
98             MEMCACHED.stop();
99         }
100         if (MEMCACHED_EXECUTABLE != null) {
101             MEMCACHED_EXECUTABLE.stop();
102         }
103     }
104 
105     @Before
106     public void setUp() throws IOException {
107 
108         // Abort tests if there is no memcached server available on localhost:11211.
109         final boolean environmentOk = isMemcachedListening();
110         if (!environmentOk) {
111             LOGGER.warn("Aborting test since no memcached server is available on localhost.");
112         }
113         Assume.assumeTrue(environmentOk);
114         final ApplicationContext context = new ClassPathXmlApplicationContext("/ticketRegistry-test.xml");
115         registry = context.getBean(registryBean, MemCacheTicketRegistry.class);
116     }
117 
118     @Test
119     public void verifyWriteGetDelete() throws Exception {
120         final String id = "ST-1234567890ABCDEFGHIJKL-crud";
121         final ServiceTicket ticket = mock(ServiceTicket.class, withSettings().serializable());
122         when(ticket.getId()).thenReturn(id);
123         registry.addTicket(ticket);
124         final ServiceTicket ticketFromRegistry = (ServiceTicket) registry.getTicket(id);
125         Assert.assertNotNull(ticketFromRegistry);
126         Assert.assertEquals(id, ticketFromRegistry.getId());
127         registry.deleteTicket(id);
128         Assert.assertNull(registry.getTicket(id));
129     }
130 
131     @Test
132     public void verifyExpiration() throws Exception {
133         final String id = "ST-1234567890ABCDEFGHIJKL-exp";
134         final ServiceTicket ticket = mock(ServiceTicket.class, withSettings().serializable());
135         when(ticket.getId()).thenReturn(id);
136         registry.addTicket(ticket);
137         Assert.assertNotNull(registry.getTicket(id, ServiceTicket.class));
138         // Sleep a little longer than service ticket expiry defined in Spring context
139         Thread.sleep(2100);
140         Assert.assertNull(registry.getTicket(id, ServiceTicket.class));
141     }
142 
143     private boolean isMemcachedListening() {
144         try (final Socket socket = new Socket("127.0.0.1", PORT)) {
145             return true;
146         } catch (final Exception e) {
147             return false;
148         }
149     }
150 
151     private static class CasRuntimeConfigBuilder extends RuntimeConfigBuilder {
152         @Override
153         public RuntimeConfigBuilder defaults(final Command command) {
154             final RuntimeConfigBuilder builder = super.defaults(command);
155 
156             final IDownloadConfig downloadConfig = new CasDownloadConfigBuilder()
157                     .defaultsForCommand(command)
158                     .progressListener(new StandardConsoleProgressListener())
159                     .build();
160             this.artifactStore().overwriteDefault(new ArtifactStoreBuilder()
161                     .defaults(command).download(downloadConfig).build());
162             return builder;
163         }
164     }
165 
166     /**
167      * Download an embedded memcached instance based on environment.
168      */
169     private static class CasDownloadConfigBuilder extends DownloadConfigBuilder {
170         @Override
171         public DownloadConfigBuilder defaults() {
172             final DownloadConfigBuilder bldr = super.defaults();
173             bldr.downloadPath("http://heli0s.darktech.org/memcached/");
174             return bldr;
175         }
176     }
