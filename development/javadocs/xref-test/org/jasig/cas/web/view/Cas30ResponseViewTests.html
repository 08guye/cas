50  /**
51   * Unit tests for {@link org.jasig.cas.web.view.Cas20ResponseView}.
52   * @author Misagh Moayyed
53   * @since 4.0.0
54   */
55  public class Cas30ResponseViewTests extends AbstractServiceValidateControllerTests {
56  
57      @Autowired
58      @Qualifier("protocolCas3ViewResolver")
59      private ViewResolver resolver;
60  
61      @Autowired
62      @Qualifier("servicesManager")
63      private ServicesManager servicesManager;
64  
65      private Map<?, ?> renderView() throws Exception{
66          final ModelAndView modelAndView = this.getModelAndViewUponServiceValidationWithSecurePgtUrl();
67          final JstlView v = (JstlView) resolver.resolveViewName(modelAndView.getViewName(), Locale.getDefault());
68          final MockHttpServletRequest req = new MockHttpServletRequest(new MockServletContext());
69          v.setServletContext(req.getServletContext());
70          req.setAttribute(RequestContext.WEB_APPLICATION_CONTEXT_ATTRIBUTE,
71                  new GenericWebApplicationContext(req.getServletContext()));
72  
73          final Cas30ResponseView view = new Cas30ResponseView(v);
74          view.setServicesManager(this.servicesManager);
75          view.setCasAttributeEncoder(new DefaultCasAttributeEncoder(this.servicesManager));
76  
77          final MockHttpServletResponse resp = new MockHttpServletResponse();
78          view.render(modelAndView.getModel(), req, resp);
79          return (Map<?, ?>) req.getAttribute(CasProtocolConstants.VALIDATION_CAS_MODEL_ATTRIBUTE_NAME_ATTRIBUTES);
80      }
81  
82      @Test
83      public void verifyViewAuthnAttributes() throws Exception {
84          final Map<?, ?> attributes = renderView();
85          assertTrue(attributes.containsKey(CasProtocolConstants.VALIDATION_CAS_MODEL_ATTRIBUTE_NAME_AUTHENTICATION_DATE));
86          assertTrue(attributes.containsKey(CasProtocolConstants.VALIDATION_CAS_MODEL_ATTRIBUTE_NAME_FROM_NEW_LOGIN));
87          assertTrue(attributes.containsKey(CasProtocolConstants.VALIDATION_REMEMBER_ME_ATTRIBUTE_NAME));
88  
89      }
90  
91      @Test
92      public void verifyPasswordAsAuthenticationAttributeCanDecrypt() throws Exception {
93          final Map<?, ?> attributes = renderView();
94          assertTrue(attributes.containsKey(CasViewConstants.MODEL_ATTRIBUTE_NAME_PRINCIPAL_CREDENTIAL));
95  
96          final String encodedPsw = (String) attributes.get(CasViewConstants.MODEL_ATTRIBUTE_NAME_PRINCIPAL_CREDENTIAL);
97          final String password = decryptCredential(encodedPsw);
98          final UsernamePasswordCredential creds = TestUtils.getCredentialsWithSameUsernameAndPassword();
99          assertEquals(password, creds.getPassword());
100     }
101 
102     @Test
103     public void verifyProxyGrantingTicketAsAuthenticationAttributeCanDecrypt() throws Exception {
104         final Map<?, ?> attributes = renderView();
105         assertTrue(attributes.containsKey(CasViewConstants.MODEL_ATTRIBUTE_NAME_PROXY_GRANTING_TICKET));
106 
107         final String encodedPgt = (String) attributes.get(CasViewConstants.MODEL_ATTRIBUTE_NAME_PROXY_GRANTING_TICKET);
108         final String pgt = decryptCredential(encodedPgt);
109         assertNotNull(pgt);
110     }
111 
112     private String decryptCredential(final String cred) {
113         try {
114             final PrivateKeyFactoryBean factory = new PrivateKeyFactoryBean();
115             factory.setAlgorithm("RSA");
116             factory.setLocation(new ClassPathResource("RSA1024Private.p8"));
117             factory.setSingleton(false);
118             final PrivateKey privateKey = factory.getObject();
119 
120             logger.debug("Initializing cipher based on [{}]", privateKey.getAlgorithm());
121             final Cipher cipher = Cipher.getInstance(privateKey.getAlgorithm());
122 
123             logger.debug("Decoding value [{}]", cred);
124             final byte[] cred64 = CompressionUtils.decodeBase64ToByteArray(cred);
125 
126             logger.debug("Initializing decrypt-mode via private key [{}]", privateKey.getAlgorithm());
127             cipher.init(Cipher.DECRYPT_MODE, privateKey);
128 
129             final byte[] cipherData = cipher.doFinal(cred64);
130             return new String(cipherData);
131         } catch (final Exception e) {
132             throw new RuntimeException(e);
133         }
134     }
