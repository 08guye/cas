45  /**
46   * Process the /validate , /serviceValidate , and /proxyValidate URL requests.
47   * <p>
48   * Obtain the Service Ticket and Service information and present them to the CAS

99      /**
100      * Overrideable method to determine which credentials to use to grant a
101      * proxy granting ticket. Default is to use the pgtUrl.
102      *
103      * @param request the HttpServletRequest object.
104      * @return the credentials or null if there was an error or no credentials
105      * provided.
106      */
107     protected Credential getServiceCredentialsFromRequest(final HttpServletRequest request) {
108         final String pgtUrl = request.getParameter(PARAMETER_PROXY_CALLBACK_URL);
109         if (StringUtils.hasText(pgtUrl)) {
110             try {
111                 return new HttpBasedServiceCredential(new URL(pgtUrl));
112             } catch (final Exception e) {
113                 logger.error("Error constructing pgtUrl", e);
114             }
115         }
116 
117         return null;
118     }

130     @Override
131     protected final ModelAndView handleRequestInternal(final HttpServletRequest request, final HttpServletResponse response)
132             throws Exception {
133         final WebApplicationService service = this.argumentExtractor.extractService(request);
134         final String serviceTicketId = service != null ? service.getArtifactId() : null;
135 
136         if (service == null || serviceTicketId == null) {
137             logger.debug("Could not identify service and/or service ticket. Service: {}, Service ticket id: {}", service, serviceTicketId);
138             return generateErrorView("INVALID_REQUEST", "INVALID_REQUEST", null);
139         }
140 
141         try {
142             final Credential serviceCredential = getServiceCredentialsFromRequest(request);
143             String proxyGrantingTicketId = null;
144             
145             if (serviceCredential != null) {
146                 try {
147                     proxyGrantingTicketId = this.centralAuthenticationService.delegateTicketGrantingTicket(serviceTicketId,
148                                 serviceCredential);
149                 } catch (final AuthenticationException e) {
150                     logger.info("Failed to authenticate service credential {}", serviceCredential);
151                 } catch (final TicketException e) {
152                     logger.error("Failed to create proxy granting ticket for {}", serviceCredential, e);
153                 }
154                 
155                 if (StringUtils.isEmpty(proxyGrantingTicketId)) {
156                     return generateErrorView("INVALID_PROXY_CALLBACK", "INVALID_PROXY_CALLBACK",
157                             new Object[] {serviceCredential.getId()});
158                 }
159             }
160 
161             final Assertion assertion = this.centralAuthenticationService.validateServiceTicket(serviceTicketId, service);
162 
163             final ValidationSpecification validationSpecification = this.getCommandClass();
164             final ServletRequestDataBinder binder = new ServletRequestDataBinder(validationSpecification, "validationSpecification");
165             initBinder(request, binder);
166             binder.bind(request);
167 
168             if (!validationSpecification.isSatisfiedBy(assertion)) {
169                 logger.debug("Service ticket [{}] does not satisfy validation specification.", serviceTicketId);
170                 return generateErrorView("INVALID_TICKET", "INVALID_TICKET_SPEC", null);
171             }
172 
173             String proxyIou = null;
174             if (serviceCredential != null && proxyGrantingTicketId != null && this.proxyHandler.canHandle(serviceCredential)) {
175                 proxyIou = this.proxyHandler.handle(serviceCredential, proxyGrantingTicketId);
176                 if (StringUtils.isEmpty(proxyIou)) {
177                     return generateErrorView("INVALID_PROXY_CALLBACK", "INVALID_PROXY_CALLBACK",
178                             new Object[] {serviceCredential.getId()});
179                 }
180             }
181 
182             onSuccessfulValidation(serviceTicketId, assertion);
183             logger.debug("Successfully validated service ticket {} for service [{}]", serviceTicketId, service.getId());
184             return generateSuccessView(assertion, proxyIou);
185         } catch (final TicketValidationException e) {
186             return generateErrorView(e.getCode(), e.getCode(),
187                     new Object[] {serviceTicketId, e.getOriginalService().getId(), service.getId()});
188         } catch (final TicketException te) {
189             return generateErrorView(te.getCode(), te.getCode(),
190                 new Object[] {serviceTicketId});
191         } catch (final UnauthorizedProxyingException e) {
192             return generateErrorView(e.getMessage(), e.getMessage(), new Object[] {service.getId()});
193         } catch (final UnauthorizedServiceException e) {
194             return generateErrorView(e.getMessage(), e.getMessage(), null);
195         }
196     }

209     /**
210      * Generate error view, set to {@link #setFailureView(String)}.
211      *
212      * @param code the code
213      * @param description the description
214      * @param args the args
215      * @return the model and view
216      */
217     private ModelAndView generateErrorView(final String code, final String description, final Object[] args) {
218         final ModelAndView modelAndView = new ModelAndView(this.failureView);
219         final String convertedDescription = getMessageSourceAccessor().getMessage(description, args, description);
220         modelAndView.addObject("code", code);
221         modelAndView.addObject("description", convertedDescription);
222 
223         return modelAndView;
224     }

226     /**
227      * Generate the success view. The result will contain the assertion and the proxy iou.
228      *
229      * @param assertion the assertion
230      * @param proxyIou the proxy iou
231      * @return the model and view, pointed to the view name set by {@link #setSuccessView(String)}
232      */
233     private ModelAndView generateSuccessView(final Assertion assertion, final String proxyIou) {
234         final ModelAndView success = new ModelAndView(this.successView);
235         success.addObject(MODEL_ASSERTION, assertion);
236         success.addObject(MODEL_PROXY_GRANTING_TICKET_IOU, proxyIou);
237         return success;
238     }
