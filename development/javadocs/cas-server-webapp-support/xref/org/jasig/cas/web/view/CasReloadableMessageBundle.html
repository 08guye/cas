19  package org.jasig.cas.web.view;
20  
21  import java.util.Locale;
22  
23  import org.apache.commons.lang3.StringUtils;
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  import org.springframework.context.support.ReloadableResourceBundleMessageSource;
27  
28  /**
29   * An extension of the {@link ReloadableResourceBundleMessageSource} whose sole concern
30   * is to print a WARN message in cases where a language key is not found in the active and
31   * default bundles.
32   * 
33   * <p>Note: By default, if a key not found in a localized bundle, Spring will auto-fallback
34   * to the default bundle that is <code>messages.properties</code>. However, if the key is also
35   * not found in the default bundle, and {@link #setUseCodeAsDefaultMessage(boolean)}
36   * is set to true, then only the requested code itself will be used as the message to display.
37   * In this case, the class will issue a WARN message instructing the caller that the bundle
38   * needs further attention. If {@link #setUseCodeAsDefaultMessage(boolean)} is set to false,
39   * only then a <code>null</code> value will be returned, which subsequently causes an instance
40   * of {@link org.springframework.context.NoSuchMessageException} to be thrown.
41   * 
42   * @author Misagh Moayyed
43   * @since 4.0
44   */
45  public class CasReloadableMessageBundle extends ReloadableResourceBundleMessageSource {
46  
47      private String[] basenames;
48      
49      private final Logger logger = LoggerFactory.getLogger(this.getClass());
50      
51      @Override
52      protected String getDefaultMessage(final String code) {
53          final String messageToReturn = super.getDefaultMessage(code);
54          if (!StringUtils.isBlank(messageToReturn) && messageToReturn.equals(code)) {
55              logger.warn("The code [{}] cannot be found in the default language bundle and will "
56                      + "be used as the message itself.", code);
57          }
58          return messageToReturn;
59      }
60  
61      @Override
62      protected String getMessageInternal(final String code, final Object[] args, final Locale locale) {
63          boolean foundCode = false;
64          
65          if (!locale.equals(Locale.ENGLISH)) {
66            for (int i = 0; !foundCode && i < this.basenames.length; i++) {
67                final String filename = this.basenames[i] + "_" + locale.getLanguage();
68                
69                logger.debug("Examining language bundle [{}] for the code [{}]", filename, code);
70                final PropertiesHolder holder = this.getProperties(filename);
71                foundCode =  holder != null && holder.getProperties() != null
72                                       && holder.getProperty(code) != null;  
73            }       
74            
75            if (!foundCode) {
76                logger.warn("The code [{}] cannot be found in the language bundle for the locale [{}]",
77                        code, locale);
78            }
79          }
80          return super.getMessageInternal(code, args, locale);
81      }
82  
83      @Override
84      public void setBasenames(final String... basenames) {
85          this.basenames = basenames;
86          super.setBasenames(basenames);
87      }
88

91  }
