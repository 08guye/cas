20  package org.jasig.cas.web.support;
21  
22  import org.apache.commons.lang3.StringUtils;
23  import org.jasig.cas.util.CipherExecutor;
24  import org.jasig.cas.util.NoOpCipherExecutor;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  
28  import javax.servlet.http.Cookie;
29  import javax.servlet.http.HttpServletRequest;
30  
31  /**
32   * The {@link DefaultCasCookieValueManager} is responsible for...
33   *
34   * @author Misagh Moayyed
35   * @since 4.1
36   */
37  public final class DefaultCasCookieValueManager implements CookieValueManager {
38      private static final Logger LOGGER = LoggerFactory.getLogger(DefaultCasCookieValueManager.class);
39      private static final String COOKIE_FIELD_SEPARATOR = "@";
40      private static final int COOKIE_FIELDS_LENGTH = 3;
41  
42      /** The cipher exec that is responsible for encryption and signing of the cookie. */
43      private CipherExecutor cipherExecutor;
44  
45      /**
46       * Instantiates a new Cas cookie value manager.
47       * Set the default cipher to do absolutely  nothing.
48       */
49      public DefaultCasCookieValueManager() {
50          this(new NoOpCipherExecutor());
51      }
52  
53      /**
54       * Instantiates a new Cas cookie value manager.
55       *
56       * @param cipherExecutor the cipher executor
57       */
58      public DefaultCasCookieValueManager(final CipherExecutor cipherExecutor) {
59          this.cipherExecutor = cipherExecutor;
60          LOGGER.debug("Using cipher [{} to encrypt and decode the cookie",
61                  this.cipherExecutor.getClass());
62      }
63  
64      @Override
65      public String buildCookieValue(final String givenCookieValue, final HttpServletRequest request) {
66          final StringBuilder builder = new StringBuilder(givenCookieValue);
67  
68          final String remoteAddr = request.getRemoteAddr();
69          if (StringUtils.isBlank(remoteAddr)) {
70              throw new IllegalStateException("Request does not specify a remote address");
71          }
72          builder.append(COOKIE_FIELD_SEPARATOR);
73          builder.append(remoteAddr);
74  
75          final String userAgent = request.getHeader("user-agent");
76          if (StringUtils.isBlank(userAgent)) {
77              throw new IllegalStateException("Request does not specify a user-agent");
78          }
79          builder.append(COOKIE_FIELD_SEPARATOR);
80          builder.append(userAgent);
81  
82          final String res = builder.toString();
83          LOGGER.debug("Encoding cookie value [{}]", res);
84          return this.cipherExecutor.encode(res);
85      }
86  
87      @Override
88      public String obtainCookieValue(final Cookie cookie, final HttpServletRequest request) {
89          final String cookieValue = this.cipherExecutor.decode(cookie.getValue());
90          LOGGER.debug("Decoded cookie value is [{}]", cookieValue);
91          if (StringUtils.isBlank(cookieValue)) {
92              LOGGER.debug("Retrieved decoded cookie value is blank. Failed to decode cookie [{}]", cookie.getName());
93              return null;
94          }
95  
96          final String[] cookieParts = cookieValue.split(COOKIE_FIELD_SEPARATOR);
97          if (cookieParts.length != COOKIE_FIELDS_LENGTH) {
98              throw new IllegalStateException("Invalid cookie. Required fields are missing");
99          }
100         final String value = cookieParts[0];
101         final String remoteAddr = cookieParts[1];
102         final String userAgent = cookieParts[2];
103 
104         if (StringUtils.isBlank(value) || StringUtils.isBlank(remoteAddr)
105                 || StringUtils.isBlank(userAgent)) {
106             throw new IllegalStateException("Invalid cookie. Required fields are empty");
107         }
108 
109         if (!remoteAddr.equals(request.getRemoteAddr())) {
110             throw new IllegalStateException("Invalid cookie. Required remote address does not match "
111                     + request.getRemoteAddr());
112         }
113 
114         if (!userAgent.equals(request.getHeader("user-agent"))) {
115             throw new IllegalStateException("Invalid cookie. Required user-agent does not match "
116                     + request.getHeader("user-agent"));
117         }
118         return value;
119     }
120 }
