24  import org.junit.runner.RunWith;

27  import java.util.Arrays;

32  /**
33   * Unit test for {@link RegisteredServiceImpl}.
34   *
35   * @author Scott Battaglia
36   * @author Marvin S. Addison
37   * @since 3.1
38   *
39   */
40  @RunWith(Parameterized.class)
41  public class RegisteredServiceImplTests {
42  
43      private RegisteredServiceImpl service;
44  
45      private String serviceToMatch;
46  
47      private boolean expected;
48  
49  
50      @Parameterized.Parameters
51      public static Collection<Object[]> getParameters() {
52          return Arrays.asList(new Object[][]{
53                  // Allow all paths on single host
54                  {
55                          newService("https://host.vt.edu/**"),
56                          "https://host.vt.edu/a/b/c?a=1&b=2",
57                          true,
58                  },
59                  // Global catch-all for HTTP
60                  {
61                          newService("http://**"),
62                          "http://host.subdomain.example.com/service",
63                          true,
64                  },
65                  // Null case
66                  {
67                          newService("https:/example.com/**"),
68                          null,
69                          false,
70                  },
71          });
72      }
73  
74  
75      public RegisteredServiceImplTests(
76              final RegisteredServiceImpl service,
77              final String serviceToMatch,
78              final boolean expectedResult) {
79          this.service = service;
80          this.serviceToMatch = serviceToMatch;
81          this.expected = expectedResult;
82      }
83  
84  
85      @Test
86      public void testMatches() throws Exception {
87          final Service testService;
88          if (serviceToMatch == null) {
89              testService = null;
90          } else {
91              testService = new MockService(serviceToMatch);
92          }
93          assertEquals(expected, service.matches(testService));
94      }
95  
96  
97      private static RegisteredServiceImpl newService(final String id) {
98          final RegisteredServiceImpl service = new RegisteredServiceImpl();
99          service.setServiceId(id);
100         return service;
101     }
