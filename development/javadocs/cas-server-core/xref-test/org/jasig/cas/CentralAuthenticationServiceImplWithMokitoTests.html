85      private static class VerifyServiceByIdMatcher extends ArgumentMatcher<Service> {
86          private String id;
87  
88          public VerifyServiceByIdMatcher(final String id) {
89              this.id = id;
90          }
91  
92          @Override
93          public boolean matches(final Object argument) {
94              final Service s = (Service) argument;
95              return s != null && s.getId().equals(this.id);
96          }

174     @Test
175     public void testChainedAuthenticationsOnValidation() throws TicketException {
176         final Service svc = TestUtils.getService(SVC2_ID);
177         final String st = this.cas.grantServiceTicket(TGT2_ID, svc);
178         assertNotNull(st);
179         
180         final Assertion assertion = this.cas.validateServiceTicket(st, svc);
181         assertNotNull(assertion);
182         
183         assertEquals(assertion.getService(), svc);
184         assertEquals(assertion.getPrimaryAuthentication().getPrincipal().getId(), PRINCIPAL);
185         assertTrue(assertion.getChainedAuthentications().size()  == 2);
186         for (int i = 0; i < assertion.getChainedAuthentications().size(); i++) {
187             final Authentication auth = assertion.getChainedAuthentications().get(i);
188             assertEquals(auth, authentication);
189         }
190     }

199     private TicketGrantingTicket createMockTicketGrantingTicket(final String id,

210         when(tgtMock.getChainedAuthentications()).thenReturn(chainedAuthnList);
211         when(svcTicket.getGrantingTicket()).thenReturn(tgtMock);

224     private RegisteredServiceProxyPolicy getServiceProxyPolicy(final boolean canProxy) {
225         if (!canProxy) {
226             return new RefuseRegisteredServiceProxyPolicy();
227         }
228         
229         return new RegexMatchingRegisteredServiceProxyPolicy(".*");
230     }
231     private RegisteredService createMockRegisteredService(final String svcId,
232             final boolean enabled, final RegisteredServiceProxyPolicy proxy) {
233         final RegisteredService mockRegSvc = mock(RegisteredService.class);
234         when(mockRegSvc.getServiceId()).thenReturn(svcId);
235         when(mockRegSvc.isEnabled()).thenReturn(enabled);
236         when(mockRegSvc.getProxyPolicy()).thenReturn(proxy);
237         when(mockRegSvc.getName()).thenReturn(svcId);
238         when(mockRegSvc.matches(argThat(new VerifyServiceByIdMatcher(svcId)))).thenReturn(true);
239         return mockRegSvc;
