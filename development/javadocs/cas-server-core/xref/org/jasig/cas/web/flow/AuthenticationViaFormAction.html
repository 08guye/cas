48  /**
49   * Action to authenticate credential and retrieve a TicketGrantingTicket for
50   * those credential. If there is a request for renew, then it also generates
51   * the Service Ticket required.

54   * @since 3.0.4

56  public class AuthenticationViaFormAction {

59      public static final String SUCCESS = "success";

61      /** Authentication succeeded with warnings from authn subsystem that should be displayed to user. */
62      public static final String SUCCESS_WITH_WARNINGS = "successWithWarnings";

96      /**
97       * Bind the request to credentials.
98       *
99       * @param context the context
100      * @param credential the credential
101      * @throws Exception the exception arising as the result of the bind op.
102      */
103     public final void doBind(final RequestContext context, final Credential credential) throws Exception {
104         final HttpServletRequest request = WebUtils.getHttpServletRequest(context);
105 
106         if (this.credentialsBinder != null && this.credentialsBinder.supports(credential.getClass())) {
107             this.credentialsBinder.bind(request, credential);
108         }
109     }

111     /**
112      * Handle the submission of credentials from the post.
113      *
114      * @param context the context
115      * @param credential the credential
116      * @param messageContext the message context
117      * @return the event
118      * @throws Exception the exception
119      */
120     public final Event submit(final RequestContext context, final Credential credential,
121             final MessageContext messageContext) throws Exception {
122         // Validate login ticket
123         final String authoritativeLoginTicket = WebUtils.getLoginTicketFromFlowScope(context);
124         final String providedLoginTicket = WebUtils.getLoginTicketFromRequest(context);
125         if (!authoritativeLoginTicket.equals(providedLoginTicket)) {
126             logger.warn("Invalid login ticket {}", providedLoginTicket);
127             messageContext.addMessage(new MessageBuilder().code("error.invalid.loginticket").build());
128             return newEvent(ERROR);
129         }
130 
131         final String ticketGrantingTicketId = WebUtils.getTicketGrantingTicketId(context);
132         final Service service = WebUtils.getService(context);
133         if (StringUtils.hasText(context.getRequestParameters().get("renew")) && ticketGrantingTicketId != null
134                 && service != null) {
135 
136             try {
137                 final String serviceTicketId = this.centralAuthenticationService.grantServiceTicket(
138                         ticketGrantingTicketId, service, credential);
139                 WebUtils.putServiceTicketInRequestScope(context, serviceTicketId);
140                 putWarnCookieIfRequestParameterPresent(context);
141                 return newEvent(WARN);
142             } catch (final AuthenticationException e) {
143                 return newEvent(AUTHENTICATION_FAILURE, e);
144             } catch (final TicketCreationException e) {
145                 logger.warn(
146                         "Invalid attempt to access service using renew=true with different credential. "
147                         + "Ending SSO session.");
148                 this.centralAuthenticationService.destroyTicketGrantingTicket(ticketGrantingTicketId);
149             } catch (final TicketException e) {
150                 return newEvent(ERROR, e);
151             }
152         }
153 
154         try {
155             final String tgtId = this.centralAuthenticationService.createTicketGrantingTicket(credential);
156             WebUtils.putTicketGrantingTicketInFlowScope(context, tgtId);
157             putWarnCookieIfRequestParameterPresent(context);
158             final TicketGrantingTicket tgt = (TicketGrantingTicket) this.ticketRegistry.getTicket(tgtId);
159             for (final Map.Entry<String, HandlerResult> entry : tgt.getAuthentication().getSuccesses().entrySet()) {
160                 for (final Message message : entry.getValue().getWarnings()) {
161                     addWarningToContext(messageContext, message);
162                 }
163             }
164             if (this.hasWarningMessages) {
165                 return newEvent(SUCCESS_WITH_WARNINGS);
166             }
167             return newEvent(SUCCESS);
168         } catch (final AuthenticationException e) {
169             return newEvent(AUTHENTICATION_FAILURE, e);
170         } catch (final Exception e) {
171             return newEvent(ERROR, e);
172         }
173     }

175     /**
176      * Put warn cookie if request parameter present.
177      *
178      * @param context the context
179      */
180     private void putWarnCookieIfRequestParameterPresent(final RequestContext context) {
181         final HttpServletResponse response = WebUtils.getHttpServletResponse(context);
182 
183         if (StringUtils.hasText(context.getExternalContext().getRequestParameterMap().get("warn"))) {
184             this.warnCookieGenerator.addCookie(response, "true");
185         } else {
186             this.warnCookieGenerator.removeCookie(response);
187         }
188     }

219     /**
220      * Set a CredentialsBinder for additional binding of the HttpServletRequest
221      * to the Credential instance, beyond our default binding of the
222      * Credential as a Form Object in Spring WebMVC parlance. By the time we
223      * invoke this CredentialsBinder, we have already engaged in default binding
224      * such that for each HttpServletRequest parameter, if there was a JavaBean
225      * property of the Credential implementation of the same name, we have set
226      * that property to be the value of the corresponding request parameter.
227      * This CredentialsBinder plugin point exists to allow consideration of
228      * things other than HttpServletRequest parameters in populating the
229      * Credential (or more sophisticated consideration of the
230      * HttpServletRequest parameters).
231      *
232      * @param credentialsBinder the credential binder to set.
233      */
234     public final void setCredentialsBinder(final CredentialsBinder credentialsBinder) {

236     }

242     /**
243      * Adds a warning message to the message context.
244      *
245      * @param context Message context.
246      * @param warning Warning message.
247      */
248     private void addWarningToContext(final MessageContext context, final Message warning) {
249         final MessageBuilder builder = new MessageBuilder()
250                 .warning()
251                 .code(warning.getCode())
252                 .defaultText(warning.getDefaultMessage())
253                 .args(warning.getParams());
254         context.addMessage(builder.build());
255         this.hasWarningMessages = true;
256     }
