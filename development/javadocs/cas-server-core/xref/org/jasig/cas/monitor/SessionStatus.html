19  package org.jasig.cas.monitor;

21  /**
22   * Provides status information about the number of SSO sessions established in CAS.
23   *
24   * @author Marvin S. Addison
25   * @since 3.5.0
26   */
27  public class SessionStatus extends Status {
28      /** Total number of SSO sessions maintained by CAS. */
29      private final int sessionCount;
30  
31      /** Total number of service tickets in CAS ticket registry. */
32      private final int serviceTicketCount;
33  
34      /**
35       * Creates a new status object with the given code.
36       *
37       * @param code Status code.
38       * @param desc Human-readable status description.
39       *
40       * @see #getCode()
41       */
42      public SessionStatus(final StatusCode code, final String desc) {
43          this(code, desc, 0, 0);
44      }
45  
46  
47      /**
48       * Creates a new status object with the given code.
49       *
50       * @param code Status code.
51       * @param desc Human-readable status description.
52       * @param sessions Number of established SSO sessions in ticket registry.
53       * @param serviceTickets Number of service tickets in ticket registry.
54       *
55       * @see #getCode()
56       */
57       public SessionStatus(final StatusCode code, final String desc, final int sessions, final int serviceTickets) {
58          super(code, desc);
59          this.sessionCount = sessions;
60          this.serviceTicketCount = serviceTickets;
61      }
62  
63  
64      /**
65       * Gets total number of SSO sessions maintained by CAS.
66       *
67       * @return Total number of SSO sessions.
68       */
69      public int getSessionCount() {
70          return this.sessionCount;
71      }
72  
73  
74      /**
75       * Gets the total number of service tickets in the CAS ticket registry.
76       *
77       * @return Total number of service tickets.
78       */
79      public int getServiceTicketCount() {
80          return this.serviceTicketCount;
81      }
82  }
