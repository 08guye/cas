19  package org.jasig.cas.authentication;
20  
21  import java.security.GeneralSecurityException;
22  
23  /**
24   * An authentication handler authenticates a single credential. In many cases credentials are authenticated by
25   * comparison with data in a system of record such as LDAP directory or database.
26   *
27   * @author Marvin S. Addison
28   * @since 4.0.0
29   */
30  public interface AuthenticationHandler {
31  
32      /**
33       * Authenticates the given credential. There are three possible outcomes of this process, and implementers
34       * MUST adhere to the following contract:
35       *
36       * <ol>
37       *     <li>Success -- return {@link HandlerResult}</li>
38       *     <li>Failure -- throw {@link GeneralSecurityException}</li>
39       *     <li>Indeterminate -- throw {@link PreventedException}</li>
40       * </ol>
41       *
42       * @param credential The credential to authenticate.
43       *
44       * @return A result object containing metadata about a successful authentication event that includes at
45       * a minimum the name of the handler that authenticated the credential and some credential metadata.
46       * The following data is optional:
47       * <ul>
48       *     <li>{@link org.jasig.cas.authentication.principal.Principal}</li>
49       *     <li>Messages issued by the handler about the credential (e.g. impending password expiration warning)</li>
50       * </ul>
51       *
52       * @throws GeneralSecurityException On authentication failures where the root cause is security related,
53       * e.g. invalid credential. Implementing classes SHOULD be as specific as possible in communicating the reason for
54       * authentication failure. Recommendations for common cases:
55       * <ul>
56       *     <li>Bad password: {@link javax.security.auth.login.FailedLoginException}</li>
57       *     <li>Expired password: {@link javax.security.auth.login.CredentialExpiredException}</li>
58       *     <li>User account expired: {@link javax.security.auth.login.AccountExpiredException}</li>
59       *     <li>User account locked: {@link javax.security.auth.login.AccountLockedException}</li>
60       *     <li>User account not found: {@link javax.security.auth.login.AccountNotFoundException}</li>
61       *     <li>Time of authentication not allowed: {@link InvalidLoginTimeException}</li>
62       *     <li>Location of authentication not allowed: {@link InvalidLoginLocationException}</li>
63       *     <li>Expired X.509 certificate: {@link java.security.cert.CertificateExpiredException}</li>
64       * </ul>
65       * @throws PreventedException On errors that prevented authentication from occurring. Implementing classes SHOULD
66       * take care to populate the cause, where applicable, with the error that prevented authentication.
67       */
68      HandlerResult authenticate(Credential credential) throws GeneralSecurityException, PreventedException;
69  
70  
71      /**
72       * Determines whether the handler has the capability to authenticate the given credential. In practical terms,
73       * the {@link #authenticate(Credential)} method MUST be capable of processing a given credential if
74       * <code>supports</code> returns true on the same credential.
75       *
76       * @param credential The credential to check.
77       *
78       * @return True if the handler supports the Credential, false otherwise.
79       */
80      boolean supports(Credential credential);
81  
82  
83      /**
84       * Gets a unique name for this authentication handler within the Spring context that contains it.
85       * For implementations that allow setting a unique name, deployers MUST take care to ensure that every
86       * handler instance has a unique name.
87       *
88       * @return Unique name within a Spring context.
89       */
90      String getName();
91  }
