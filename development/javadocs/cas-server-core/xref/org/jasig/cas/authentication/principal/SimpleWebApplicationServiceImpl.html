19  package org.jasig.cas.authentication.principal;
20  
21  import java.util.HashMap;
22  import java.util.Map;
23  
24  import javax.servlet.http.HttpServletRequest;
25  
26  import org.jasig.cas.authentication.principal.Response.ResponseType;
27  import org.springframework.util.StringUtils;
28  
29  /**
30   * Represents a service which wishes to use the CAS protocol.
31   *
32   * @author Scott Battaglia
33   * @since 3.1
34   */
35  public final class SimpleWebApplicationServiceImpl extends AbstractWebApplicationService {
36  
37      private static final String CONST_PARAM_SERVICE = "service";
38  
39      private static final String CONST_PARAM_TARGET_SERVICE = "targetService";
40  
41      private static final String CONST_PARAM_TICKET = "ticket";
42  
43      private static final String CONST_PARAM_METHOD = "method";
44  
45      private final ResponseType responseType;
46  
47      private static final long serialVersionUID = 8334068957483758042L;
48  
49      /**
50       * Instantiates a new simple web application service impl.
51       *
52       * @param id the id
53       */
54      public SimpleWebApplicationServiceImpl(final String id) {
55          this(id, id, null, null);
56      }
57  
58      /**
59       * Instantiates a new simple web application service impl.
60       *
61       * @param id the id
62       * @param originalUrl the original url
63       * @param artifactId the artifact id
64       * @param responseType the response type
65       */
66      private SimpleWebApplicationServiceImpl(final String id,
67          final String originalUrl, final String artifactId,
68          final ResponseType responseType) {
69          super(id, originalUrl, artifactId);
70          this.responseType = responseType;
71      }
72  
73      /**
74       * Creates the service from the request.
75       *
76       * @param request the request
77       * @return the simple web application service impl
78       */
79      public static SimpleWebApplicationServiceImpl createServiceFrom(
80          final HttpServletRequest request) {
81          final String targetService = request.getParameter(CONST_PARAM_TARGET_SERVICE);
82          final String service = request.getParameter(CONST_PARAM_SERVICE);
83          final String serviceAttribute = (String) request.getAttribute(CONST_PARAM_SERVICE);
84          final String method = request.getParameter(CONST_PARAM_METHOD);
85          final String serviceToUse;
86          if (StringUtils.hasText(targetService)) {
87              serviceToUse = targetService;
88          } else if (StringUtils.hasText(service)) {
89              serviceToUse = service;
90          } else {
91              serviceToUse = serviceAttribute;
92          }
93  
94          if (!StringUtils.hasText(serviceToUse)) {
95              return null;
96          }
97  
98          final String id = cleanupUrl(serviceToUse);
99          final String artifactId = request.getParameter(CONST_PARAM_TICKET);
100 
101         return new SimpleWebApplicationServiceImpl(id, serviceToUse,
102             artifactId, "POST".equals(method) ? ResponseType.POST
103                 : ResponseType.REDIRECT);
104     }
105 
106     @Override
107     public Response getResponse(final String ticketId) {
108         final Map<String, String> parameters = new HashMap<String, String>();
109 
110         if (StringUtils.hasText(ticketId)) {
111             parameters.put(CONST_PARAM_TICKET, ticketId);
112         }
113 
114         if (ResponseType.POST == this.responseType) {
115             return Response.getPostResponse(getOriginalUrl(), parameters);
116         }
117         return Response.getRedirectResponse(getOriginalUrl(), parameters);
118     }
