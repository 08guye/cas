20  package org.jasig.cas.authentication.support;
21  
22  import org.apache.commons.lang3.StringUtils;
23  import org.jasig.cas.services.RegisteredService;
24  import org.jasig.cas.services.ServicesManager;
25  import org.jasig.cas.util.services.RegisteredServiceCipherExecutor;
26  import org.jasig.cas.services.web.view.CasViewConstants;
27  import org.springframework.stereotype.Component;
28  
29  import java.util.Map;
30  
31  /**
32   * The default implementation of the attribute
33   * encoder that will use a per-service key-pair
34   * to encrypt the credential password and PGT
35   * when available. All other attributes remain in
36   * place.
37   *
38   * @author Misagh Moayyed
39   * @since 4.1
40   */
41  @Component("casAttributeEncoder")
42  public class DefaultCasAttributeEncoder extends AbstractCasAttributeEncoder {
43  
44      /**
45       * Instantiates a new Default cas attribute encoder.
46       */
47      protected DefaultCasAttributeEncoder() {
48          super();
49      }
50  
51      /**
52       * Instantiates a new Default cas attribute encoder.
53       *
54       * @param servicesManager the services manager
55       */
56      public DefaultCasAttributeEncoder(final ServicesManager servicesManager) {
57          super(servicesManager);
58      }
59  
60      /**
61       * Instantiates a new Default cas attribute encoder.
62       *
63       * @param servicesManager the services manager
64       * @param cipherExecutor the cipher executor
65       */
66      public DefaultCasAttributeEncoder(final ServicesManager servicesManager,
67                                        final RegisteredServiceCipherExecutor cipherExecutor) {
68          super(servicesManager, cipherExecutor);
69      }
70  
71      /**
72       * Encode and encrypt credential password using the public key
73       * supplied by the service. The result is base64 encoded
74       * and put into the attributes collection again, overwriting
75       * the previous value.
76       *
77       * @param attributes the attributes
78       * @param cachedAttributesToEncode the cached attributes to encode
79       * @param cipher the cipher
80       * @param registeredService the registered service
81       */
82      protected final void encodeAndEncryptCredentialPassword(final Map<String, Object> attributes,
83                                                        final Map<String, String> cachedAttributesToEncode,
84                                                        final RegisteredServiceCipherExecutor cipher,
85                                                        final RegisteredService registeredService) {
86          encryptAndEncodeAndPutIntoAttributesMap(attributes, cachedAttributesToEncode,
87                  CasViewConstants.MODEL_ATTRIBUTE_NAME_PRINCIPAL_CREDENTIAL,
88                  cipher, registeredService);
89      }
90  
91      /**
92       * Encode and encrypt pgt.
93       *
94       * @param attributes the attributes
95       * @param cachedAttributesToEncode the cached attributes to encode
96       * @param cipher the cipher
97       * @param registeredService the registered service
98       */
99      protected final void encodeAndEncryptProxyGrantingTicket(final Map<String, Object> attributes,
100                                                        final Map<String, String> cachedAttributesToEncode,
101                                                        final RegisteredServiceCipherExecutor cipher,
102                                                        final RegisteredService registeredService) {
103         encryptAndEncodeAndPutIntoAttributesMap(attributes, cachedAttributesToEncode,
104                 CasViewConstants.MODEL_ATTRIBUTE_NAME_PROXY_GRANTING_TICKET, cipher, registeredService);
105     }
106 
107     /**
108      * Encrypt, encode and put the attribute into attributes map.
109      *
110      * @param attributes the attributes
111      * @param cachedAttributesToEncode the cached attributes to encode
112      * @param cachedAttributeName the cached attribute name
113      * @param cipher the cipher
114      * @param registeredService the registered service
115      */
116     protected final void encryptAndEncodeAndPutIntoAttributesMap(final Map<String, Object> attributes,
117                                                            final Map<String, String> cachedAttributesToEncode,
118                                                            final String cachedAttributeName,
119                                                            final RegisteredServiceCipherExecutor cipher,
120                                                            final RegisteredService registeredService) {
121         final String cachedAttribute = cachedAttributesToEncode.remove(cachedAttributeName);
122         if (StringUtils.isNotBlank(cachedAttribute)) {
123             logger.debug("Retrieved [{}] as a cached model attribute...", cachedAttributeName);
124             final String encodedValue = cipher.encode(cachedAttribute, registeredService);
125             if (StringUtils.isNotBlank(encodedValue)) {
126                 attributes.put(cachedAttributeName, encodedValue);
127                 logger.debug("Encrypted and encoded [{}] as an attribute to [{}].",
128                         cachedAttributeName, encodedValue);
129             }
130         } else {
131             logger.debug("[{}] is not available as a cached model attribute to encrypt...", cachedAttributeName);
132         }
133     }
134 
135     @Override
136     protected void encodeAttributesInternal(final Map<String, Object> attributes,
137                                             final Map<String, String> cachedAttributesToEncode,
138                                             final RegisteredServiceCipherExecutor cipher,
139                                             final RegisteredService registeredService) {
140         encodeAndEncryptCredentialPassword(attributes, cachedAttributesToEncode, cipher, registeredService);
141         encodeAndEncryptProxyGrantingTicket(attributes, cachedAttributesToEncode, cipher, registeredService);
142     }
143 }
