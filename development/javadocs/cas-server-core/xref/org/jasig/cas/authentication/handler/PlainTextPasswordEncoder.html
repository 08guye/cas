21  /**
22   * Default password encoder for the case where no password encoder is needed.
23   * Encoding results in the same password that was passed in.
24   *
25   * @author Scott Battaglia
26  
27   * @since 3.0.0
28   */
29  public final class PlainTextPasswordEncoder implements PasswordEncoder {
30  
31      @Override
32      public String encode(final String password) {
33          return password;
34      }
35  }
