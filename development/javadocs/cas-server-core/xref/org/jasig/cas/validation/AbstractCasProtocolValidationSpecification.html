19  package org.jasig.cas.validation;
20  
21  /**
22   * Base validation specification for the CAS protocol. This specification checks
23   * for the presence of renew=true and if requested, succeeds only if ticket
24   * validation is occurring from a new login.
25   *
26   * @author Scott Battaglia
27   * @since 3.0
28   */
29  public abstract class AbstractCasProtocolValidationSpecification implements ValidationSpecification {
30  
31      /** The default value for the renew attribute is false. */
32      private static final boolean DEFAULT_RENEW = false;
33  
34      /** Denotes whether we should always authenticate or not. */
35      private boolean renew;
36  
37      /**
38       * Instantiates a new abstract cas protocol validation specification.
39       */
40      public AbstractCasProtocolValidationSpecification() {
41          this.renew = DEFAULT_RENEW;
42      }
43  
44      /**
45       * Instantiates a new abstract cas protocol validation specification.
46       *
47       * @param renew the renew
48       */
49      public AbstractCasProtocolValidationSpecification(final boolean renew) {
50          this.renew = renew;
51      }
52  
53      /**
54       * Method to set the renew requirement.
55       *
56       * @param renew The renew value we want.
57       */
58      public final void setRenew(final boolean renew) {
59          this.renew = renew;
60      }
61  
62      /**
63       * Method to determine if we require renew to be true.
64       *
65       * @return true if renew is required, false otherwise.
66       */
67      public final boolean isRenew() {
68          return this.renew;
69      }
70  
71      /* (non-Javadoc)
72       * @see org.jasig.cas.validation.ValidationSpecification#isSatisfiedBy(org.jasig.cas.validation.Assertion)
73       */
74      @Override
75      public final boolean isSatisfiedBy(final Assertion assertion) {
76          return isSatisfiedByInternal(assertion)
77              && ((!this.renew) || (assertion.isFromNewLogin() && this.renew));
78      }
79  
80      /**
81       * Template method to allow for additional checks by subclassed methods
82       * without needing to call super.isSatisfiedBy(...).
83       * @param assertion the assertion
84       * @return true, if the subclass implementation is satisfied by the assertion
85       */
86      protected abstract boolean isSatisfiedByInternal(final Assertion assertion);
87  }
