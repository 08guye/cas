19  package org.jasig.cas.authentication;
20  
21  import org.jasig.cas.authentication.principal.Credentials;
22  import org.jasig.cas.authentication.principal.UsernamePasswordCredentials;
23  
24  /**
25   * Converts a CAS 4.0 username/password credential into a CAS 3.0 username/password credential.
26   *
27   * @author Marvin S. Addison
28   * @since 4.0
29   */
30  public class UsernamePasswordCredentialsAdapter implements CredentialsAdapter {
31      @Override
32      public Credentials convert(final Credential credential) {
33          if (!(credential instanceof UsernamePasswordCredential)) {
34              throw new IllegalArgumentException(credential + " not supported.");
35          }
36          final UsernamePasswordCredential original = (UsernamePasswordCredential) credential;
37          final UsernamePasswordCredentials old = new UsernamePasswordCredentials();
38          old.setUsername(original.getUsername());
39          old.setPassword(original.getPassword());
40          return old;
41      }
42  }
