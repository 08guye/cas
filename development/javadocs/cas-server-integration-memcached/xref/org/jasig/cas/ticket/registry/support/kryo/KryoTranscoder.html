54  /**
55   * {@link net.spy.memcached.MemcachedClient} transcoder implementation based on Kryo fast serialization framework
56   * suited for efficient serialization of tickets.
57   *
58   * @author Marvin S. Addison
59   */
60  public class KryoTranscoder implements Transcoder<Object> {
61  
62      /** Kryo serializer. */
63      private final Kryo kryo = new Kryo();
64  
65      /** Logging instance. */
66      private final Logger logger = LoggerFactory.getLogger(getClass());
67  
68      /** Maximum size of single encoded object in bytes. */
69      private final int bufferSize;
70  
71      /** Map of class to serializer that handles it. */
72      private Map<Class<?>, Serializer> serializerMap;
73  
74  
75      /**
76       * Creates a Kryo-based transcoder.
77       *
78       * @param initialBufferSize Initial size for buffer holding encoded object data.
79       */
80      public KryoTranscoder(final int initialBufferSize) {
81          bufferSize = initialBufferSize;
82      }
83  
84  
85      /**
86       * Sets a map of additional types that should be regisetered with Kryo,
87       * for example GoogleAccountsService and OpenIdService.
88       *
89       * @param map Map of class to the serializer instance that handles it.
90       */
91      public void setSerializerMap(final Map<Class<?>, Serializer> map) {
92          this.serializerMap = map;
93      }
94  
95      public void initialize() {
96          // Register types we know about and do not require external configuration
97          kryo.register(ArrayList.class);
98          kryo.register(BasicCredentialMetaData.class);
99          kryo.register(Class.class, new ClassSerializer(kryo));
100         kryo.register(Date.class, new DateSerializer());
101         kryo.register(HardTimeoutExpirationPolicy.class);
102         kryo.register(HashMap.class);
103         kryo.register(HandlerResult.class);
104         kryo.register(ImmutableAuthentication.class);
105         kryo.register(MultiTimeUseOrTimeoutExpirationPolicy.class);
106         kryo.register(NeverExpiresExpirationPolicy.class);
107         kryo.register(RememberMeDelegatingExpirationPolicy.class);
108         kryo.register(ServiceTicketImpl.class);
109         kryo.register(SimpleWebApplicationServiceImpl.class, new SimpleWebApplicationServiceSerializer(kryo));
110         kryo.register(ThrottledUseAndTimeoutExpirationPolicy.class);
111         kryo.register(TicketGrantingTicketExpirationPolicy.class);
112         kryo.register(TicketGrantingTicketImpl.class);
113         kryo.register(TimeoutExpirationPolicy.class);
114         kryo.register(URL.class, new URLSerializer(kryo));
115 
116         // Register other types
117         if (serializerMap != null) {
118             for (final Class<?> clazz : serializerMap.keySet()) {
119                 kryo.register(clazz, serializerMap.get(clazz));
120             }
121         }
122 
123         // Catchall for any classes not explicitly registered
124         kryo.setRegistrationOptional(true);
125     }

131      * @param d Data to decode.

139     public CachedData encode(final Object o) {
140         final byte[] bytes = encodeToBytes(o);
141         return new CachedData(0, bytes, bytes.length);
142     }
143 
144 
145     public Object decode(final CachedData d) {
146         return kryo.readClassAndObject(ByteBuffer.wrap(d.getData()));
147     }

154      */

156         return CachedData.MAX_SIZE;
157     }

161      * Gets the kryo object that provides encoding and decoding services for this instance.

164      */
165     public Kryo getKryo() {
166         return kryo;
167     }

170     /**
171      * Encodes the given object using registered Kryo serializers.  Provides explicit buffer overflow protection, but
172      * careful buffer sizing should be employed to reduce the need for this facility.
173      *
174      * @param o Object to encode.
175      *
176      * @return Encoded bytes.
177      */
178     private byte[] encodeToBytes(final Object o) {
179         int factor = 1;
180         byte[] result = null;
181         ByteBuffer buffer = Kryo.getContext().getBuffer(bufferSize * factor);
182         while (result == null) {
183             try {
184                 kryo.writeClassAndObject(buffer, o);
185                 result = new byte[buffer.flip().limit()];
186                 buffer.get(result);
187             } catch (final SerializationException e) {
188                 Throwable rootCause = e;
189                 while (rootCause.getCause() != null) {
190                     rootCause = rootCause.getCause();
191                 }
192                 if (rootCause instanceof BufferOverflowException) {
193                     buffer = ByteBuffer.allocate(bufferSize * ++factor);
194                     logger.warn("Buffer overflow while encoding {}", o);
195                 } else {
196                     throw e;
197                 }
198             }
199         }
200         return result;
201     }
