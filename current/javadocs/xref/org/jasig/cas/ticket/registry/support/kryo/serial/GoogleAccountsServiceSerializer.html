30  /**
31   * Serializer for {@link GoogleAccountsService}.
32   *
33   * @author Marvin S. Addison
34   */
35  public final class GoogleAccountsServiceSerializer extends AbstractWebApplicationServiceSerializer<GoogleAccountsService> {
36  
37      private static final Constructor CONSTRUCTOR;
38  
39      private final PrivateKey privateKey;
40      private final PublicKey publicKey;
41      private final String alternateUsername;
42  
43      static {
44          try {
45              CONSTRUCTOR = GoogleAccountsService.class.getDeclaredConstructor(
46                      String.class,
47                      String.class,
48                      String.class,
49                      String.class,
50                      String.class,
51                      PrivateKey.class,
52                      PublicKey.class,
53                      String.class);
54              CONSTRUCTOR.setAccessible(true);
55          } catch (final NoSuchMethodException e) {
56              throw new IllegalStateException("Expected constructor signature not found.", e);
57          }
58      }
59  
60      public GoogleAccountsServiceSerializer(
61              final Kryo kryo,
62              final FieldHelper helper,
63              final PublicKey publicKey,
64              final PrivateKey privateKey,
65              final String alternateUsername) {
66  
67          super(kryo, helper);
68          this.publicKey = publicKey;
69          this.privateKey = privateKey;
70          this.alternateUsername = alternateUsername;
71      }
72  
73      public void write(final ByteBuffer buffer, final GoogleAccountsService service) {
74          super.write(buffer, service);
75          kryo.writeObject(buffer, fieldHelper.getFieldValue(service, "requestId"));
76          kryo.writeObject(buffer, fieldHelper.getFieldValue(service, "relayState"));
77      }
78  
79      protected GoogleAccountsService createService(
80              final ByteBuffer buffer,
81              final String id,
82              final String originalUrl,
83              final String artifactId) {
84  
85          final String requestId = kryo.readObject(buffer, String.class);
86          final String relayState = kryo.readObject(buffer, String.class);
87          try {
88              final GoogleAccountsService service = (GoogleAccountsService) CONSTRUCTOR.newInstance(
89                      id,
90                      originalUrl,
91                      artifactId,
92                      relayState,
93                      requestId,
94                      privateKey,
95                      publicKey,
96                      alternateUsername);
97              return service;
98          } catch (final Exception e) {
99              throw new IllegalStateException("Error creating SamlService", e);
100         }
101     }
