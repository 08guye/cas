42  /**
43   * Performs CRL-based revocation checking by consulting resources defined in
44   * the CRLDistributionPoints extension field on the certificate.  Although RFC
45   * 2459 allows the distribution point name to have arbitrary meaning, this class
46   * expects the name to define an absolute URL, which is the most common
47   * implementation.  This implementation caches CRL resources fetched from remote
48   * URLs to improve performance by avoiding CRL fetching on every revocation
49   * check.
50   *
51   * @author Marvin S. Addison
52   * @since 3.4.6
53   *
54   */
55  public class CRLDistributionPointRevocationChecker extends AbstractCRLRevocationChecker {
56  
57      /** CRL cache. */
58      private final Cache crlCache;
59  
60  
61      /**
62       * Creates a new instance that uses the given cache instance for CRL caching.
63       *
64       * @param crlCache Cache for CRL data.
65       */
66      public CRLDistributionPointRevocationChecker(final Cache crlCache) {
67          if (crlCache == null) {
68              throw new IllegalArgumentException("Cache cannot be null.");
69          }
70          this.crlCache = crlCache;
71      }
72  
73      /**
74       * {@inheritDoc}
75       * @see AbstractCRLRevocationChecker#getCRL(X509Certificate)
76       */
77      @Override
78      protected X509CRL getCRL(final X509Certificate cert) {
79          final URL[] urls = getDistributionPoints(cert);
80          logger.debug(String.format(
81                  "Distribution points for %s: %s.",
82                  CertUtils.toString(cert), Arrays.asList(urls)));
83  
84          Element item;
85          for (URL url : urls) {
86              item = this.crlCache.get(url);
87              if (item != null) {
88                  logger.debug("Found CRL in cache for {}", CertUtils.toString(cert));
89                  return (X509CRL) item.getObjectValue();
90              }
91          }
92  
93          // Try all distribution points and stop at first fetch that succeeds
94          X509CRL crl = null;
95          for (int i = 0; i < urls.length && crl == null; i++) {
96              logger.info("Attempting to fetch CRL at {}", urls[i]);
97              try {
98                  crl = CertUtils.fetchCRL(new UrlResource(urls[i]));
99                  logger.info("Success. Caching fetched CRL.");
100                 this.crlCache.put(new Element(urls[i], crl));
101             } catch (final Exception e) {
102                 logger.error("Error fetching CRL at {}", urls[i], e);
103             }
104         }
105 
106         return crl;
107     }
108 
109     private URL[] getDistributionPoints(final X509Certificate cert) {
110         final DistributionPointList points;
111         try {
112             points = new ExtensionReader(cert).readCRLDistributionPoints();
113         } catch (final Exception e) {
114             logger.error(
115                     "Error reading CRLDistributionPoints extension field on " + CertUtils.toString(cert), e);
116             return new URL[0];
117         }
118 
119         final List<URL> urls = new ArrayList<URL>();
120         for (DistributionPoint point : points.getItems()) {
121             final Object location = point.getDistributionPoint();
122             if (location instanceof String) {
123                 addURL(urls, (String) location);
124             } else if (location instanceof GeneralNameList) {
125                 for (GeneralName gn : ((GeneralNameList) location).getItems()) {
126                     addURL(urls, gn.getName());
127                 }
128             } else {
129                 logger.warn("{} not supported. String or GeneralNameList expected.", location);
130             }
131         }
132 
133         return urls.toArray(new URL[urls.size()]);
134     }
135 
136     private void addURL(final List<URL> list, final String uriString) {
137         try {
138             // Build URI by components to facilitate proper encoding of querystring
139             // e.g. http://example.com:8085/ca?action=crl&issuer=CN=CAS Test User CA
140             final URL url = new URL(uriString);
141             final URI uri = new URI(url.getProtocol(), url.getAuthority(), url.getPath(), url.getQuery(), null);
142             list.add(uri.toURL());
143         } catch (final Exception e) {
144             logger.warn("{} is not a valid distribution point URI.", uriString);
145         }
146     }
