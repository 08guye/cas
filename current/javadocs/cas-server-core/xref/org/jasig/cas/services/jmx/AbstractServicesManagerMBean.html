20  package org.jasig.cas.services.jmx;
21  
22  import org.jasig.cas.services.RegisteredService;
23  import org.jasig.cas.services.RegisteredServiceImpl;
24  import org.jasig.cas.services.ServicesManager;
25  import org.springframework.jmx.export.annotation.ManagedAttribute;
26  import org.springframework.jmx.export.annotation.ManagedOperation;
27  import org.springframework.jmx.export.annotation.ManagedOperationParameter;
28  import org.springframework.util.Assert;
29  
30  import javax.validation.constraints.NotNull;
31  import java.util.ArrayList;
32  import java.util.List;
33  
34  /**
35   * Abstract base class to support both the {@link org.jasig.cas.services.ServicesManager} and the
36   * {@link org.jasig.cas.services.ReloadableServicesManager}.
37   *
38   * @author <a href="mailto:tobias.trelle@proximity.de">Tobias Trelle</a>
39   * @author Scott Battaglia
40   * @since 3.4.4
41   */
42  public abstract class AbstractServicesManagerMBean<T extends ServicesManager> {
43  
44      @NotNull
45      private T servicesManager;
46  
47      protected AbstractServicesManagerMBean(final T svcMgr) {
48          this.servicesManager = svcMgr;
49      }
50  
51      protected final T getServicesManager() {
52          return this.servicesManager;
53      }
54  
55      @ManagedAttribute(description = "Retrieves the list of Registered Services in a slightly friendlier output.")
56      public final List<String> getRegisteredServicesAsStrings() {
57          final List<String> services = new ArrayList<String>();
58  
59          for (final RegisteredService r : this.servicesManager.getAllServices()) {
60          services.add(new StringBuilder().append("id: ").append(r.getId())
61                  .append(" name: ").append(r.getName())
62                  .append(" enabled: ").append(r.isEnabled())
63                  .append(" ssoEnabled: ").append(r.isSsoEnabled())
64                  .append(" serviceId: ").append(r.getServiceId())
65                  .toString());
66          }
67  
68          return services;
69      }
70  
71      @ManagedOperation(description = "Can remove a service based on its identifier.")
72      @ManagedOperationParameter(name="id", description = "the identifier to remove")
73      public final RegisteredService removeService(final long id) {
74          return this.servicesManager.delete(id);
75      }
76  
77      @ManagedOperation(description = "Disable a service by id.")
78      @ManagedOperationParameter(name="id", description = "the identifier to disable")
79      public final void disableService(final long id) {
80          changeEnabledState(id, false);
81      }
82  
83      @ManagedOperation(description = "Enable a service by its id.")
84      @ManagedOperationParameter(name="id", description = "the identifier to enable.")
85      public final void enableService(final long id) {
86          changeEnabledState(id, true);
87      }
88  
89      private void changeEnabledState(final long id, final boolean newState) {
90          final RegisteredService r = this.servicesManager.findServiceBy(id);
91          Assert.notNull(r, "invalid RegisteredService id");
92  
93          // we screwed up our APIs in older versions of CAS, so we need to CAST this to do anything useful.
94          ((RegisteredServiceImpl) r).setEnabled(newState);
95          this.servicesManager.save(r);
96      }
97  }
