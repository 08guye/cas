19  package org.jasig.cas.adaptors.jdbc;
20  
21  import java.security.GeneralSecurityException;
22  
23  import org.jasig.cas.authentication.HandlerResult;
24  import org.jasig.cas.authentication.PreventedException;
25  import org.jasig.cas.authentication.UsernamePasswordCredential;
26  import org.jasig.cas.authentication.principal.SimplePrincipal;
27  import org.springframework.dao.DataAccessException;
28  import org.springframework.dao.IncorrectResultSizeDataAccessException;
29  
30  import javax.security.auth.login.AccountNotFoundException;
31  import javax.security.auth.login.FailedLoginException;
32  import javax.validation.constraints.NotNull;
33  
34  /**
35   * Class that if provided a query that returns a password (parameter of query
36   * must be username) will compare that password to a translated version of the
37   * password provided by the user. If they match, then authentication succeeds.
38   * Default password translator is plaintext translator.
39   *
40   * @author Scott Battaglia
41   * @author Dmitriy Kopylenko
42   * @author Marvin S. Addison
43   *
44   * @since 3.0
45   */
46  public class QueryDatabaseAuthenticationHandler extends AbstractJdbcUsernamePasswordAuthenticationHandler {
47  
48      @NotNull
49      private String sql;
50  
51      /** {@inheritDoc} */
52      @Override
53      protected final HandlerResult authenticateUsernamePasswordInternal(final UsernamePasswordCredential credential)
54              throws GeneralSecurityException, PreventedException {
55  
56          final String username = credential.getUsername();
57          final String encryptedPassword = this.getPasswordEncoder().encode(credential.getPassword());
58          try {
59              final String dbPassword = getJdbcTemplate().queryForObject(this.sql, String.class, username);
60              if (!dbPassword.equals(encryptedPassword)) {
61                  throw new FailedLoginException("Password does not match value on record.");
62              }
63          } catch (final IncorrectResultSizeDataAccessException e) {
64              if (e.getActualSize() == 0) {
65                  throw new AccountNotFoundException(username + " not found with SQL query");
66              } else {
67                  throw new FailedLoginException("Multiple records found for " + username);
68              }
69          } catch (final DataAccessException e) {
70              throw new PreventedException("SQL exception while executing query for " + username, e);
71          }
72          return createHandlerResult(credential, new SimplePrincipal(username), null);
73      }
74  
75      /**
76       * @param sql The sql to set.
77       */
78      public void setSql(final String sql) {
79          this.sql = sql;
80      }
81  }
