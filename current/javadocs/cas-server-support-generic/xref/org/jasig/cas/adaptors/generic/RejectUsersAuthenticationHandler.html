19  package org.jasig.cas.adaptors.generic;
20  
21  import java.security.GeneralSecurityException;
22  import java.util.List;
23  
24  import org.jasig.cas.authentication.PreventedException;
25  import org.jasig.cas.authentication.handler.support.AbstractUsernamePasswordAuthenticationHandler;
26  import org.jasig.cas.authentication.principal.Principal;
27  import org.jasig.cas.authentication.principal.SimplePrincipal;
28  
29  import javax.security.auth.login.FailedLoginException;
30  import javax.validation.constraints.NotNull;
31  
32  /**
33   * AuthenticationHandler which fails to authenticate a user purporting to be one
34   * of the blocked usernames, and blindly authenticates all other users.
35   * <p>
36   * Note that RejectUsersAuthenticationHandler throws an exception when the user
37   * is found in the map. This is done to indicate that this is an extreme case
38   * and any AuthenticationManager checking the RejectUsersAuthenticationHandler
39   * should not continue checking other Authentication Handlers on the failure of
40   * RejectUsersAuthenticationHandler to authenticate someone.
41   *
42   * @author Scott Battaglia
43   * @since 3.0
44   */
45  public class RejectUsersAuthenticationHandler extends AbstractUsernamePasswordAuthenticationHandler {
46  
47      /** The collection of users to reject. */
48      @NotNull
49      private List<String> users;
50  
51      protected final Principal authenticateUsernamePasswordInternal(final String username, final String password)
52              throws GeneralSecurityException, PreventedException {
53  
54          if (this.users.contains(username)) {
55              throw new FailedLoginException();
56          }
57  
58          return new SimplePrincipal(username);
59      }
60  
61      /**
62       * Set the Collection of usernames which we will fail to authenticate.
63       *
64       * @param users The Collection of usernames we should not authenticate.
65       */
66      public final void setUsers(final List<String> users) {
67          this.users = users;
68      }
69  }
